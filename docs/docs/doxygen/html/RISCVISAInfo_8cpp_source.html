138 // This function finds the first character that doesn't belong to a version
139 // (e.g. zba1p0 is extension 'zba' of version '1p0'). So the function will
140 // consume [0-9]*p[0-9]* starting from the backward. An extension name will not
141 // end with a digit or the letter 'p', so this function will parse correctly.
142 // NOTE: This function is NOT able to take empty strings or strings that only
143 // have version numbers and no extension name. It assumes the extension name
144 // will be at least more than one character.
145 static size_t findFirstNonVersionCharacter(StringRef Ext) {
146  assert(!Ext.empty() &&
147  "Already guarded by if-statement in ::parseArchString");
148 
149  int Pos = Ext.size() - 1;
150  while (Pos > 0 && isDigit(Ext[Pos]))
151  Pos--;
152  if (Pos > 0 && Ext[Pos] == 'p' && isDigit(Ext[Pos - 1])) {
153  Pos--;
154  while (Pos > 0 && isDigit(Ext[Pos]))
155  Pos--;
156  }
157  return Pos;
158 }

170 static std::optional<RISCVExtensionVersion>
171 findDefaultVersion(StringRef ExtName) {
172  // Find default version of an extension.
173  // TODO: We might set default version based on profile or ISA spec.
174  for (auto &ExtInfo : {ArrayRef(SupportedExtensions),
175  ArrayRef(SupportedExperimentalExtensions)}) {
176  auto ExtensionInfoIterator = llvm::find_if(ExtInfo, FindByName(ExtName));
177 
178  if (ExtensionInfoIterator == ExtInfo.end()) {
179  continue;
180  }
181  return ExtensionInfoIterator->Version;
182  }
183  return std::nullopt;

245  auto FindByNameAndVersion = [=](const RISCVSupportedExtension &ExtInfo) {
246  return ExtInfo.Name == Ext && (MajorVersion == ExtInfo.Version.Major) &&

262 // Get the rank for single-letter extension, lower value meaning higher
263 // priority.
264 static int singleLetterExtensionRank(char Ext) {
265  switch (Ext) {
266  case 'i':
267  return -2;
268  case 'e':
269  return -1;
270  default:
271  break;
272  }
273 
274  size_t Pos = AllStdExts.find(Ext);
275  int Rank;
276  if (Pos == StringRef::npos)
277  // If we got an unknown extension letter, then give it an alphabetical
278  // order, but after all known standard extensions.
279  Rank = AllStdExts.size() + (Ext - 'a');
280  else
281  Rank = Pos;
282 
283  return Rank;
284 }

286 // Get the rank for multi-letter extension, lower value meaning higher
287 // priority/order in canonical order.
288 static int multiLetterExtensionRank(const std::string &ExtName) {
289  assert(ExtName.length() >= 2);
290  int HighOrder;
291  int LowOrder = 0;
292  // The order between multi-char extensions: s -> h -> z -> x.
293  char ExtClass = ExtName[0];
294  switch (ExtClass) {
295  case 's':
296  HighOrder = 0;
297  break;
298  case 'z':
299  HighOrder = 1;
300  // `z` extension must be sorted by canonical order of second letter.
301  // e.g. zmx has higher rank than zax.
302  LowOrder = singleLetterExtensionRank(ExtName[1]);
303  break;
304  case 'x':
305  HighOrder = 2;
306  break;
307  default:
308  llvm_unreachable("Unknown prefix for multi-char extension");
309  return -1;
310  }
311 
312  return (HighOrder << 8) + LowOrder;
313 }

315 // Compare function for extension.
316 // Only compare the extension name, ignore version comparison.
317 bool RISCVISAInfo::compareExtension(const std::string &LHS,
318  const std::string &RHS) {
319  size_t LHSLen = LHS.length();
320  size_t RHSLen = RHS.length();
321  if (LHSLen == 1 && RHSLen != 1)
322  return true;
323 
324  if (LHSLen != 1 && RHSLen == 1)
325  return false;
326 
327  if (LHSLen == 1 && RHSLen == 1)
328  return singleLetterExtensionRank(LHS[0]) <
329  singleLetterExtensionRank(RHS[0]);
330 
331  // Both are multi-char ext here.
332  int LHSRank = multiLetterExtensionRank(LHS);
333  int RHSRank = multiLetterExtensionRank(RHS);
334  if (LHSRank != RHSRank)
335  return LHSRank < RHSRank;
336 
337  // If the rank is same, it must be sorted by lexicographic order.
338  return LHS < RHS;
339 }

341 void RISCVISAInfo::toFeatures(
342  std::vector<StringRef> &Features,
343  llvm::function_ref<StringRef(const Twine &)> StrAlloc,
344  bool AddAllExtensions) const {
345  for (auto const &Ext : Exts) {
346  StringRef ExtName = Ext.first;
347 
348  if (ExtName == "i")
349  continue;
350 
351  if (isExperimentalExtension(ExtName)) {
352  Features.push_back(StrAlloc("+experimental-" + ExtName));
353  } else {
354  Features.push_back(StrAlloc("+" + ExtName));
355  }
356  }
357  if (AddAllExtensions) {
358  for (const RISCVSupportedExtension &Ext : SupportedExtensions) {
359  if (Exts.count(Ext.Name))
360  continue;
361  Features.push_back(StrAlloc(Twine("-") + Ext.Name));
362  }
363 
364  for (const RISCVSupportedExtension &Ext : SupportedExperimentalExtensions) {
365  if (Exts.count(Ext.Name))
366  continue;
367  Features.push_back(StrAlloc(Twine("-experimental-") + Ext.Name));
368  }
369  }
370 }

372 // Extensions may have a version number, and may be separated by
373 // an underscore '_' e.g.: rv32i2_m2.
374 // Version number is divided into major and minor version numbers,
375 // separated by a 'p'. If the minor version is 0 then 'p0' can be
376 // omitted from the version string. E.g., rv32i2p0, rv32i2, rv32i2p1.
377 static Error getExtensionVersion(StringRef Ext, StringRef In, unsigned &Major,
378  unsigned &Minor, unsigned &ConsumeLength,
379  bool EnableExperimentalExtension,
380  bool ExperimentalExtensionVersionCheck) {
381  StringRef MajorStr, MinorStr;
382  Major = 0;
383  Minor = 0;
384  ConsumeLength = 0;
385  MajorStr = In.take_while(isDigit);
386  In = In.substr(MajorStr.size());
387 
388  if (!MajorStr.empty() && In.consume_front("p")) {
389  MinorStr = In.take_while(isDigit);
390  In = In.substr(MajorStr.size() + MinorStr.size() - 1);
391 
392  // Expected 'p' to be followed by minor version number.
393  if (MinorStr.empty()) {
394  return createStringError(
395  errc::invalid_argument,
396  "minor version number missing after 'p' for extension '" + Ext + "'");
397  }
398  }
399 
400  if (!MajorStr.empty() && MajorStr.getAsInteger(10, Major))
401  return createStringError(
402  errc::invalid_argument,
403  "Failed to parse major version number for extension '" + Ext + "'");
404 
405  if (!MinorStr.empty() && MinorStr.getAsInteger(10, Minor))
406  return createStringError(
407  errc::invalid_argument,
408  "Failed to parse minor version number for extension '" + Ext + "'");
409 
410  ConsumeLength = MajorStr.size();
411 
412  if (!MinorStr.empty())
413  ConsumeLength += MinorStr.size() + 1 /*'p'*/;
414 
415  // Expected multi-character extension with version number to have no
416  // subsequent characters (i.e. must either end string or be followed by
417  // an underscore).
418  if (Ext.size() > 1 && In.size()) {
419  std::string Error =
420  "multi-character extensions must be separated by underscores";
421  return createStringError(errc::invalid_argument, Error);
422  }
423 
424  // If experimental extension, require use of current version number number
425  if (auto ExperimentalExtension = isExperimentalExtension(Ext)) {
426  if (!EnableExperimentalExtension) {
427  std::string Error = "requires '-menable-experimental-extensions' for "
428  "experimental extension '" +
429  Ext.str() + "'";
430  return createStringError(errc::invalid_argument, Error);
431  }
432 
433  if (ExperimentalExtensionVersionCheck &&
434  (MajorStr.empty() && MinorStr.empty())) {
435  std::string Error =
436  "experimental extension requires explicit version number `" +
437  Ext.str() + "`";
438  return createStringError(errc::invalid_argument, Error);
439  }
440 
441  auto SupportedVers = *ExperimentalExtension;
442  if (ExperimentalExtensionVersionCheck &&
443  (Major != SupportedVers.Major || Minor != SupportedVers.Minor)) {
444  std::string Error = "unsupported version number " + MajorStr.str();
445  if (!MinorStr.empty())
446  Error += "." + MinorStr.str();
447  Error += " for experimental extension '" + Ext.str() +
448  "' (this compiler supports " + utostr(SupportedVers.Major) +
449  "." + utostr(SupportedVers.Minor) + ")";
450  return createStringError(errc::invalid_argument, Error);
451  }
452  return Error::success();
453  }
454 
455  // Exception rule for `g`, we don't have clear version scheme for that on
456  // ISA spec.
457  if (Ext == "g")
458  return Error::success();
459 
460  if (MajorStr.empty() && MinorStr.empty()) {
461  if (auto DefaultVersion = findDefaultVersion(Ext)) {
462  Major = DefaultVersion->Major;
463  Minor = DefaultVersion->Minor;
464  }
465  // No matter found or not, return success, assume other place will
466  // verify.
467  return Error::success();
468  }
469 
470  if (RISCVISAInfo::isSupportedExtension(Ext, Major, Minor))
471  return Error::success();
472 
473  std::string Error = "unsupported version number " + std::string(MajorStr);
474  if (!MinorStr.empty())
475  Error += "." + MinorStr.str();
476  Error += " for extension '" + Ext.str() + "'";
477  return createStringError(errc::invalid_argument, Error);
478 }

480 llvm::Expected<std::unique_ptr<RISCVISAInfo>>
481 RISCVISAInfo::parseFeatures(unsigned XLen,
482  const std::vector<std::string> &Features) {
483  assert(XLen == 32 || XLen == 64);
484  std::unique_ptr<RISCVISAInfo> ISAInfo(new RISCVISAInfo(XLen));
485 
486  for (auto &Feature : Features) {
487  StringRef ExtName = Feature;
488  bool Experimental = false;
489  assert(ExtName.size() > 1 && (ExtName[0] == '+' || ExtName[0] == '-'));
490  bool Add = ExtName[0] == '+';
491  ExtName = ExtName.drop_front(1); // Drop '+' or '-'
492  Experimental = stripExperimentalPrefix(ExtName);
493  auto ExtensionInfos = Experimental
494  ? ArrayRef(SupportedExperimentalExtensions)
495  : ArrayRef(SupportedExtensions);
496  auto ExtensionInfoIterator =
497  llvm::find_if(ExtensionInfos, FindByName(ExtName));
498 
499  // Not all features is related to ISA extension, like `relax` or
500  // `save-restore`, skip those feature.
501  if (ExtensionInfoIterator == ExtensionInfos.end())
502  continue;
503 
504  if (Add)
505  ISAInfo->addExtension(ExtName, ExtensionInfoIterator->Version.Major,
506  ExtensionInfoIterator->Version.Minor);
507  else
508  ISAInfo->Exts.erase(ExtName.str());
509  }
510 
511  return RISCVISAInfo::postProcessAndChecking(std::move(ISAInfo));
512 }

514 llvm::Expected<std::unique_ptr<RISCVISAInfo>>
515 RISCVISAInfo::parseArchString(StringRef Arch, bool EnableExperimentalExtension,
516  bool ExperimentalExtensionVersionCheck,
517  bool IgnoreUnknown) {
518  // RISC-V ISA strings must be lowercase.
519  if (llvm::any_of(Arch, isupper)) {
520  return createStringError(errc::invalid_argument,
521  "string must be lowercase");
522  }
523 
524  bool HasRV64 = Arch.startswith("rv64");
525  // ISA string must begin with rv32 or rv64.
526  if (!(Arch.startswith("rv32") || HasRV64) || (Arch.size() < 5)) {
527  return createStringError(errc::invalid_argument,
528  "string must begin with rv32{i,e,g} or rv64{i,g}");
529  }
530 
531  unsigned XLen = HasRV64 ? 64 : 32;
532  std::unique_ptr<RISCVISAInfo> ISAInfo(new RISCVISAInfo(XLen));
533 
534  // The canonical order specified in ISA manual.
535  // Ref: Table 22.1 in RISC-V User-Level ISA V2.2
536  StringRef StdExts = AllStdExts;
537  char Baseline = Arch[4];
538 
539  // First letter should be 'e', 'i' or 'g'.
540  switch (Baseline) {
541  default:
542  return createStringError(errc::invalid_argument,
543  "first letter should be 'e', 'i' or 'g'");
544  case 'e': {
545  // Extension 'e' is not allowed in rv64.
546  if (HasRV64)
547  return createStringError(
548  errc::invalid_argument,
549  "standard user-level extension 'e' requires 'rv32'");
550  break;
551  }
552  case 'i':
553  break;
554  case 'g':
555  // g = imafd
556  StdExts = StdExts.drop_front(4);
557  break;
558  }
559 
560  // Skip rvxxx
561  StringRef Exts = Arch.substr(5);
562 
563  // Remove multi-letter standard extensions, non-standard extensions and
564  // supervisor-level extensions. They have 'z', 'x', 's', 'sx' prefixes.
565  // Parse them at the end.
566  // Find the very first occurrence of 's', 'x' or 'z'.
567  StringRef OtherExts;
568  size_t Pos = Exts.find_first_of("zsx");
569  if (Pos != StringRef::npos) {
570  OtherExts = Exts.substr(Pos);
571  Exts = Exts.substr(0, Pos);
572  }
573 
574  unsigned Major, Minor, ConsumeLength;
575  if (auto E = getExtensionVersion(std::string(1, Baseline), Exts, Major, Minor,
576  ConsumeLength, EnableExperimentalExtension,
577  ExperimentalExtensionVersionCheck))
578  return std::move(E);
579 
580  if (Baseline == 'g') {
581  // No matter which version is given to `g`, we always set imafd to default
582  // version since the we don't have clear version scheme for that on
583  // ISA spec.
584  for (const auto *Ext : {"i", "m", "a", "f", "d"})
585  if (auto Version = findDefaultVersion(Ext))
586  ISAInfo->addExtension(Ext, Version->Major, Version->Minor);
587  else
588  llvm_unreachable("Default extension version not found?");
589  } else
590  // Baseline is `i` or `e`
591  ISAInfo->addExtension(std::string(1, Baseline), Major, Minor);
592 
593  // Consume the base ISA version number and any '_' between rvxxx and the
594  // first extension
595  Exts = Exts.drop_front(ConsumeLength);
596  Exts.consume_front("_");
597 
598  // TODO: Use version number when setting target features
599 
600  auto StdExtsItr = StdExts.begin();
601  auto StdExtsEnd = StdExts.end();
602  auto GoToNextExt = [](StringRef::iterator &I, unsigned ConsumeLength) {
603  I += 1 + ConsumeLength;
604  if (*I == '_')
605  ++I;
606  };
607  for (auto I = Exts.begin(), E = Exts.end(); I != E;) {
608  char C = *I;
609 
610  // Check ISA extensions are specified in the canonical order.
611  while (StdExtsItr != StdExtsEnd && *StdExtsItr != C)
612  ++StdExtsItr;
613 
614  if (StdExtsItr == StdExtsEnd) {
615  // Either c contains a valid extension but it was not given in
616  // canonical order or it is an invalid extension.
617  if (StdExts.contains(C)) {
618  return createStringError(
619  errc::invalid_argument,
620  "standard user-level extension not given in canonical order '%c'",
621  C);
622  }
623 
624  return createStringError(errc::invalid_argument,
625  "invalid standard user-level extension '%c'", C);
626  }
627 
628  // Move to next char to prevent repeated letter.
629  ++StdExtsItr;
630 
631  std::string Next;
632  unsigned Major, Minor, ConsumeLength;
633  if (std::next(I) != E)
634  Next = std::string(std::next(I), E);
635  if (auto E = getExtensionVersion(std::string(1, C), Next, Major, Minor,
636  ConsumeLength, EnableExperimentalExtension,
637  ExperimentalExtensionVersionCheck)) {
638  if (IgnoreUnknown) {
639  consumeError(std::move(E));
640  GoToNextExt(I, ConsumeLength);
641  continue;
642  }
643  return std::move(E);
644  }
645 
646  // The order is OK, then push it into features.
647  // TODO: Use version number when setting target features
648  // Currently LLVM supports only "mafdcvh".
649  if (!isSupportedExtension(StringRef(&C, 1))) {
650  if (IgnoreUnknown) {
651  GoToNextExt(I, ConsumeLength);
652  continue;
653  }
654  return createStringError(errc::invalid_argument,
655  "unsupported standard user-level extension '%c'",
656  C);
657  }
658  ISAInfo->addExtension(std::string(1, C), Major, Minor);
659 
660  // Consume full extension name and version, including any optional '_'
661  // between this extension and the next
662  GoToNextExt(I, ConsumeLength);
663  }
664 
665  // Handle other types of extensions other than the standard
666  // general purpose and standard user-level extensions.
667  // Parse the ISA string containing non-standard user-level
668  // extensions, standard supervisor-level extensions and
669  // non-standard supervisor-level extensions.
670  // These extensions start with 'z', 'x', 's', 'sx' prefixes, follow a
671  // canonical order, might have a version number (major, minor)
672  // and are separated by a single underscore '_'.
673  // Set the hardware features for the extensions that are supported.
674 
675  // Multi-letter extensions are seperated by a single underscore
676  // as described in RISC-V User-Level ISA V2.2.
677  SmallVector<StringRef, 8> Split;
678  OtherExts.split(Split, '_');
679 
680  SmallVector<StringRef, 8> AllExts;
681  std::array<StringRef, 4> Prefix{"z", "x", "s", "sx"};
682  auto I = Prefix.begin();
683  auto E = Prefix.end();
684  if (Split.size() > 1 || Split[0] != "") {
685  for (StringRef Ext : Split) {
686  if (Ext.empty())
687  return createStringError(errc::invalid_argument,
688  "extension name missing after separator '_'");
689 
690  StringRef Type = getExtensionType(Ext);
691  StringRef Desc = getExtensionTypeDesc(Ext);
692  auto Pos = findFirstNonVersionCharacter(Ext) + 1;
693  StringRef Name(Ext.substr(0, Pos));
694  StringRef Vers(Ext.substr(Pos));
695 
696  if (Type.empty()) {
697  if (IgnoreUnknown)
698  continue;
699  return createStringError(errc::invalid_argument,
700  "invalid extension prefix '" + Ext + "'");
701  }
702 
703  // Check ISA extensions are specified in the canonical order.
704  while (I != E && *I != Type)
705  ++I;
706 
707  if (I == E) {
708  if (IgnoreUnknown)
709  continue;
710  return createStringError(errc::invalid_argument,
711  "%s not given in canonical order '%s'",
712  Desc.str().c_str(), Ext.str().c_str());
713  }
714 
715  if (!IgnoreUnknown && Name.size() == Type.size()) {
716  return createStringError(errc::invalid_argument,
717  "%s name missing after '%s'",
718  Desc.str().c_str(), Type.str().c_str());
719  }
720 
721  unsigned Major, Minor, ConsumeLength;
722  if (auto E = getExtensionVersion(Name, Vers, Major, Minor, ConsumeLength,
723  EnableExperimentalExtension,
724  ExperimentalExtensionVersionCheck)) {
725  if (IgnoreUnknown) {
726  consumeError(std::move(E));
727  continue;
728  }
729  return std::move(E);
730  }
731 
732  // Check if duplicated extension.
733  if (!IgnoreUnknown && llvm::is_contained(AllExts, Name)) {
734  return createStringError(errc::invalid_argument, "duplicated %s '%s'",
735  Desc.str().c_str(), Name.str().c_str());
736  }
737 
738  ISAInfo->addExtension(Name, Major, Minor);
739  // Extension format is correct, keep parsing the extensions.
740  // TODO: Save Type, Name, Major, Minor to avoid parsing them later.
741  AllExts.push_back(Name);
742  }
743  }
744 
745  for (auto Ext : AllExts) {
746  if (!isSupportedExtension(Ext)) {
747  StringRef Desc = getExtensionTypeDesc(getExtensionType(Ext));
748  return createStringError(errc::invalid_argument, "unsupported %s '%s'",
749  Desc.str().c_str(), Ext.str().c_str());
750  }
751  }
752 
753  return RISCVISAInfo::postProcessAndChecking(std::move(ISAInfo));
754 }

756 Error RISCVISAInfo::checkDependency() {
757  bool IsRv32 = XLen == 32;
758  bool HasE = Exts.count("e") != 0;
759  bool HasD = Exts.count("d") != 0;
760  bool HasF = Exts.count("f") != 0;
761  bool HasZfinx = Exts.count("zfinx") != 0;
762  bool HasZdinx = Exts.count("zdinx") != 0;
763  bool HasVector = Exts.count("zve32x") != 0;
764  bool HasZve32f = Exts.count("zve32f") != 0;
765  bool HasZve64d = Exts.count("zve64d") != 0;
766  bool HasZvl = MinVLen != 0;
767 
768  if (HasE && !IsRv32)
769  return createStringError(
770  errc::invalid_argument,
771  "standard user-level extension 'e' requires 'rv32'");
772 
773  // It's illegal to specify the 'd' (double-precision floating point)
774  // extension without also specifying the 'f' (single precision
775  // floating-point) extension.
776  // TODO: This has been removed in later specs, which specify that D implies F
777  if (HasD && !HasF)
778  return createStringError(errc::invalid_argument,
779  "d requires f extension to also be specified");
780 
781  if (HasZve32f && !HasF && !HasZfinx)
782  return createStringError(
783  errc::invalid_argument,
784  "zve32f requires f or zfinx extension to also be specified");
785 
786  if (HasZve64d && !HasD && !HasZdinx)
787  return createStringError(
788  errc::invalid_argument,
789  "zve64d requires d or zdinx extension to also be specified");
790 
791  if (Exts.count("zvfh") && !Exts.count("zfh") && !Exts.count("zfhmin") &&
792  !Exts.count("zhinx") && !Exts.count("zhinxmin"))
793  return createStringError(
794  errc::invalid_argument,
795  "zvfh requires zfh, zfhmin, zhinx or zhinxmin extension to also be "
796  "specified");
797 
798  if (HasZvl && !HasVector)
799  return createStringError(
800  errc::invalid_argument,
801  "zvl*b requires v or zve* extension to also be specified");
802 
803  // Additional dependency checks.
804  // TODO: The 'q' extension requires rv64.
805  // TODO: It is illegal to specify 'e' extensions with 'f' and 'd'.
806 
807  return Error::success();
808 }

883 void RISCVISAInfo::updateImplication() {
884  bool HasE = Exts.count("e") != 0;
885  bool HasI = Exts.count("i") != 0;
886 
887  // If not in e extension and i extension does not exist, i extension is
888  // implied
889  if (!HasE && !HasI) {
890  auto Version = findDefaultVersion("i");
891  addExtension("i", Version->Major, Version->Minor);
892  }
893 
894  assert(llvm::is_sorted(ImpliedExts) && "Table not sorted by Name");
895 
896  // This loop may execute over 1 iteration since implication can be layered
897  // Exits loop if no more implication is applied
898  SmallSetVector<StringRef, 16> WorkList;
899  for (auto const &Ext : Exts)
900  WorkList.insert(Ext.first);
901 
902  while (!WorkList.empty()) {
903  StringRef ExtName = WorkList.pop_back_val();
904  auto I = llvm::lower_bound(ImpliedExts, ExtName);
905  if (I != std::end(ImpliedExts) && I->Name == ExtName) {
906  for (const char *ImpliedExt : I->Exts) {
907  if (WorkList.count(ImpliedExt))
908  continue;
909  if (Exts.count(ImpliedExt))
910  continue;
911  auto Version = findDefaultVersion(ImpliedExt);
912  addExtension(ImpliedExt, Version->Major, Version->Minor);
913  WorkList.insert(ImpliedExt);
914  }
915  }
916  }
917 }
