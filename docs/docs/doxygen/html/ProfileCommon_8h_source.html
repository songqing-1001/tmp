100 /// This is called when a count is seen in the profile.
101 void ProfileSummaryBuilder::addCount(uint64_t Count) {
102  TotalCount += Count;
103  if (Count > MaxCount)
104  MaxCount = Count;
105  NumCounts++;
106  CountFrequencies[Count]++;
107 }
