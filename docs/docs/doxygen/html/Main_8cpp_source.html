76 /// Create a dependency file for `-d` option.
77 ///
78 /// This functionality is really only for the benefit of the build system.
79 /// It is similar to GCC's `-M*` family of options.
80 static int createDependencyFile(const TGParser &Parser, const char *argv0) {
81  if (OutputFilename == "-")
82  return reportError(argv0, "the option -d must be used together with -o\n");
83 
84  std::error_code EC;
85  ToolOutputFile DepOut(DependFilename, EC, sys::fs::OF_Text);
86  if (EC)
87  return reportError(argv0, "error opening " + DependFilename + ":" +
88  EC.message() + "\n");
89  DepOut.os() << OutputFilename << ":";
90  for (const auto &Dep : Parser.getDependencies()) {
91  DepOut.os() << ' ' << Dep;
92  }
93  DepOut.os() << "\n";
94  DepOut.keep();
95  return 0;
96 }

98 int llvm::TableGenMain(const char *argv0, TableGenMainFn *MainFn) {
99  RecordKeeper Records;
100 
101  if (TimePhases)
102  Records.startPhaseTiming();
103 
104  // Parse the input file.
105 
106  Records.startTimer("Parse, build records");
107  ErrorOr<std::unique_ptr<MemoryBuffer>> FileOrErr =
108  MemoryBuffer::getFileOrSTDIN(InputFilename, /*IsText=*/true);
109  if (std::error_code EC = FileOrErr.getError())
110  return reportError(argv0, "Could not open input file '" + InputFilename +
111  "': " + EC.message() + "\n");
112 
113  Records.saveInputFilename(InputFilename);
114 
115  // Tell SrcMgr about this buffer, which is what TGParser will pick up.
116  SrcMgr.AddNewSourceBuffer(std::move(*FileOrErr), SMLoc());
117 
118  // Record the location of the include directory so that the lexer can find
119  // it later.
120  SrcMgr.setIncludeDirs(IncludeDirs);
121 
122  TGParser Parser(SrcMgr, MacroNames, Records, NoWarnOnUnusedTemplateArgs);
123 
124  if (Parser.ParseFile())
125  return 1;
126  Records.stopTimer();
127 
128  // Write output to memory.
129  Records.startBackendTimer("Backend overall");
130  std::string OutString;
131  raw_string_ostream Out(OutString);
132  unsigned status = MainFn(Out, Records);
133  Records.stopBackendTimer();
134  if (status)
135  return 1;
136 
137  // Always write the depfile, even if the main output hasn't changed.
138  // If it's missing, Ninja considers the output dirty. If this was below
139  // the early exit below and someone deleted the .inc.d file but not the .inc
140  // file, tablegen would never write the depfile.
141  if (!DependFilename.empty()) {
142  if (int Ret = createDependencyFile(Parser, argv0))
143  return Ret;
144  }
145 
146  Records.startTimer("Write output");
147  bool WriteFile = true;
148  if (WriteIfChanged) {
149  // Only updates the real output file if there are any differences.
150  // This prevents recompilation of all the files depending on it if there
151  // aren't any.
152  if (auto ExistingOrErr =
153  MemoryBuffer::getFile(OutputFilename, /*IsText=*/true))
154  if (std::move(ExistingOrErr.get())->getBuffer() == Out.str())
155  WriteFile = false;
156  }
157  if (WriteFile) {
158  std::error_code EC;
159  ToolOutputFile OutFile(OutputFilename, EC, sys::fs::OF_Text);
160  if (EC)
161  return reportError(argv0, "error opening " + OutputFilename + ": " +
162  EC.message() + "\n");
163  OutFile.os() << Out.str();
164  if (ErrorsPrinted == 0)
165  OutFile.keep();
166  }
167 
168  Records.stopTimer();
169  Records.stopPhaseTiming();
170 
171  if (ErrorsPrinted > 0)
172  return reportError(argv0, Twine(ErrorsPrinted) + " errors.\n");
173  return 0;
174 }
