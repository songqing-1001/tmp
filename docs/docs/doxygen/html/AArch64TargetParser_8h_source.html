259 // Information about a specific architecture, e.g. V8.1-A
260 struct ArchInfo {
261  VersionTuple Version; // Architecture version, major + minor.
262  ArchProfile Profile; // Architecuture profile
263  StringRef Name; // Human readable name, e.g. "armv8.1-a"
264  StringRef ArchFeature; // Command line feature flag, e.g. +v8a
265  uint64_t DefaultExts; // bitfield of default extensions ArchExtKind
266 
267  bool operator==(const ArchInfo &Other) const {
268  return this->Name == Other.Name;
269  }
270  bool operator!=(const ArchInfo &Other) const {
271  return this->Name != Other.Name;
272  }
273 
274  // Defines the following partial order, indicating when an architecture is
275  // a superset of another:
276  //
277  // v9.4a > v9.3a > v9.3a > v9.3a > v9a;
278  // v v v v v
279  // v8.9a > v8.8a > v8.7a > v8.6a > v8.5a > v8.4a > ... > v8a;
280  //
281  // v8r has no relation to anything. This is used to determine which
282  // features to enable for a given architecture. See
283  // AArch64TargetInfo::setFeatureEnabled.
284  bool implies(const ArchInfo &Other) const {
285  if (this->Profile != Other.Profile)
286  return false; // ARMV8R
287  if (this->Version.getMajor() == Other.Version.getMajor()) {
288  return this->Version > Other.Version;
289  }
290  if (this->Version.getMajor() == 9 && Other.Version.getMajor() == 8) {
291  return this->Version.getMinor().value() + 5 >=
292  Other.Version.getMinor().value();
293  }
294  return false;
295  }

297  // Return ArchFeature without the leading "+".
298  StringRef getSubArch() const { return ArchFeature.substr(1); }

301  static std::optional<ArchInfo> findBySubArch(StringRef SubArch);
