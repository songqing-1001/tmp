31 bool DwarfStreamer::init(Triple TheTriple,
32  StringRef Swift5ReflectionSegmentName) {
33  std::string ErrorStr;
34  std::string TripleName;
35  StringRef Context = "dwarf streamer init";
36 
37  // Get the target.
38  const Target *TheTarget =
39  TargetRegistry::lookupTarget(TripleName, TheTriple, ErrorStr);
40  if (!TheTarget)
41  return error(ErrorStr, Context), false;
42  TripleName = TheTriple.getTriple();
43 
44  // Create all the MC Objects.
45  MRI.reset(TheTarget->createMCRegInfo(TripleName));
46  if (!MRI)
47  return error(Twine("no register info for target ") + TripleName, Context),
48  false;
49 
50  MCTargetOptions MCOptions = mc::InitMCTargetOptionsFromFlags();
51  MAI.reset(TheTarget->createMCAsmInfo(*MRI, TripleName, MCOptions));
52  if (!MAI)
53  return error("no asm info for target " + TripleName, Context), false;
54 
55  MSTI.reset(TheTarget->createMCSubtargetInfo(TripleName, "", ""));
56  if (!MSTI)
57  return error("no subtarget info for target " + TripleName, Context), false;
58 
59  MC.reset(new MCContext(TheTriple, MAI.get(), MRI.get(), MSTI.get(), nullptr,
60  nullptr, true, Swift5ReflectionSegmentName));
61  MOFI.reset(TheTarget->createMCObjectFileInfo(*MC, /*PIC=*/false, false));
62  MC->setObjectFileInfo(MOFI.get());
63 
64  MAB = TheTarget->createMCAsmBackend(*MSTI, *MRI, MCOptions);
65  if (!MAB)
66  return error("no asm backend for target " + TripleName, Context), false;
67 
68  MII.reset(TheTarget->createMCInstrInfo());
69  if (!MII)
70  return error("no instr info info for target " + TripleName, Context), false;
71 
72  MCE = TheTarget->createMCCodeEmitter(*MII, *MC);
73  if (!MCE)
74  return error("no code emitter for target " + TripleName, Context), false;
75 
76  switch (OutFileType) {
77  case OutputFileType::Assembly: {
78  MIP = TheTarget->createMCInstPrinter(TheTriple, MAI->getAssemblerDialect(),
79  *MAI, *MII, *MRI);
80  MS = TheTarget->createAsmStreamer(
81  *MC, std::make_unique<formatted_raw_ostream>(OutFile), true, true, MIP,
82  std::unique_ptr<MCCodeEmitter>(MCE), std::unique_ptr<MCAsmBackend>(MAB),
83  true);
84  break;
85  }
86  case OutputFileType::Object: {
87  MS = TheTarget->createMCObjectStreamer(
88  TheTriple, *MC, std::unique_ptr<MCAsmBackend>(MAB),
89  MAB->createObjectWriter(OutFile), std::unique_ptr<MCCodeEmitter>(MCE),
90  *MSTI, MCOptions.MCRelaxAll, MCOptions.MCIncrementalLinkerCompatible,
91  /*DWARFMustBeAtTheEnd*/ false);
92  break;
93  }
94  }
95 
96  if (!MS)
97  return error("no object streamer for target " + TripleName, Context), false;
98 
99  // Finally create the AsmPrinter we'll use to emit the DIEs.
100  TM.reset(TheTarget->createTargetMachine(TripleName, "", "", TargetOptions(),
101  std::nullopt));
102  if (!TM)
103  return error("no target machine for target " + TripleName, Context), false;
104 
105  Asm.reset(TheTarget->createAsmPrinter(*TM, std::unique_ptr<MCStreamer>(MS)));
106  if (!Asm)
107  return error("no asm printer for target " + TripleName, Context), false;
108  Asm->setDwarfUsesRelocationsAcrossSections(false);
109 
110  RangesSectionSize = 0;
111  LocSectionSize = 0;
112  LineSectionSize = 0;
113  FrameSectionSize = 0;
114  DebugInfoSectionSize = 0;
115  MacInfoSectionSize = 0;
116  MacroSectionSize = 0;
117 
118  return true;
119 }

128 /// Emit the compilation unit header for \p Unit in the debug_info section.
129 ///
130 /// A Dwarf 4 section header is encoded as:
131 /// uint32_t Unit length (omitting this field)
132 /// uint16_t Version
133 /// uint32_t Abbreviation table offset
134 /// uint8_t Address size
135 /// Leading to a total of 11 bytes.
136 ///
137 /// A Dwarf 5 section header is encoded as:
138 /// uint32_t Unit length (omitting this field)
139 /// uint16_t Version
140 /// uint8_t Unit type
141 /// uint8_t Address size
142 /// uint32_t Abbreviation table offset
143 /// Leading to a total of 12 bytes.
144 void DwarfStreamer::emitCompileUnitHeader(CompileUnit &Unit,
145  unsigned DwarfVersion) {
146  switchToDebugInfoSection(DwarfVersion);
147 
148  /// The start of the unit within its section.
149  Unit.setLabelBegin(Asm->createTempSymbol("cu_begin"));
150  Asm->OutStreamer->emitLabel(Unit.getLabelBegin());
151 
152  // Emit size of content not including length itself. The size has already
153  // been computed in CompileUnit::computeOffsets(). Subtract 4 to that size to
154  // account for the length field.
155  Asm->emitInt32(Unit.getNextUnitOffset() - Unit.getStartOffset() - 4);
156  Asm->emitInt16(DwarfVersion);
157 
158  if (DwarfVersion >= 5) {
159  Asm->emitInt8(dwarf::DW_UT_compile);
160  Asm->emitInt8(Unit.getOrigUnit().getAddressByteSize());
161  // We share one abbreviations table across all units so it's always at the
162  // start of the section.
163  Asm->emitInt32(0);
164  DebugInfoSectionSize += 12;
165  } else {
166  // We share one abbreviations table across all units so it's always at the
167  // start of the section.
168  Asm->emitInt32(0);
169  Asm->emitInt8(Unit.getOrigUnit().getAddressByteSize());
170  DebugInfoSectionSize += 11;
171  }
172 
173  // Remember this CU.
174  EmittedUnits.push_back({Unit.getUniqueID(), Unit.getLabelBegin()});
175 }

249  AccelTable<DWARF5AccelTableStaticData> &Table) {
250  if (EmittedUnits.empty())
251  return;

253  // Build up data structures needed to emit this section.

255  DenseMap<unsigned, size_t> UniqueIdToCuMap;
256  unsigned Id = 0;
257  for (auto &CU : EmittedUnits) {

266  [&UniqueIdToCuMap](const DWARF5AccelTableStaticData &Entry) {

323 void DwarfStreamer::emitDwarfDebugArangesTable(
324  const CompileUnit &Unit, const AddressRanges &LinkedRanges) {
325  unsigned AddressSize = Unit.getOrigUnit().getAddressByteSize();
326 
327  // Make .debug_aranges to be current section.
328  MS->switchSection(MC->getObjectFileInfo()->getDwarfARangesSection());
329 
330  // Emit Header.
331  MCSymbol *BeginLabel = Asm->createTempSymbol("Barange");
332  MCSymbol *EndLabel = Asm->createTempSymbol("Earange");
333 
334  unsigned HeaderSize =
335  sizeof(int32_t) + // Size of contents (w/o this field
336  sizeof(int16_t) + // DWARF ARange version number
337  sizeof(int32_t) + // Offset of CU in the .debug_info section
338  sizeof(int8_t) + // Pointer Size (in bytes)
339  sizeof(int8_t); // Segment Size (in bytes)
340 
341  unsigned TupleSize = AddressSize * 2;
342  unsigned Padding = offsetToAlignment(HeaderSize, Align(TupleSize));
343 
344  Asm->emitLabelDifference(EndLabel, BeginLabel, 4); // Arange length
345  Asm->OutStreamer->emitLabel(BeginLabel);
346  Asm->emitInt16(dwarf::DW_ARANGES_VERSION); // Version number
347  Asm->emitInt32(Unit.getStartOffset()); // Corresponding unit's offset
348  Asm->emitInt8(AddressSize); // Address size
349  Asm->emitInt8(0); // Segment size
350 
351  Asm->OutStreamer->emitFill(Padding, 0x0);
352 
353  // Emit linked ranges.
354  for (const AddressRange &Range : LinkedRanges) {
355  MS->emitIntValue(Range.start(), AddressSize);
356  MS->emitIntValue(Range.end() - Range.start(), AddressSize);
357  }
358 
359  // Emit terminator.
360  Asm->OutStreamer->emitIntValue(0, AddressSize);
361  Asm->OutStreamer->emitIntValue(0, AddressSize);
362  Asm->OutStreamer->emitLabel(EndLabel);
363 }

365 void DwarfStreamer::emitDwarfDebugRangesTableFragment(
366  const CompileUnit &Unit, const AddressRanges &LinkedRanges) {
367  unsigned AddressSize = Unit.getOrigUnit().getAddressByteSize();
368 
369  // Make .debug_ranges to be current section.
370  MS->switchSection(MC->getObjectFileInfo()->getDwarfRangesSection());
371 
372  // Emit ranges.
373  uint64_t BaseAddress = 0;
374  if (std::optional<uint64_t> LowPC = Unit.getLowPc())
375  BaseAddress = *LowPC;
376 
377  for (const AddressRange &Range : LinkedRanges) {
378  MS->emitIntValue(Range.start() - BaseAddress, AddressSize);
379  MS->emitIntValue(Range.end() - BaseAddress, AddressSize);
380 
381  RangesSectionSize += AddressSize;
382  RangesSectionSize += AddressSize;
383  }
384 
385  // Add the terminator entry.
386  MS->emitIntValue(0, AddressSize);
387  MS->emitIntValue(0, AddressSize);
388 
389  RangesSectionSize += AddressSize;
390  RangesSectionSize += AddressSize;
391 }

416 /// Emit location lists for \p Unit and update attributes to point to the new
417 /// entries.
418 void DwarfStreamer::emitLocationsForUnit(
419  const CompileUnit &Unit, DWARFContext &Dwarf,
420  std::function<void(StringRef, SmallVectorImpl<uint8_t> &)> ProcessExpr) {
421  const auto &Attributes = Unit.getLocationAttributes();
422 
423  if (Attributes.empty())
424  return;
425 
426  MS->switchSection(MC->getObjectFileInfo()->getDwarfLocSection());
427 
428  unsigned AddressSize = Unit.getOrigUnit().getAddressByteSize();
429  uint64_t BaseAddressMarker = (AddressSize == 8)
430  ? std::numeric_limits<uint64_t>::max()
431  : std::numeric_limits<uint32_t>::max();
432  const DWARFSection &InputSec = Dwarf.getDWARFObj().getLocSection();
433  DataExtractor Data(InputSec.Data, Dwarf.isLittleEndian(), AddressSize);
434  DWARFUnit &OrigUnit = Unit.getOrigUnit();
435  auto OrigUnitDie = OrigUnit.getUnitDIE(false);
436  int64_t UnitPcOffset = 0;
437  if (auto OrigLowPc =
438  dwarf::toAddress(OrigUnitDie.find(dwarf::DW_AT_low_pc))) {
439  assert(Unit.getLowPc());
440  UnitPcOffset = int64_t(*OrigLowPc) - *Unit.getLowPc();
441  }
442 
443  SmallVector<uint8_t, 32> Buffer;
444  for (const auto &Attr : Attributes) {
445  uint64_t Offset = Attr.first.get();
446  Attr.first.set(LocSectionSize);
447  // This is the quantity to add to the old location address to get
448  // the correct address for the new one.
449  int64_t LocPcOffset = Attr.second + UnitPcOffset;
450  while (Data.isValidOffset(Offset)) {
451  uint64_t Low = Data.getUnsigned(&Offset, AddressSize);
452  uint64_t High = Data.getUnsigned(&Offset, AddressSize);
453  LocSectionSize += 2 * AddressSize;
454  // End of list entry.
455  if (Low == 0 && High == 0) {
456  Asm->OutStreamer->emitIntValue(0, AddressSize);
457  Asm->OutStreamer->emitIntValue(0, AddressSize);
458  break;
459  }
460  // Base address selection entry.
461  if (Low == BaseAddressMarker) {
462  Asm->OutStreamer->emitIntValue(BaseAddressMarker, AddressSize);
463  Asm->OutStreamer->emitIntValue(High + Attr.second, AddressSize);
464  LocPcOffset = 0;
465  continue;
466  }
467  // Location list entry.
468  Asm->OutStreamer->emitIntValue(Low + LocPcOffset, AddressSize);
469  Asm->OutStreamer->emitIntValue(High + LocPcOffset, AddressSize);
470  uint64_t Length = Data.getU16(&Offset);
471  Asm->OutStreamer->emitIntValue(Length, 2);
472  // Copy the bytes into to the buffer, process them, emit them.
473  Buffer.reserve(Length);
474  Buffer.resize(0);
475  StringRef Input = InputSec.Data.substr(Offset, Length);
476  ProcessExpr(Input, Buffer);
477  Asm->OutStreamer->emitBytes(
478  StringRef((const char *)Buffer.data(), Length));
479  Offset += Length;
480  LocSectionSize += Length + 2;
481  }
482  }
483 }

485 void DwarfStreamer::emitLineTableForUnit(MCDwarfLineTableParams Params,
486  StringRef PrologueBytes,
487  unsigned MinInstLength,
488  std::vector<DWARFDebugLine::Row> &Rows,
489  unsigned PointerSize) {
490  // Switch to the section where the table will be emitted into.
491  MS->switchSection(MC->getObjectFileInfo()->getDwarfLineSection());
492  MCSymbol *LineStartSym = MC->createTempSymbol();
493  MCSymbol *LineEndSym = MC->createTempSymbol();
494 
495  // The first 4 bytes is the total length of the information for this
496  // compilation unit (not including these 4 bytes for the length).
497  Asm->emitLabelDifference(LineEndSym, LineStartSym, 4);
498  Asm->OutStreamer->emitLabel(LineStartSym);
499  // Copy Prologue.
500  MS->emitBytes(PrologueBytes);
501  LineSectionSize += PrologueBytes.size() + 4;
502 
503  SmallString<128> EncodingBuffer;
504  raw_svector_ostream EncodingOS(EncodingBuffer);
505 
506  if (Rows.empty()) {
507  // We only have the dummy entry, dsymutil emits an entry with a 0
508  // address in that case.
509  MCDwarfLineAddr::Encode(*MC, Params, std::numeric_limits<int64_t>::max(), 0,
510  EncodingOS);
511  MS->emitBytes(EncodingOS.str());
512  LineSectionSize += EncodingBuffer.size();
513  MS->emitLabel(LineEndSym);
514  return;
515  }
516 
517  // Line table state machine fields
518  unsigned FileNum = 1;
519  unsigned LastLine = 1;
520  unsigned Column = 0;
521  unsigned IsStatement = 1;
522  unsigned Isa = 0;
523  uint64_t Address = -1ULL;
524 
525  unsigned RowsSinceLastSequence = 0;
526 
527  for (DWARFDebugLine::Row &Row : Rows) {
528  int64_t AddressDelta;
529  if (Address == -1ULL) {
530  MS->emitIntValue(dwarf::DW_LNS_extended_op, 1);
531  MS->emitULEB128IntValue(PointerSize + 1);
532  MS->emitIntValue(dwarf::DW_LNE_set_address, 1);
533  MS->emitIntValue(Row.Address.Address, PointerSize);
534  LineSectionSize += 2 + PointerSize + getULEB128Size(PointerSize + 1);
535  AddressDelta = 0;
536  } else {
537  AddressDelta = (Row.Address.Address - Address) / MinInstLength;
538  }
539 
540  // FIXME: code copied and transformed from MCDwarf.cpp::EmitDwarfLineTable.
541  // We should find a way to share this code, but the current compatibility
542  // requirement with classic dsymutil makes it hard. Revisit that once this
543  // requirement is dropped.
544 
545  if (FileNum != Row.File) {
546  FileNum = Row.File;
547  MS->emitIntValue(dwarf::DW_LNS_set_file, 1);
548  MS->emitULEB128IntValue(FileNum);
549  LineSectionSize += 1 + getULEB128Size(FileNum);
550  }
551  if (Column != Row.Column) {
552  Column = Row.Column;
553  MS->emitIntValue(dwarf::DW_LNS_set_column, 1);
554  MS->emitULEB128IntValue(Column);
555  LineSectionSize += 1 + getULEB128Size(Column);
556  }
557 
558  // FIXME: We should handle the discriminator here, but dsymutil doesn't
559  // consider it, thus ignore it for now.
560 
561  if (Isa != Row.Isa) {
562  Isa = Row.Isa;
563  MS->emitIntValue(dwarf::DW_LNS_set_isa, 1);
564  MS->emitULEB128IntValue(Isa);
565  LineSectionSize += 1 + getULEB128Size(Isa);
566  }
567  if (IsStatement != Row.IsStmt) {
568  IsStatement = Row.IsStmt;
569  MS->emitIntValue(dwarf::DW_LNS_negate_stmt, 1);
570  LineSectionSize += 1;
571  }
572  if (Row.BasicBlock) {
573  MS->emitIntValue(dwarf::DW_LNS_set_basic_block, 1);
574  LineSectionSize += 1;
575  }
576 
577  if (Row.PrologueEnd) {
578  MS->emitIntValue(dwarf::DW_LNS_set_prologue_end, 1);
579  LineSectionSize += 1;
580  }
581 
582  if (Row.EpilogueBegin) {
583  MS->emitIntValue(dwarf::DW_LNS_set_epilogue_begin, 1);
584  LineSectionSize += 1;
585  }
586

588  if (!Row.EndSequence) {

595  RowsSinceLastSequence++;

598  MS->emitIntValue(dwarf::DW_LNS_advance_line, 1);
599  MS->emitSLEB128IntValue(LineDelta);

601  }
602  if (AddressDelta) {

608  0, EncodingOS);

610  LineSectionSize += EncodingBuffer.size();
611  EncodingBuffer.resize(0);
612  Address = -1ULL;
613  LastLine = FileNum = IsStatement = 1;
614  RowsSinceLastSequence = Column = Isa = 0;
615  }
616  }
617 
618  if (RowsSinceLastSequence) {
619  MCDwarfLineAddr::Encode(*MC, Params, std::numeric_limits<int64_t>::max(), 0,
620  EncodingOS);
621  MS->emitBytes(EncodingOS.str());
622  LineSectionSize += EncodingBuffer.size();
623  EncodingBuffer.resize(0);
624  }
625 
626  MS->emitLabel(LineEndSym);
627 }

629 /// Copy the debug_line over to the updated binary while unobfuscating the file
630 /// names and directories.
631 void DwarfStreamer::translateLineTable(DataExtractor Data, uint64_t Offset) {
632  MS->switchSection(MC->getObjectFileInfo()->getDwarfLineSection());
633  StringRef Contents = Data.getData();
634 
635  // We have to deconstruct the line table header, because it contains to
636  // length fields that will need to be updated when we change the length of
637  // the files and directories in there.
638  unsigned UnitLength = Data.getU32(&Offset);
639  uint64_t UnitEnd = Offset + UnitLength;
640  MCSymbol *BeginLabel = MC->createTempSymbol();
641  MCSymbol *EndLabel = MC->createTempSymbol();
642  unsigned Version = Data.getU16(&Offset);
643 
644  if (Version > 5) {
645  warn("Unsupported line table version: dropping contents and not "
646  "unobfsucating line table.");
647  return;
648  }
649 
650  Asm->emitLabelDifference(EndLabel, BeginLabel, 4);
651  Asm->OutStreamer->emitLabel(BeginLabel);
652  Asm->emitInt16(Version);
653  LineSectionSize += 6;
654 
655  MCSymbol *HeaderBeginLabel = MC->createTempSymbol();
656  MCSymbol *HeaderEndLabel = MC->createTempSymbol();
657  Asm->emitLabelDifference(HeaderEndLabel, HeaderBeginLabel, 4);
658  Asm->OutStreamer->emitLabel(HeaderBeginLabel);
659  Offset += 4;
660  LineSectionSize += 4;
661 
662  uint64_t AfterHeaderLengthOffset = Offset;
663  // Skip to the directories.
664  Offset += (Version >= 4) ? 5 : 4;
665  unsigned OpcodeBase = Data.getU8(&Offset);
666  Offset += OpcodeBase - 1;
667  Asm->OutStreamer->emitBytes(Contents.slice(AfterHeaderLengthOffset, Offset));
668  LineSectionSize += Offset - AfterHeaderLengthOffset;
669 
670  // Offset points to the first directory.
671  while (const char *Dir = Data.getCStr(&Offset)) {
672  if (Dir[0] == 0)
673  break;
674 
675  StringRef Translated = Translator(Dir);
676  Asm->OutStreamer->emitBytes(Translated);
677  Asm->emitInt8(0);
678  LineSectionSize += Translated.size() + 1;
679  }
680  Asm->emitInt8(0);
681  LineSectionSize += 1;
682 
683  while (const char *File = Data.getCStr(&Offset)) {
684  if (File[0] == 0)
685  break;
686 
687  StringRef Translated = Translator(File);
688  Asm->OutStreamer->emitBytes(Translated);
689  Asm->emitInt8(0);
690  LineSectionSize += Translated.size() + 1;
691 
692  uint64_t OffsetBeforeLEBs = Offset;
693  Asm->emitULEB128(Data.getULEB128(&Offset));
694  Asm->emitULEB128(Data.getULEB128(&Offset));
695  Asm->emitULEB128(Data.getULEB128(&Offset));
696  LineSectionSize += Offset - OffsetBeforeLEBs;
697  }
698  Asm->emitInt8(0);
699  LineSectionSize += 1;
700 
701  Asm->OutStreamer->emitLabel(HeaderEndLabel);
702 
703  // Copy the actual line table program over.
704  Asm->OutStreamer->emitBytes(Contents.slice(Offset, UnitEnd));
705  LineSectionSize += UnitEnd - Offset;
706 
707  Asm->OutStreamer->emitLabel(EndLabel);
708  Offset = UnitEnd;
709 }

712 /// Unit into \p Sec. The data is provided in \p Names.
713 void DwarfStreamer::emitPubSectionForUnit(
714  MCSection *Sec, StringRef SecName, const CompileUnit &Unit,
715  const std::vector<CompileUnit::AccelInfo> &Names) {

720  Asm->OutStreamer->switchSection(Sec);
721  MCSymbol *BeginLabel = Asm->createTempSymbol("pub" + SecName + "_begin");
722  MCSymbol *EndLabel = Asm->createTempSymbol("pub" + SecName + "_end");

725  // Emit the pubnames for this compilation unit.
726  for (const auto &Name : Names) {

728  continue;

732  Asm->emitLabelDifference(EndLabel, BeginLabel, 4); // Length

735  Asm->emitInt32(Unit.getStartOffset()); // Unit offset
736  Asm->emitInt32(Unit.getNextUnitOffset() - Unit.getStartOffset()); // Size

738  }
739  Asm->emitInt32(Name.Die->getOffset());

741  // Emit the string itself.
742  Asm->OutStreamer->emitBytes(Name.Name.getString());
743  // Emit a null terminator.
744  Asm->emitInt8(0);
745  }
746 
747  if (!HeaderEmitted)
748  return;
749  Asm->emitInt32(0); // End marker.
750  Asm->OutStreamer->emitLabel(EndLabel);

805 void DwarfStreamer::emitMacroTableImpl(const DWARFDebugMacro *MacroTable,
806  const Offset2UnitMap &UnitMacroMap,
807  OffsetsStringPool &StringPool,
808  uint64_t &OutOffset) {
809  bool DefAttributeIsReported = false;
810  bool UndefAttributeIsReported = false;
811  bool ImportAttributeIsReported = false;
812  for (const DWARFDebugMacro::MacroList &List : MacroTable->MacroLists) {
813  Offset2UnitMap::const_iterator UnitIt = UnitMacroMap.find(List.Offset);
814  if (UnitIt == UnitMacroMap.end()) {
815  warn(formatv(
816  "couldn`t find compile unit for the macro table with offset = {0:x}",
817  List.Offset));
818  continue;
819  }
820 
821  // Skip macro table if the unit was not cloned.
822  DIE *OutputUnitDIE = UnitIt->second->getOutputUnitDIE();
823  if (OutputUnitDIE == nullptr)
824  continue;
825 
826  // Update macro attribute of cloned compile unit with the proper offset to
827  // the macro table.
828  bool hasDWARFv5Header = false;
829  for (auto &V : OutputUnitDIE->values()) {
830  if (V.getAttribute() == dwarf::DW_AT_macro_info) {
831  V = DIEValue(V.getAttribute(), V.getForm(), DIEInteger(OutOffset));
832  break;
833  } else if (V.getAttribute() == dwarf::DW_AT_macros) {
834  hasDWARFv5Header = true;
835  V = DIEValue(V.getAttribute(), V.getForm(), DIEInteger(OutOffset));
836  break;
837  }
838  }
839 
840  // Write DWARFv5 header.
841  if (hasDWARFv5Header) {
842  // Write header version.
843  MS->emitIntValue(List.Header.Version, sizeof(List.Header.Version));
844  OutOffset += sizeof(List.Header.Version);
845 
846  uint8_t Flags = List.Header.Flags;
847 
848  // Check for OPCODE_OPERANDS_TABLE.
849  if (Flags &
850  DWARFDebugMacro::HeaderFlagMask::MACRO_OPCODE_OPERANDS_TABLE) {
851  Flags &= ~DWARFDebugMacro::HeaderFlagMask::MACRO_OPCODE_OPERANDS_TABLE;
852  warn("opcode_operands_table is not supported yet.");
853  }
854 
855  // Check for DEBUG_LINE_OFFSET.
856  std::optional<uint64_t> StmtListOffset;
857  if (Flags & DWARFDebugMacro::HeaderFlagMask::MACRO_DEBUG_LINE_OFFSET) {
858  // Get offset to the line table from the cloned compile unit.
859  for (auto &V : OutputUnitDIE->values()) {
860  if (V.getAttribute() == dwarf::DW_AT_stmt_list) {
861  StmtListOffset = V.getDIEInteger().getValue();
862  break;
863  }
864  }
865 
866  if (!StmtListOffset) {
867  Flags &= ~DWARFDebugMacro::HeaderFlagMask::MACRO_DEBUG_LINE_OFFSET;
868  warn("couldn`t find line table for macro table.");
869  }
870  }
871 
872  // Write flags.
873  MS->emitIntValue(Flags, sizeof(Flags));
874  OutOffset += sizeof(Flags);
875 
876  // Write offset to line table.
877  if (StmtListOffset) {
878  MS->emitIntValue(*StmtListOffset, List.Header.getOffsetByteSize());
879  OutOffset += List.Header.getOffsetByteSize();
880  }
881  }
882 
883  // Write macro entries.
884  for (const DWARFDebugMacro::Entry &MacroEntry : List.Macros) {
885  if (MacroEntry.Type == 0) {
886  OutOffset += MS->emitULEB128IntValue(MacroEntry.Type);
887  continue;
888  }
889 
890  uint8_t MacroType = MacroEntry.Type;
891  switch (MacroType) {
892  default: {
893  bool HasVendorSpecificExtension =
894  (!hasDWARFv5Header && MacroType == dwarf::DW_MACINFO_vendor_ext) ||
895  (hasDWARFv5Header && (MacroType >= dwarf::DW_MACRO_lo_user &&
896  MacroType <= dwarf::DW_MACRO_hi_user));
897 
898  if (HasVendorSpecificExtension) {
899  // Write macinfo type.
900  MS->emitIntValue(MacroType, 1);
901  OutOffset++;
902 
903  // Write vendor extension constant.
904  OutOffset += MS->emitULEB128IntValue(MacroEntry.ExtConstant);
905 
906  // Write vendor extension string.
907  StringRef String = MacroEntry.ExtStr;
908  MS->emitBytes(String);
909  MS->emitIntValue(0, 1);
910  OutOffset += String.size() + 1;
911  } else
912  warn("unknown macro type. skip.");
913  } break;
914  // debug_macro and debug_macinfo share some common encodings.
915  // DW_MACRO_define == DW_MACINFO_define
916  // DW_MACRO_undef == DW_MACINFO_undef
917  // DW_MACRO_start_file == DW_MACINFO_start_file
918  // DW_MACRO_end_file == DW_MACINFO_end_file
919  // For readibility/uniformity we are using DW_MACRO_*.
920  case dwarf::DW_MACRO_define:
921  case dwarf::DW_MACRO_undef: {
922  // Write macinfo type.
923  MS->emitIntValue(MacroType, 1);
924  OutOffset++;
925 
926  // Write source line.
927  OutOffset += MS->emitULEB128IntValue(MacroEntry.Line);
928 
929  // Write macro string.
930  StringRef String = MacroEntry.MacroStr;
931  MS->emitBytes(String);
932  MS->emitIntValue(0, 1);
933  OutOffset += String.size() + 1;
934  } break;
935  case dwarf::DW_MACRO_define_strp:
936  case dwarf::DW_MACRO_undef_strp:
937  case dwarf::DW_MACRO_define_strx:
938  case dwarf::DW_MACRO_undef_strx: {
939  assert(UnitIt->second->getOrigUnit().getVersion() >= 5);
940 
941  // DW_MACRO_*_strx forms are not supported currently.
942  // Convert to *_strp.
943  switch (MacroType) {
944  case dwarf::DW_MACRO_define_strx: {
945  MacroType = dwarf::DW_MACRO_define_strp;
946  if (!DefAttributeIsReported) {
947  warn("DW_MACRO_define_strx unsupported yet. Convert to "
948  "DW_MACRO_define_strp.");
949  DefAttributeIsReported = true;
950  }
951  } break;
952  case dwarf::DW_MACRO_undef_strx: {
953  MacroType = dwarf::DW_MACRO_undef_strp;
954  if (!UndefAttributeIsReported) {
955  warn("DW_MACRO_undef_strx unsupported yet. Convert to "
956  "DW_MACRO_undef_strp.");
957  UndefAttributeIsReported = true;
958  }
959  } break;
960  default:
961  // Nothing to do.
962  break;
963  }
964 
965  // Write macinfo type.
966  MS->emitIntValue(MacroType, 1);
967  OutOffset++;
968 
969  // Write source line.
970  OutOffset += MS->emitULEB128IntValue(MacroEntry.Line);
971 
972  // Write macro string.
973  DwarfStringPoolEntryRef EntryRef =
974  StringPool.getEntry(MacroEntry.MacroStr);
975  MS->emitIntValue(EntryRef.getOffset(), List.Header.getOffsetByteSize());
976  OutOffset += List.Header.getOffsetByteSize();
977  break;
978  }
979  case dwarf::DW_MACRO_start_file: {
980  // Write macinfo type.
981  MS->emitIntValue(MacroType, 1);
982  OutOffset++;
983  // Write source line.
984  OutOffset += MS->emitULEB128IntValue(MacroEntry.Line);
985  // Write source file id.
986  OutOffset += MS->emitULEB128IntValue(MacroEntry.File);
987  } break;
988  case dwarf::DW_MACRO_end_file: {
989  // Write macinfo type.
990  MS->emitIntValue(MacroType, 1);
991  OutOffset++;
992  } break;
993  case dwarf::DW_MACRO_import:
994  case dwarf::DW_MACRO_import_sup: {
995  if (!ImportAttributeIsReported) {
996  warn("DW_MACRO_import and DW_MACRO_import_sup are unsupported yet. "
997  "remove.");
998  ImportAttributeIsReported = true;
999  }
1000  } break;
1001  }
1002  }
1003  }
