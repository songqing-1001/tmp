222 void NVPTXTargetMachine::registerPassBuilderCallbacks(PassBuilder &PB) {
223  PB.registerPipelineParsingCallback(
224  [](StringRef PassName, FunctionPassManager &PM,
225  ArrayRef<PassBuilder::PipelineElement>) {
226  if (PassName == "nvvm-reflect") {
227  PM.addPass(NVVMReflectPass());
228  return true;
229  }
230  if (PassName == "nvvm-intr-range") {
231  PM.addPass(NVVMIntrRangePass());
232  return true;
233  }
234  return false;
235  });
236 
237  PB.registerAnalysisRegistrationCallback([](FunctionAnalysisManager &FAM) {
238  FAM.registerPass([&] { return NVPTXAA(); });
239  });
240 
241  PB.registerParseAACallback([](StringRef AAName, AAManager &AAM) {
242  if (AAName == "nvptx-aa") {
243  AAM.registerFunctionAnalysis<NVPTXAA>();
244  return true;
245  }
246  return false;
247  });
248 
249  PB.registerPipelineStartEPCallback(
250  [this](ModulePassManager &PM, OptimizationLevel Level) {
251  FunctionPassManager FPM;
252  FPM.addPass(NVVMReflectPass(Subtarget.getSmVersion()));
253  // FIXME: NVVMIntrRangePass is causing numerical discrepancies,
254  // investigate and re-enable.
255  // FPM.addPass(NVVMIntrRangePass(Subtarget.getSmVersion()));
256  PM.addPass(createModuleToFunctionPassAdaptor(std::move(FPM)));
257  });

265 std::pair<const Value *, unsigned>
266 NVPTXTargetMachine::getPredicatedAddrSpace(const Value *V) const {
267  if (auto *II = dyn_cast<IntrinsicInst>(V)) {
268  switch (II->getIntrinsicID()) {
269  case Intrinsic::nvvm_isspacep_const:
270  return std::make_pair(II->getArgOperand(0), llvm::ADDRESS_SPACE_CONST);
271  case Intrinsic::nvvm_isspacep_global:
272  return std::make_pair(II->getArgOperand(0), llvm::ADDRESS_SPACE_GLOBAL);
273  case Intrinsic::nvvm_isspacep_local:
274  return std::make_pair(II->getArgOperand(0), llvm::ADDRESS_SPACE_LOCAL);
275  case Intrinsic::nvvm_isspacep_shared:
276  return std::make_pair(II->getArgOperand(0), llvm::ADDRESS_SPACE_SHARED);
277  default:
278  break;
279  }
280  }
281  return std::make_pair(nullptr, -1);
282 }

300 void NVPTXPassConfig::addStraightLineScalarOptimizationPasses() {
301  addPass(createSeparateConstOffsetFromGEPPass());
302  addPass(createSpeculativeExecutionPass());
303  // ReassociateGEPs exposes more opportunites for SLSR. See
304  // the example in reassociate-geps-and-slsr.ll.
305  addPass(createStraightLineStrengthReducePass());
306  // SeparateConstOffsetFromGEP and SLSR creates common expressions which GVN or
307  // EarlyCSE can reuse. GVN generates significantly better code than EarlyCSE
308  // for some of our benchmarks.
309  addEarlyCSEOrGVNPass();
310  // Run NaryReassociate after EarlyCSE/GVN to be more effective.
311  addPass(createNaryReassociatePass());
312  // NaryReassociate on GEPs creates redundant common expressions, so run
313  // EarlyCSE after it.
314  addPass(createEarlyCSEPass());
315 }

317 void NVPTXPassConfig::addIRPasses() {
318  // The following passes are known to not play well with virtual regs hanging
319  // around after register allocation (which in our case, is *all* registers).
320  // We explicitly disable them here. We do, however, need some functionality
321  // of the PrologEpilogCodeInserter pass, so we emulate that behavior in the
322  // NVPTXPrologEpilog pass (see NVPTXPrologEpilogPass.cpp).
323  disablePass(&PrologEpilogCodeInserterID);
324  disablePass(&MachineLateInstrsCleanupID);
325  disablePass(&MachineCopyPropagationID);
326  disablePass(&TailDuplicateID);
327  disablePass(&StackMapLivenessID);
328  disablePass(&LiveDebugValuesID);
329  disablePass(&PostRAMachineSinkingID);
330  disablePass(&PostRASchedulerID);
331  disablePass(&FuncletLayoutID);
332  disablePass(&PatchableFunctionID);
333  disablePass(&ShrinkWrapID);
334 
335  addPass(createNVPTXAAWrapperPass());
336  addPass(createExternalAAWrapperPass([](Pass &P, Function &, AAResults &AAR) {
337  if (auto *WrapperPass = P.getAnalysisIfAvailable<NVPTXAAWrapperPass>())
338  AAR.addAAResult(WrapperPass->getResult());
339  }));
340 
341  // NVVMReflectPass is added in addEarlyAsPossiblePasses, so hopefully running
342  // it here does nothing. But since we need it for correctness when lowering
343  // to NVPTX, run it here too, in case whoever built our pass pipeline didn't
344  // call addEarlyAsPossiblePasses.
345  const NVPTXSubtarget &ST = *getTM<NVPTXTargetMachine>().getSubtargetImpl();
346  addPass(createNVVMReflectPass(ST.getSmVersion()));
347 
348  if (getOptLevel() != CodeGenOpt::None)
349  addPass(createNVPTXImageOptimizerPass());
350  addPass(createNVPTXAssignValidGlobalNamesPass());
351  addPass(createGenericToNVVMPass());
352 
353  // NVPTXLowerArgs is required for correctness and should be run right
354  // before the address space inference passes.
355  addPass(createNVPTXLowerArgsPass());
356  if (getOptLevel() != CodeGenOpt::None) {
357  addAddressSpaceInferencePasses();
358  addStraightLineScalarOptimizationPasses();
359  }
360 
361  addPass(createAtomicExpandPass());
362 
363  // === LSR and other generic IR passes ===
364  TargetPassConfig::addIRPasses();
365  // EarlyCSE is not always strong enough to clean up what LSR produces. For
366  // example, GVN can combine
367  //
368  // %0 = add %a, %b
369  // %1 = add %b, %a
370  //
371  // and
372  //
373  // %0 = shl nsw %a, 2
374  // %1 = shl %a, 2
375  //
376  // but EarlyCSE can do neither of them.
377  if (getOptLevel() != CodeGenOpt::None) {
378  addEarlyCSEOrGVNPass();
379  if (!DisableLoadStoreVectorizer)
380  addPass(createLoadStoreVectorizerPass());
381  addPass(createSROAPass());
382  }
383 }

444 void NVPTXPassConfig::addMachineSSAOptimization() {
445  // Pre-ra tail duplication.
446  if (addPass(&EarlyTailDuplicateID))
447  printAndVerify("After Pre-RegAlloc TailDuplicate");
448 
449  // Optimize PHIs before DCE: removing dead PHI cycles may make more
450  // instructions dead.
451  addPass(&OptimizePHIsID);
452 
453  // This pass merges large allocas. StackSlotColoring is a different pass
454  // which merges spill slots.
455  addPass(&StackColoringID);
456 
457  // If the target requests it, assign local variables to stack slots relative
458  // to one another and simplify frame index references where possible.
459  addPass(&LocalStackSlotAllocationID);
460 
461  // With optimization, dead code should already be eliminated. However
462  // there is one known exception: lowered code for arguments that are only
463  // used by tail calls, where the tail calls reuse the incoming stack
464  // arguments directly (see t11 in test/CodeGen/X86/sibcall.ll).
465  addPass(&DeadMachineInstructionElimID);
466  printAndVerify("After codegen DCE pass");
467 
468  // Allow targets to insert passes that improve instruction level parallelism,
469  // like if-conversion. Such passes will typically need dominator trees and
470  // loop info, just like LICM and CSE below.
471  if (addILPOpts())
472  printAndVerify("After ILP optimizations");
473 
474  addPass(&EarlyMachineLICMID);
475  addPass(&MachineCSEID);
476 
477  addPass(&MachineSinkingID);
478  printAndVerify("After Machine LICM, CSE and Sinking passes");
479 
480  addPass(&PeepholeOptimizerID);
481  printAndVerify("After codegen peephole optimization pass");
482 }
