62  emitULEB128(Offset, "", ULEB128PadSize);

115  // Add some empty comments to keep the Buffer and Comments vectors aligned
116  // with each other.
117  for (size_t i = 1; i < Length; ++i)
118  Comments.push_back("");

120  }
