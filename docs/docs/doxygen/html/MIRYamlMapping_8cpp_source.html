30 Expected<int> FrameIndex::getFI(const llvm::MachineFrameInfo &MFI) const {
31  int FI = this->FI;
32  if (IsFixed) {
33  if (unsigned(FI) >= MFI.getNumFixedObjects())

40  return make_error<StringError>(formatv("invalid frame index {0}", FI).str(),
