203 void VPInstruction::generateInstruction(VPTransformState &State,
204  unsigned Part) {
205  IRBuilderBase &Builder = State.Builder;
206  Builder.SetCurrentDebugLocation(DL);
207 
208  if (Instruction::isBinaryOp(getOpcode())) {
209  Value *A = State.get(getOperand(0), Part);
210  Value *B = State.get(getOperand(1), Part);
211  Value *V =
212  Builder.CreateBinOp((Instruction::BinaryOps)getOpcode(), A, B, Name);
213  State.set(this, V, Part);
214  return;
215  }
216 
217  switch (getOpcode()) {
218  case VPInstruction::Not: {
219  Value *A = State.get(getOperand(0), Part);
220  Value *V = Builder.CreateNot(A, Name);
221  State.set(this, V, Part);
222  break;
223  }
224  case VPInstruction::ICmpULE: {
225  Value *IV = State.get(getOperand(0), Part);
226  Value *TC = State.get(getOperand(1), Part);
227  Value *V = Builder.CreateICmpULE(IV, TC, Name);
228  State.set(this, V, Part);
229  break;
230  }
231  case Instruction::Select: {
232  Value *Cond = State.get(getOperand(0), Part);
233  Value *Op1 = State.get(getOperand(1), Part);
234  Value *Op2 = State.get(getOperand(2), Part);
235  Value *V = Builder.CreateSelect(Cond, Op1, Op2, Name);
236  State.set(this, V, Part);
237  break;
238  }
239  case VPInstruction::ActiveLaneMask: {
240  // Get first lane of vector induction variable.
241  Value *VIVElem0 = State.get(getOperand(0), VPIteration(Part, 0));
242  // Get the original loop tripcount.
243  Value *ScalarTC = State.get(getOperand(1), Part);
244 
245  auto *Int1Ty = Type::getInt1Ty(Builder.getContext());
246  auto *PredTy = VectorType::get(Int1Ty, State.VF);
247  Instruction *Call = Builder.CreateIntrinsic(
248  Intrinsic::get_active_lane_mask, {PredTy, ScalarTC->getType()},
249  {VIVElem0, ScalarTC}, nullptr, Name);
250  State.set(this, Call, Part);
251  break;
252  }
253  case VPInstruction::FirstOrderRecurrenceSplice: {
254  // Generate code to combine the previous and current values in vector v3.
255  //
256  // vector.ph:
257  // v_init = vector(..., ..., ..., a[-1])
258  // br vector.body
259  //
260  // vector.body
261  // i = phi [0, vector.ph], [i+4, vector.body]
262  // v1 = phi [v_init, vector.ph], [v2, vector.body]
263  // v2 = a[i, i+1, i+2, i+3];
264  // v3 = vector(v1(3), v2(0, 1, 2))
265 
266  // For the first part, use the recurrence phi (v1), otherwise v2.
267  auto *V1 = State.get(getOperand(0), 0);
268  Value *PartMinus1 = Part == 0 ? V1 : State.get(getOperand(1), Part - 1);
269  if (!PartMinus1->getType()->isVectorTy()) {
270  State.set(this, PartMinus1, Part);
271  } else {
272  Value *V2 = State.get(getOperand(1), Part);
273  State.set(this, Builder.CreateVectorSplice(PartMinus1, V2, -1, Name),
274  Part);
275  }
276  break;
277  }
278  case VPInstruction::CanonicalIVIncrement:
279  case VPInstruction::CanonicalIVIncrementNUW: {
280  Value *Next = nullptr;
281  if (Part == 0) {
282  bool IsNUW = getOpcode() == VPInstruction::CanonicalIVIncrementNUW;
283  auto *Phi = State.get(getOperand(0), 0);
284  // The loop step is equal to the vectorization factor (num of SIMD
285  // elements) times the unroll factor (num of SIMD instructions).
286  Value *Step =
287  createStepForVF(Builder, Phi->getType(), State.VF, State.UF);
288  Next = Builder.CreateAdd(Phi, Step, Name, IsNUW, false);
289  } else {
290  Next = State.get(this, 0);
291  }
292 
293  State.set(this, Next, Part);
294  break;
295  }
296 
297  case VPInstruction::CanonicalIVIncrementForPart:
298  case VPInstruction::CanonicalIVIncrementForPartNUW: {
299  bool IsNUW = getOpcode() == VPInstruction::CanonicalIVIncrementForPartNUW;
300  auto *IV = State.get(getOperand(0), VPIteration(0, 0));
301  if (Part == 0) {
302  State.set(this, IV, Part);
303  break;
304  }
305 
306  // The canonical IV is incremented by the vectorization factor (num of SIMD
307  // elements) times the unroll part.
308  Value *Step = createStepForVF(Builder, IV->getType(), State.VF, Part);
309  Value *Next = Builder.CreateAdd(IV, Step, Name, IsNUW, false);
310  State.set(this, Next, Part);
311  break;
312  }
313  case VPInstruction::BranchOnCond: {
314  if (Part != 0)
315  break;
316 
317  Value *Cond = State.get(getOperand(0), VPIteration(Part, 0));
318  VPRegionBlock *ParentRegion = getParent()->getParent();
319  VPBasicBlock *Header = ParentRegion->getEntryBasicBlock();
320 
321  // Replace the temporary unreachable terminator with a new conditional
322  // branch, hooking it up to backward destination for exiting blocks now and
323  // to forward destination(s) later when they are created.
324  BranchInst *CondBr =
325  Builder.CreateCondBr(Cond, Builder.GetInsertBlock(), nullptr);
326 
327  if (getParent()->isExiting())
328  CondBr->setSuccessor(1, State.CFG.VPBB2IRBB[Header]);
329 
330  CondBr->setSuccessor(0, nullptr);
331  Builder.GetInsertBlock()->getTerminator()->eraseFromParent();
332  break;
333  }
334  case VPInstruction::BranchOnCount: {
335  if (Part != 0)
336  break;
337  // First create the compare.
338  Value *IV = State.get(getOperand(0), Part);
339  Value *TC = State.get(getOperand(1), Part);
340  Value *Cond = Builder.CreateICmpEQ(IV, TC);
341 
342  // Now create the branch.
343  auto *Plan = getParent()->getPlan();
344  VPRegionBlock *TopRegion = Plan->getVectorLoopRegion();
345  VPBasicBlock *Header = TopRegion->getEntry()->getEntryBasicBlock();
346 
347  // Replace the temporary unreachable terminator with a new conditional
348  // branch, hooking it up to backward destination (the header) now and to the
349  // forward destination (the exit/middle block) later when it is created.
350  // Note that CreateCondBr expects a valid BB as first argument, so we need
351  // to set it to nullptr later.
352  BranchInst *CondBr = Builder.CreateCondBr(Cond, Builder.GetInsertBlock(),
353  State.CFG.VPBB2IRBB[Header]);
354  CondBr->setSuccessor(0, nullptr);
355  Builder.GetInsertBlock()->getTerminator()->eraseFromParent();
356  break;
357  }
358  default:
359  llvm_unreachable("Unsupported opcode for instruction");
360  }
361 }

451 void VPWidenCallRecipe::execute(VPTransformState &State) {
452  auto &CI = *cast<CallInst>(getUnderlyingInstr());
453  assert(!isa<DbgInfoIntrinsic>(CI) &&
454  "DbgInfoIntrinsic should have been dropped during VPlan construction");
455  State.setDebugLocFromInst(&CI);
456 
457  for (unsigned Part = 0; Part < State.UF; ++Part) {
458  SmallVector<Type *, 2> TysForDecl = {CI.getType()};
459  SmallVector<Value *, 4> Args;
460  for (const auto &I : enumerate(operands())) {
461  // Some intrinsics have a scalar argument - don't replace it with a
462  // vector.
463  Value *Arg;
464  if (VectorIntrinsicID == Intrinsic::not_intrinsic ||
465  !isVectorIntrinsicWithScalarOpAtArg(VectorIntrinsicID, I.index()))
466  Arg = State.get(I.value(), Part);
467  else
468  Arg = State.get(I.value(), VPIteration(0, 0));
469  if (isVectorIntrinsicWithOverloadTypeAtArg(VectorIntrinsicID, I.index()))
470  TysForDecl.push_back(Arg->getType());
471  Args.push_back(Arg);
472  }
473 
474  Function *VectorF;
475  if (VectorIntrinsicID != Intrinsic::not_intrinsic) {
476  // Use vector version of the intrinsic.
477  if (State.VF.isVector())
478  TysForDecl[0] =
479  VectorType::get(CI.getType()->getScalarType(), State.VF);
480  Module *M = State.Builder.GetInsertBlock()->getModule();
481  VectorF = Intrinsic::getDeclaration(M, VectorIntrinsicID, TysForDecl);
482  assert(VectorF && "Can't retrieve vector intrinsic.");
483  } else {
484 #ifndef NDEBUG
485  assert(Variant != nullptr && "Can't create vector function.");
486 #endif
487  VectorF = Variant;
488  }
489 
490  SmallVector<OperandBundleDef, 1> OpBundles;
491  CI.getOperandBundlesAsDefs(OpBundles);
492  CallInst *V = State.Builder.CreateCall(VectorF, Args, OpBundles);
493 
494  if (isa<FPMathOperator>(V))
495  V->copyFastMathFlags(&CI);
496 
497  State.set(this, V, Part);
498  State.addMetadata(V, &CI);
499  }

564 void VPWidenRecipe::execute(VPTransformState &State) {
565  auto &I = *cast<Instruction>(getUnderlyingValue());
566  auto &Builder = State.Builder;
567  switch (I.getOpcode()) {
568  case Instruction::Call:
569  case Instruction::Br:
570  case Instruction::PHI:
571  case Instruction::GetElementPtr:
572  case Instruction::Select:
573  llvm_unreachable("This instruction is handled by a different recipe.");
574  case Instruction::UDiv:
575  case Instruction::SDiv:
576  case Instruction::SRem:
577  case Instruction::URem:
578  case Instruction::Add:
579  case Instruction::FAdd:
580  case Instruction::Sub:
581  case Instruction::FSub:
582  case Instruction::FNeg:
583  case Instruction::Mul:
584  case Instruction::FMul:
585  case Instruction::FDiv:
586  case Instruction::FRem:
587  case Instruction::Shl:
588  case Instruction::LShr:
589  case Instruction::AShr:
590  case Instruction::And:
591  case Instruction::Or:
592  case Instruction::Xor: {
593  // Just widen unops and binops.
594  State.setDebugLocFromInst(&I);
595 
596  for (unsigned Part = 0; Part < State.UF; ++Part) {
597  SmallVector<Value *, 2> Ops;
598  for (VPValue *VPOp : operands())
599  Ops.push_back(State.get(VPOp, Part));
600 
601  Value *V = Builder.CreateNAryOp(I.getOpcode(), Ops);
602 
603  if (auto *VecOp = dyn_cast<Instruction>(V)) {
604  VecOp->copyIRFlags(&I);
605 
606  // If the instruction is vectorized and was in a basic block that needed
607  // predication, we can't propagate poison-generating flags (nuw/nsw,
608  // exact, etc.). The control flow has been linearized and the
609  // instruction is no longer guarded by the predicate, which could make
610  // the flag properties to no longer hold.
611  if (State.MayGeneratePoisonRecipes.contains(this))
612  VecOp->dropPoisonGeneratingFlags();
613  }
614 
615  // Use this vector value for all users of the original instruction.
616  State.set(this, V, Part);
617  State.addMetadata(V, &I);
618  }
619 
620  break;
621  }
622  case Instruction::Freeze: {
623  State.setDebugLocFromInst(&I);
624 
625  for (unsigned Part = 0; Part < State.UF; ++Part) {
626  Value *Op = State.get(getOperand(0), Part);
627 
628  Value *Freeze = Builder.CreateFreeze(Op);
629  State.set(this, Freeze, Part);
630  }
631  break;
632  }
633  case Instruction::ICmp:
634  case Instruction::FCmp: {
635  // Widen compares. Generate vector compares.
636  bool FCmp = (I.getOpcode() == Instruction::FCmp);
637  auto *Cmp = cast<CmpInst>(&I);
638  State.setDebugLocFromInst(Cmp);
639  for (unsigned Part = 0; Part < State.UF; ++Part) {
640  Value *A = State.get(getOperand(0), Part);
641  Value *B = State.get(getOperand(1), Part);
642  Value *C = nullptr;
643  if (FCmp) {
644  // Propagate fast math flags.
645  IRBuilder<>::FastMathFlagGuard FMFG(Builder);
646  Builder.setFastMathFlags(Cmp->getFastMathFlags());
647  C = Builder.CreateFCmp(Cmp->getPredicate(), A, B);
648  } else {
649  C = Builder.CreateICmp(Cmp->getPredicate(), A, B);
650  }
651  State.set(this, C, Part);
652  State.addMetadata(C, &I);
653  }
654 
655  break;
656  }
657 
658  case Instruction::ZExt:
659  case Instruction::SExt:
660  case Instruction::FPToUI:
661  case Instruction::FPToSI:
662  case Instruction::FPExt:
663  case Instruction::PtrToInt:
664  case Instruction::IntToPtr:
665  case Instruction::SIToFP:
666  case Instruction::UIToFP:
667  case Instruction::Trunc:
668  case Instruction::FPTrunc:
669  case Instruction::BitCast: {
670  auto *CI = cast<CastInst>(&I);
671  State.setDebugLocFromInst(CI);
672 
673  /// Vectorize casts.
674  Type *DestTy = (State.VF.isScalar())
675  ? CI->getType()
676  : VectorType::get(CI->getType(), State.VF);
677 
678  for (unsigned Part = 0; Part < State.UF; ++Part) {
679  Value *A = State.get(getOperand(0), Part);
680  Value *Cast = Builder.CreateCast(CI->getOpcode(), A, DestTy);
681  State.set(this, Cast, Part);
682  State.addMetadata(Cast, &I);
683  }
684  break;
685  }
686  default:
687  // This instruction is not vectorized by simple widening.
688  LLVM_DEBUG(dbgs() << "LV: Found an unhandled instruction: " << I);
689  llvm_unreachable("Unhandled instruction!");
690  } // end of switch.

753 void VPWidenGEPRecipe::execute(VPTransformState &State) {
754  auto *GEP = cast<GetElementPtrInst>(getUnderlyingInstr());
755  // Construct a vector GEP by widening the operands of the scalar GEP as
756  // necessary. We mark the vector GEP 'inbounds' if appropriate. A GEP
757  // results in a vector of pointers when at least one operand of the GEP
758  // is vector-typed. Thus, to keep the representation compact, we only use
759  // vector-typed operands for loop-varying values.
760 
761  if (State.VF.isVector() && IsPtrLoopInvariant && IsIndexLoopInvariant.all()) {
762  // If we are vectorizing, but the GEP has only loop-invariant operands,
763  // the GEP we build (by only using vector-typed operands for
764  // loop-varying values) would be a scalar pointer. Thus, to ensure we
765  // produce a vector of pointers, we need to either arbitrarily pick an
766  // operand to broadcast, or broadcast a clone of the original GEP.
767  // Here, we broadcast a clone of the original.
768  //
769  // TODO: If at some point we decide to scalarize instructions having
770  // loop-invariant operands, this special case will no longer be
771  // required. We would add the scalarization decision to
772  // collectLoopScalars() and teach getVectorValue() to broadcast
773  // the lane-zero scalar value.
774  auto *Clone = State.Builder.Insert(GEP->clone());
775  for (unsigned Part = 0; Part < State.UF; ++Part) {
776  Value *EntryPart = State.Builder.CreateVectorSplat(State.VF, Clone);
777  State.set(this, EntryPart, Part);
778  State.addMetadata(EntryPart, GEP);
779  }
780  } else {
781  // If the GEP has at least one loop-varying operand, we are sure to
782  // produce a vector of pointers. But if we are only unrolling, we want
783  // to produce a scalar GEP for each unroll part. Thus, the GEP we
784  // produce with the code below will be scalar (if VF == 1) or vector
785  // (otherwise). Note that for the unroll-only case, we still maintain
786  // values in the vector mapping with initVector, as we do for other
787  // instructions.
788  for (unsigned Part = 0; Part < State.UF; ++Part) {
789  // The pointer operand of the new GEP. If it's loop-invariant, we
790  // won't broadcast it.
791  auto *Ptr = IsPtrLoopInvariant
792  ? State.get(getOperand(0), VPIteration(0, 0))
793  : State.get(getOperand(0), Part);
794 
795  // Collect all the indices for the new GEP. If any index is
796  // loop-invariant, we won't broadcast it.
797  SmallVector<Value *, 4> Indices;
798  for (unsigned I = 1, E = getNumOperands(); I < E; I++) {
799  VPValue *Operand = getOperand(I);
800  if (IsIndexLoopInvariant[I - 1])
801  Indices.push_back(State.get(Operand, VPIteration(0, 0)));
802  else
803  Indices.push_back(State.get(Operand, Part));
804  }
805 
806  // If the GEP instruction is vectorized and was in a basic block that
807  // needed predication, we can't propagate the poison-generating 'inbounds'
808  // flag. The control flow has been linearized and the GEP is no longer
809  // guarded by the predicate, which could make the 'inbounds' properties to
810  // no longer hold.
811  bool IsInBounds =
812  GEP->isInBounds() && State.MayGeneratePoisonRecipes.count(this) == 0;
813 
814  // Create the new GEP. Note that this GEP may be a scalar if VF == 1,
815  // but it should be a vector, otherwise.
816  auto *NewGEP = State.Builder.CreateGEP(GEP->getSourceElementType(), Ptr,
817  Indices, "", IsInBounds);
818  assert((State.VF.isScalar() || NewGEP->getType()->isVectorTy()) &&
819  "NewGEP is not a pointer vector");
820  State.set(this, NewGEP, Part);
821  State.addMetadata(NewGEP, GEP);
822  }
823  }
824 }

842 void VPBlendRecipe::execute(VPTransformState &State) {
843  State.setDebugLocFromInst(Phi);
844  // We know that all PHIs in non-header blocks are converted into
845  // selects, so we don't have to worry about the insertion order and we
846  // can just use the builder.
847  // At this point we generate the predication tree. There may be
848  // duplications since this is a simple recursive scan, but future
849  // optimizations will clean it up.
850 
851  unsigned NumIncoming = getNumIncomingValues();
852 
853  // Generate a sequence of selects of the form:
854  // SELECT(Mask3, In3,
855  // SELECT(Mask2, In2,
856  // SELECT(Mask1, In1,
857  // In0)))
858  // Note that Mask0 is never used: lanes for which no path reaches this phi and
859  // are essentially undef are taken from In0.
860  VectorParts Entry(State.UF);
861  for (unsigned In = 0; In < NumIncoming; ++In) {
862  for (unsigned Part = 0; Part < State.UF; ++Part) {
863  // We might have single edge PHIs (blocks) - use an identity
864  // 'select' for the first PHI operand.
865  Value *In0 = State.get(getIncomingValue(In), Part);
866  if (In == 0)
867  Entry[Part] = In0; // Initialize with the first incoming value.
868  else {
869  // Select between the current value and the previous incoming edge
870  // based on the incoming mask.
871  Value *Cond = State.get(getMask(In), Part);
872  Entry[Part] =
873  State.Builder.CreateSelect(Cond, In0, Entry[Part], "predphi");
874  }
875  }
876  }
877  for (unsigned Part = 0; Part < State.UF; ++Part)
878  State.set(this, Entry[Part], Part);
879 }

974 void VPPredInstPHIRecipe::execute(VPTransformState &State) {
975  assert(State.Instance && "Predicated instruction PHI works per instance.");
976  Instruction *ScalarPredInst =
977  cast<Instruction>(State.get(getOperand(0), *State.Instance));
978  BasicBlock *PredicatedBB = ScalarPredInst->getParent();
979  BasicBlock *PredicatingBB = PredicatedBB->getSinglePredecessor();
980  assert(PredicatingBB && "Predicated block has no single predecessor.");
981  assert(isa<VPReplicateRecipe>(getOperand(0)) &&
982  "operand must be VPReplicateRecipe");
983 
984  // By current pack/unpack logic we need to generate only a single phi node: if
985  // a vector value for the predicated instruction exists at this point it means
986  // the instruction has vector users only, and a phi for the vector value is
987  // needed. In this case the recipe of the predicated instruction is marked to
988  // also do that packing, thereby "hoisting" the insert-element sequence.
989  // Otherwise, a phi node for the scalar value is needed.
990  unsigned Part = State.Instance->Part;
991  if (State.hasVectorValue(getOperand(0), Part)) {
992  Value *VectorValue = State.get(getOperand(0), Part);
993  InsertElementInst *IEI = cast<InsertElementInst>(VectorValue);
994  PHINode *VPhi = State.Builder.CreatePHI(IEI->getType(), 2);
995  VPhi->addIncoming(IEI->getOperand(0), PredicatingBB); // Unmodified vector.
996  VPhi->addIncoming(IEI, PredicatedBB); // New vector with inserted element.
997  if (State.hasVectorValue(this, Part))
998  State.reset(this, VPhi, Part);
999  else
1000  State.set(this, VPhi, Part);
1001  // NOTE: Currently we need to update the value of the operand, so the next
1002  // predicated iteration inserts its generated value in the correct vector.
1003  State.reset(getOperand(0), VPhi, Part);
1004  } else {
1005  Type *PredInstType = getOperand(0)->getUnderlyingValue()->getType();
1006  PHINode *Phi = State.Builder.CreatePHI(PredInstType, 2);
1007  Phi->addIncoming(PoisonValue::get(ScalarPredInst->getType()),
1008  PredicatingBB);
1009  Phi->addIncoming(ScalarPredInst, PredicatedBB);
1010  if (State.hasScalarValue(this, *State.Instance))
1011  State.reset(this, Phi, *State.Instance);
1012  else
1013  State.set(this, Phi, *State.Instance);
1014  // NOTE: Currently we need to update the value of the operand, so the next
1015  // predicated iteration inserts its generated value in the correct vector.
1016  State.reset(getOperand(0), Phi, *State.Instance);
1017  }

1185 void VPReductionPHIRecipe::execute(VPTransformState &State) {
1186  PHINode *PN = cast<PHINode>(getUnderlyingValue());
1187  auto &Builder = State.Builder;
1188 
1189  // In order to support recurrences we need to be able to vectorize Phi nodes.
1190  // Phi nodes have cycles, so we need to vectorize them in two stages. This is
1191  // stage #1: We create a new vector PHI node with no incoming edges. We'll use
1192  // this value when we vectorize all of the instructions that use the PHI.
1193  bool ScalarPHI = State.VF.isScalar() || IsInLoop;
1194  Type *VecTy =
1195  ScalarPHI ? PN->getType() : VectorType::get(PN->getType(), State.VF);
1196 
1197  BasicBlock *HeaderBB = State.CFG.PrevBB;
1198  assert(State.CurrentVectorLoop->getHeader() == HeaderBB &&
1199  "recipe must be in the vector loop header");
1200  unsigned LastPartForNewPhi = isOrdered() ? 1 : State.UF;
1201  for (unsigned Part = 0; Part < LastPartForNewPhi; ++Part) {
1202  Value *EntryPart =
1203  PHINode::Create(VecTy, 2, "vec.phi", &*HeaderBB->getFirstInsertionPt());
1204  State.set(this, EntryPart, Part);
1205  }
1206 
1207  BasicBlock *VectorPH = State.CFG.getPreheaderBBFor(this);
1208 
1209  // Reductions do not have to start at zero. They can start with
1210  // any loop invariant values.
1211  VPValue *StartVPV = getStartValue();
1212  Value *StartV = StartVPV->getLiveInIRValue();
1213 
1214  Value *Iden = nullptr;
1215  RecurKind RK = RdxDesc.getRecurrenceKind();
1216  if (RecurrenceDescriptor::isMinMaxRecurrenceKind(RK) ||
1217  RecurrenceDescriptor::isSelectCmpRecurrenceKind(RK)) {
1218  // MinMax reduction have the start value as their identify.
1219  if (ScalarPHI) {
1220  Iden = StartV;
1221  } else {
1222  IRBuilderBase::InsertPointGuard IPBuilder(Builder);
1223  Builder.SetInsertPoint(VectorPH->getTerminator());
1224  StartV = Iden =
1225  Builder.CreateVectorSplat(State.VF, StartV, "minmax.ident");
1226  }
1227  } else {
1228  Iden = RdxDesc.getRecurrenceIdentity(RK, VecTy->getScalarType(),
1229  RdxDesc.getFastMathFlags());
1230 
1231  if (!ScalarPHI) {
1232  Iden = Builder.CreateVectorSplat(State.VF, Iden);
1233  IRBuilderBase::InsertPointGuard IPBuilder(Builder);
1234  Builder.SetInsertPoint(VectorPH->getTerminator());
1235  Constant *Zero = Builder.getInt32(0);
1236  StartV = Builder.CreateInsertElement(Iden, StartV, Zero);
1237  }
1238  }
1239 
1240  for (unsigned Part = 0; Part < LastPartForNewPhi; ++Part) {
1241  Value *EntryPart = State.get(this, Part);
1242  // Make sure to add the reduction start value only to the
1243  // first unroll part.
1244  Value *StartVal = (Part == 0) ? StartV : Iden;
1245  cast<PHINode>(EntryPart)->addIncoming(StartVal, VectorPH);
1246  }
1247 }
