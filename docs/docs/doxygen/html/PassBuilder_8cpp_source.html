508 /// This performs customized parsing of pass name with parameters.
509 ///
510 /// We do not need parametrization of passes in textual pipeline very often,
511 /// yet on a rare occasion ability to specify parameters right there can be
512 /// useful.
513 ///
514 /// \p Name - parameterized specification of a pass from a textual pipeline
515 /// is a string in a form of :
516 /// PassName '<' parameter-list '>'
517 ///
518 /// Parameter list is being parsed by the parser callable argument, \p Parser,
519 /// It takes a string-ref of parameters and returns either StringError or a
520 /// parameter list in a form of a custom parameters type, all wrapped into
521 /// Expected<> template class.
522 ///
523 template <typename ParametersParseCallableT>
524 auto parsePassParameters(ParametersParseCallableT &&Parser, StringRef Name,
525  StringRef PassName) -> decltype(Parser(StringRef{})) {
526  using ParametersT = typename decltype(Parser(StringRef{}))::value_type;
527 
528  StringRef Params = Name;
529  if (!Params.consume_front(PassName)) {
530  assert(false &&
531  "unable to strip pass name from parametrized pass specification");
532  }
533  if (!Params.empty() &&
534  (!Params.consume_front("<") || !Params.consume_back(">"))) {
535  assert(false && "invalid format for parametrized pass name");
536  }
537 
538  Expected<ParametersT> Result = Parser(Params);
539  assert((Result || Result.template errorIsA<StringError>()) &&
540  "Pass parameter parser can only return StringErrors.");
541  return Result;
542 }

1111 std::optional<std::vector<PassBuilder::PipelineElement>>
1112 PassBuilder::parsePipelineText(StringRef Text) {
1113  std::vector<PipelineElement> ResultPipeline;
1114 
1115  SmallVector<std::vector<PipelineElement> *, 4> PipelineStack = {
1116  &ResultPipeline};
1117  for (;;) {
1118  std::vector<PipelineElement> &Pipeline = *PipelineStack.back();
1119  size_t Pos = Text.find_first_of(",()");
1120  Pipeline.push_back({Text.substr(0, Pos), {}});
1121 
1122  // If we have a single terminating name, we're done.
1123  if (Pos == Text.npos)
1124  break;
1125 
1126  char Sep = Text[Pos];
1127  Text = Text.substr(Pos + 1);
1128  if (Sep == ',')
1129  // Just a name ending in a comma, continue.
1130  continue;
1131 
1132  if (Sep == '(') {
1133  // Push the inner pipeline onto the stack to continue processing.
1134  PipelineStack.push_back(&Pipeline.back().InnerPipeline);
1135  continue;
1136  }

1140  // empty strings in the pipeline.
1141  do {

1143  if (PipelineStack.size() == 1)
1144  return std::nullopt;
1145 
1146  PipelineStack.pop_back();
1147  } while (Text.consume_front(")"));
1148 
1149  // Check if we've finished parsing.
1150  if (Text.empty())
1151  break;
1152 
1153  // Otherwise, the end of an inner pipeline always has to be followed by
1154  // a comma, and then we can continue.
1155  if (!Text.consume_front(","))
1156  return std::nullopt;
1157  }
1158 
1159  if (PipelineStack.size() > 1)
1160  // Unbalanced paretheses.
1161  return std::nullopt;
1162 
1163  assert(PipelineStack.back() == &ResultPipeline &&
1164  "Wrong pipeline at the bottom of the stack!");
1165  return {std::move(ResultPipeline)};
1166 }
