146 /// Return the datalayout string of a subtarget.
147 static std::string getDataLayoutString(const Triple &T) {
148  bool is64Bit = T.getArch() == Triple::ppc64 || T.getArch() == Triple::ppc64le;
149  std::string Ret;
150 
151  // Most PPC* platforms are big endian, PPC(64)LE is little endian.
152  if (isLittleEndianTriple(T))
153  Ret = "e";
154  else
155  Ret = "E";
156 
157  Ret += DataLayout::getManglingComponent(T);
158 
159  // PPC32 has 32 bit pointers. The PS3 (OS Lv2) is a PPC64 machine with 32 bit
160  // pointers.
161  if (!is64Bit || T.getOS() == Triple::Lv2)
162  Ret += "-p:32:32";
163 
164  // Note, the alignment values for f64 and i64 on ppc64 in Darwin
165  // documentation are wrong; these are correct (i.e. "what gcc does").
166  Ret += "-i64:64";
167 
168  // PPC64 has 32 and 64 bit registers, PPC32 has only 32 bit ones.
169  if (is64Bit)
170  Ret += "-n32:64";
171  else
172  Ret += "-n32";
173 
174  // Specify the vector alignment explicitly. For v256i1 and v512i1, the
175  // calculated alignment would be 256*alignment(i1) and 512*alignment(i1),
176  // which is 256 and 512 bytes - way over aligned.
177  if (is64Bit && (T.isOSAIX() || T.isOSLinux()))
178  Ret += "-S128-v256:256:256-v512:512:512";
179 
180  return Ret;
181 }

183 static std::string computeFSAdditions(StringRef FS, CodeGenOpt::Level OL,
184  const Triple &TT) {
185  std::string FullFS = std::string(FS);
186 
187  // Make sure 64-bit features are available when CPUname is generic
188  if (TT.getArch() == Triple::ppc64 || TT.getArch() == Triple::ppc64le) {
189  if (!FullFS.empty())
190  FullFS = "+64bit," + FullFS;
191  else
192  FullFS = "+64bit";
193  }
194 
195  if (OL >= CodeGenOpt::Default) {
196  if (!FullFS.empty())
197  FullFS = "+crbits," + FullFS;
198  else
199  FullFS = "+crbits";
200  }
201 
202  if (OL != CodeGenOpt::None) {
203  if (!FullFS.empty())
204  FullFS = "+invariant-function-descriptors," + FullFS;
205  else
206  FullFS = "+invariant-function-descriptors";
207  }
208 
209  if (TT.isOSAIX()) {
210  if (!FullFS.empty())
211  FullFS = "+aix," + FullFS;
212  else
213  FullFS = "+aix";
214  }
215 
216  return FullFS;
217 }

226 static PPCTargetMachine::PPCABI computeTargetABI(const Triple &TT,
227  const TargetOptions &Options) {
228  if (Options.MCOptions.getABIName().startswith("elfv1"))
229  return PPCTargetMachine::PPC_ABI_ELFv1;
230  else if (Options.MCOptions.getABIName().startswith("elfv2"))
231  return PPCTargetMachine::PPC_ABI_ELFv2;
232 
233  assert(Options.MCOptions.getABIName().empty() &&
234  "Unknown target-abi option!");
235 
236  switch (TT.getArch()) {
237  case Triple::ppc64le:
238  return PPCTargetMachine::PPC_ABI_ELFv2;
239  case Triple::ppc64:
240  return PPCTargetMachine::PPC_ABI_ELFv1;
241  default:
242  return PPCTargetMachine::PPC_ABI_UNKNOWN;
243  }
244 }

246 static Reloc::Model getEffectiveRelocModel(const Triple &TT,
247  std::optional<Reloc::Model> RM) {
248  assert((!TT.isOSAIX() || !RM || *RM == Reloc::PIC_) &&
249  "Invalid relocation model for AIX.");
250 
251  if (RM)
252  return *RM;
253 
254  // Big Endian PPC and AIX default to PIC.
255  if (TT.getArch() == Triple::ppc64 || TT.isOSAIX())
256  return Reloc::PIC_;
257 
258  // Rest are static by default.
259  return Reloc::Static;
260 }

262 static CodeModel::Model
263 getEffectivePPCCodeModel(const Triple &TT, std::optional<CodeModel::Model> CM,
264  bool JIT) {
265  if (CM) {
266  if (*CM == CodeModel::Tiny)
267  report_fatal_error("Target does not support the tiny CodeModel", false);
268  if (*CM == CodeModel::Kernel)
269  report_fatal_error("Target does not support the kernel CodeModel", false);
270  return *CM;
271  }
272 
273  if (JIT)
274  return CodeModel::Small;
275  if (TT.isOSAIX())
276  return CodeModel::Small;
277 
278  assert(TT.isOSBinFormatELF() && "All remaining PPC OSes are ELF based.");
279 
280  if (TT.isArch32Bit())
281  return CodeModel::Small;
282 
283  assert(TT.isArch64Bit() && "Unsupported PPC architecture.");
284  return CodeModel::Medium;
285 }

288 static ScheduleDAGInstrs *createPPCMachineScheduler(MachineSchedContext *C) {
289  const PPCSubtarget &ST = C->MF->getSubtarget<PPCSubtarget>();
290  ScheduleDAGMILive *DAG =
291  new ScheduleDAGMILive(C, ST.usePPCPreRASchedStrategy() ?
292  std::make_unique<PPCPreRASchedStrategy>(C) :
293  std::make_unique<GenericScheduler>(C));
294  // add DAG Mutations here.
295  DAG->addMutation(createCopyConstrainDAGMutation(DAG->TII, DAG->TRI));
296  if (ST.hasStoreFusion())
297  DAG->addMutation(createStoreClusterDAGMutation(DAG->TII, DAG->TRI));
298  if (ST.hasFusion())
299  DAG->addMutation(createPowerPCMacroFusionDAGMutation());
300 
301  return DAG;
302 }

316  return DAG;

341 const PPCSubtarget *
342 PPCTargetMachine::getSubtargetImpl(const Function &F) const {
343  Attribute CPUAttr = F.getFnAttribute("target-cpu");
344  Attribute TuneAttr = F.getFnAttribute("tune-cpu");
345  Attribute FSAttr = F.getFnAttribute("target-features");
346 
347  std::string CPU =
348  CPUAttr.isValid() ? CPUAttr.getValueAsString().str() : TargetCPU;
349  std::string TuneCPU =
350  TuneAttr.isValid() ? TuneAttr.getValueAsString().str() : CPU;
351  std::string FS =
352  FSAttr.isValid() ? FSAttr.getValueAsString().str() : TargetFS;
353 
354  // FIXME: This is related to the code below to reset the target options,
355  // we need to know whether or not the soft float flag is set on the
356  // function before we can generate a subtarget. We also need to use
357  // it as a key for the subtarget since that can be the only difference
358  // between two functions.
359  bool SoftFloat = F.getFnAttribute("use-soft-float").getValueAsBool();
360  // If the soft float attribute is set on the function turn on the soft float
361  // subtarget feature.
362  if (SoftFloat)
363  FS += FS.empty() ? "-hard-float" : ",-hard-float";
364 
365  auto &I = SubtargetMap[CPU + TuneCPU + FS];
366  if (!I) {
367  // This needs to be done before we create a new subtarget since any
368  // creation will depend on the TM and the code generation flags on the
369  // function that reside in TargetOptions.
370  resetTargetOptions(F);
371  I = std::make_unique<PPCSubtarget>(
372  TargetTriple, CPU, TuneCPU,
373  // FIXME: It would be good to have the subtarget additions here
374  // not necessary. Anything that turns them on/off (overrides) ends
375  // up being put at the end of the feature string, but the defaults
376  // shouldn't require adding them. Fixing this means pulling Feature64Bit
377  // out of most of the target cpus in the .td file and making it set only
378  // as part of initialization via the TargetTriple.
379  computeFSAdditions(FS, getOptLevel(), getTargetTriple()), *this);
380  }
381  return I.get();
382 }

436 void PPCPassConfig::addIRPasses() {
437  if (TM->getOptLevel() != CodeGenOpt::None)
438  addPass(createPPCBoolRetToIntPass());
439  addPass(createAtomicExpandPass());
440 
441  // Lower generic MASSV routines to PowerPC subtarget-specific entries.
442  addPass(createPPCLowerMASSVEntriesPass());
443 
444  // Generate PowerPC target-specific entries for scalar math functions
445  // that are available in IBM MASS (scalar) library.
446  if (TM->getOptLevel() == CodeGenOpt::Aggressive &&
447  EnablePPCGenScalarMASSEntries) {
448  TM->Options.PPCGenScalarMASSEntries = EnablePPCGenScalarMASSEntries;
449  addPass(createPPCGenScalarMASSEntriesPass());
450  }
451 
452  // If explicitly requested, add explicit data prefetch intrinsics.
453  if (EnablePrefetch.getNumOccurrences() > 0)
454  addPass(createLoopDataPrefetchPass());
455 
456  if (TM->getOptLevel() >= CodeGenOpt::Default && EnableGEPOpt) {
457  // Call SeparateConstOffsetFromGEP pass to extract constants within indices
458  // and lower a GEP with multiple indices to either arithmetic operations or
459  // multiple GEPs with single index.
460  addPass(createSeparateConstOffsetFromGEPPass(true));
461  // Call EarlyCSE pass to find and remove subexpressions in the lowered
462  // result.
463  addPass(createEarlyCSEPass());
464  // Do loop invariant code motion in case part of the lowered result is
465  // invariant.
466  addPass(createLICMPass());
467  }
468 
469  TargetPassConfig::addIRPasses();
470 }

504 void PPCPassConfig::addMachineSSAOptimization() {
505  // Run CTR loops pass before any cfg modification pass to prevent the
506  // canonical form of hardware loop from being destroied.
507  if (!DisableCTRLoops && getOptLevel() != CodeGenOpt::None)
508  addPass(createPPCCTRLoopsPass());
509 
510  // PPCBranchCoalescingPass need to be done before machine sinking
511  // since it merges empty blocks.
512  if (EnableBranchCoalescing && getOptLevel() != CodeGenOpt::None)
513  addPass(createPPCBranchCoalescingPass());
514  TargetPassConfig::addMachineSSAOptimization();
515  // For little endian, remove where possible the vector swap instructions
516  // introduced at code generation to normalize vector element order.
517  if (TM->getTargetTriple().getArch() == Triple::ppc64le &&
518  !DisableVSXSwapRemoval)
519  addPass(createPPCVSXSwapRemovalPass());
520  // Reduce the number of cr-logical ops.
521  if (ReduceCRLogical && getOptLevel() != CodeGenOpt::None)
522  addPass(createPPCReduceCRLogicalsPass());
523  // Target-specific peephole cleanups performed after instruction
524  // selection.
525  if (!DisableMIPeephole) {
526  addPass(createPPCMIPeepholePass());
527  addPass(&DeadMachineInstructionElimID);
528  }
529 }

531 void PPCPassConfig::addPreRegAlloc() {
532  if (getOptLevel() != CodeGenOpt::None) {
533  initializePPCVSXFMAMutatePass(*PassRegistry::getPassRegistry());
534  insertPass(VSXFMAMutateEarly ? &RegisterCoalescerID : &MachineSchedulerID,
535  &PPCVSXFMAMutateID);
536  }
537 
538  // FIXME: We probably don't need to run these for -fPIE.
539  if (getPPCTargetMachine().isPositionIndependent()) {
540  // FIXME: LiveVariables should not be necessary here!
541  // PPCTLSDynamicCallPass uses LiveIntervals which previously dependent on
542  // LiveVariables. This (unnecessary) dependency has been removed now,
543  // however a stage-2 clang build fails without LiveVariables computed here.
544  addPass(&LiveVariablesID);
545  addPass(createPPCTLSDynamicCallPass());
546  }
547  if (EnableExtraTOCRegDeps)
548  addPass(createPPCTOCRegDepsPass());
549 
550  if (getOptLevel() != CodeGenOpt::None)
551  addPass(&MachinePipelinerID);
552 }
