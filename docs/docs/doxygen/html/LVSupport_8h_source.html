137 // Double map data structure.
138 template <typename FirstKeyType, typename SecondKeyType, typename ValueType>
139 class LVDoubleMap {
140  static_assert(std::is_pointer<ValueType>::value,
141  "ValueType must be a pointer.");
142  using LVSecondMapType = std::map<SecondKeyType, ValueType>;
143  using LVFirstMapType =
144  std::map<FirstKeyType, std::unique_ptr<LVSecondMapType>>;
145  using LVAuxMapType = std::map<SecondKeyType, FirstKeyType>;
146  using LVValueTypes = std::vector<ValueType>;
147  LVFirstMapType FirstMap;
148  LVAuxMapType AuxMap;
149 
150 public:
151  void add(FirstKeyType FirstKey, SecondKeyType SecondKey, ValueType Value) {
152  typename LVFirstMapType::iterator FirstIter = FirstMap.find(FirstKey);
153  if (FirstIter == FirstMap.end()) {
154  auto SecondMapSP = std::make_unique<LVSecondMapType>();
155  SecondMapSP->emplace(SecondKey, Value);
156  FirstMap.emplace(FirstKey, std::move(SecondMapSP));
157  } else {
158  LVSecondMapType *SecondMap = FirstIter->second.get();
159  if (SecondMap->find(SecondKey) == SecondMap->end())
160  SecondMap->emplace(SecondKey, Value);
161  }
162 
163  typename LVAuxMapType::iterator AuxIter = AuxMap.find(SecondKey);
164  if (AuxIter == AuxMap.end()) {
165  AuxMap.emplace(SecondKey, FirstKey);
166  }
167  }
168 
169  LVSecondMapType *findMap(FirstKeyType FirstKey) const {
170  typename LVFirstMapType::const_iterator FirstIter = FirstMap.find(FirstKey);
171  if (FirstIter == FirstMap.end())
172  return nullptr;
173 
174  return FirstIter->second.get();
175  }
176 
177  ValueType find(FirstKeyType FirstKey, SecondKeyType SecondKey) const {
178  LVSecondMapType *SecondMap = findMap(FirstKey);
179  if (!SecondMap)
180  return nullptr;
181 
182  typename LVSecondMapType::const_iterator SecondIter =
183  SecondMap->find(SecondKey);
184  return (SecondIter != SecondMap->end()) ? SecondIter->second : nullptr;
185  }
186 
187  ValueType find(SecondKeyType SecondKey) const {
188  typename LVAuxMapType::const_iterator AuxIter = AuxMap.find(SecondKey);
189  if (AuxIter == AuxMap.end())
190  return nullptr;
191  return find(AuxIter->second, SecondKey);
192  }
193 
194  // Return a vector with all the 'ValueType' values.
195  LVValueTypes find() const {
196  LVValueTypes Values;
197  if (FirstMap.empty())
198  return Values;
199  for (typename LVFirstMapType::const_reference FirstEntry : FirstMap) {
200  LVSecondMapType &SecondMap = *FirstEntry.second;
201  for (typename LVSecondMapType::const_reference SecondEntry : SecondMap)
202  Values.push_back(SecondEntry.second);
203  }
204  return Values;
205  }
206 };
