23 void X86InstrPostProcess::setMemBarriers(std::unique_ptr<Instruction> &Inst,
24  const MCInst &MCI) {

26  case X86::MFENCE:

28  Inst->setStoreBarrier(true);
29  break;
30  case X86::LFENCE:
31  Inst->setLoadBarrier(true);
32  break;
33  case X86::SFENCE:
34  Inst->setStoreBarrier(true);

36  }
37 }
