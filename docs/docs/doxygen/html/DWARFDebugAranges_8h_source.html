38  struct Range {

40  : LowPC(LowPC), Length(HighPC - LowPC), CUOffset(CUOffset) {}
41 
42  void setHighPC(uint64_t HighPC) {
43  if (HighPC == -1ULL || HighPC <= LowPC)
44  Length = 0;
45  else
46  Length = HighPC - LowPC;
47  }

49  uint64_t HighPC() const {
