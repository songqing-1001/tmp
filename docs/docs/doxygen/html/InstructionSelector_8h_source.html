50 /// Container class for CodeGen predicate results.
51 /// This is convenient because std::bitset does not have a constructor
52 /// with an initializer list of set bits.
53 ///
54 /// Each InstructionSelector subclass should define a PredicateBitset class
55 /// with:
56 /// const unsigned MAX_SUBTARGET_PREDICATES = 192;
57 /// using PredicateBitset = PredicateBitsetImpl<MAX_SUBTARGET_PREDICATES>;
58 /// and updating the constant to suit the target. Tablegen provides a suitable
59 /// definition for the predicates in use in <Target>GenGlobalISel.inc when
60 /// GET_GLOBALISEL_PREDICATE_BITSET is defined.
61 template <std::size_t MaxPredicates>
62 class PredicateBitsetImpl : public std::bitset<MaxPredicates> {
63 public:
64  // Cannot inherit constructors because it's not supported by VC++..
65  PredicateBitsetImpl() = default;
66 
67  PredicateBitsetImpl(const std::bitset<MaxPredicates> &B)
68  : std::bitset<MaxPredicates>(B) {}
69 
70  PredicateBitsetImpl(std::initializer_list<unsigned> Init) {
71  for (auto I : Init)
72  std::bitset<MaxPredicates>::set(I);
73  }
74 };

438  /// Select the (possibly generic) instruction \p I to only use target-specific
439  /// opcodes. It is OK to insert multiple instructions, but they cannot be
440  /// generic pre-isel instructions.
441  ///
442  /// \returns whether selection succeeded.
443  /// \pre I.getParent() && I.getParent()->getParent()
444  /// \post
445  /// if returns true:
446  /// for I in all mutated/inserted instructions:
447  /// !isPreISelGenericOpcode(I.getOpcode())
448  virtual bool select(MachineInstr &I) = 0;
449 
450  CodeGenCoverage *CoverageInfo = nullptr;
451  GISelKnownBits *KnownBits = nullptr;
452  MachineFunction *MF = nullptr;
453  ProfileSummaryInfo *PSI = nullptr;
454  BlockFrequencyInfo *BFI = nullptr;
455  // For some predicates, we need to track the current MBB.
456  MachineBasicBlock *CurMBB = nullptr;
457 
458  virtual void setupGeneratedPerFunctionState(MachineFunction &MF) {
459  llvm_unreachable("TableGen should have emitted implementation");
460  }

464  CodeGenCoverage &covinfo, ProfileSummaryInfo *psi,

467  KnownBits = KB;
