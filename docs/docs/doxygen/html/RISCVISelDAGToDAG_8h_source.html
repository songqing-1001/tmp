115  // Return the RISC-V condition code that matches the given DAG integer
116  // condition code. The CondCode must be one of those supported by the RISC-V
117  // ISA (see translateSetCCForBranch).
118  static RISCVCC::CondCode getRISCVCCForIntCC(ISD::CondCode CC) {
119  switch (CC) {
120  default:
121  llvm_unreachable("Unsupported CondCode");
122  case ISD::SETEQ:
123  return RISCVCC::COND_EQ;
124  case ISD::SETNE:
125  return RISCVCC::COND_NE;
126  case ISD::SETLT:
127  return RISCVCC::COND_LT;
128  case ISD::SETGE:
129  return RISCVCC::COND_GE;
130  case ISD::SETULT:
131  return RISCVCC::COND_LTU;
132  case ISD::SETUGE:
133  return RISCVCC::COND_GEU;
134  }
135  }
