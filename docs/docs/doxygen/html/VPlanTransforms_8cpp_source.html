26 void VPlanTransforms::VPInstructionsToVPRecipes(
27  Loop *OrigLoop, VPlanPtr &Plan,
28  function_ref<const InductionDescriptor *(PHINode *)>
29  GetIntOrFpInductionDescriptor,
30  SmallPtrSetImpl<Instruction *> &DeadInstructions, ScalarEvolution &SE,
31  const TargetLibraryInfo &TLI) {
32 
33  ReversePostOrderTraversal<VPBlockDeepTraversalWrapper<VPBlockBase *>> RPOT(
34  Plan->getEntry());
35  for (VPBasicBlock *VPBB : VPBlockUtils::blocksOnly<VPBasicBlock>(RPOT)) {
36  VPRecipeBase *Term = VPBB->getTerminator();
37  auto EndIter = Term ? Term->getIterator() : VPBB->end();
38  // Introduce each ingredient into VPlan.
39  for (VPRecipeBase &Ingredient :
40  make_early_inc_range(make_range(VPBB->begin(), EndIter))) {
41 
42  VPValue *VPV = Ingredient.getVPSingleValue();
43  Instruction *Inst = cast<Instruction>(VPV->getUnderlyingValue());
44  if (DeadInstructions.count(Inst)) {
45  VPValue DummyValue;
46  VPV->replaceAllUsesWith(&DummyValue);
47  Ingredient.eraseFromParent();
48  continue;
49  }
50 
51  VPRecipeBase *NewRecipe = nullptr;
52  if (auto *VPPhi = dyn_cast<VPWidenPHIRecipe>(&Ingredient)) {
53  auto *Phi = cast<PHINode>(VPPhi->getUnderlyingValue());
54  if (const auto *II = GetIntOrFpInductionDescriptor(Phi)) {
55  VPValue *Start = Plan->getOrAddVPValue(II->getStartValue());
56  VPValue *Step =
57  vputils::getOrCreateVPValueForSCEVExpr(*Plan, II->getStep(), SE);
58  NewRecipe =
59  new VPWidenIntOrFpInductionRecipe(Phi, Start, Step, *II, true);
60  } else {
61  Plan->addVPValue(Phi, VPPhi);
62  continue;
63  }
64  } else {
65  assert(isa<VPInstruction>(&Ingredient) &&
66  "only VPInstructions expected here");
67  assert(!isa<PHINode>(Inst) && "phis should be handled above");
68  // Create VPWidenMemoryInstructionRecipe for loads and stores.
69  if (LoadInst *Load = dyn_cast<LoadInst>(Inst)) {
70  NewRecipe = new VPWidenMemoryInstructionRecipe(
71  *Load, Plan->getOrAddVPValue(getLoadStorePointerOperand(Inst)),
72  nullptr /*Mask*/, false /*Consecutive*/, false /*Reverse*/);
73  } else if (StoreInst *Store = dyn_cast<StoreInst>(Inst)) {
74  NewRecipe = new VPWidenMemoryInstructionRecipe(
75  *Store, Plan->getOrAddVPValue(getLoadStorePointerOperand(Inst)),
76  Plan->getOrAddVPValue(Store->getValueOperand()), nullptr /*Mask*/,
77  false /*Consecutive*/, false /*Reverse*/);
78  } else if (GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(Inst)) {
79  NewRecipe = new VPWidenGEPRecipe(
80  GEP, Plan->mapToVPValues(GEP->operands()), OrigLoop);
81  } else if (CallInst *CI = dyn_cast<CallInst>(Inst)) {
82  NewRecipe =
83  new VPWidenCallRecipe(*CI, Plan->mapToVPValues(CI->args()),
84  getVectorIntrinsicIDForCall(CI, &TLI));
85  } else if (SelectInst *SI = dyn_cast<SelectInst>(Inst)) {
86  bool InvariantCond =
87  SE.isLoopInvariant(SE.getSCEV(SI->getOperand(0)), OrigLoop);
88  NewRecipe = new VPWidenSelectRecipe(
89  *SI, Plan->mapToVPValues(SI->operands()), InvariantCond);
90  } else {
91  NewRecipe =
92  new VPWidenRecipe(*Inst, Plan->mapToVPValues(Inst->operands()));
93  }
94  }
95 
96  NewRecipe->insertBefore(&Ingredient);
97  if (NewRecipe->getNumDefinedValues() == 1)
98  VPV->replaceAllUsesWith(NewRecipe->getVPSingleValue());
99  else
100  assert(NewRecipe->getNumDefinedValues() == 0 &&
101  "Only recpies with zero or one defined values expected");
102  Ingredient.eraseFromParent();
103  Plan->removeVPValueFor(Inst);
104  for (auto *Def : NewRecipe->definedValues()) {
105  Plan->addVPValue(Inst, Def);
106  }
107  }
108  }
109 }

111 bool VPlanTransforms::sinkScalarOperands(VPlan &Plan) {
112  auto Iter = vp_depth_first_deep(Plan.getEntry());
113  bool Changed = false;
114  // First, collect the operands of all recipes in replicate blocks as seeds for
115  // sinking.
116  SetVector<std::pair<VPBasicBlock *, VPRecipeBase *>> WorkList;
117  for (VPRegionBlock *VPR : VPBlockUtils::blocksOnly<VPRegionBlock>(Iter)) {
118  VPBasicBlock *EntryVPBB = VPR->getEntryBasicBlock();
119  if (!VPR->isReplicator() || EntryVPBB->getSuccessors().size() != 2)
120  continue;
121  VPBasicBlock *VPBB = dyn_cast<VPBasicBlock>(EntryVPBB->getSuccessors()[0]);
122  if (!VPBB || VPBB->getSingleSuccessor() != VPR->getExitingBasicBlock())
123  continue;
124  for (auto &Recipe : *VPBB) {
125  for (VPValue *Op : Recipe.operands())
126  if (auto *Def = Op->getDefiningRecipe())
127  WorkList.insert(std::make_pair(VPBB, Def));
128  }
129  }
130 
131  bool ScalarVFOnly = Plan.hasScalarVFOnly();
132  // Try to sink each replicate or scalar IV steps recipe in the worklist.
133  for (unsigned I = 0; I != WorkList.size(); ++I) {
134  VPBasicBlock *SinkTo;
135  VPRecipeBase *SinkCandidate;
136  std::tie(SinkTo, SinkCandidate) = WorkList[I];
137  if (SinkCandidate->getParent() == SinkTo ||
138  SinkCandidate->mayHaveSideEffects() ||
139  SinkCandidate->mayReadOrWriteMemory())
140  continue;
141  if (auto *RepR = dyn_cast<VPReplicateRecipe>(SinkCandidate)) {
142  if (!ScalarVFOnly && RepR->isUniform())
143  continue;
144  } else if (!isa<VPScalarIVStepsRecipe>(SinkCandidate))
145  continue;
146 
147  bool NeedsDuplicating = false;
148  // All recipe users of the sink candidate must be in the same block SinkTo
149  // or all users outside of SinkTo must be uniform-after-vectorization (
150  // i.e., only first lane is used) . In the latter case, we need to duplicate
151  // SinkCandidate.
152  auto CanSinkWithUser = [SinkTo, &NeedsDuplicating,
153  SinkCandidate](VPUser *U) {
154  auto *UI = dyn_cast<VPRecipeBase>(U);
155  if (!UI)
156  return false;
157  if (UI->getParent() == SinkTo)
158  return true;
159  NeedsDuplicating =
160  UI->onlyFirstLaneUsed(SinkCandidate->getVPSingleValue());
161  // We only know how to duplicate VPRecipeRecipes for now.
162  return NeedsDuplicating && isa<VPReplicateRecipe>(SinkCandidate);
163  };
164  if (!all_of(SinkCandidate->getVPSingleValue()->users(), CanSinkWithUser))
165  continue;
166 
167  if (NeedsDuplicating) {
168  if (ScalarVFOnly)
169  continue;
170  Instruction *I = cast<Instruction>(
171  cast<VPReplicateRecipe>(SinkCandidate)->getUnderlyingValue());
172  auto *Clone =
173  new VPReplicateRecipe(I, SinkCandidate->operands(), true, false);
174  // TODO: add ".cloned" suffix to name of Clone's VPValue.
175 
176  Clone->insertBefore(SinkCandidate);
177  for (auto *U : to_vector(SinkCandidate->getVPSingleValue()->users())) {
178  auto *UI = cast<VPRecipeBase>(U);
179  if (UI->getParent() == SinkTo)
180  continue;
181 
182  for (unsigned Idx = 0; Idx != UI->getNumOperands(); Idx++) {
183  if (UI->getOperand(Idx) != SinkCandidate->getVPSingleValue())
184  continue;
185  UI->setOperand(Idx, Clone);
186  }
187  }
188  }
189  SinkCandidate->moveBefore(*SinkTo, SinkTo->getFirstNonPhi());
190  for (VPValue *Op : SinkCandidate->operands())
191  if (auto *Def = Op->getDefiningRecipe())
192  WorkList.insert(std::make_pair(SinkTo, Def));
193  Changed = true;
194  }
195  return Changed;
196 }

209 /// If \p R is a triangle region, return the 'then' block of the triangle.
210 static VPBasicBlock *getPredicatedThenBlock(VPRegionBlock *R) {
211  auto *EntryBB = cast<VPBasicBlock>(R->getEntry());
212  if (EntryBB->getNumSuccessors() != 2)
213  return nullptr;
214 
215  auto *Succ0 = dyn_cast<VPBasicBlock>(EntryBB->getSuccessors()[0]);
216  auto *Succ1 = dyn_cast<VPBasicBlock>(EntryBB->getSuccessors()[1]);
217  if (!Succ0 || !Succ1)
218  return nullptr;
219 
220  if (Succ0->getNumSuccessors() + Succ1->getNumSuccessors() != 1)
221  return nullptr;
222  if (Succ0->getSingleSuccessor() == Succ1)
223  return Succ0;
224  if (Succ1->getSingleSuccessor() == Succ0)
225  return Succ1;
226  return nullptr;

229 bool VPlanTransforms::mergeReplicateRegionsIntoSuccessors(VPlan &Plan) {
230  SetVector<VPRegionBlock *> DeletedRegions;
231 
232  // Collect replicate regions followed by an empty block, followed by another
233  // replicate region with matching masks to process front. This is to avoid
234  // iterator invalidation issues while merging regions.
235  SmallVector<VPRegionBlock *, 8> WorkList;
236  for (VPRegionBlock *Region1 : VPBlockUtils::blocksOnly<VPRegionBlock>(
237  vp_depth_first_deep(Plan.getEntry()))) {
238  if (!Region1->isReplicator())
239  continue;
240  auto *MiddleBasicBlock =
241  dyn_cast_or_null<VPBasicBlock>(Region1->getSingleSuccessor());
242  if (!MiddleBasicBlock || !MiddleBasicBlock->empty())
243  continue;
244 
245  auto *Region2 =
246  dyn_cast_or_null<VPRegionBlock>(MiddleBasicBlock->getSingleSuccessor());
247  if (!Region2 || !Region2->isReplicator())
248  continue;
249 
250  VPValue *Mask1 = getPredicatedMask(Region1);
251  VPValue *Mask2 = getPredicatedMask(Region2);
252  if (!Mask1 || Mask1 != Mask2)
253  continue;
254 
255  assert(Mask1 && Mask2 && "both region must have conditions");
256  WorkList.push_back(Region1);
257  }
258 
259  // Move recipes from Region1 to its successor region, if both are triangles.
260  for (VPRegionBlock *Region1 : WorkList) {
261  if (DeletedRegions.contains(Region1))
262  continue;
263  auto *MiddleBasicBlock = cast<VPBasicBlock>(Region1->getSingleSuccessor());
264  auto *Region2 = cast<VPRegionBlock>(MiddleBasicBlock->getSingleSuccessor());
265 
266  VPBasicBlock *Then1 = getPredicatedThenBlock(Region1);
267  VPBasicBlock *Then2 = getPredicatedThenBlock(Region2);
268  if (!Then1 || !Then2)
269  continue;
270 
271  // Note: No fusion-preventing memory dependencies are expected in either
272  // region. Such dependencies should be rejected during earlier dependence
273  // checks, which guarantee accesses can be re-ordered for vectorization.
274  //
275  // Move recipes to the successor region.
276  for (VPRecipeBase &ToMove : make_early_inc_range(reverse(*Then1)))
277  ToMove.moveBefore(*Then2, Then2->getFirstNonPhi());
278 
279  auto *Merge1 = cast<VPBasicBlock>(Then1->getSingleSuccessor());
280  auto *Merge2 = cast<VPBasicBlock>(Then2->getSingleSuccessor());
281 
282  // Move VPPredInstPHIRecipes from the merge block to the successor region's
283  // merge block. Update all users inside the successor region to use the
284  // original values.
285  for (VPRecipeBase &Phi1ToMove : make_early_inc_range(reverse(*Merge1))) {
286  VPValue *PredInst1 =
287  cast<VPPredInstPHIRecipe>(&Phi1ToMove)->getOperand(0);
288  VPValue *Phi1ToMoveV = Phi1ToMove.getVPSingleValue();
289  for (VPUser *U : to_vector(Phi1ToMoveV->users())) {
290  auto *UI = dyn_cast<VPRecipeBase>(U);
291  if (!UI || UI->getParent() != Then2)
292  continue;
293  for (unsigned I = 0, E = U->getNumOperands(); I != E; ++I) {
294  if (Phi1ToMoveV != U->getOperand(I))
295  continue;
296  U->setOperand(I, PredInst1);
297  }
298  }
299 
300  Phi1ToMove.moveBefore(*Merge2, Merge2->begin());
301  }
302 
303  // Finally, remove the first region.
304  for (VPBlockBase *Pred : make_early_inc_range(Region1->getPredecessors())) {
305  VPBlockUtils::disconnectBlocks(Pred, Region1);
306  VPBlockUtils::connectBlocks(Pred, MiddleBasicBlock);
307  }
308  VPBlockUtils::disconnectBlocks(Region1, MiddleBasicBlock);
309  DeletedRegions.insert(Region1);
310  }
311 
312  for (VPRegionBlock *ToDelete : DeletedRegions)
313  delete ToDelete;
314  return !DeletedRegions.empty();
315 }

317 bool VPlanTransforms::mergeBlocksIntoPredecessors(VPlan &Plan) {
318  SmallVector<VPBasicBlock *> WorkList;
319  for (VPBasicBlock *VPBB : VPBlockUtils::blocksOnly<VPBasicBlock>(
320  vp_depth_first_deep(Plan.getEntry()))) {
321  auto *PredVPBB =
322  dyn_cast_or_null<VPBasicBlock>(VPBB->getSinglePredecessor());
323  if (PredVPBB && PredVPBB->getNumSuccessors() == 1)
324  WorkList.push_back(VPBB);
325  }
326 
327  for (VPBasicBlock *VPBB : WorkList) {
328  VPBasicBlock *PredVPBB = cast<VPBasicBlock>(VPBB->getSinglePredecessor());
329  for (VPRecipeBase &R : make_early_inc_range(*VPBB))
330  R.moveBefore(*PredVPBB, PredVPBB->end());
331  VPBlockUtils::disconnectBlocks(PredVPBB, VPBB);
332  auto *ParentRegion = cast_or_null<VPRegionBlock>(VPBB->getParent());
333  if (ParentRegion && ParentRegion->getExiting() == VPBB)
334  ParentRegion->setExiting(PredVPBB);
335  for (auto *Succ : to_vector(VPBB->successors())) {
336  VPBlockUtils::disconnectBlocks(VPBB, Succ);
337  VPBlockUtils::connectBlocks(PredVPBB, Succ);
338  }
339  delete VPBB;
340  }
341  return !WorkList.empty();
342 }

344 void VPlanTransforms::removeRedundantInductionCasts(VPlan &Plan) {
345  for (auto &Phi : Plan.getVectorLoopRegion()->getEntryBasicBlock()->phis()) {
346  auto *IV = dyn_cast<VPWidenIntOrFpInductionRecipe>(&Phi);
347  if (!IV || IV->getTruncInst())
348  continue;
349 
350  // A sequence of IR Casts has potentially been recorded for IV, which
351  // *must be bypassed* when the IV is vectorized, because the vectorized IV
352  // will produce the desired casted value. This sequence forms a def-use
353  // chain and is provided in reverse order, ending with the cast that uses
354  // the IV phi. Search for the recipe of the last cast in the chain and
355  // replace it with the original IV. Note that only the final cast is
356  // expected to have users outside the cast-chain and the dead casts left
357  // over will be cleaned up later.
358  auto &Casts = IV->getInductionDescriptor().getCastInsts();
359  VPValue *FindMyCast = IV;
360  for (Instruction *IRCast : reverse(Casts)) {
361  VPRecipeBase *FoundUserCast = nullptr;
362  for (auto *U : FindMyCast->users()) {
363  auto *UserCast = cast<VPRecipeBase>(U);
364  if (UserCast->getNumDefinedValues() == 1 &&
365  UserCast->getVPSingleValue()->getUnderlyingValue() == IRCast) {
366  FoundUserCast = UserCast;
367  break;
368  }
369  }
370  FindMyCast = FoundUserCast->getVPSingleValue();
371  }
372  FindMyCast->replaceAllUsesWith(IV);
373  }
374 }

376 void VPlanTransforms::removeRedundantCanonicalIVs(VPlan &Plan) {
377  VPCanonicalIVPHIRecipe *CanonicalIV = Plan.getCanonicalIV();
378  VPWidenCanonicalIVRecipe *WidenNewIV = nullptr;
379  for (VPUser *U : CanonicalIV->users()) {
380  WidenNewIV = dyn_cast<VPWidenCanonicalIVRecipe>(U);
381  if (WidenNewIV)
382  break;
383  }
384 
385  if (!WidenNewIV)
386  return;
387 
388  VPBasicBlock *HeaderVPBB = Plan.getVectorLoopRegion()->getEntryBasicBlock();
389  for (VPRecipeBase &Phi : HeaderVPBB->phis()) {
390  auto *WidenOriginalIV = dyn_cast<VPWidenIntOrFpInductionRecipe>(&Phi);
391 
392  if (!WidenOriginalIV || !WidenOriginalIV->isCanonical() ||
393  WidenOriginalIV->getScalarType() != WidenNewIV->getScalarType())
394  continue;
395 
396  // Replace WidenNewIV with WidenOriginalIV if WidenOriginalIV provides
397  // everything WidenNewIV's users need. That is, WidenOriginalIV will
398  // generate a vector phi or all users of WidenNewIV demand the first lane
399  // only.
400  if (WidenOriginalIV->needsVectorIV() ||
401  vputils::onlyFirstLaneUsed(WidenNewIV)) {
402  WidenNewIV->replaceAllUsesWith(WidenOriginalIV);
403  WidenNewIV->eraseFromParent();
404  return;
405  }
406  }

409 void VPlanTransforms::removeDeadRecipes(VPlan &Plan) {
410  ReversePostOrderTraversal<VPBlockDeepTraversalWrapper<VPBlockBase *>> RPOT(
411  Plan.getEntry());
412 
413  for (VPBasicBlock *VPBB : reverse(VPBlockUtils::blocksOnly<VPBasicBlock>(RPOT))) {
414  // The recipes in the block are processed in reverse order, to catch chains
415  // of dead recipes.
416  for (VPRecipeBase &R : make_early_inc_range(reverse(*VPBB))) {
417  if (R.mayHaveSideEffects() || any_of(R.definedValues(), [](VPValue *V) {
418  return V->getNumUsers() > 0;
419  }))
420  continue;
421  R.eraseFromParent();
422  }
423  }

426 void VPlanTransforms::optimizeInductions(VPlan &Plan, ScalarEvolution &SE) {
427  SmallVector<VPRecipeBase *> ToRemove;
428  VPBasicBlock *HeaderVPBB = Plan.getVectorLoopRegion()->getEntryBasicBlock();
429  bool HasOnlyVectorVFs = !Plan.hasVF(ElementCount::getFixed(1));
430  for (VPRecipeBase &Phi : HeaderVPBB->phis()) {
431  auto *WideIV = dyn_cast<VPWidenIntOrFpInductionRecipe>(&Phi);
432  if (!WideIV)
433  continue;
434  if (HasOnlyVectorVFs && none_of(WideIV->users(), [WideIV](VPUser *U) {
435  return U->usesScalars(WideIV);
436  }))
437  continue;
438 
439  auto IP = HeaderVPBB->getFirstNonPhi();
440  VPCanonicalIVPHIRecipe *CanonicalIV = Plan.getCanonicalIV();
441  Type *ResultTy = WideIV->getPHINode()->getType();
442  if (Instruction *TruncI = WideIV->getTruncInst())
443  ResultTy = TruncI->getType();
444  const InductionDescriptor &ID = WideIV->getInductionDescriptor();
445  VPValue *Step =
446  vputils::getOrCreateVPValueForSCEVExpr(Plan, ID.getStep(), SE);
447  VPValue *BaseIV = CanonicalIV;
448  if (!CanonicalIV->isCanonical(ID, ResultTy)) {
449  BaseIV = new VPDerivedIVRecipe(ID, WideIV->getStartValue(), CanonicalIV,
450  Step, ResultTy);
451  HeaderVPBB->insert(BaseIV->getDefiningRecipe(), IP);
452  }
453 
454  VPScalarIVStepsRecipe *Steps = new VPScalarIVStepsRecipe(ID, BaseIV, Step);
455  HeaderVPBB->insert(Steps, IP);
456 
457  // Update scalar users of IV to use Step instead. Use SetVector to ensure
458  // the list of users doesn't contain duplicates.
459  SetVector<VPUser *> Users(WideIV->user_begin(), WideIV->user_end());
460  for (VPUser *U : Users) {
461  if (HasOnlyVectorVFs && !U->usesScalars(WideIV))
462  continue;
463  for (unsigned I = 0, E = U->getNumOperands(); I != E; I++) {
464  if (U->getOperand(I) != WideIV)
465  continue;
466  U->setOperand(I, Steps);
467  }
468  }
469  }
470 }

498 void VPlanTransforms::optimizeForVFAndUF(VPlan &Plan, ElementCount BestVF,
499  unsigned BestUF,
500  PredicatedScalarEvolution &PSE) {
501  assert(Plan.hasVF(BestVF) && "BestVF is not available in Plan");
502  assert(Plan.hasUF(BestUF) && "BestUF is not available in Plan");
503  VPBasicBlock *ExitingVPBB =
504  Plan.getVectorLoopRegion()->getExitingBasicBlock();
505  auto *Term = dyn_cast<VPInstruction>(&ExitingVPBB->back());
506  // Try to simplify the branch condition if TC <= VF * UF when preparing to
507  // execute the plan for the main vector loop. We only do this if the
508  // terminator is:
509  // 1. BranchOnCount, or
510  // 2. BranchOnCond where the input is Not(ActiveLaneMask).
511  if (!Term || (Term->getOpcode() != VPInstruction::BranchOnCount &&
512  (Term->getOpcode() != VPInstruction::BranchOnCond ||
513  !canSimplifyBranchOnCond(Term))))
514  return;
515 
516  Type *IdxTy =
517  Plan.getCanonicalIV()->getStartValue()->getLiveInIRValue()->getType();
518  const SCEV *TripCount = createTripCountSCEV(IdxTy, PSE);
519  ScalarEvolution &SE = *PSE.getSE();
520  const SCEV *C =
521  SE.getConstant(TripCount->getType(), BestVF.getKnownMinValue() * BestUF);
522  if (TripCount->isZero() ||
523  !SE.isKnownPredicate(CmpInst::ICMP_ULE, TripCount, C))
524  return;
525 
526  LLVMContext &Ctx = SE.getContext();
527  auto *BOC =
528  new VPInstruction(VPInstruction::BranchOnCond,
529  {Plan.getOrAddExternalDef(ConstantInt::getTrue(Ctx))});
530  Term->eraseFromParent();
531  ExitingVPBB->appendRecipe(BOC);
532  Plan.setVF(BestVF);
533  Plan.setUF(BestUF);
534  // TODO: Further simplifications are possible
535  // 1. Replace inductions with constants.
536  // 2. Replace vector loop region with VPBasicBlock.
537 }

552 static bool properlyDominates(const VPRecipeBase *A, const VPRecipeBase *B,
553  VPDominatorTree &VPDT) {
554  auto LocalComesBefore = [](const VPRecipeBase *A, const VPRecipeBase *B) {
555  for (auto &R : *A->getParent()) {
556  if (&R == A)
557  return true;
558  if (&R == B)
559  return false;
560  }
561  llvm_unreachable("recipe not found");
562  };
563  const VPBlockBase *ParentA = A->getParent();
564  const VPBlockBase *ParentB = B->getParent();
565  if (ParentA == ParentB)
566  return LocalComesBefore(A, B);
567 
568  const VPRegionBlock *RegionA =
569  GetReplicateRegion(const_cast<VPRecipeBase *>(A));
570  const VPRegionBlock *RegionB =
571  GetReplicateRegion(const_cast<VPRecipeBase *>(B));
572  if (RegionA)
573  ParentA = RegionA->getExiting();
574  if (RegionB)
575  ParentB = RegionB->getExiting();
576  return VPDT.properlyDominates(ParentA, ParentB);
577 }

579 // Sink users of \p FOR after the recipe defining the previous value \p Previous
580 // of the recurrence.
581 static void
582 sinkRecurrenceUsersAfterPrevious(VPFirstOrderRecurrencePHIRecipe *FOR,
583  VPRecipeBase *Previous,
584  VPDominatorTree &VPDT) {
585  // Collect recipes that need sinking.
586  SmallVector<VPRecipeBase *> WorkList;
587  SmallPtrSet<VPRecipeBase *, 8> Seen;
588  Seen.insert(Previous);
589  auto TryToPushSinkCandidate = [&](VPRecipeBase *SinkCandidate) {
590  assert(
591  SinkCandidate != Previous &&
592  "The previous value cannot depend on the users of the recurrence phi.");
593  if (isa<VPHeaderPHIRecipe>(SinkCandidate) ||
594  !Seen.insert(SinkCandidate).second ||
595  properlyDominates(Previous, SinkCandidate, VPDT))
596  return;
597 
598  WorkList.push_back(SinkCandidate);
599  };
600 
601  // Recursively sink users of FOR after Previous.
602  WorkList.push_back(FOR);
603  for (unsigned I = 0; I != WorkList.size(); ++I) {
604  VPRecipeBase *Current = WorkList[I];
605  assert(Current->getNumDefinedValues() == 1 &&
606  "only recipes with a single defined value expected");
607  for (VPUser *User : Current->getVPSingleValue()->users()) {
608  if (auto *R = dyn_cast<VPRecipeBase>(User))
609  TryToPushSinkCandidate(R);
610  }
611  }
612 
613  // Keep recipes to sink ordered by dominance so earlier instructions are
614  // processed first.
615  sort(WorkList, [&VPDT](const VPRecipeBase *A, const VPRecipeBase *B) {
616  return properlyDominates(A, B, VPDT);
617  });
618 
619  for (VPRecipeBase *SinkCandidate : WorkList) {
620  // VPPredInstPHIRecipes don't need sinking, because they will be sunk when
621  // sinking the containing replicate region.
622  if (isa<VPPredInstPHIRecipe>(SinkCandidate) || SinkCandidate == FOR)
623  continue;
624 
625  VPRecipeBase *Target = Previous;
626  Previous = SinkCandidate;
627  auto *TargetRegion = GetReplicateRegion(Target);
628  auto *SinkRegion = GetReplicateRegion(SinkCandidate);
629  if (!SinkRegion) {
630  // If the sink source is not a replicate region, sink the recipe
631  // directly.
632  if (TargetRegion) {
633  // The target is in a replication region, make sure to move Sink to
634  // the block after it, not into the replication region itself.
635  VPBasicBlock *NextBlock =
636  cast<VPBasicBlock>(TargetRegion->getSuccessors().front());
637  SinkCandidate->moveBefore(*NextBlock, NextBlock->getFirstNonPhi());
638  } else
639  SinkCandidate->moveAfter(Target);
640  continue;
641  }
642  // The sink source is in a replicate region. Unhook the region from the
643  // CFG.
644  auto *SinkPred = SinkRegion->getSinglePredecessor();
645  auto *SinkSucc = SinkRegion->getSingleSuccessor();
646  VPBlockUtils::disconnectBlocks(SinkPred, SinkRegion);
647  VPBlockUtils::disconnectBlocks(SinkRegion, SinkSucc);
648  VPBlockUtils::connectBlocks(SinkPred, SinkSucc);
649 
650  if (TargetRegion) {
651  // The target recipe is also in a replicate region, move the sink
652  // region after the target region.
653  auto *TargetSucc = TargetRegion->getSingleSuccessor();
654  VPBlockUtils::disconnectBlocks(TargetRegion, TargetSucc);
655  VPBlockUtils::connectBlocks(TargetRegion, SinkRegion);
656  VPBlockUtils::connectBlocks(SinkRegion, TargetSucc);
657  } else {
658  // The sink source is in a replicate region, we need to move the whole
659  // replicate region, which should only contain a single recipe in the
660  // main block.
661  auto *SplitBlock =
662  Target->getParent()->splitAt(std::next(Target->getIterator()));
663 
664  auto *SplitPred = SplitBlock->getSinglePredecessor();
665 
666  VPBlockUtils::disconnectBlocks(SplitPred, SplitBlock);
667  VPBlockUtils::connectBlocks(SplitPred, SinkRegion);
668  VPBlockUtils::connectBlocks(SinkRegion, SplitBlock);
669  }
670  // We modified the CFG, update dominator tree.
671  VPDT.recalculate(*SinkRegion->getPlan());
672  }
673 }

675 void VPlanTransforms::adjustFixedOrderRecurrences(VPlan &Plan,
676  VPBuilder &Builder) {
677  VPDominatorTree VPDT;
678  VPDT.recalculate(Plan);
679 
680  SmallVector<VPFirstOrderRecurrencePHIRecipe *> RecurrencePhis;
681  for (VPRecipeBase &R :
682  Plan.getVectorLoopRegion()->getEntry()->getEntryBasicBlock()->phis())
683  if (auto *FOR = dyn_cast<VPFirstOrderRecurrencePHIRecipe>(&R))
684  RecurrencePhis.push_back(FOR);
685 
686  for (VPFirstOrderRecurrencePHIRecipe *FOR : RecurrencePhis) {
687  SmallPtrSet<VPFirstOrderRecurrencePHIRecipe *, 4> SeenPhis;
688  VPRecipeBase *Previous = FOR->getBackedgeValue()->getDefiningRecipe();
689  // Fixed-order recurrences do not contain cycles, so this loop is guaranteed
690  // to terminate.
691  while (auto *PrevPhi =
692  dyn_cast_or_null<VPFirstOrderRecurrencePHIRecipe>(Previous)) {
693  assert(PrevPhi->getParent() == FOR->getParent());
694  assert(SeenPhis.insert(PrevPhi).second);
695  Previous = PrevPhi->getBackedgeValue()->getDefiningRecipe();
696  }
697 
698  sinkRecurrenceUsersAfterPrevious(FOR, Previous, VPDT);
699 
700  // Introduce a recipe to combine the incoming and previous values of a
701  // fixed-order recurrence.
702  VPBasicBlock *InsertBlock = Previous->getParent();
703  auto *Region = GetReplicateRegion(Previous);
704  if (Region)
705  InsertBlock = dyn_cast<VPBasicBlock>(Region->getSingleSuccessor());
706  if (!InsertBlock) {
707  InsertBlock = new VPBasicBlock(Region->getName() + ".succ");
708  VPBlockUtils::insertBlockAfter(InsertBlock, Region);
709  }
710  if (Region || isa<VPHeaderPHIRecipe>(Previous))
711  Builder.setInsertPoint(InsertBlock, InsertBlock->getFirstNonPhi());
712  else
713  Builder.setInsertPoint(InsertBlock, std::next(Previous->getIterator()));
714 
715  auto *RecurSplice = cast<VPInstruction>(
716  Builder.createNaryOp(VPInstruction::FirstOrderRecurrenceSplice,
717  {FOR, FOR->getBackedgeValue()}));
718 
719  FOR->replaceAllUsesWith(RecurSplice);
720  // Set the first operand of RecurSplice to FOR again, after replacing
721  // all users.
722  RecurSplice->setOperand(0, FOR);
723  }
