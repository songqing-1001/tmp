79 bool AMDGPUPostLegalizerCombinerHelper::matchFMinFMaxLegacy(
80  MachineInstr &MI, FMinFMaxLegacyInfo &Info) {
81  // FIXME: Type predicate on pattern
82  if (MRI.getType(MI.getOperand(0).getReg()) != LLT::scalar(32))
83  return false;
84 
85  Register Cond = MI.getOperand(1).getReg();
86  if (!MRI.hasOneNonDBGUse(Cond) ||
87  !mi_match(Cond, MRI,
88  m_GFCmp(m_Pred(Info.Pred), m_Reg(Info.LHS), m_Reg(Info.RHS))))
89  return false;
90 
91  Info.True = MI.getOperand(2).getReg();
92  Info.False = MI.getOperand(3).getReg();
93 
94  // TODO: Handle case where the the selected value is an fneg and the compared
95  // constant is the negation of the selected value.
96  if (!(Info.LHS == Info.True && Info.RHS == Info.False) &&
97  !(Info.LHS == Info.False && Info.RHS == Info.True))
98  return false;
99 
100  switch (Info.Pred) {
101  case CmpInst::FCMP_FALSE:
102  case CmpInst::FCMP_OEQ:
103  case CmpInst::FCMP_ONE:
104  case CmpInst::FCMP_ORD:
105  case CmpInst::FCMP_UNO:
106  case CmpInst::FCMP_UEQ:
107  case CmpInst::FCMP_UNE:
108  case CmpInst::FCMP_TRUE:
109  return false;
110  default:
111  return true;
112  }
113 }

115 void AMDGPUPostLegalizerCombinerHelper::applySelectFCmpToFMinToFMaxLegacy(
116  MachineInstr &MI, const FMinFMaxLegacyInfo &Info) {
117  B.setInstrAndDebugLoc(MI);
118  auto buildNewInst = [&MI, this](unsigned Opc, Register X, Register Y) {
119  B.buildInstr(Opc, {MI.getOperand(0)}, {X, Y}, MI.getFlags());
120  };
121 
122  switch (Info.Pred) {
123  case CmpInst::FCMP_ULT:
124  case CmpInst::FCMP_ULE:
125  if (Info.LHS == Info.True)
126  buildNewInst(AMDGPU::G_AMDGPU_FMIN_LEGACY, Info.RHS, Info.LHS);
127  else
128  buildNewInst(AMDGPU::G_AMDGPU_FMAX_LEGACY, Info.LHS, Info.RHS);
129  break;
130  case CmpInst::FCMP_OLE:
131  case CmpInst::FCMP_OLT: {
132  // We need to permute the operands to get the correct NaN behavior. The
133  // selected operand is the second one based on the failing compare with NaN,
134  // so permute it based on the compare type the hardware uses.
135  if (Info.LHS == Info.True)
136  buildNewInst(AMDGPU::G_AMDGPU_FMIN_LEGACY, Info.LHS, Info.RHS);
137  else
138  buildNewInst(AMDGPU::G_AMDGPU_FMAX_LEGACY, Info.RHS, Info.LHS);
139  break;
140  }
141  case CmpInst::FCMP_UGE:
142  case CmpInst::FCMP_UGT: {
143  if (Info.LHS == Info.True)
144  buildNewInst(AMDGPU::G_AMDGPU_FMAX_LEGACY, Info.RHS, Info.LHS);
145  else
146  buildNewInst(AMDGPU::G_AMDGPU_FMIN_LEGACY, Info.LHS, Info.RHS);
147  break;
148  }
149  case CmpInst::FCMP_OGT:
150  case CmpInst::FCMP_OGE: {
151  if (Info.LHS == Info.True)
152  buildNewInst(AMDGPU::G_AMDGPU_FMAX_LEGACY, Info.LHS, Info.RHS);
153  else
154  buildNewInst(AMDGPU::G_AMDGPU_FMIN_LEGACY, Info.RHS, Info.LHS);
155  break;
156  }
157  default:
158  llvm_unreachable("predicate should not have matched");
159  }
160 
161  MI.eraseFromParent();
162 }

164 bool AMDGPUPostLegalizerCombinerHelper::matchUCharToFloat(MachineInstr &MI) {
165  Register DstReg = MI.getOperand(0).getReg();
166 
167  // TODO: We could try to match extracting the higher bytes, which would be
168  // easier if i8 vectors weren't promoted to i32 vectors, particularly after
169  // types are legalized. v4i8 -> v4f32 is probably the only case to worry
170  // about in practice.
171  LLT Ty = MRI.getType(DstReg);
172  if (Ty == LLT::scalar(32) || Ty == LLT::scalar(16)) {
173  Register SrcReg = MI.getOperand(1).getReg();
174  unsigned SrcSize = MRI.getType(SrcReg).getSizeInBits();
175  assert(SrcSize == 16 || SrcSize == 32 || SrcSize == 64);
176  const APInt Mask = APInt::getHighBitsSet(SrcSize, SrcSize - 8);
177  return Helper.getKnownBits()->maskedValueIsZero(SrcReg, Mask);
178  }
179 
180  return false;
181 }

183 void AMDGPUPostLegalizerCombinerHelper::applyUCharToFloat(MachineInstr &MI) {
184  B.setInstrAndDebugLoc(MI);
185 
186  const LLT S32 = LLT::scalar(32);
187 
188  Register DstReg = MI.getOperand(0).getReg();
189  Register SrcReg = MI.getOperand(1).getReg();
190  LLT Ty = MRI.getType(DstReg);
191  LLT SrcTy = MRI.getType(SrcReg);
192  if (SrcTy != S32)
193  SrcReg = B.buildAnyExtOrTrunc(S32, SrcReg).getReg(0);
194 
195  if (Ty == S32) {
196  B.buildInstr(AMDGPU::G_AMDGPU_CVT_F32_UBYTE0, {DstReg},
197  {SrcReg}, MI.getFlags());
198  } else {
199  auto Cvt0 = B.buildInstr(AMDGPU::G_AMDGPU_CVT_F32_UBYTE0, {S32},
200  {SrcReg}, MI.getFlags());
201  B.buildFPTrunc(DstReg, Cvt0, MI.getFlags());
202  }
203 
204  MI.eraseFromParent();
205 }

207 bool AMDGPUPostLegalizerCombinerHelper::matchRcpSqrtToRsq(
208  MachineInstr &MI, std::function<void(MachineIRBuilder &)> &MatchInfo) {
209 
210  auto getRcpSrc = [=](const MachineInstr &MI) {
211  MachineInstr *ResMI = nullptr;
212  if (MI.getOpcode() == TargetOpcode::G_INTRINSIC &&
213  MI.getIntrinsicID() == Intrinsic::amdgcn_rcp)
214  ResMI = MRI.getVRegDef(MI.getOperand(2).getReg());
215 
216  return ResMI;
217  };
218 
219  auto getSqrtSrc = [=](const MachineInstr &MI) {
220  MachineInstr *SqrtSrcMI = nullptr;
221  auto Match =
222  mi_match(MI.getOperand(0).getReg(), MRI, m_GFSqrt(m_MInstr(SqrtSrcMI)));
223  (void)Match;
224  return SqrtSrcMI;
225  };
226 
227  MachineInstr *RcpSrcMI = nullptr, *SqrtSrcMI = nullptr;
228  // rcp(sqrt(x))
229  if ((RcpSrcMI = getRcpSrc(MI)) && (SqrtSrcMI = getSqrtSrc(*RcpSrcMI))) {
230  MatchInfo = [SqrtSrcMI, &MI](MachineIRBuilder &B) {
231  B.buildIntrinsic(Intrinsic::amdgcn_rsq, {MI.getOperand(0)}, false)
232  .addUse(SqrtSrcMI->getOperand(0).getReg())
233  .setMIFlags(MI.getFlags());
234  };
235  return true;
236  }
237 
238  // sqrt(rcp(x))
239  if ((SqrtSrcMI = getSqrtSrc(MI)) && (RcpSrcMI = getRcpSrc(*SqrtSrcMI))) {
240  MatchInfo = [RcpSrcMI, &MI](MachineIRBuilder &B) {
241  B.buildIntrinsic(Intrinsic::amdgcn_rsq, {MI.getOperand(0)}, false)
242  .addUse(RcpSrcMI->getOperand(0).getReg())
243  .setMIFlags(MI.getFlags());
244  };
245  return true;
246  }
247 
248  return false;
249 }

251 bool AMDGPUPostLegalizerCombinerHelper::matchCvtF32UByteN(
252  MachineInstr &MI, CvtF32UByteMatchInfo &MatchInfo) {
253  Register SrcReg = MI.getOperand(1).getReg();
254 
255  // Look through G_ZEXT.
256  bool IsShr = mi_match(SrcReg, MRI, m_GZExt(m_Reg(SrcReg)));
257 
258  Register Src0;
259  int64_t ShiftAmt;
260  IsShr = mi_match(SrcReg, MRI, m_GLShr(m_Reg(Src0), m_ICst(ShiftAmt)));
261  if (IsShr || mi_match(SrcReg, MRI, m_GShl(m_Reg(Src0), m_ICst(ShiftAmt)))) {
262  const unsigned Offset = MI.getOpcode() - AMDGPU::G_AMDGPU_CVT_F32_UBYTE0;
263 
264  unsigned ShiftOffset = 8 * Offset;
265  if (IsShr)
266  ShiftOffset += ShiftAmt;
267  else
268  ShiftOffset -= ShiftAmt;
269 
270  MatchInfo.CvtVal = Src0;
271  MatchInfo.ShiftOffset = ShiftOffset;
272  return ShiftOffset < 32 && ShiftOffset >= 8 && (ShiftOffset % 8) == 0;
273  }
274 
275  // TODO: Simplify demanded bits.
276  return false;
277 }

279 void AMDGPUPostLegalizerCombinerHelper::applyCvtF32UByteN(
280  MachineInstr &MI, const CvtF32UByteMatchInfo &MatchInfo) {
281  B.setInstrAndDebugLoc(MI);
282  unsigned NewOpc = AMDGPU::G_AMDGPU_CVT_F32_UBYTE0 + MatchInfo.ShiftOffset / 8;
283 
284  const LLT S32 = LLT::scalar(32);
285  Register CvtSrc = MatchInfo.CvtVal;
286  LLT SrcTy = MRI.getType(MatchInfo.CvtVal);
287  if (SrcTy != S32) {
288  assert(SrcTy.isScalar() && SrcTy.getSizeInBits() >= 8);
289  CvtSrc = B.buildAnyExt(S32, CvtSrc).getReg(0);
290  }
291 
292  assert(MI.getOpcode() != NewOpc);
293  B.buildInstr(NewOpc, {MI.getOperand(0)}, {CvtSrc}, MI.getFlags());
294  MI.eraseFromParent();
295 }

355 bool AMDGPUPostLegalizerCombinerInfo::combine(GISelChangeObserver &Observer,
356  MachineInstr &MI,
357  MachineIRBuilder &B) const {
358  AMDGPUCombinerHelper Helper(Observer, B, /*IsPreLegalize*/ false, KB, MDT,
359  LInfo);
360  AMDGPUPostLegalizerCombinerHelper PostLegalizerHelper(B, Helper);
361  AMDGPUGenPostLegalizerCombinerHelper Generated(
362  GeneratedRuleCfg, Helper, PostLegalizerHelper, Subtarget);
363 
364  if (Generated.tryCombineAll(Observer, MI, B))
365  return true;
366 
367  switch (MI.getOpcode()) {
368  case TargetOpcode::G_SHL:
369  case TargetOpcode::G_LSHR:
370  case TargetOpcode::G_ASHR:
371  // On some subtargets, 64-bit shift is a quarter rate instruction. In the
372  // common case, splitting this into a move and a 32-bit shift is faster and
373  // the same code size.
374  return Helper.tryCombineShiftToUnmerge(MI, 32);
375  }
376 
377  return false;
378 }

423 bool AMDGPUPostLegalizerCombiner::runOnMachineFunction(MachineFunction &MF) {
424  if (MF.getProperties().hasProperty(
425  MachineFunctionProperties::Property::FailedISel))
426  return false;
427  auto *TPC = &getAnalysis<TargetPassConfig>();
428  const Function &F = MF.getFunction();
429  bool EnableOpt =
430  MF.getTarget().getOptLevel() != CodeGenOpt::None && !skipFunction(F);
431 
432  const GCNSubtarget &ST = MF.getSubtarget<GCNSubtarget>();
433  const AMDGPULegalizerInfo *LI
434  = static_cast<const AMDGPULegalizerInfo *>(ST.getLegalizerInfo());
435 
436  GISelKnownBits *KB = &getAnalysis<GISelKnownBitsAnalysis>().get(MF);
437  MachineDominatorTree *MDT =
438  IsOptNone ? nullptr : &getAnalysis<MachineDominatorTree>();
439  AMDGPUPostLegalizerCombinerInfo PCInfo(ST, EnableOpt, F.hasOptSize(),
440  F.hasMinSize(), LI, KB, MDT);
441  Combiner C(PCInfo, TPC);
442  return C.combineMachineInstrs(MF, /*CSEInfo*/ nullptr);
443 }
