82 static bool dependsOnLocalPhi(const Loop *L, const Value *Cond,
83  unsigned Depth = 0) {
84  const Instruction *I = dyn_cast<Instruction>(Cond);
85  if (!I)
86  return false;
87 
88  for (const Value *V : I->operand_values()) {
89  if (!L->contains(I))
90  continue;
91  if (const PHINode *PHI = dyn_cast<PHINode>(V)) {
92  if (llvm::none_of(L->getSubLoops(), [PHI](const Loop* SubLoop) {
93  return SubLoop->contains(PHI); }))
94  return true;
95  } else if (Depth < 10 && dependsOnLocalPhi(L, V, Depth+1))
96  return true;
97  }
98  return false;
99 }

107 void AMDGPUTTIImpl::getUnrollingPreferences(Loop *L, ScalarEvolution &SE,
108  TTI::UnrollingPreferences &UP,
109  OptimizationRemarkEmitter *ORE) {
110  const Function &F = *L->getHeader()->getParent();
111  UP.Threshold =
112  F.getFnAttributeAsParsedInteger("amdgpu-unroll-threshold", 300);
113  UP.MaxCount = std::numeric_limits<unsigned>::max();
114  UP.Partial = true;
115 
116  // Conditional branch in a loop back edge needs 3 additional exec
117  // manipulations in average.
118  UP.BEInsns += 3;
119 
120  // TODO: Do we want runtime unrolling?
121 
122  // Maximum alloca size than can fit registers. Reserve 16 registers.
123  const unsigned MaxAlloca = (256 - 16) * 4;
124  unsigned ThresholdPrivate = UnrollThresholdPrivate;
125  unsigned ThresholdLocal = UnrollThresholdLocal;
126 
127  // If this loop has the amdgpu.loop.unroll.threshold metadata we will use the
128  // provided threshold value as the default for Threshold
129  if (MDNode *LoopUnrollThreshold =
130  findOptionMDForLoop(L, "amdgpu.loop.unroll.threshold")) {
131  if (LoopUnrollThreshold->getNumOperands() == 2) {
132  ConstantInt *MetaThresholdValue = mdconst::extract_or_null<ConstantInt>(
133  LoopUnrollThreshold->getOperand(1));
134  if (MetaThresholdValue) {
135  // We will also use the supplied value for PartialThreshold for now.
136  // We may introduce additional metadata if it becomes necessary in the
137  // future.
138  UP.Threshold = MetaThresholdValue->getSExtValue();
139  UP.PartialThreshold = UP.Threshold;
140  ThresholdPrivate = std::min(ThresholdPrivate, UP.Threshold);
141  ThresholdLocal = std::min(ThresholdLocal, UP.Threshold);
142  }
143  }
144  }
145 
146  unsigned MaxBoost = std::max(ThresholdPrivate, ThresholdLocal);
147  for (const BasicBlock *BB : L->getBlocks()) {
148  const DataLayout &DL = BB->getModule()->getDataLayout();
149  unsigned LocalGEPsSeen = 0;
150 
151  if (llvm::any_of(L->getSubLoops(), [BB](const Loop* SubLoop) {
152  return SubLoop->contains(BB); }))
153  continue; // Block belongs to an inner loop.
154 
155  for (const Instruction &I : *BB) {
156  // Unroll a loop which contains an "if" statement whose condition
157  // defined by a PHI belonging to the loop. This may help to eliminate
158  // if region and potentially even PHI itself, saving on both divergence
159  // and registers used for the PHI.
160  // Add a small bonus for each of such "if" statements.
161  if (const BranchInst *Br = dyn_cast<BranchInst>(&I)) {
162  if (UP.Threshold < MaxBoost && Br->isConditional()) {
163  BasicBlock *Succ0 = Br->getSuccessor(0);
164  BasicBlock *Succ1 = Br->getSuccessor(1);
165  if ((L->contains(Succ0) && L->isLoopExiting(Succ0)) ||
166  (L->contains(Succ1) && L->isLoopExiting(Succ1)))
167  continue;
168  if (dependsOnLocalPhi(L, Br->getCondition())) {
169  UP.Threshold += UnrollThresholdIf;
170  LLVM_DEBUG(dbgs() << "Set unroll threshold " << UP.Threshold
171  << " for loop:\n"
172  << *L << " due to " << *Br << '\n');
173  if (UP.Threshold >= MaxBoost)
174  return;
175  }
176  }
177  continue;
178  }
179 
180  const GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(&I);
181  if (!GEP)
182  continue;
183 
184  unsigned AS = GEP->getAddressSpace();
185  unsigned Threshold = 0;
186  if (AS == AMDGPUAS::PRIVATE_ADDRESS)
187  Threshold = ThresholdPrivate;
188  else if (AS == AMDGPUAS::LOCAL_ADDRESS || AS == AMDGPUAS::REGION_ADDRESS)
189  Threshold = ThresholdLocal;
190  else
191  continue;
192 
193  if (UP.Threshold >= Threshold)
194  continue;
195 
196  if (AS == AMDGPUAS::PRIVATE_ADDRESS) {
197  const Value *Ptr = GEP->getPointerOperand();
198  const AllocaInst *Alloca =
199  dyn_cast<AllocaInst>(getUnderlyingObject(Ptr));
200  if (!Alloca || !Alloca->isStaticAlloca())
201  continue;
202  Type *Ty = Alloca->getAllocatedType();
203  unsigned AllocaSize = Ty->isSized() ? DL.getTypeAllocSize(Ty) : 0;
204  if (AllocaSize > MaxAlloca)
205  continue;
206  } else if (AS == AMDGPUAS::LOCAL_ADDRESS ||
207  AS == AMDGPUAS::REGION_ADDRESS) {
208  LocalGEPsSeen++;
209  // Inhibit unroll for local memory if we have seen addressing not to
210  // a variable, most likely we will be unable to combine it.
211  // Do not unroll too deep inner loops for local memory to give a chance
212  // to unroll an outer loop for a more important reason.
213  if (LocalGEPsSeen > 1 || L->getLoopDepth() > 2 ||
214  (!isa<GlobalVariable>(GEP->getPointerOperand()) &&
215  !isa<Argument>(GEP->getPointerOperand())))
216  continue;
217  LLVM_DEBUG(dbgs() << "Allow unroll runtime for loop:\n"
218  << *L << " due to LDS use.\n");
219  UP.Runtime = UnrollRuntimeLocal;
220  }
221 
222  // Check if GEP depends on a value defined by this loop itself.
223  bool HasLoopDef = false;
224  for (const Value *Op : GEP->operands()) {
225  const Instruction *Inst = dyn_cast<Instruction>(Op);
226  if (!Inst || L->isLoopInvariant(Op))
227  continue;
228 
229  if (llvm::any_of(L->getSubLoops(), [Inst](const Loop* SubLoop) {
230  return SubLoop->contains(Inst); }))
231  continue;
232  HasLoopDef = true;
233  break;
234  }
235  if (!HasLoopDef)
236  continue;
237 
238  // We want to do whatever we can to limit the number of alloca
239  // instructions that make it through to the code generator. allocas
240  // require us to use indirect addressing, which is slow and prone to
241  // compiler bugs. If this loop does an address calculation on an
242  // alloca ptr, then we want to use a higher than normal loop unroll
243  // threshold. This will give SROA a better chance to eliminate these
244  // allocas.
245  //
246  // We also want to have more unrolling for local memory to let ds
247  // instructions with different offsets combine.
248  //
249  // Don't use the maximum allowed value here as it will make some
250  // programs way too big.
251  UP.Threshold = Threshold;
252  LLVM_DEBUG(dbgs() << "Set unroll threshold " << Threshold
253  << " for loop:\n"
254  << *L << " due to " << *GEP << '\n');
255  if (UP.Threshold >= MaxBoost)
256  return;
257  }
258 
259  // If we got a GEP in a small BB from inner loop then increase max trip
260  // count to analyze for better estimation cost in unroll
261  if (L->isInnermost() && BB->size() < UnrollMaxBlockToAnalyze)
262  UP.MaxIterationsCountToAnalyze = 32;
263  }
264 }

405 Type *GCNTTIImpl::getMemcpyLoopLoweringType(
406  LLVMContext &Context, Value *Length, unsigned SrcAddrSpace,
407  unsigned DestAddrSpace, unsigned SrcAlign, unsigned DestAlign,
408  std::optional<uint32_t> AtomicElementSize) const {
409 
410  if (AtomicElementSize)
411  return Type::getIntNTy(Context, *AtomicElementSize * 8);
412 
413  unsigned MinAlign = std::min(SrcAlign, DestAlign);
414 
415  // A (multi-)dword access at an address == 2 (mod 4) will be decomposed by the
416  // hardware into byte accesses. If you assume all alignments are equally
417  // probable, it's more efficient on average to use short accesses for this
418  // case.
419  if (MinAlign == 2)
420  return Type::getInt16Ty(Context);
421 
422  // Not all subtargets have 128-bit DS instructions, and we currently don't
423  // form them by default.

425  SrcAddrSpace == AMDGPUAS::REGION_ADDRESS ||
426  DestAddrSpace == AMDGPUAS::LOCAL_ADDRESS ||

428  return FixedVectorType::get(Type::getInt32Ty(Context), 2);

430 
431  // Global memory works best with 16-byte accesses. Private memory will also
432  // hit this, although they'll be decomposed.
433  return FixedVectorType::get(Type::getInt32Ty(Context), 4);
434 }

486 bool GCNTTIImpl::getTgtMemIntrinsic(IntrinsicInst *Inst,
487  MemIntrinsicInfo &Info) const {
488  switch (Inst->getIntrinsicID()) {
489  case Intrinsic::amdgcn_atomic_inc:
490  case Intrinsic::amdgcn_atomic_dec:
491  case Intrinsic::amdgcn_ds_ordered_add:
492  case Intrinsic::amdgcn_ds_ordered_swap:
493  case Intrinsic::amdgcn_ds_fadd:
494  case Intrinsic::amdgcn_ds_fmin:
495  case Intrinsic::amdgcn_ds_fmax: {
496  auto *Ordering = dyn_cast<ConstantInt>(Inst->getArgOperand(2));
497  auto *Volatile = dyn_cast<ConstantInt>(Inst->getArgOperand(4));
498  if (!Ordering || !Volatile)
499  return false; // Invalid.
500 
501  unsigned OrderingVal = Ordering->getZExtValue();
502  if (OrderingVal > static_cast<unsigned>(AtomicOrdering::SequentiallyConsistent))
503  return false;
504 
505  Info.PtrVal = Inst->getArgOperand(0);
506  Info.Ordering = static_cast<AtomicOrdering>(OrderingVal);
507  Info.ReadMem = true;
508  Info.WriteMem = true;
509  Info.IsVolatile = !Volatile->isZero();
510  return true;
511  }
512  default:
513  return false;
514  }

517 InstructionCost GCNTTIImpl::getArithmeticInstrCost(
518  unsigned Opcode, Type *Ty, TTI::TargetCostKind CostKind,
519  TTI::OperandValueInfo Op1Info, TTI::OperandValueInfo Op2Info,
520  ArrayRef<const Value *> Args,
521  const Instruction *CxtI) {
522 
523  // Legalize the type.
524  std::pair<InstructionCost, MVT> LT = getTypeLegalizationCost(Ty);
525  int ISD = TLI->InstructionOpcodeToISD(Opcode);
526 
527  // Because we don't have any legal vector operations, but the legal types, we
528  // need to account for split vectors.
529  unsigned NElts = LT.second.isVector() ?
530  LT.second.getVectorNumElements() : 1;
531 
532  MVT::SimpleValueType SLT = LT.second.getScalarType().SimpleTy;
533 
534  switch (ISD) {
535  case ISD::SHL:
536  case ISD::SRL:
537  case ISD::SRA:
538  if (SLT == MVT::i64)
539  return get64BitInstrCost(CostKind) * LT.first * NElts;
540 
541  if (ST->has16BitInsts() && SLT == MVT::i16)
542  NElts = (NElts + 1) / 2;
543 
544  // i32
545  return getFullRateInstrCost() * LT.first * NElts;
546  case ISD::ADD:
547  case ISD::SUB:
548  case ISD::AND:
549  case ISD::OR:
550  case ISD::XOR:
551  if (SLT == MVT::i64) {
552  // and, or and xor are typically split into 2 VALU instructions.
553  return 2 * getFullRateInstrCost() * LT.first * NElts;
554  }
555 
556  if (ST->has16BitInsts() && SLT == MVT::i16)
557  NElts = (NElts + 1) / 2;
558 
559  return LT.first * NElts * getFullRateInstrCost();
560  case ISD::MUL: {
561  const int QuarterRateCost = getQuarterRateInstrCost(CostKind);
562  if (SLT == MVT::i64) {
563  const int FullRateCost = getFullRateInstrCost();
564  return (4 * QuarterRateCost + (2 * 2) * FullRateCost) * LT.first * NElts;
565  }
566 
567  if (ST->has16BitInsts() && SLT == MVT::i16)
568  NElts = (NElts + 1) / 2;
569 
570  // i32
571  return QuarterRateCost * NElts * LT.first;
572  }
573  case ISD::FMUL:
574  // Check possible fuse {fadd|fsub}(a,fmul(b,c)) and return zero cost for
575  // fmul(b,c) supposing the fadd|fsub will get estimated cost for the whole
576  // fused operation.
577  if (CxtI && CxtI->hasOneUse())
578  if (const auto *FAdd = dyn_cast<BinaryOperator>(*CxtI->user_begin())) {
579  const int OPC = TLI->InstructionOpcodeToISD(FAdd->getOpcode());
580  if (OPC == ISD::FADD || OPC == ISD::FSUB) {
581  if (ST->hasMadMacF32Insts() && SLT == MVT::f32 && !HasFP32Denormals)
582  return TargetTransformInfo::TCC_Free;
583  if (ST->has16BitInsts() && SLT == MVT::f16 && !HasFP64FP16Denormals)
584  return TargetTransformInfo::TCC_Free;
585 
586  // Estimate all types may be fused with contract/unsafe flags
587  const TargetOptions &Options = TLI->getTargetMachine().Options;
588  if (Options.AllowFPOpFusion == FPOpFusion::Fast ||
589  Options.UnsafeFPMath ||
590  (FAdd->hasAllowContract() && CxtI->hasAllowContract()))
591  return TargetTransformInfo::TCC_Free;
592  }
593  }
594  [[fallthrough]];
595  case ISD::FADD:
596  case ISD::FSUB:
597  if (ST->hasPackedFP32Ops() && SLT == MVT::f32)
598  NElts = (NElts + 1) / 2;
599  if (SLT == MVT::f64)
600  return LT.first * NElts * get64BitInstrCost(CostKind);
601 
602  if (ST->has16BitInsts() && SLT == MVT::f16)
603  NElts = (NElts + 1) / 2;
604 
605  if (SLT == MVT::f32 || SLT == MVT::f16)
606  return LT.first * NElts * getFullRateInstrCost();
607  break;
608  case ISD::FDIV:
609  case ISD::FREM:
610  // FIXME: frem should be handled separately. The fdiv in it is most of it,
611  // but the current lowering is also not entirely correct.
612  if (SLT == MVT::f64) {
613  int Cost = 7 * get64BitInstrCost(CostKind) +
614  getQuarterRateInstrCost(CostKind) +
615  3 * getHalfRateInstrCost(CostKind);
616  // Add cost of workaround.
617  if (!ST->hasUsableDivScaleConditionOutput())
618  Cost += 3 * getFullRateInstrCost();
619 
620  return LT.first * Cost * NElts;
621  }
622 
623  if (!Args.empty() && match(Args[0], PatternMatch::m_FPOne())) {
624  // TODO: This is more complicated, unsafe flags etc.
625  if ((SLT == MVT::f32 && !HasFP32Denormals) ||
626  (SLT == MVT::f16 && ST->has16BitInsts())) {
627  return LT.first * getQuarterRateInstrCost(CostKind) * NElts;
628  }
629  }
630 
631  if (SLT == MVT::f16 && ST->has16BitInsts()) {
632  // 2 x v_cvt_f32_f16
633  // f32 rcp
634  // f32 fmul
635  // v_cvt_f16_f32
636  // f16 div_fixup
637  int Cost =
638  4 * getFullRateInstrCost() + 2 * getQuarterRateInstrCost(CostKind);
639  return LT.first * Cost * NElts;
640  }
641 
642  if (SLT == MVT::f32 || SLT == MVT::f16) {
643  // 4 more v_cvt_* insts without f16 insts support
644  int Cost = (SLT == MVT::f16 ? 14 : 10) * getFullRateInstrCost() +
645  1 * getQuarterRateInstrCost(CostKind);
646 
647  if (!HasFP32Denormals) {
648  // FP mode switches.
649  Cost += 2 * getFullRateInstrCost();
650  }
651 
652  return LT.first * NElts * Cost;
653  }
654  break;
655  case ISD::FNEG:
656  // Use the backend' estimation. If fneg is not free each element will cost
657  // one additional instruction.
658  return TLI->isFNegFree(SLT) ? 0 : NElts;
659  default:
660  break;
661  }
662 
663  return BaseT::getArithmeticInstrCost(Opcode, Ty, CostKind, Op1Info, Op2Info,
664  Args, CxtI);
665 }

888 /// \returns true if the result of the value could potentially be
889 /// different across workitems in a wavefront.
890 bool GCNTTIImpl::isSourceOfDivergence(const Value *V) const {
891  if (const Argument *A = dyn_cast<Argument>(V))
892  return !AMDGPU::isArgPassedInSGPR(A);
893 
894  // Loads from the private and flat address spaces are divergent, because
895  // threads can execute the load instruction with the same inputs and get
896  // different results.
897  //
898  // All other loads are not divergent, because if threads issue loads with the
899  // same arguments, they will always get the same result.
900  if (const LoadInst *Load = dyn_cast<LoadInst>(V))
901  return Load->getPointerAddressSpace() == AMDGPUAS::PRIVATE_ADDRESS ||
902  Load->getPointerAddressSpace() == AMDGPUAS::FLAT_ADDRESS;
903 
904  // Atomics are divergent because they are executed sequentially: when an
905  // atomic operation refers to the same address in each thread, then each
906  // thread after the first sees the value written by the previous thread as
907  // original value.
908  if (isa<AtomicRMWInst>(V) || isa<AtomicCmpXchgInst>(V))
909  return true;
910 
911  if (const IntrinsicInst *Intrinsic = dyn_cast<IntrinsicInst>(V)) {
912  if (Intrinsic->getIntrinsicID() == Intrinsic::read_register)
913  return isReadRegisterSourceOfDivergence(Intrinsic);
914 
915  return AMDGPU::isIntrinsicSourceOfDivergence(Intrinsic->getIntrinsicID());
916  }
917 
918  // Assume all function calls are a source of divergence.
919  if (const CallInst *CI = dyn_cast<CallInst>(V)) {
920  if (CI->isInlineAsm())
921  return isInlineAsmSourceOfDivergence(CI);
922  return true;
923  }
924 
925  // Assume all function calls are a source of divergence.
926  if (isa<InvokeInst>(V))
927  return true;
928 
929  return false;
930 }

932 bool GCNTTIImpl::isAlwaysUniform(const Value *V) const {
933  if (const IntrinsicInst *Intrinsic = dyn_cast<IntrinsicInst>(V))
934  return AMDGPU::isIntrinsicAlwaysUniform(Intrinsic->getIntrinsicID());
935 
936  if (const CallInst *CI = dyn_cast<CallInst>(V)) {
937  if (CI->isInlineAsm())
938  return !isInlineAsmSourceOfDivergence(CI);
939  return false;
940  }
941 
942  // In most cases TID / wavefrontsize is uniform.
943  //
944  // However, if a kernel has uneven dimesions we can have a value of
945  // workitem-id-x divided by the wavefrontsize non-uniform. For example
946  // dimensions (65, 2) will have workitems with address (64, 0) and (0, 1)
947  // packed into a same wave which gives 1 and 0 after the division by 64
948  // respectively.
949  //
950  // FIXME: limit it to 1D kernels only, although that shall be possible
951  // to perform this optimization is the size of the X dimension is a power
952  // of 2, we just do not currently have infrastructure to query it.
953  using namespace llvm::PatternMatch;
954  uint64_t C;
955  if (match(V, m_LShr(m_Intrinsic<Intrinsic::amdgcn_workitem_id_x>(),
956  m_ConstantInt(C))) ||
957  match(V, m_AShr(m_Intrinsic<Intrinsic::amdgcn_workitem_id_x>(),
958  m_ConstantInt(C)))) {
959  const Function *F = cast<Instruction>(V)->getFunction();
960  return C >= ST->getWavefrontSizeLog2() &&
961  ST->getMaxWorkitemID(*F, 1) == 0 && ST->getMaxWorkitemID(*F, 2) == 0;
962  }
963 
964  Value *Mask;
965  if (match(V, m_c_And(m_Intrinsic<Intrinsic::amdgcn_workitem_id_x>(),
966  m_Value(Mask)))) {
967  const Function *F = cast<Instruction>(V)->getFunction();
968  const DataLayout &DL = F->getParent()->getDataLayout();
969  return computeKnownBits(Mask, DL).countMinTrailingZeros() >=
970  ST->getWavefrontSizeLog2() &&
971  ST->getMaxWorkitemID(*F, 1) == 0 && ST->getMaxWorkitemID(*F, 2) == 0;
972  }
973 
974  const ExtractValueInst *ExtValue = dyn_cast<ExtractValueInst>(V);
975  if (!ExtValue)
976  return false;
977 
978  const CallInst *CI = dyn_cast<CallInst>(ExtValue->getOperand(0));
979  if (!CI)
980  return false;
981 
982  if (const IntrinsicInst *Intrinsic = dyn_cast<IntrinsicInst>(CI)) {
983  switch (Intrinsic->getIntrinsicID()) {
984  default:
985  return false;
986  case Intrinsic::amdgcn_if:
987  case Intrinsic::amdgcn_else: {
988  ArrayRef<unsigned> Indices = ExtValue->getIndices();
989  return Indices.size() == 1 && Indices[0] == 1;
990  }
991  }
992  }
993 
994  // If we have inline asm returning mixed SGPR and VGPR results, we inferred
995  // divergent for the overall struct return. We need to override it in the
996  // case we're extracting an SGPR component here.
997  if (CI->isInlineAsm())
998  return !isInlineAsmSourceOfDivergence(CI, ExtValue->getIndices());
999 
1000  return false;
1001 }

1029  case Intrinsic::amdgcn_atomic_dec:

1070  DL.getPointerSizeInBits(NewAS) != 32)

1091  case Intrinsic::amdgcn_flat_atomic_fmax:
1092  case Intrinsic::amdgcn_flat_atomic_fmin: {

1098  II->setArgOperand(0, NewV);
1099  II->setCalledFunction(NewDecl);

1102  default:
