501 static StringRef computeDataLayout(const Triple &TT) {

503  // 32-bit pointers.
504  return "e-p:32:32-i64:64-v16:16-v24:32-v32:32-v48:64-v96:128"

511  "-i64:64-v16:16-v24:32-v32:32-v48:64-v96:128"

742 unsigned AMDGPUTargetMachine::getAssumedAddrSpace(const Value *V) const {
743  const auto *LD = dyn_cast<LoadInst>(V);
744  if (!LD)
745  return AMDGPUAS::UNKNOWN_ADDRESS_SPACE;
746 
747  // It must be a generic pointer loaded.
748  assert(V->getType()->isPointerTy() &&
749  V->getType()->getPointerAddressSpace() == AMDGPUAS::FLAT_ADDRESS);
750 
751  const auto *Ptr = LD->getPointerOperand();
752  if (Ptr->getType()->getPointerAddressSpace() != AMDGPUAS::CONSTANT_ADDRESS)
753  return AMDGPUAS::UNKNOWN_ADDRESS_SPACE;
754  // For a generic pointer loaded from the constant memory, it could be assumed
755  // as a global pointer since the constant memory is only populated on the
756  // host side. As implied by the offload programming model, only global
757  // pointers could be referenced on the host side.
758  return AMDGPUAS::GLOBAL_ADDRESS;
759 }

761 std::pair<const Value *, unsigned>
762 AMDGPUTargetMachine::getPredicatedAddrSpace(const Value *V) const {
763  if (auto *II = dyn_cast<IntrinsicInst>(V)) {
764  switch (II->getIntrinsicID()) {
765  case Intrinsic::amdgcn_is_shared:
766  return std::pair(II->getArgOperand(0), AMDGPUAS::LOCAL_ADDRESS);
767  case Intrinsic::amdgcn_is_private:
768  return std::pair(II->getArgOperand(0), AMDGPUAS::PRIVATE_ADDRESS);
769  default:
770  break;
771  }
772  return std::pair(nullptr, -1);
773  }
774  // Check the global pointer predication based on
775  // (!is_share(p) && !is_private(p)). Note that logic 'and' is commutative and
776  // the order of 'is_shared' and 'is_private' is not significant.
777  Value *Ptr;
778  if (match(
779  const_cast<Value *>(V),
780  m_c_And(m_Not(m_Intrinsic<Intrinsic::amdgcn_is_shared>(m_Value(Ptr))),
781  m_Not(m_Intrinsic<Intrinsic::amdgcn_is_private>(
782  m_Deferred(Ptr))))))
783  return std::pair(Ptr, AMDGPUAS::GLOBAL_ADDRESS);
784 
785  return std::pair(nullptr, -1);
786 }

951 void AMDGPUPassConfig::addIRPasses() {
952  const AMDGPUTargetMachine &TM = getAMDGPUTargetMachine();
953 
954  // There is no reason to run these.
955  disablePass(&StackMapLivenessID);
956  disablePass(&FuncletLayoutID);
957  disablePass(&PatchableFunctionID);
958 
959  addPass(createAMDGPUPrintfRuntimeBinding());
960  addPass(createAMDGPUCtorDtorLoweringLegacyPass());
961 
962  // A call to propagate attributes pass in the backend in case opt was not run.
963  addPass(createAMDGPUPropagateAttributesEarlyPass(&TM));
964 
965  addPass(createAMDGPULowerIntrinsicsPass());
966 
967  // Function calls are not supported, so make sure we inline everything.
968  addPass(createAMDGPUAlwaysInlinePass());
969  addPass(createAlwaysInlinerLegacyPass());
970  // We need to add the barrier noop pass, otherwise adding the function
971  // inlining pass will cause all of the PassConfigs passes to be run
972  // one function at a time, which means if we have a module with two
973  // functions, then we will generate code for the first function
974  // without ever running any passes on the second.
975  addPass(createBarrierNoopPass());
976 
977  // Handle uses of OpenCL image2d_t, image3d_t and sampler_t arguments.
978  if (TM.getTargetTriple().getArch() == Triple::r600)
979  addPass(createR600OpenCLImageTypeLoweringPass());
980 
981  // Replace OpenCL enqueued block function pointers with global variables.
982  addPass(createAMDGPUOpenCLEnqueuedBlockLoweringPass());
983 
984  // Can increase LDS used by kernel so runs before PromoteAlloca
985  if (EnableLowerModuleLDS) {
986  // The pass "amdgpu-replace-lds-use-with-pointer" need to be run before the
987  // pass "amdgpu-lower-module-lds", and also it required to be run only if
988  // "amdgpu-lower-module-lds" pass is enabled.
989  if (EnableLDSReplaceWithPointer)
990  addPass(createAMDGPUReplaceLDSUseWithPointerPass());
991 
992  addPass(createAMDGPULowerModuleLDSPass());
993  }
994 
995  if (TM.getOptLevel() > CodeGenOpt::None)
996  addPass(createInferAddressSpacesPass());
997 
998  addPass(createAtomicExpandPass());
999 
1000  if (TM.getOptLevel() > CodeGenOpt::None) {
1001  addPass(createAMDGPUPromoteAlloca());
1002 
1003  if (EnableSROA)
1004  addPass(createSROAPass());
1005  if (isPassEnabled(EnableScalarIRPasses))
1006  addStraightLineScalarOptimizationPasses();
1007 
1008  if (EnableAMDGPUAliasAnalysis) {
1009  addPass(createAMDGPUAAWrapperPass());
1010  addPass(createExternalAAWrapperPass([](Pass &P, Function &,
1011  AAResults &AAR) {
1012  if (auto *WrapperPass = P.getAnalysisIfAvailable<AMDGPUAAWrapperPass>())
1013  AAR.addAAResult(WrapperPass->getResult());
1014  }));
1015  }

1018  // TODO: May want to move later or split into an early and late one.
1019  addPass(createAMDGPUCodeGenPreparePass());
1020  }
1021  }

1026  // example, GVN can combine
1027  //
1028  // %0 = add %a, %b
1029  // %1 = add %b, %a

1031  // and
1032  //
1033  // %0 = shl nsw %a, 2
