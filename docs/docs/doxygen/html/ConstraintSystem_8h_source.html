74  bool addVariableRow(ArrayRef<int64_t> R) {
75  assert(Constraints.empty() || R.size() == NumVariables);
76  // If all variable coefficients are 0, the constraint does not provide any
77  // usable information.
78  if (all_of(ArrayRef(R).drop_front(1), [](int64_t C) { return C == 0; }))
79  return false;
80 
81  SmallVector<Entry, 4> NewRow;
82  for (const auto &[Idx, C] : enumerate(R)) {
83  if (C == 0)
84  continue;
85  auto A = std::abs(C);
86  GCD = APIntOps::GreatestCommonDivisor({32, (uint32_t)A}, {32, GCD})
87  .getZExtValue();
88 
89  NewRow.emplace_back(C, Idx);
90  }
91  if (Constraints.empty())
92  NumVariables = R.size();
93  Constraints.push_back(std::move(NewRow));
94  return true;
95  }
