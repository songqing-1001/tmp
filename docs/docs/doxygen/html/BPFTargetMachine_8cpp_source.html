105 void BPFTargetMachine::registerPassBuilderCallbacks(PassBuilder &PB) {
106  PB.registerPipelineStartEPCallback(
107  [=](ModulePassManager &MPM, OptimizationLevel) {
108  FunctionPassManager FPM;
109  FPM.addPass(BPFAbstractMemberAccessPass(this));
110  FPM.addPass(BPFPreserveDITypePass());
111  FPM.addPass(BPFIRPeepholePass());
112  MPM.addPass(createModuleToFunctionPassAdaptor(std::move(FPM)));
113  });
114  PB.registerPeepholeEPCallback([=](FunctionPassManager &FPM,
115  OptimizationLevel Level) {
116  FPM.addPass(SimplifyCFGPass(SimplifyCFGOptions().hoistCommonInsts(true)));
117  });
118  PB.registerPipelineEarlySimplificationEPCallback(
119  [=](ModulePassManager &MPM, OptimizationLevel) {
120  MPM.addPass(BPFAdjustOptPass());
121  });
122 }

142 void BPFPassConfig::addMachineSSAOptimization() {
143  addPass(createBPFMISimplifyPatchablePass());
144 
145  // The default implementation must be called first as we want eBPF
146  // Peephole ran at last.
147  TargetPassConfig::addMachineSSAOptimization();
148 
149  const BPFSubtarget *Subtarget = getBPFTargetMachine().getSubtargetImpl();
150  if (!DisableMIPeephole) {
151  if (Subtarget->getHasAlu32())
152  addPass(createBPFMIPeepholePass());
153  addPass(createBPFMIPeepholeTruncElimPass());
154  }
155 }
