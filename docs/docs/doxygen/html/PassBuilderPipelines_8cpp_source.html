314 // TODO: Investigate the cost/benefit of tail call elimination on debugging.
315 FunctionPassManager
316 PassBuilder::buildO1FunctionSimplificationPipeline(OptimizationLevel Level,
317  ThinOrFullLTOPhase Phase) {
318 
319  FunctionPassManager FPM;
320 
321  if (CountCGSCCVisits)
322  FPM.addPass(CountVisitsPass());
323 
324  // Form SSA out of local memory accesses after breaking apart aggregates into
325  // scalars.
326  FPM.addPass(SROAPass(SROAOptions::ModifyCFG));
327 
328  // Catch trivial redundancies
329  FPM.addPass(EarlyCSEPass(true /* Enable mem-ssa. */));
330 
331  // Hoisting of scalars and load expressions.
332  FPM.addPass(
333  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
334  FPM.addPass(InstCombinePass());
335 
336  FPM.addPass(LibCallsShrinkWrapPass());
337 
338  invokePeepholeEPCallbacks(FPM, Level);
339 
340  FPM.addPass(
341  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
342 
343  // Form canonically associated expression trees, and simplify the trees using
344  // basic mathematical properties. For example, this will form (nearly)
345  // minimal multiplication trees.
346  FPM.addPass(ReassociatePass());
347 
348  // Add the primary loop simplification pipeline.
349  // FIXME: Currently this is split into two loop pass pipelines because we run
350  // some function passes in between them. These can and should be removed
351  // and/or replaced by scheduling the loop pass equivalents in the correct
352  // positions. But those equivalent passes aren't powerful enough yet.
353  // Specifically, `SimplifyCFGPass` and `InstCombinePass` are currently still
354  // used. We have `LoopSimplifyCFGPass` which isn't yet powerful enough yet to
355  // fully replace `SimplifyCFGPass`, and the closest to the other we have is
356  // `LoopInstSimplify`.
357  LoopPassManager LPM1, LPM2;
358 
359  // Simplify the loop body. We do this initially to clean up after other loop
360  // passes run, either when iterating on a loop or on inner loops with
361  // implications on the outer loop.
362  LPM1.addPass(LoopInstSimplifyPass());
363  LPM1.addPass(LoopSimplifyCFGPass());
364 
365  // Try to remove as much code from the loop header as possible,
366  // to reduce amount of IR that will have to be duplicated. However,
367  // do not perform speculative hoisting the first time as LICM
368  // will destroy metadata that may not need to be destroyed if run
369  // after loop rotation.
370  // TODO: Investigate promotion cap for O1.
371  LPM1.addPass(LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
372  /*AllowSpeculation=*/false));
373 
374  LPM1.addPass(LoopRotatePass(/* Disable header duplication */ true,
375  isLTOPreLink(Phase)));
376  // TODO: Investigate promotion cap for O1.
377  LPM1.addPass(LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
378  /*AllowSpeculation=*/true));
379  LPM1.addPass(SimpleLoopUnswitchPass());
380  if (EnableLoopFlatten)
381  LPM1.addPass(LoopFlattenPass());
382 
383  LPM2.addPass(LoopIdiomRecognizePass());
384  LPM2.addPass(IndVarSimplifyPass());
385 
386  for (auto &C : LateLoopOptimizationsEPCallbacks)
387  C(LPM2, Level);
388 
389  LPM2.addPass(LoopDeletionPass());
390 
391  if (EnableLoopInterchange)
392  LPM2.addPass(LoopInterchangePass());
393 
394  // Do not enable unrolling in PreLinkThinLTO phase during sample PGO
395  // because it changes IR to makes profile annotation in back compile
396  // inaccurate. The normal unroller doesn't pay attention to forced full unroll
397  // attributes so we need to make sure and allow the full unroll pass to pay
398  // attention to it.
399  if (Phase != ThinOrFullLTOPhase::ThinLTOPreLink || !PGOOpt ||
400  PGOOpt->Action != PGOOptions::SampleUse)
401  LPM2.addPass(LoopFullUnrollPass(Level.getSpeedupLevel(),
402  /* OnlyWhenForced= */ !PTO.LoopUnrolling,
403  PTO.ForgetAllSCEVInLoopUnroll));
404 
405  for (auto &C : LoopOptimizerEndEPCallbacks)
406  C(LPM2, Level);
407 
408  // We provide the opt remark emitter pass for LICM to use. We only need to do
409  // this once as it is immutable.
410  FPM.addPass(
411  RequireAnalysisPass<OptimizationRemarkEmitterAnalysis, Function>());
412  FPM.addPass(createFunctionToLoopPassAdaptor(std::move(LPM1),
413  /*UseMemorySSA=*/true,
414  /*UseBlockFrequencyInfo=*/true));
415  FPM.addPass(
416  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
417  FPM.addPass(InstCombinePass());
418  // The loop passes in LPM2 (LoopFullUnrollPass) do not preserve MemorySSA.
419  // *All* loop passes must preserve it, in order to be able to use it.
420  FPM.addPass(createFunctionToLoopPassAdaptor(std::move(LPM2),
421  /*UseMemorySSA=*/false,
422  /*UseBlockFrequencyInfo=*/false));
423 
424  // Delete small array after loop unroll.
425  FPM.addPass(SROAPass(SROAOptions::ModifyCFG));
426 
427  // Specially optimize memory movement as it doesn't look like dataflow in SSA.
428  FPM.addPass(MemCpyOptPass());
429 
430  // Sparse conditional constant propagation.
431  // FIXME: It isn't clear why we do this *after* loop passes rather than
432  // before...
433  FPM.addPass(SCCPPass());
434 
435  // Delete dead bit computations (instcombine runs after to fold away the dead
436  // computations, and then ADCE will run later to exploit any new DCE
437  // opportunities that creates).
438  FPM.addPass(BDCEPass());
439 
440  // Run instcombine after redundancy and dead bit elimination to exploit
441  // opportunities opened up by them.
442  FPM.addPass(InstCombinePass());
443  invokePeepholeEPCallbacks(FPM, Level);
444 
445  FPM.addPass(CoroElidePass());
446 
447  for (auto &C : ScalarOptimizerLateEPCallbacks)
448  C(FPM, Level);
449 
450  // Finally, do an expensive DCE pass to catch all the dead code exposed by
451  // the simplifications and basic cleanup after all the simplifications.
452  // TODO: Investigate if this is too expensive.
453  FPM.addPass(ADCEPass());
454  FPM.addPass(
455  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
456  FPM.addPass(InstCombinePass());
457  invokePeepholeEPCallbacks(FPM, Level);
458 
459  return FPM;
460 }

462 FunctionPassManager
463 PassBuilder::buildFunctionSimplificationPipeline(OptimizationLevel Level,
464  ThinOrFullLTOPhase Phase) {
465  assert(Level != OptimizationLevel::O0 && "Must request optimizations!");
466 
467  // The O1 pipeline has a separate pipeline creation function to simplify
468  // construction readability.
469  if (Level.getSpeedupLevel() == 1)
470  return buildO1FunctionSimplificationPipeline(Level, Phase);
471 
472  FunctionPassManager FPM;
473 
474  if (CountCGSCCVisits)
475  FPM.addPass(CountVisitsPass());
476 
477  // Form SSA out of local memory accesses after breaking apart aggregates into
478  // scalars.
479  FPM.addPass(SROAPass(SROAOptions::ModifyCFG));
480 
481  // Catch trivial redundancies
482  FPM.addPass(EarlyCSEPass(true /* Enable mem-ssa. */));
483  if (EnableKnowledgeRetention)
484  FPM.addPass(AssumeSimplifyPass());
485 
486  // Hoisting of scalars and load expressions.
487  if (EnableGVNHoist)
488  FPM.addPass(GVNHoistPass());
489 
490  // Global value numbering based sinking.
491  if (EnableGVNSink) {
492  FPM.addPass(GVNSinkPass());
493  FPM.addPass(
494  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
495  }
496 
497  // Speculative execution if the target has divergent branches; otherwise nop.
498  FPM.addPass(SpeculativeExecutionPass(/* OnlyIfDivergentTarget =*/true));
499 
500  // Optimize based on known information about branches, and cleanup afterward.
501  FPM.addPass(JumpThreadingPass());
502  FPM.addPass(CorrelatedValuePropagationPass());
503 
504  FPM.addPass(
505  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
506  FPM.addPass(InstCombinePass());
507  if (Level == OptimizationLevel::O3)
508  FPM.addPass(AggressiveInstCombinePass());
509 
510  if (EnableConstraintElimination)
511  FPM.addPass(ConstraintEliminationPass());
512 
513  if (!Level.isOptimizingForSize())
514  FPM.addPass(LibCallsShrinkWrapPass());
515 
516  invokePeepholeEPCallbacks(FPM, Level);
517 
518  // For PGO use pipeline, try to optimize memory intrinsics such as memcpy
519  // using the size value profile. Don't perform this when optimizing for size.
520  if (PGOOpt && PGOOpt->Action == PGOOptions::IRUse &&
521  !Level.isOptimizingForSize())
522  FPM.addPass(PGOMemOPSizeOpt());
523 
524  FPM.addPass(TailCallElimPass());
525  FPM.addPass(
526  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
527 
528  // Form canonically associated expression trees, and simplify the trees using
529  // basic mathematical properties. For example, this will form (nearly)
530  // minimal multiplication trees.
531  FPM.addPass(ReassociatePass());
532 
533  // Add the primary loop simplification pipeline.
534  // FIXME: Currently this is split into two loop pass pipelines because we run
535  // some function passes in between them. These can and should be removed
536  // and/or replaced by scheduling the loop pass equivalents in the correct
537  // positions. But those equivalent passes aren't powerful enough yet.
538  // Specifically, `SimplifyCFGPass` and `InstCombinePass` are currently still
539  // used. We have `LoopSimplifyCFGPass` which isn't yet powerful enough yet to
540  // fully replace `SimplifyCFGPass`, and the closest to the other we have is
541  // `LoopInstSimplify`.
542  LoopPassManager LPM1, LPM2;
543 
544  // Simplify the loop body. We do this initially to clean up after other loop
545  // passes run, either when iterating on a loop or on inner loops with
546  // implications on the outer loop.
547  LPM1.addPass(LoopInstSimplifyPass());
548  LPM1.addPass(LoopSimplifyCFGPass());
549 
550  // Try to remove as much code from the loop header as possible,
551  // to reduce amount of IR that will have to be duplicated. However,
552  // do not perform speculative hoisting the first time as LICM
553  // will destroy metadata that may not need to be destroyed if run
554  // after loop rotation.
555  // TODO: Investigate promotion cap for O1.
556  LPM1.addPass(LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
557  /*AllowSpeculation=*/false));
558 
559  // Disable header duplication in loop rotation at -Oz.
560  LPM1.addPass(
561  LoopRotatePass(Level != OptimizationLevel::Oz, isLTOPreLink(Phase)));
562  // TODO: Investigate promotion cap for O1.
563  LPM1.addPass(LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
564  /*AllowSpeculation=*/true));
565  LPM1.addPass(
566  SimpleLoopUnswitchPass(/* NonTrivial */ Level == OptimizationLevel::O3));
567  if (EnableLoopFlatten)
568  LPM1.addPass(LoopFlattenPass());
569 
570  LPM2.addPass(LoopIdiomRecognizePass());
571  LPM2.addPass(IndVarSimplifyPass());
572 
573  for (auto &C : LateLoopOptimizationsEPCallbacks)
574  C(LPM2, Level);
575 
576  LPM2.addPass(LoopDeletionPass());
577 
578  if (EnableLoopInterchange)
579  LPM2.addPass(LoopInterchangePass());
580 
581  // Do not enable unrolling in PreLinkThinLTO phase during sample PGO
582  // because it changes IR to makes profile annotation in back compile
583  // inaccurate. The normal unroller doesn't pay attention to forced full unroll
584  // attributes so we need to make sure and allow the full unroll pass to pay
585  // attention to it.
586  if (Phase != ThinOrFullLTOPhase::ThinLTOPreLink || !PGOOpt ||
587  PGOOpt->Action != PGOOptions::SampleUse)
588  LPM2.addPass(LoopFullUnrollPass(Level.getSpeedupLevel(),
589  /* OnlyWhenForced= */ !PTO.LoopUnrolling,
590  PTO.ForgetAllSCEVInLoopUnroll));
591 
592  for (auto &C : LoopOptimizerEndEPCallbacks)
593  C(LPM2, Level);
594 
595  // We provide the opt remark emitter pass for LICM to use. We only need to do
596  // this once as it is immutable.
597  FPM.addPass(
598  RequireAnalysisPass<OptimizationRemarkEmitterAnalysis, Function>());
599  FPM.addPass(createFunctionToLoopPassAdaptor(std::move(LPM1),
600  /*UseMemorySSA=*/true,
601  /*UseBlockFrequencyInfo=*/true));
602  FPM.addPass(
603  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
604  FPM.addPass(InstCombinePass());
605  // The loop passes in LPM2 (LoopIdiomRecognizePass, IndVarSimplifyPass,
606  // LoopDeletionPass and LoopFullUnrollPass) do not preserve MemorySSA.
607  // *All* loop passes must preserve it, in order to be able to use it.
608  FPM.addPass(createFunctionToLoopPassAdaptor(std::move(LPM2),
609  /*UseMemorySSA=*/false,
610  /*UseBlockFrequencyInfo=*/false));
611 
612  // Delete small array after loop unroll.
613  FPM.addPass(SROAPass(SROAOptions::ModifyCFG));
614 
615  // Try vectorization/scalarization transforms that are both improvements
616  // themselves and can allow further folds with GVN and InstCombine.
617  FPM.addPass(VectorCombinePass(/*TryEarlyFoldsOnly=*/true));
618 
619  // Eliminate redundancies.
620  FPM.addPass(MergedLoadStoreMotionPass());
621  if (RunNewGVN)
622  FPM.addPass(NewGVNPass());
623  else
624  FPM.addPass(GVNPass());
625 
626  // Sparse conditional constant propagation.
627  // FIXME: It isn't clear why we do this *after* loop passes rather than
628  // before...
629  FPM.addPass(SCCPPass());
630 
631  // Delete dead bit computations (instcombine runs after to fold away the dead
632  // computations, and then ADCE will run later to exploit any new DCE
633  // opportunities that creates).
634  FPM.addPass(BDCEPass());
635 
636  // Run instcombine after redundancy and dead bit elimination to exploit
637  // opportunities opened up by them.
638  FPM.addPass(InstCombinePass());
639  invokePeepholeEPCallbacks(FPM, Level);
640 
641  // Re-consider control flow based optimizations after redundancy elimination,
642  // redo DCE, etc.
643  if (EnableDFAJumpThreading && Level.getSizeLevel() == 0)
644  FPM.addPass(DFAJumpThreadingPass());
645 
646  FPM.addPass(JumpThreadingPass());
647  FPM.addPass(CorrelatedValuePropagationPass());
648 
649  // Finally, do an expensive DCE pass to catch all the dead code exposed by
650  // the simplifications and basic cleanup after all the simplifications.
651  // TODO: Investigate if this is too expensive.
652  FPM.addPass(ADCEPass());
653 
654  // Specially optimize memory movement as it doesn't look like dataflow in SSA.
655  FPM.addPass(MemCpyOptPass());
656 
657  FPM.addPass(DSEPass());
658  FPM.addPass(createFunctionToLoopPassAdaptor(
659  LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
660  /*AllowSpeculation=*/true),
661  /*UseMemorySSA=*/true, /*UseBlockFrequencyInfo=*/true));
662 
663  FPM.addPass(CoroElidePass());
664 
665  for (auto &C : ScalarOptimizerLateEPCallbacks)
666  C(FPM, Level);
667 
668  FPM.addPass(SimplifyCFGPass(SimplifyCFGOptions()
669  .convertSwitchRangeToICmp(true)
670  .hoistCommonInsts(true)
671  .sinkCommonInsts(true)));
672  FPM.addPass(InstCombinePass());
673  invokePeepholeEPCallbacks(FPM, Level);
674 
675  return FPM;

787 ModuleInlinerWrapperPass
788 PassBuilder::buildInlinerPipeline(OptimizationLevel Level,
789  ThinOrFullLTOPhase Phase) {
790  InlineParams IP;
791  if (PTO.InlinerThreshold == -1)
792  IP = getInlineParamsFromOptLevel(Level);
793  else
794  IP = getInlineParams(PTO.InlinerThreshold);
795  // For PreLinkThinLTO + SamplePGO, set hot-caller threshold to 0 to
796  // disable hot callsite inline (as much as possible [1]) because it makes
797  // profile annotation in the backend inaccurate.
798  //
799  // [1] Note the cost of a function could be below zero due to erased
800  // prologue / epilogue.
801  if (Phase == ThinOrFullLTOPhase::ThinLTOPreLink && PGOOpt &&
802  PGOOpt->Action == PGOOptions::SampleUse)
803  IP.HotCallSiteThreshold = 0;
804 
805  if (PGOOpt)
806  IP.EnableDeferral = EnablePGOInlineDeferral;
807 
808  ModuleInlinerWrapperPass MIWP(IP, PerformMandatoryInliningsFirst,
809  InlineContext{Phase, InlinePass::CGSCCInliner},
810  UseInlineAdvisor, MaxDevirtIterations);
811 
812  // Require the GlobalsAA analysis for the module so we can query it within
813  // the CGSCC pipeline.
814  MIWP.addModulePass(RequireAnalysisPass<GlobalsAA, Module>());
815  // Invalidate AAManager so it can be recreated and pick up the newly available
816  // GlobalsAA.
817  MIWP.addModulePass(
818  createModuleToFunctionPassAdaptor(InvalidateAnalysisPass<AAManager>()));
819 
820  // Require the ProfileSummaryAnalysis for the module so we can query it within
821  // the inliner pass.
822  MIWP.addModulePass(RequireAnalysisPass<ProfileSummaryAnalysis, Module>());
823 
824  // Now begin the main postorder CGSCC pipeline.
825  // FIXME: The current CGSCC pipeline has its origins in the legacy pass
826  // manager and trying to emulate its precise behavior. Much of this doesn't
827  // make a lot of sense and we should revisit the core CGSCC structure.
828  CGSCCPassManager &MainCGPipeline = MIWP.getPM();
829 
830  // Note: historically, the PruneEH pass was run first to deduce nounwind and
831  // generally clean up exception handling overhead. It isn't clear this is
832  // valuable as the inliner doesn't currently care whether it is inlining an
833  // invoke or a call.
834 
835  if (AttributorRun & AttributorRunOption::CGSCC)
836  MainCGPipeline.addPass(AttributorCGSCCPass());
837 
838  // Now deduce any function attributes based in the current code.
839  MainCGPipeline.addPass(PostOrderFunctionAttrsPass());
840 
841  // When at O3 add argument promotion to the pass pipeline.
842  // FIXME: It isn't at all clear why this should be limited to O3.
843  if (Level == OptimizationLevel::O3)
844  MainCGPipeline.addPass(ArgumentPromotionPass());
845 
846  // Try to perform OpenMP specific optimizations. This is a (quick!) no-op if
847  // there are no OpenMP runtime calls present in the module.
848  if (Level == OptimizationLevel::O2 || Level == OptimizationLevel::O3)
849  MainCGPipeline.addPass(OpenMPOptCGSCCPass());
850 
851  for (auto &C : CGSCCOptimizerLateEPCallbacks)
852  C(MainCGPipeline, Level);
853 
854  // Lastly, add the core function simplification pipeline nested inside the
855  // CGSCC walk.
856  MainCGPipeline.addPass(createCGSCCToFunctionPassAdaptor(
857  buildFunctionSimplificationPipeline(Level, Phase),
858  PTO.EagerlyInvalidateAnalyses, /*NoRerun=*/true));
859 
860  MainCGPipeline.addPass(CoroSplitPass(Level != OptimizationLevel::O0));
861 
862  MIWP.addLateModulePass(createModuleToFunctionPassAdaptor(
863  InvalidateAnalysisPass<ShouldNotRunFunctionPassesAnalysis>()));
864 
865  return MIWP;
866 }

871  ModulePassManager MPM;

907 ModulePassManager
908 PassBuilder::buildModuleSimplificationPipeline(OptimizationLevel Level,
909  ThinOrFullLTOPhase Phase) {
910  ModulePassManager MPM;
911 
912  // Place pseudo probe instrumentation as the first pass of the pipeline to
913  // minimize the impact of optimization changes.
914  if (PGOOpt && PGOOpt->PseudoProbeForProfiling &&
915  Phase != ThinOrFullLTOPhase::ThinLTOPostLink)
916  MPM.addPass(SampleProfileProbePass(TM));
917 
918  bool HasSampleProfile = PGOOpt && (PGOOpt->Action == PGOOptions::SampleUse);
919 
920  // In ThinLTO mode, when flattened profile is used, all the available
921  // profile information will be annotated in PreLink phase so there is
922  // no need to load the profile again in PostLink.
923  bool LoadSampleProfile =
924  HasSampleProfile &&
925  !(FlattenedProfileUsed && Phase == ThinOrFullLTOPhase::ThinLTOPostLink);
926 
927  // During the ThinLTO backend phase we perform early indirect call promotion
928  // here, before globalopt. Otherwise imported available_externally functions
929  // look unreferenced and are removed. If we are going to load the sample
930  // profile then defer until later.
931  // TODO: See if we can move later and consolidate with the location where
932  // we perform ICP when we are loading a sample profile.
933  // TODO: We pass HasSampleProfile (whether there was a sample profile file
934  // passed to the compile) to the SamplePGO flag of ICP. This is used to
935  // determine whether the new direct calls are annotated with prof metadata.
936  // Ideally this should be determined from whether the IR is annotated with
937  // sample profile, and not whether the a sample profile was provided on the
938  // command line. E.g. for flattened profiles where we will not be reloading
939  // the sample profile in the ThinLTO backend, we ideally shouldn't have to
940  // provide the sample profile file.
941  if (Phase == ThinOrFullLTOPhase::ThinLTOPostLink && !LoadSampleProfile)
942  MPM.addPass(PGOIndirectCallPromotion(true /* InLTO */, HasSampleProfile));
943 
944  // Do basic inference of function attributes from known properties of system
945  // libraries and other oracles.
946  MPM.addPass(InferFunctionAttrsPass());
947  MPM.addPass(CoroEarlyPass());
948 
949  // Create an early function pass manager to cleanup the output of the
950  // frontend.
951  FunctionPassManager EarlyFPM;
952  // Lower llvm.expect to metadata before attempting transforms.
953  // Compare/branch metadata may alter the behavior of passes like SimplifyCFG.
954  EarlyFPM.addPass(LowerExpectIntrinsicPass());
955  EarlyFPM.addPass(SimplifyCFGPass());
956  EarlyFPM.addPass(SROAPass(SROAOptions::ModifyCFG));
957  EarlyFPM.addPass(EarlyCSEPass());
958  if (Level == OptimizationLevel::O3)
959  EarlyFPM.addPass(CallSiteSplittingPass());
960 
961  // In SamplePGO ThinLTO backend, we need instcombine before profile annotation
962  // to convert bitcast to direct calls so that they can be inlined during the
963  // profile annotation prepration step.
964  // More details about SamplePGO design can be found in:
965  // https://research.google.com/pubs/pub45290.html
966  // FIXME: revisit how SampleProfileLoad/Inliner/ICP is structured.
967  if (LoadSampleProfile)
968  EarlyFPM.addPass(InstCombinePass());
969  MPM.addPass(createModuleToFunctionPassAdaptor(std::move(EarlyFPM),
970  PTO.EagerlyInvalidateAnalyses));
971 
972  if (LoadSampleProfile) {
973  // Annotate sample profile right after early FPM to ensure freshness of
974  // the debug info.
975  MPM.addPass(SampleProfileLoaderPass(PGOOpt->ProfileFile,
976  PGOOpt->ProfileRemappingFile, Phase));
977  // Cache ProfileSummaryAnalysis once to avoid the potential need to insert
978  // RequireAnalysisPass for PSI before subsequent non-module passes.
979  MPM.addPass(RequireAnalysisPass<ProfileSummaryAnalysis, Module>());
980  // Do not invoke ICP in the LTOPrelink phase as it makes it hard
981  // for the profile annotation to be accurate in the LTO backend.
982  if (Phase != ThinOrFullLTOPhase::ThinLTOPreLink &&
983  Phase != ThinOrFullLTOPhase::FullLTOPreLink)
984  // We perform early indirect call promotion here, before globalopt.
985  // This is important for the ThinLTO backend phase because otherwise
986  // imported available_externally functions look unreferenced and are
987  // removed.
988  MPM.addPass(
989  PGOIndirectCallPromotion(true /* IsInLTO */, true /* SamplePGO */));
990  }
991 
992  // Try to perform OpenMP specific optimizations on the module. This is a
993  // (quick!) no-op if there are no OpenMP runtime calls present in the module.
994  if (Level != OptimizationLevel::O0)
995  MPM.addPass(OpenMPOptPass());
996 
997  if (AttributorRun & AttributorRunOption::MODULE)
998  MPM.addPass(AttributorPass());
999 
1000  // Lower type metadata and the type.test intrinsic in the ThinLTO
1001  // post link pipeline after ICP. This is to enable usage of the type
1002  // tests in ICP sequences.
1003  if (Phase == ThinOrFullLTOPhase::ThinLTOPostLink)
1004  MPM.addPass(LowerTypeTestsPass(nullptr, nullptr, true));
1005 
1006  for (auto &C : PipelineEarlySimplificationEPCallbacks)
1007  C(MPM, Level);
1008 
1009  // Interprocedural constant propagation now that basic cleanup has occurred
1010  // and prior to optimizing globals.
1011  // FIXME: This position in the pipeline hasn't been carefully considered in
1012  // years, it should be re-analyzed.
1013  MPM.addPass(IPSCCPPass(IPSCCPOptions(/*AllowFuncSpec=*/
1014  Level != OptimizationLevel::Os &&
1015  Level != OptimizationLevel::Oz)));
1016 
1017  // Attach metadata to indirect call sites indicating the set of functions
1018  // they may target at run-time. This should follow IPSCCP.
1019  MPM.addPass(CalledValuePropagationPass());
1020 
1021  // Optimize globals to try and fold them into constants.
1022  MPM.addPass(GlobalOptPass());
1023 
1024  // Promote any localized globals to SSA registers.
1025  // FIXME: Should this instead by a run of SROA?
1026  // FIXME: We should probably run instcombine and simplifycfg afterward to
1027  // delete control flows that are dead once globals have been folded to
1028  // constants.
1029  MPM.addPass(createModuleToFunctionPassAdaptor(PromotePass()));
1030 
1031  // Create a small function pass pipeline to cleanup after all the global
1032  // optimizations.
1033  FunctionPassManager GlobalCleanupPM;
1034  GlobalCleanupPM.addPass(InstCombinePass());
1035  invokePeepholeEPCallbacks(GlobalCleanupPM, Level);
1036 
1037  GlobalCleanupPM.addPass(
1038  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
1039  MPM.addPass(createModuleToFunctionPassAdaptor(std::move(GlobalCleanupPM),
1040  PTO.EagerlyInvalidateAnalyses));
1041 
1042  // Add all the requested passes for instrumentation PGO, if requested.
1043  if (PGOOpt && Phase != ThinOrFullLTOPhase::ThinLTOPostLink &&
1044  (PGOOpt->Action == PGOOptions::IRInstr ||
1045  PGOOpt->Action == PGOOptions::IRUse)) {
1046  addPGOInstrPasses(MPM, Level,
1047  /* RunProfileGen */ PGOOpt->Action == PGOOptions::IRInstr,
1048  /* IsCS */ false, PGOOpt->ProfileFile,
1049  PGOOpt->ProfileRemappingFile, Phase, PGOOpt->FS);
1050  MPM.addPass(PGOIndirectCallPromotion(false, false));
1051  }
1052  if (PGOOpt && Phase != ThinOrFullLTOPhase::ThinLTOPostLink &&
1053  PGOOpt->CSAction == PGOOptions::CSIRInstr)
1054  MPM.addPass(PGOInstrumentationGenCreateVar(PGOOpt->CSProfileGenFile));
1055 
1056  // Synthesize function entry counts for non-PGO compilation.
1057  if (EnableSyntheticCounts && !PGOOpt)
1058  MPM.addPass(SyntheticCountsPropagation());
1059 
1060  if (EnableModuleInliner)
1061  MPM.addPass(buildModuleInlinerPipeline(Level, Phase));
1062  else
1063  MPM.addPass(buildInlinerPipeline(Level, Phase));
1064 
1065  // Remove any dead arguments exposed by cleanups, constant folding globals,
1066  // and argument promotion.
1067  MPM.addPass(DeadArgumentEliminationPass());
1068 
1069  MPM.addPass(CoroCleanupPass());
1070 
1071  if (EnableMemProfiler && Phase != ThinOrFullLTOPhase::ThinLTOPreLink) {
1072  MPM.addPass(createModuleToFunctionPassAdaptor(MemProfilerPass()));
1073  MPM.addPass(ModuleMemProfilerPass());
1074  }
1075 
1076  return MPM;
1077 }

1086  // The vectorizer may have significantly shortened a loop body; unroll
1087  // again. Unroll small loops to hide loop backedge latency and saturate any
1088  // parallel execution resources of an out-of-order processor. We also then
1089  // need to clean up redundancies and loop invariant code.
1090  // FIXME: It would be really good to use a loop-integrated instruction
1091  // combiner for cleanup here so that the unrolling and LICM can be pipelined
1092  // across the loop nests.
1093  // We do UnrollAndJam in a separate LPM to ensure it happens before unroll
1094  if (EnableUnrollAndJam && PTO.LoopUnrolling)
1095  FPM.addPass(createFunctionToLoopPassAdaptor(
1096  LoopUnrollAndJamPass(Level.getSpeedupLevel())));
1097  FPM.addPass(LoopUnrollPass(LoopUnrollOptions(
1098  Level.getSpeedupLevel(), /*OnlyWhenForced=*/!PTO.LoopUnrolling,
1099  PTO.ForgetAllSCEVInLoopUnroll)));
1100  FPM.addPass(WarnMissedTransformationsPass());
1101  // Now that we are done with loop unrolling, be it either by LoopVectorizer,
1102  // or LoopUnroll passes, some variable-offset GEP's into alloca's could have
1103  // become constant-offset, thus enabling SROA and alloca promotion. Do so.
1104  // NOTE: we are very late in the pipeline, and we don't have any LICM
1105  // or SimplifyCFG passes scheduled after us, that would cleanup
1106  // the CFG mess this may created if allowed to modify CFG, so forbid that.
1107  FPM.addPass(SROAPass(SROAOptions::PreserveCFG));
1108  }
1109 
1110  if (!IsFullLTO) {
1111  // Eliminate loads by forwarding stores from the previous iteration to loads
1112  // of the current iteration.
1113  FPM.addPass(LoopLoadEliminationPass());
1114  }
1115  // Cleanup after the loop optimization passes.
1116  FPM.addPass(InstCombinePass());

1119  ExtraVectorPassManager ExtraPasses;
1120  // At higher optimization levels, try to clean up any runtime overlap and
1121  // alignment checks inserted by the vectorizer. We want to track correlated
1122  // runtime checks for two inner loops in the same outer loop, fold any
1123  // common computations, hoist loop-invariant aspects out of any outer loop,
1124  // and unswitch the runtime checks if possible. Once hoisted, we may have
1125  // dead (or speculatable) control flows or more combining opportunities.
1126  ExtraPasses.addPass(EarlyCSEPass());
1127  ExtraPasses.addPass(CorrelatedValuePropagationPass());
1128  ExtraPasses.addPass(InstCombinePass());
1129  LoopPassManager LPM;
1130  LPM.addPass(LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
1131  /*AllowSpeculation=*/true));
1132  LPM.addPass(SimpleLoopUnswitchPass(/* NonTrivial */ Level ==
1133  OptimizationLevel::O3));
1134  ExtraPasses.addPass(
1135  RequireAnalysisPass<OptimizationRemarkEmitterAnalysis, Function>());
1136  ExtraPasses.addPass(
1137  createFunctionToLoopPassAdaptor(std::move(LPM), /*UseMemorySSA=*/true,
1138  /*UseBlockFrequencyInfo=*/true));
1139  ExtraPasses.addPass(
1140  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
1141  ExtraPasses.addPass(InstCombinePass());
1142  FPM.addPass(std::move(ExtraPasses));
1143  }
1144 
1145  // Now that we've formed fast to execute loop structures, we do further
1146  // optimizations. These are run afterward as they might block doing complex
1147  // analyses and transforms such as what are needed for loop vectorization.
1148 
1149  // Cleanup after loop vectorization, etc. Simplification passes like CVP and
1150  // GVN, loop transforms, and others have already run, so it's now better to
1151  // convert to more optimized IR using more aggressive simplify CFG options.
1152  // The extra sinking transform can create larger basic blocks, so do this
1153  // before SLP vectorization.
1154  FPM.addPass(SimplifyCFGPass(SimplifyCFGOptions()
1155  .forwardSwitchCondToPhi(true)
1156  .convertSwitchRangeToICmp(true)
1157  .convertSwitchToLookupTable(true)
1158  .needCanonicalLoops(false)
1159  .hoistCommonInsts(true)
1160  .sinkCommonInsts(true)));
1161 
1162  if (IsFullLTO) {
1163  FPM.addPass(SCCPPass());
1164  FPM.addPass(InstCombinePass());
1165  FPM.addPass(BDCEPass());
1166  }
1167 
1168  // Optimize parallel scalar instruction chains into SIMD instructions.
1169  if (PTO.SLPVectorization) {
1170  FPM.addPass(SLPVectorizerPass());
1171  if (Level.getSpeedupLevel() > 1 && ExtraVectorizerPasses) {
1172  FPM.addPass(EarlyCSEPass());
1173  }
1174  }
1175  // Enhance/cleanup vector code.
1176  FPM.addPass(VectorCombinePass());
1177 
1178  if (!IsFullLTO) {
1179  FPM.addPass(InstCombinePass());
1180  // Unroll small loops to hide loop backedge latency and saturate any
1181  // parallel execution resources of an out-of-order processor. We also then
1182  // need to clean up redundancies and loop invariant code.
1183  // FIXME: It would be really good to use a loop-integrated instruction
1184  // combiner for cleanup here so that the unrolling and LICM can be pipelined
1185  // across the loop nests.
1186  // We do UnrollAndJam in a separate LPM to ensure it happens before unroll
1187  if (EnableUnrollAndJam && PTO.LoopUnrolling) {
1188  FPM.addPass(createFunctionToLoopPassAdaptor(
1189  LoopUnrollAndJamPass(Level.getSpeedupLevel())));
1190  }
1191  FPM.addPass(LoopUnrollPass(LoopUnrollOptions(
1192  Level.getSpeedupLevel(), /*OnlyWhenForced=*/!PTO.LoopUnrolling,
1193  PTO.ForgetAllSCEVInLoopUnroll)));
1194  FPM.addPass(WarnMissedTransformationsPass());
1195  // Now that we are done with loop unrolling, be it either by LoopVectorizer,
1196  // or LoopUnroll passes, some variable-offset GEP's into alloca's could have
1197  // become constant-offset, thus enabling SROA and alloca promotion. Do so.
1198  // NOTE: we are very late in the pipeline, and we don't have any LICM
1199  // or SimplifyCFG passes scheduled after us, that would cleanup
1200  // the CFG mess this may created if allowed to modify CFG, so forbid that.
1201  FPM.addPass(SROAPass(SROAOptions::PreserveCFG));
1202  FPM.addPass(InstCombinePass());
1203  FPM.addPass(
1204  RequireAnalysisPass<OptimizationRemarkEmitterAnalysis, Function>());
1205  FPM.addPass(createFunctionToLoopPassAdaptor(
1206  LICMPass(PTO.LicmMssaOptCap, PTO.LicmMssaNoAccForPromotionCap,
1207  /*AllowSpeculation=*/true),
1208  /*UseMemorySSA=*/true, /*UseBlockFrequencyInfo=*/true));
1209  }
1210 
1211  // Now that we've vectorized and unrolled loops, we may have more refined
1212  // alignment information, try to re-derive it here.
1213  FPM.addPass(AlignmentFromAssumptionsPass());
1214 
1215  if (IsFullLTO)
1216  FPM.addPass(InstCombinePass());

1221  ThinOrFullLTOPhase LTOPhase) {
1222  const bool LTOPreLink = (LTOPhase == ThinOrFullLTOPhase::ThinLTOPreLink ||
1223  LTOPhase == ThinOrFullLTOPhase::FullLTOPreLink);
1224  ModulePassManager MPM;
1225 
1226  // Optimize globals now that the module is fully simplified.
1227  MPM.addPass(GlobalOptPass());
1228  MPM.addPass(GlobalDCEPass());
1229 
1230  // Run partial inlining pass to partially inline functions that have
1231  // large bodies.
1232  if (RunPartialInlining)
1233  MPM.addPass(PartialInlinerPass());
1234 
1235  // Remove avail extern fns and globals definitions since we aren't compiling
1236  // an object file for later LTO. For LTO we want to preserve these so they
1237  // are eligible for inlining at link-time. Note if they are unreferenced they
1238  // will be removed by GlobalDCE later, so this only impacts referenced
1239  // available externally globals. Eventually they will be suppressed during
1240  // codegen, but eliminating here enables more opportunity for GlobalDCE as it
1241  // may make globals referenced by available external functions dead and saves
1242  // running remaining passes on the eliminated functions. These should be
1243  // preserved during prelinking for link-time inlining decisions.
1244  if (!LTOPreLink)
1245  MPM.addPass(EliminateAvailableExternallyPass());
1246 
1247  if (EnableOrderFileInstrumentation)
1248  MPM.addPass(InstrOrderFilePass());
1249 
1250  // Do RPO function attribute inference across the module to forward-propagate
1251  // attributes where applicable.
1252  // FIXME: Is this really an optimization rather than a canonicalization?
1253  MPM.addPass(ReversePostOrderFunctionAttrsPass());
1254 
1255  // Do a post inline PGO instrumentation and use pass. This is a context
1256  // sensitive PGO pass. We don't want to do this in LTOPreLink phrase as
1257  // cross-module inline has not been done yet. The context sensitive
1258  // instrumentation is after all the inlines are done.
1259  if (!LTOPreLink && PGOOpt) {

1262  /* IsCS */ true, PGOOpt->CSProfileGenFile,
1263  PGOOpt->ProfileRemappingFile, LTOPhase, PGOOpt->FS);
1264  else if (PGOOpt->CSAction == PGOOptions::CSIRUse)
1265  addPGOInstrPasses(MPM, Level, /* RunProfileGen */ false,
1266  /* IsCS */ true, PGOOpt->ProfileFile,
1267  PGOOpt->ProfileRemappingFile, LTOPhase, PGOOpt->FS);
1268  }
1269 
1270  // Re-compute GlobalsAA here prior to function passes. This is particularly
1271  // useful as the above will have inlined, DCE'ed, and function-attr
1272  // propagated everything. We should at this point have a reasonably minimal
1273  // and richly annotated call graph. By computing aliasing and mod/ref
1274  // information for all local globals here, the late loop passes and notably
1275  // the vectorizer will be able to use them to help recognize vectorizable
1276  // memory operations.
1277  MPM.addPass(RecomputeGlobalsAAPass());
1278 
1279  for (auto &C : OptimizerEarlyEPCallbacks)
1280  C(MPM, Level);
1281

1283  OptimizePM.addPass(Float2IntPass());
1284  OptimizePM.addPass(LowerConstantIntrinsicsPass());
1285 
1286  if (EnableMatrix) {
1287  OptimizePM.addPass(LowerMatrixIntrinsicsPass());
1288  OptimizePM.addPass(EarlyCSEPass());
1289  }
1290 
1291  if (EnableCHR && Level == OptimizationLevel::O3 && PGOOpt &&
1292  (PGOOpt->Action == PGOOptions::IRUse ||
1293  PGOOpt->Action == PGOOptions::SampleUse))
1294  OptimizePM.addPass(ControlHeightReductionPass());
1295 
1296  // FIXME: We need to run some loop optimizations to re-rotate loops after
1297  // simplifycfg and others undo their rotation.
1298 
1299  // Optimize the loop execution. These passes operate on entire loop nests
1300  // rather than on each loop in an inside-out manner, and so they are actually
1301  // function passes.
1302 
1303  for (auto &C : VectorizerStartEPCallbacks)
1304  C(OptimizePM, Level);
1305 
1306  LoopPassManager LPM;
1307  // First rotate loops that may have been un-rotated by prior passes.
1308  // Disable header duplication at -Oz.
1309  LPM.addPass(LoopRotatePass(Level != OptimizationLevel::Oz, LTOPreLink));
1310  // Some loops may have become dead by now. Try to delete them.
1311  // FIXME: see discussion in https://reviews.llvm.org/D112851,
1312  // this may need to be revisited once we run GVN before loop deletion
1313  // in the simplification pipeline.
1314  LPM.addPass(LoopDeletionPass());
1315  OptimizePM.addPass(createFunctionToLoopPassAdaptor(
1316  std::move(LPM), /*UseMemorySSA=*/false, /*UseBlockFrequencyInfo=*/false));
1317 
1318  // Distribute loops to allow partial vectorization. I.e. isolate dependences
1319  // into separate loop that would otherwise inhibit vectorization. This is
1320  // currently only performed for loops marked with the metadata
1321  // llvm.loop.distribute=true or when -enable-loop-distribute is specified.
1322  OptimizePM.addPass(LoopDistributePass());
1323 
1324  // Populates the VFABI attribute with the scalar-to-vector mappings
1325  // from the TargetLibraryInfo.
1326  OptimizePM.addPass(InjectTLIMappings());
1327 
1328  addVectorPasses(Level, OptimizePM, /* IsFullLTO */ false);
1329 
1330  // LoopSink pass sinks instructions hoisted by LICM, which serves as a
1331  // canonicalization pass that enables other optimizations. As a result,
1332  // LoopSink pass needs to be a very late IR pass to avoid undoing LICM
1333  // result too early.
1334  OptimizePM.addPass(LoopSinkPass());
1335 
1336  // And finally clean up LCSSA form before generating code.
1337  OptimizePM.addPass(InstSimplifyPass());
1338 
1339  // This hoists/decomposes div/rem ops. It should run after other sink/hoist
1340  // passes to avoid re-sinking, but before SimplifyCFG because it can allow
1341  // flattening of blocks.
1342  OptimizePM.addPass(DivRemPairsPass());
1343 
1344  // Try to annotate calls that were created during optimization.
1345  OptimizePM.addPass(TailCallElimPass());
1346 
1347  // LoopSink (and other loop passes since the last simplifyCFG) might have
1348  // resulted in single-entry-single-exit or empty blocks. Clean up the CFG.
1349  OptimizePM.addPass(
1350  SimplifyCFGPass(SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
1351 
1352  // Add the core optimizing pipeline.
1353  MPM.addPass(createModuleToFunctionPassAdaptor(std::move(OptimizePM),
1354  PTO.EagerlyInvalidateAnalyses));
1355 
1356  for (auto &C : OptimizerLastEPCallbacks)
1357  C(MPM, Level);
1358 
1359  // Split out cold code. Splitting is done late to avoid hiding context from
1360  // other optimizations and inadvertently regressing performance. The tradeoff
1361  // is that this has a higher code size cost than splitting early.
1362  if (EnableHotColdSplit && !LTOPreLink)
1363  MPM.addPass(HotColdSplittingPass());
1364 
1365  // Search the code for similar regions of code. If enough similar regions can
1366  // be found where extracting the regions into their own function will decrease
1367  // the size of the program, we extract the regions, a deduplicate the
1368  // structurally similar regions.
1369  if (EnableIROutliner)
1370  MPM.addPass(IROutlinerPass());
1371 
1372  // Merge functions if requested.
1373  if (PTO.MergeFunctions)
1374  MPM.addPass(MergeFunctionsPass());
1375 
1376  // Now we need to do some global optimization transforms.
1377  // FIXME: It would seem like these should come first in the optimization
1378  // pipeline and maybe be the bottom of the canonicalization pipeline? Weird
1379  // ordering here.
1380  MPM.addPass(GlobalDCEPass());
1381  MPM.addPass(ConstantMergePass());
1382 
1383  if (PTO.CallGraphProfile && !LTOPreLink)
1384  MPM.addPass(CGProfilePass());
1385 
1386  // TODO: Relative look table converter pass caused an issue when full lto is
1387  // enabled. See https://reviews.llvm.org/D94355 for more details.
1388  // Until the issue fixed, disable this pass during pre-linking phase.
1389  if (!LTOPreLink)
1390  MPM.addPass(RelLookupTableConverterPass());
1391 
1392  return MPM;
1393 }

1600  // Cache ProfileSummaryAnalysis once to avoid the potential need to insert

1613  MPM.addPass(ForceFunctionAttrsPass());

1616  // libraries and other oracles.
1617  MPM.addPass(InferFunctionAttrsPass());

1621  CallSiteSplittingPass(), PTO.EagerlyInvalidateAnalyses));

1624  // left by the earlier promotion pass that promotes intra-module targets.
