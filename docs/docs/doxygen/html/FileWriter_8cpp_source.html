70 void FileWriter::alignTo(size_t Align) {
71  off_t Offset = OS.tell();
72  off_t AlignedOffset = (Offset + Align - 1) / Align * Align;
73  if (AlignedOffset == Offset)

75  off_t PadCount = AlignedOffset - Offset;
