40 /// AddressesMap represents information about valid addresses used
41 /// by debug information. Valid addresses are those which points to

52  /// Checks that the specified variable \p DIE references live code section.
53  /// Allowed kind of input die: DW_TAG_variable, DW_TAG_constant.

55  virtual bool isLiveVariable(const DWARFDie &DIE,
56  CompileUnit::DIEInfo &Info) = 0;
57 
58  /// Checks that the specified subprogram \p DIE references live code section.
59  /// Allowed kind of input die: DW_TAG_subprogram, DW_TAG_label.

61  virtual bool isLiveSubprogram(const DWARFDie &DIE,
62  CompileUnit::DIEInfo &Info) = 0;
63

65  /// account that Data is at \p BaseOffset in the .debug_info section.

76  virtual void clear() = 0;

237 /// The core of the Dwarf linking logic.
238 ///
239 /// The generation of the dwarf information from the object files will be
240 /// driven by the selection of 'root DIEs', which are DIEs that
241 /// describe variables or functions that resolves to the corresponding
242 /// code section(and thus have entries in the Addresses map). All the debug
243 /// information that will be generated(the DIEs, but also the line
244 /// tables, ranges, ...) is derived from that set of root DIEs.
245 ///
246 /// The root DIEs are identified because they contain relocations that
247 /// points to code section(the low_pc for a function, the location for
248 /// a variable). These relocations are called ValidRelocs in the
249 /// AddressesInfo and are gathered as a very first step when we start
250 /// processing a object file.
251 class DWARFLinker {
252 public:
253  DWARFLinker(DwarfEmitter *Emitter,
254  DwarfLinkerClient ClientID = DwarfLinkerClient::General)
255  : TheDwarfEmitter(Emitter), DwarfLinkerClientID(ClientID) {}
