331 /// isVREVMask - Check if a vector shuffle corresponds to a VREV
332 /// instruction with the specified blocksize. (The order of the elements
333 /// within each block of the vector is reversed.)
334 inline bool isVREVMask(ArrayRef<int> M, EVT VT, unsigned BlockSize) {
335  assert((BlockSize == 16 || BlockSize == 32 || BlockSize == 64) &&
336  "Only possible block sizes for VREV are: 16, 32, 64");
337 
338  unsigned EltSz = VT.getScalarSizeInBits();
339  if (EltSz != 8 && EltSz != 16 && EltSz != 32)
340  return false;
341 
342  unsigned BlockElts = M[0] + 1;
343  // If the first shuffle index is UNDEF, be optimistic.
344  if (M[0] < 0)
345  BlockElts = BlockSize / EltSz;
346 
347  if (BlockSize <= EltSz || BlockSize != BlockElts * EltSz)
348  return false;
349 
350  for (unsigned i = 0, e = M.size(); i < e; ++i) {
351  if (M[i] < 0)
352  continue; // ignore UNDEF indices
353  if ((unsigned)M[i] != (i - i % BlockElts) + (BlockElts - 1 - i % BlockElts))
354  return false;
355  }
356 
357  return true;
358 }
