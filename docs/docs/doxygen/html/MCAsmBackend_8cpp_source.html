29 std::unique_ptr<MCObjectWriter>
30 MCAsmBackend::createObjectWriter(raw_pwrite_stream &OS) const {
31  auto TW = createObjectTargetWriter();
32  switch (TW->getFormat()) {
33  case Triple::ELF:
34  return createELFObjectWriter(cast<MCELFObjectTargetWriter>(std::move(TW)), OS,
35  Endian == support::little);
36  case Triple::MachO:
37  return createMachObjectWriter(cast<MCMachObjectTargetWriter>(std::move(TW)),
38  OS, Endian == support::little);
39  case Triple::COFF:
40  return createWinCOFFObjectWriter(
41  cast<MCWinCOFFObjectTargetWriter>(std::move(TW)), OS);
42  case Triple::SPIRV:
43  return createSPIRVObjectWriter(
44  cast<MCSPIRVObjectTargetWriter>(std::move(TW)), OS);
45  case Triple::Wasm:
46  return createWasmObjectWriter(cast<MCWasmObjectTargetWriter>(std::move(TW)),
47  OS);
48  case Triple::XCOFF:
49  return createXCOFFObjectWriter(
50  cast<MCXCOFFObjectTargetWriter>(std::move(TW)), OS);
51  case Triple::DXContainer:
52  return createDXContainerObjectWriter(
53  cast<MCDXContainerTargetWriter>(std::move(TW)), OS);
54  default:
55  llvm_unreachable("unexpected object format");
56  }
57 }

59 std::unique_ptr<MCObjectWriter>
60 MCAsmBackend::createDwoObjectWriter(raw_pwrite_stream &OS,
61  raw_pwrite_stream &DwoOS) const {
62  auto TW = createObjectTargetWriter();
63  switch (TW->getFormat()) {
64  case Triple::ELF:
65  return createELFDwoObjectWriter(
66  cast<MCELFObjectTargetWriter>(std::move(TW)), OS, DwoOS,
67  Endian == support::little);
68  case Triple::Wasm:
69  return createWasmDwoObjectWriter(
70  cast<MCWasmObjectTargetWriter>(std::move(TW)), OS, DwoOS);
71  default:
72  report_fatal_error("dwo only supported with ELF and Wasm");
73  }
