52 CtorDtorIterator::Element CtorDtorIterator::operator*() const {
53  ConstantStruct *CS = dyn_cast<ConstantStruct>(InitList->getOperand(I));
54  assert(CS && "Unrecognized type in llvm.global_ctors/llvm.global_dtors");
55 
56  Constant *FuncC = CS->getOperand(1);
57  Function *Func = nullptr;
58 
59  // Extract function pointer, pulling off any casts.
60  while (FuncC) {
61  if (Function *F = dyn_cast_or_null<Function>(FuncC)) {
62  Func = F;
63  break;
64  } else if (ConstantExpr *CE = dyn_cast_or_null<ConstantExpr>(FuncC)) {
65  if (CE->isCast())
66  FuncC = CE->getOperand(0);
67  else
68  break;
69  } else {
70  // This isn't anything we recognize. Bail out with Func left set to null.
71  break;
72  }
73  }
74 
75  auto *Priority = cast<ConstantInt>(CS->getOperand(0));
76  Value *Data = CS->getNumOperands() == 3 ? CS->getOperand(2) : nullptr;
77  if (Data && !isa<GlobalValue>(Data))
78  Data = nullptr;
79  return Element(Priority->getZExtValue(), Func, Data);
80 }

115  if (CtorDtors.empty())
116  return;

118  MangleAndInterner Mangle(

142 Error CtorDtorRunner::run() {
143  using CtorDtorTy = void (*)();
144 
145  SymbolLookupSet LookupSet;
146  for (auto &KV : CtorDtorsByPriority)
147  for (auto &Name : KV.second)
148  LookupSet.add(Name);
149  assert(!LookupSet.containsDuplicates() &&
150  "Ctor/Dtor list contains duplicates");
151 
152  auto &ES = JD.getExecutionSession();
153  if (auto CtorDtorMap = ES.lookup(
154  makeJITDylibSearchOrder(&JD, JITDylibLookupFlags::MatchAllSymbols),
155  std::move(LookupSet))) {
156  for (auto &KV : CtorDtorsByPriority) {
157  for (auto &Name : KV.second) {
158  assert(CtorDtorMap->count(Name) && "No entry for Name");
159  auto CtorDtor = reinterpret_cast<CtorDtorTy>(
160  static_cast<uintptr_t>((*CtorDtorMap)[Name].getAddress()));
161  CtorDtor();
162  }
163  }
164  CtorDtorsByPriority.clear();
165  return Error::success();
166  } else
167  return CtorDtorMap.takeError();
168 }

239 Error DynamicLibrarySearchGenerator::tryToGenerate(
240  LookupState &LS, LookupKind K, JITDylib &JD,
241  JITDylibLookupFlags JDLookupFlags, const SymbolLookupSet &Symbols) {
242  orc::SymbolMap NewSymbols;
243 
244  bool HasGlobalPrefix = (GlobalPrefix != '\0');
245 
246  for (auto &KV : Symbols) {
247  auto &Name = KV.first;
248 
249  if ((*Name).empty())
250  continue;
251 
252  if (Allow && !Allow(Name))
253  continue;
254 
255  if (HasGlobalPrefix && (*Name).front() != GlobalPrefix)
256  continue;
257 
258  std::string Tmp((*Name).data() + HasGlobalPrefix,
259  (*Name).size() - HasGlobalPrefix);
260  if (void *Addr = Dylib.getAddressOfSymbol(Tmp.c_str())) {
261  NewSymbols[Name] = JITEvaluatedSymbol(
262  static_cast<JITTargetAddress>(reinterpret_cast<uintptr_t>(Addr)),
263  JITSymbolFlags::Exported);
264  }
265  }
266 
267  if (NewSymbols.empty())
268  return Error::success();
269 
270  return JD.define(absoluteSymbols(std::move(NewSymbols)));

289

395 Error StaticLibraryDefinitionGenerator::tryToGenerate(
396  LookupState &LS, LookupKind K, JITDylib &JD,
397  JITDylibLookupFlags JDLookupFlags, const SymbolLookupSet &Symbols) {
398  // Don't materialize symbols from static archives unless this is a static
399  // lookup.
400  if (K != LookupKind::Static)
401  return Error::success();
402 
403  // Bail out early if we've already freed the archive.
404  if (!Archive)
405  return Error::success();
406 
407  DenseSet<std::pair<StringRef, StringRef>> ChildBufferInfos;
408 
409  for (const auto &KV : Symbols) {
410  const auto &Name = KV.first;
411  if (!ObjectFilesMap.count(Name))
412  continue;
413  auto ChildBuffer = ObjectFilesMap[Name];
414  ChildBufferInfos.insert(
415  {ChildBuffer.getBuffer(), ChildBuffer.getBufferIdentifier()});
416  }
417 
418  for (auto ChildBufferInfo : ChildBufferInfos) {
419  MemoryBufferRef ChildBufferRef(ChildBufferInfo.first,
420  ChildBufferInfo.second);
421 
422  auto I = GetObjFileInterface(L.getExecutionSession(), ChildBufferRef);
423  if (!I)
424  return I.takeError();
425 
426  if (auto Err = L.add(JD, MemoryBuffer::getMemBuffer(ChildBufferRef, false),
427  std::move(*I)))
428  return Err;
429  }
430 
431  return Error::success();
432 }

434 Error StaticLibraryDefinitionGenerator::buildObjectFilesMap() {
435  DenseMap<uint64_t, MemoryBufferRef> MemoryBuffers;
436  DenseSet<uint64_t> Visited;
437  DenseSet<uint64_t> Excluded;
438  for (auto &S : Archive->symbols()) {
439  StringRef SymName = S.getName();
440  auto Member = S.getMember();
441  if (!Member)
442  return Member.takeError();
443  auto DataOffset = Member->getDataOffset();
444  if (!Visited.count(DataOffset)) {
445  Visited.insert(DataOffset);
446  auto Child = Member->getAsBinary();
447  if (!Child)
448  return Child.takeError();
449  if ((*Child)->isCOFFImportFile()) {
450  ImportedDynamicLibraries.insert((*Child)->getFileName().str());
451  Excluded.insert(DataOffset);
452  continue;
453  }
454  MemoryBuffers[DataOffset] = (*Child)->getMemoryBufferRef();
455  }
456  if (!Excluded.count(DataOffset))
457  ObjectFilesMap[L.getExecutionSession().intern(SymName)] =
458  MemoryBuffers[DataOffset];
459  }
460 
461  return Error::success();
462 }

464 Expected<std::pair<size_t, size_t>>
465 StaticLibraryDefinitionGenerator::getSliceRangeForArch(
466  object::MachOUniversalBinary &UB, const Triple &TT) {
467 
468  for (const auto &Obj : UB.objects()) {
469  auto ObjTT = Obj.getTriple();
470  if (ObjTT.getArch() == TT.getArch() &&
471  ObjTT.getSubArch() == TT.getSubArch() &&
472  (TT.getVendor() == Triple::UnknownVendor ||
473  ObjTT.getVendor() == TT.getVendor())) {
474  // We found a match. Return the range for the slice.
475  return std::make_pair(Obj.getOffset(), Obj.getSize());
476  }
477  }
478 
479  return make_error<StringError>(Twine("Universal binary ") + UB.getFileName() +
480  " does not contain a slice for " +
481  TT.str(),
482  inconvertibleErrorCode());
483 }

505 Error DLLImportDefinitionGenerator::tryToGenerate(
506  LookupState &LS, LookupKind K, JITDylib &JD,
507  JITDylibLookupFlags JDLookupFlags, const SymbolLookupSet &Symbols) {
508  JITDylibSearchOrder LinkOrder;
509  JD.withLinkOrderDo([&](const JITDylibSearchOrder &LO) {
510  LinkOrder.reserve(LO.size());
511  for (auto &KV : LO) {
512  if (KV.first == &JD)
513  continue;
514  LinkOrder.push_back(KV);
515  }
516  });
517 
518  // FIXME: if regular symbol name start with __imp_ we have to issue lookup of
519  // both __imp_ and stripped name and use the lookup information to resolve the
520  // real symbol name.
521  SymbolLookupSet LookupSet;
522  DenseMap<StringRef, SymbolLookupFlags> ToLookUpSymbols;
523  for (auto &KV : Symbols) {
524  StringRef Deinterned = *KV.first;
525  if (Deinterned.startswith(getImpPrefix()))
526  Deinterned = Deinterned.drop_front(StringRef(getImpPrefix()).size());
527  // Don't degrade the required state
528  if (ToLookUpSymbols.count(Deinterned) &&
529  ToLookUpSymbols[Deinterned] == SymbolLookupFlags::RequiredSymbol)
530  continue;
531  ToLookUpSymbols[Deinterned] = KV.second;
532  }
533 
534  for (auto &KV : ToLookUpSymbols)
535  LookupSet.add(ES.intern(KV.first), KV.second);
536 
537  auto Resolved =
538  ES.lookup(LinkOrder, LookupSet, LookupKind::DLSym, SymbolState::Resolved);
539  if (!Resolved)
540  return Resolved.takeError();
541 
542  auto G = createStubsGraph(*Resolved);
543  if (!G)
544  return G.takeError();
545  return L.add(JD, std::move(*G));
546 }

572 Expected<std::unique_ptr<jitlink::LinkGraph>>
573 DLLImportDefinitionGenerator::createStubsGraph(const SymbolMap &Resolved) {
574  Triple TT = ES.getExecutorProcessControl().getTargetTriple();
575  auto PointerSize = getTargetEndianness(TT);
576  if (!PointerSize)
577  return PointerSize.takeError();
578  auto Endianness = getTargetEndianness(TT);
579  if (!Endianness)
580  return Endianness.takeError();
581 
582  auto G = std::make_unique<jitlink::LinkGraph>(
583  "<DLLIMPORT_STUBS>", TT, *PointerSize, *Endianness,
584  jitlink::getGenericEdgeKindName);
585  jitlink::Section &Sec =
586  G->createSection(getSectionName(), MemProt::Read | MemProt::Exec);
587 
588  for (auto &KV : Resolved) {
589  jitlink::Symbol &Target = G->addAbsoluteSymbol(
590  *KV.first, ExecutorAddr(KV.second.getAddress()), *PointerSize,
591  jitlink::Linkage::Strong, jitlink::Scope::Local, false);
592 
593  // Create __imp_ symbol
594  jitlink::Symbol &Ptr =
595  jitlink::x86_64::createAnonymousPointer(*G, Sec, &Target);
596  auto NameCopy = G->allocateContent(Twine(getImpPrefix()) + *KV.first);
597  StringRef NameCopyRef = StringRef(NameCopy.data(), NameCopy.size());
598  Ptr.setName(NameCopyRef);
599  Ptr.setLinkage(jitlink::Linkage::Strong);
600  Ptr.setScope(jitlink::Scope::Default);
601 
602  // Create PLT stub
603  // FIXME: check PLT stub of data symbol is not accessed
604  jitlink::Block &StubBlock =
605  jitlink::x86_64::createPointerJumpStubBlock(*G, Sec, Ptr);
606  G->addDefinedSymbol(StubBlock, 0, *KV.first, StubBlock.getSize(),
607  jitlink::Linkage::Strong, jitlink::Scope::Default, true,
608  false);
609  }
610 
611  return std::move(G);
612 }
