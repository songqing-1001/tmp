24 void LVSymbolTable::add(StringRef Name, LVScope *Function,
25  LVSectionIndex SectionIndex) {
26  std::string SymbolName(Name);
27  if (SymbolNames.find(SymbolName) == SymbolNames.end()) {
28  SymbolNames.emplace(
29  std::piecewise_construct, std::forward_as_tuple(SymbolName),
30  std::forward_as_tuple(Function, 0, SectionIndex, false));
31  } else {
32  // Update a recorded entry with its logical scope and section index.
33  SymbolNames[SymbolName].Scope = Function;
34  if (SectionIndex)
35  SymbolNames[SymbolName].SectionIndex = SectionIndex;
36  }
37 
38  if (Function && SymbolNames[SymbolName].IsComdat)
39  Function->setIsComdat();
40 
41  LLVM_DEBUG({ print(dbgs()); });
42 }

61 LVSectionIndex LVSymbolTable::update(LVScope *Function) {
62  LVSectionIndex SectionIndex = getReader().getDotTextSectionIndex();
63  StringRef Name = Function->getLinkageName();
64  if (Name.empty())
65  Name = Function->getName();
66  std::string SymbolName(Name);
67 
68  if (SymbolName.empty() || (SymbolNames.find(SymbolName) == SymbolNames.end()))
69  return SectionIndex;
70 
71  // Update a recorded entry with its logical scope, only if the scope has
72  // ranges. That is the case when in DWARF there are 2 DIEs connected via
73  // the DW_AT_specification.
74  if (Function->getHasRanges()) {
75  SymbolNames[SymbolName].Scope = Function;
76  SectionIndex = SymbolNames[SymbolName].SectionIndex;
77  } else {
78  SectionIndex = UndefinedSectionIndex;
79  }
80 
81  if (SymbolNames[SymbolName].IsComdat)
82  Function->setIsComdat();
83 
84  LLVM_DEBUG({ print(dbgs()); });
85  return SectionIndex;
86 }

147 void LVBinaryReader::mapVirtualAddress(const object::ObjectFile &Obj) {
148  for (const object::SectionRef &Section : Obj.sections()) {
149  if (!Section.isText() || Section.isVirtual() || !Section.getSize())
150  continue;
151 
152  // Record section information required for symbol resolution.
153  // Note: The section index returned by 'getIndex()' is one based.
154  Sections.emplace(Section.getIndex(), Section);
155  addSectionAddress(Section);
156 
157  // Identify the ".text" section.
158  Expected<StringRef> SectionNameOrErr = Section.getName();
159  if (!SectionNameOrErr) {
160  consumeError(SectionNameOrErr.takeError());
161  continue;
162  }
163  if ((*SectionNameOrErr).equals(".text") ||
164  (*SectionNameOrErr).equals(".code"))
165  DotTextSectionIndex = Section.getIndex();
166  }
167 
168  // Process the symbol table.
169  mapRangeAddress(Obj);
170 
171  LLVM_DEBUG({
172  dbgs() << "\nSections Information:\n";
173  for (LVSections::reference Entry : Sections) {
174  LVSectionIndex SectionIndex = Entry.first;
175  const object::SectionRef Section = Entry.second;
176  Expected<StringRef> SectionNameOrErr = Section.getName();
177  if (!SectionNameOrErr)
178  consumeError(SectionNameOrErr.takeError());
179  dbgs() << "\nIndex: " << format_decimal(SectionIndex, 3)
180  << " Name: " << *SectionNameOrErr << "\n"
181  << "Size: " << hexValue(Section.getSize()) << "\n"
182  << "VirtualAddress: " << hexValue(VirtualAddress) << "\n"
183  << "SectionAddress: " << hexValue(Section.getAddress()) << "\n";
184  }
185  dbgs() << "\nObject Section Information:\n";
186  for (LVSectionAddresses::const_reference Entry : SectionAddresses)
187  dbgs() << "[" << hexValue(Entry.first) << ":"
188  << hexValue(Entry.first + Entry.second.getSize())
189  << "] Size: " << hexValue(Entry.second.getSize()) << "\n";
190  });
191 }

193 Error LVBinaryReader::loadGenericTargetInfo(StringRef TheTriple,
194  StringRef TheFeatures) {
195  std::string TargetLookupError;
196  const Target *TheTarget =
197  TargetRegistry::lookupTarget(std::string(TheTriple), TargetLookupError);
198  if (!TheTarget)
199  return createStringError(errc::invalid_argument, TargetLookupError.c_str());
200 
201  // Register information.
202  MCRegisterInfo *RegisterInfo = TheTarget->createMCRegInfo(TheTriple);
203  if (!RegisterInfo)
204  return createStringError(errc::invalid_argument,
205  "no register info for target " + TheTriple);
206  MRI.reset(RegisterInfo);
207 
208  // Assembler properties and features.
209  MCTargetOptions MCOptions;
210  MCAsmInfo *AsmInfo(TheTarget->createMCAsmInfo(*MRI, TheTriple, MCOptions));
211  if (!AsmInfo)
212  return createStringError(errc::invalid_argument,
213  "no assembly info for target " + TheTriple);
214  MAI.reset(AsmInfo);
215 
216  // Target subtargets.
217  StringRef CPU;
218  MCSubtargetInfo *SubtargetInfo(
219  TheTarget->createMCSubtargetInfo(TheTriple, CPU, TheFeatures));
220  if (!SubtargetInfo)
221  return createStringError(errc::invalid_argument,
222  "no subtarget info for target " + TheTriple);
223  STI.reset(SubtargetInfo);
224 
225  // Instructions Info.
226  MCInstrInfo *InstructionInfo(TheTarget->createMCInstrInfo());
227  if (!InstructionInfo)
228  return createStringError(errc::invalid_argument,
229  "no instruction info for target " + TheTriple);
230  MII.reset(InstructionInfo);
231 
232  MC = std::make_unique<MCContext>(Triple(TheTriple), MAI.get(), MRI.get(),
233  STI.get());
234 
235  // Assembler.
236  MCDisassembler *DisAsm(TheTarget->createMCDisassembler(*STI, *MC));
237  if (!DisAsm)
238  return createStringError(errc::invalid_argument,
239  "no disassembler for target " + TheTriple);
240  MD.reset(DisAsm);
241 
242  MCInstPrinter *InstructionPrinter(TheTarget->createMCInstPrinter(
243  Triple(TheTriple), AsmInfo->getAssemblerDialect(), *MAI, *MII, *MRI));
244  if (!InstructionPrinter)
245  return createStringError(errc::invalid_argument,
246  "no target assembly language printer for target " +
247  TheTriple);
248  MIP.reset(InstructionPrinter);
249  InstructionPrinter->setPrintImmHex(true);
250 
251  return Error::success();
252 }

254 Expected<std::pair<uint64_t, object::SectionRef>>
255 LVBinaryReader::getSection(LVScope *Scope, LVAddress Address,
256  LVSectionIndex SectionIndex) {
257  // Return the 'text' section with the code for this logical scope.
258  // COFF: SectionIndex is zero. Use 'SectionAddresses' data.
259  // ELF: SectionIndex is the section index in the file.
260  if (SectionIndex) {
261  LVSections::iterator Iter = Sections.find(SectionIndex);
262  if (Iter == Sections.end()) {
263  return createStringError(errc::invalid_argument,
264  "invalid section index for: '%s'",
265  Scope->getName().str().c_str());
266  }
267  const object::SectionRef Section = Iter->second;
268  return std::make_pair(Section.getAddress(), Section);
269  }
270 
271  // Ensure a valid starting address for the public names.
272  LVSectionAddresses::const_iterator Iter =
273  SectionAddresses.upper_bound(Address);
274  if (Iter == SectionAddresses.begin())
275  return createStringError(errc::invalid_argument,
276  "invalid section address for: '%s'",
277  Scope->getName().str().c_str());
278 
279  // Get section that contains the code for this function.
280  Iter = SectionAddresses.lower_bound(Address);
281  if (Iter != SectionAddresses.begin())
282  --Iter;
283  return std::make_pair(Iter->first, Iter->second);
284 }

310 Error LVBinaryReader::createInstructions(LVScope *Scope,
311  LVSectionIndex SectionIndex,
312  const LVNameInfo &NameInfo) {
313  assert(Scope && "Scope is null.");
314 
315  // Skip stripped functions.
316  if (Scope->getIsDiscarded())
317  return Error::success();
318 
319  // Find associated address and size for the given function entry point.
320  LVAddress Address = NameInfo.first;
321  uint64_t Size = NameInfo.second;
322 
323  LLVM_DEBUG({
324  dbgs() << "\nPublic Name instructions: '" << Scope->getName() << "' / '"
325  << Scope->getLinkageName() << "'\n"
326  << "DIE Offset: " << hexValue(Scope->getOffset()) << " Range: ["
327  << hexValue(Address) << ":" << hexValue(Address + Size) << "]\n";
328  });
329 
330  Expected<std::pair<uint64_t, const object::SectionRef>> SectionOrErr =
331  getSection(Scope, Address, SectionIndex);
332  if (!SectionOrErr)
333  return SectionOrErr.takeError();
334  const object::SectionRef Section = (*SectionOrErr).second;
335  uint64_t SectionAddress = (*SectionOrErr).first;
336 
337  Expected<StringRef> SectionContentsOrErr = Section.getContents();
338  if (!SectionContentsOrErr)
339  return SectionOrErr.takeError();
340 
341  // There are cases where the section size is smaller than the [LowPC,HighPC]
342  // range; it causes us to decode invalid addresses. The recorded size in the
343  // logical scope is one less than the real size.
344  LLVM_DEBUG({
345  dbgs() << " Size: " << hexValue(Size)
346  << ", Section Size: " << hexValue(Section.getSize()) << "\n";
347  });
348  Size = std::min(Size + 1, Section.getSize());
349 
350  ArrayRef<uint8_t> Bytes = arrayRefFromStringRef(*SectionContentsOrErr);
351  uint64_t Offset = Address - SectionAddress;
352  uint8_t const *Begin = Bytes.data() + Offset;
353  uint8_t const *End = Bytes.data() + Offset + Size;
354 
355  LLVM_DEBUG({
356  Expected<StringRef> SectionNameOrErr = Section.getName();
357  if (!SectionNameOrErr)
358  consumeError(SectionNameOrErr.takeError());
359  else
360  dbgs() << "Section Index: " << hexValue(Section.getIndex()) << " ["
361  << hexValue((uint64_t)Section.getAddress()) << ":"
362  << hexValue((uint64_t)Section.getAddress() + Section.getSize(), 10)
363  << "] Name: '" << *SectionNameOrErr << "'\n"
364  << "Begin: " << hexValue((uint64_t)Begin)
365  << ", End: " << hexValue((uint64_t)End) << "\n";
366  });
367 
368  // Address for first instruction line.
369  LVAddress FirstAddress = Address;
370  auto InstructionsSP = std::make_unique<LVLines>();
371  LVLines &Instructions = *InstructionsSP;
372  DiscoveredLines.emplace_back(std::move(InstructionsSP));
373 
374  while (Begin < End) {
375  MCInst Instruction;
376  uint64_t BytesConsumed = 0;
377  SmallVector<char, 64> InsnStr;
378  raw_svector_ostream Annotations(InsnStr);
379  MCDisassembler::DecodeStatus const S =
380  MD->getInstruction(Instruction, BytesConsumed,
381  ArrayRef<uint8_t>(Begin, End), Address, outs());
382  switch (S) {
383  case MCDisassembler::Fail:
384  LLVM_DEBUG({ dbgs() << "Invalid instruction\n"; });
385  if (BytesConsumed == 0)
386  // Skip invalid bytes
387  BytesConsumed = 1;
388  break;
389  case MCDisassembler::SoftFail:
390  LLVM_DEBUG({ dbgs() << "Potentially undefined instruction:"; });
391  LLVM_FALLTHROUGH;
392  case MCDisassembler::Success: {
393  std::string Buffer;
394  raw_string_ostream Stream(Buffer);
395  StringRef AnnotationsStr = Annotations.str();
396  MIP->printInst(&Instruction, Address, AnnotationsStr, *STI, Stream);
397  LLVM_DEBUG({
398  std::string BufferCodes;
399  raw_string_ostream StreamCodes(BufferCodes);
400  StreamCodes << format_bytes(
401  ArrayRef<uint8_t>(Begin, Begin + BytesConsumed), std::nullopt, 16,
402  16);
403  dbgs() << "[" << hexValue((uint64_t)Begin) << "] "
404  << "Size: " << format_decimal(BytesConsumed, 2) << " ("
405  << formatv("{0}",
406  fmt_align(StreamCodes.str(), AlignStyle::Left, 32))
407  << ") " << hexValue((uint64_t)Address) << ": " << Stream.str()
408  << "\n";
409  });
410  // Here we add logical lines to the Instructions. Later on,
411  // the 'processLines()' function will move each created logical line
412  // to its enclosing logical scope, using the debug ranges information
413  // and they will be released when its scope parent is deleted.
414  LVLineAssembler *Line = createLineAssembler();
415  Line->setAddress(Address);
416  Line->setName(StringRef(Stream.str()).trim());
417  Instructions.push_back(Line);
418  break;
419  }
420  }
421  Address += BytesConsumed;
422  Begin += BytesConsumed;
423  }
424 
425  LLVM_DEBUG({
426  size_t Index = 0;
427  dbgs() << "\nSectionIndex: " << format_decimal(SectionIndex, 3)
428  << " Scope DIE: " << hexValue(Scope->getOffset()) << "\n"
429  << "Address: " << hexValue(FirstAddress)
430  << format(" - Collected instructions lines: %d\n",
431  Instructions.size());
432  for (const LVLine *Line : Instructions)
433  dbgs() << format_decimal(++Index, 5) << ": "
434  << hexValue(Line->getOffset()) << ", (" << Line->getName()
435  << ")\n";
436  });
437 
438  // The scope in the assembler names is linked to its own instructions.
439  ScopeInstructions.add(SectionIndex, Scope, &Instructions);
440  AssemblerMappings.add(SectionIndex, FirstAddress, Scope);
441 
442  return Error::success();
443 }

457 Error LVBinaryReader::createInstructions() {
458  if (!options().getPrintInstructions())
459  return Error::success();
460 
461  LLVM_DEBUG({
462  size_t Index = 1;
463  dbgs() << "\nPublic Names (Scope):\n";
464  for (LVPublicNames::const_reference Name : CompileUnit->getPublicNames()) {
465  LVScope *Scope = Name.first;
466  const LVNameInfo &NameInfo = Name.second;
467  LVAddress Address = NameInfo.first;
468  uint64_t Size = NameInfo.second;
469  dbgs() << format_decimal(Index++, 5) << ": "
470  << "DIE Offset: " << hexValue(Scope->getOffset()) << " Range: ["
471  << hexValue(Address) << ":" << hexValue(Address + Size) << "] "
472  << "Name: '" << Scope->getName() << "' / '"
473  << Scope->getLinkageName() << "'\n";
474  }
475  });
476 
477  // For each public name in the current compile unit, create the line
478  // records that represent the executable instructions.
479  for (LVPublicNames::const_reference Name : CompileUnit->getPublicNames()) {
480  LVScope *Scope = Name.first;
481  // The symbol table extracted from the object file always contains a
482  // non-empty name (linkage name). However, the logical scope does not
483  // guarantee to have a name for the linkage name (main is one case).
484  // For those cases, set the linkage name the same as the name.
485  if (!Scope->getLinkageNameIndex())
486  Scope->setLinkageName(Scope->getName());
487  LVSectionIndex SectionIndex = getSymbolTableIndex(Scope->getLinkageName());
488  if (Error Err = createInstructions(Scope, SectionIndex, Name.second))
489  return Err;
490  }
491 
492  return Error::success();
493 }

495 // During the traversal of the debug information sections, we created the
496 // logical lines representing the disassembled instructions from the text
497 // section and the logical lines representing the line records from the
498 // debug line section. Using the ranges associated with the logical scopes,
499 // we will allocate those logical lines to their logical scopes.
500 void LVBinaryReader::processLines(LVLines *DebugLines,
501  LVSectionIndex SectionIndex,
502  LVScope *Function) {
503  assert(DebugLines && "DebugLines is null.");
504 
505  // Just return if this compilation unit does not have any line records
506  // and no instruction lines were created.
507  if (DebugLines->empty() && !options().getPrintInstructions())
508  return;
509 
510  // Merge the debug lines and instruction lines using their text address;
511  // the logical line representing the debug line record is followed by the
512  // line(s) representing the disassembled instructions, whose addresses are
513  // equal or greater that the line address and less than the address of the
514  // next debug line record.
515  LLVM_DEBUG({
516  size_t Index = 1;
517  size_t PerLine = 4;
518  dbgs() << format("\nProcess debug lines: %d\n", DebugLines->size());
519  for (const LVLine *Line : *DebugLines) {
520  dbgs() << format_decimal(Index, 5) << ": " << hexValue(Line->getOffset())
521  << ", (" << Line->getLineNumber() << ")"
522  << ((Index % PerLine) ? " " : "\n");
523  ++Index;
524  }
525  dbgs() << ((Index % PerLine) ? "\n" : "");
526  });
527 
528  bool TraverseLines = true;
529  LVLines::iterator Iter = DebugLines->begin();
530  while (TraverseLines && Iter != DebugLines->end()) {
531  uint64_t DebugAddress = (*Iter)->getAddress();
532 
533  // Get the function with an entry point that matches this line and
534  // its associated assembler entries. In the case of COMDAT, the input
535  // 'Function' is not null. Use it to find its address ranges.
536  LVScope *Scope = Function;
537  if (!Function) {
538  Scope = AssemblerMappings.find(SectionIndex, DebugAddress);
539  if (!Scope) {
540  ++Iter;
541  continue;
542  }
543  }
544 
545  // Get the associated instructions for the found 'Scope'.
546  LVLines InstructionLines;
547  LVLines *Lines = ScopeInstructions.find(SectionIndex, Scope);
548  if (Lines)
549  InstructionLines = std::move(*Lines);
550 
551  LLVM_DEBUG({
552  size_t Index = 0;
553  dbgs() << "\nSectionIndex: " << format_decimal(SectionIndex, 3)
554  << " Scope DIE: " << hexValue(Scope->getOffset()) << "\n"
555  << format("Process instruction lines: %d\n",
556  InstructionLines.size());
557  for (const LVLine *Line : InstructionLines)
558  dbgs() << format_decimal(++Index, 5) << ": "
559  << hexValue(Line->getOffset()) << ", (" << Line->getName()
560  << ")\n";
561  });
562 
563  // Continue with next debug line if there are not instructions lines.
564  if (InstructionLines.empty()) {
565  ++Iter;
566  continue;
567  }
568 
569  for (LVLine *InstructionLine : InstructionLines) {
570  uint64_t InstructionAddress = InstructionLine->getAddress();
571  LLVM_DEBUG({
572  dbgs() << "Instruction address: " << hexValue(InstructionAddress)
573  << "\n";
574  });
575  if (TraverseLines) {
576  while (Iter != DebugLines->end()) {
577  DebugAddress = (*Iter)->getAddress();
578  LLVM_DEBUG({
579  bool IsDebug = (*Iter)->getIsLineDebug();
580  dbgs() << "Line " << (IsDebug ? "dbg:" : "ins:") << " ["
581  << hexValue(DebugAddress) << "]";
582  if (IsDebug)
583  dbgs() << format(" %d", (*Iter)->getLineNumber());
584  dbgs() << "\n";
585  });
586  // Instruction address before debug line.
587  if (InstructionAddress < DebugAddress) {
588  LLVM_DEBUG({
589  dbgs() << "Inserted instruction address: "
590  << hexValue(InstructionAddress) << " before line: "
591  << format("%d", (*Iter)->getLineNumber()) << " ["
592  << hexValue(DebugAddress) << "]\n";
593  });
594  Iter = DebugLines->insert(Iter, InstructionLine);
595  // The returned iterator points to the inserted instruction.
596  // Skip it and point to the line acting as reference.
597  ++Iter;
598  break;
599  }
600  ++Iter;
601  }
602  if (Iter == DebugLines->end()) {
603  // We have reached the end of the source lines and the current
604  // instruction line address is greater than the last source line.
605  TraverseLines = false;
606  DebugLines->push_back(InstructionLine);
607  }
608  } else {
609  DebugLines->push_back(InstructionLine);
610  }
611  }
612  }
613 
614  LLVM_DEBUG({
615  dbgs() << format("Lines after merge: %d\n", DebugLines->size());
616  size_t Index = 0;
617  for (const LVLine *Line : *DebugLines) {
618  dbgs() << format_decimal(++Index, 5) << ": "
619  << hexValue(Line->getOffset()) << ", ("
620  << ((Line->getIsLineDebug())
621  ? Line->lineNumberAsStringStripped(/*ShowZero=*/true)
622  : Line->getName())
623  << ")\n";
624  }
625  });
626 
627  // If this compilation unit does not have line records, traverse its scopes
628  // and take any collected instruction lines as the working set in order
629  // to move them to their associated scope.
630  if (DebugLines->empty()) {
631  if (const LVScopes *Scopes = CompileUnit->getScopes())
632  for (LVScope *Scope : *Scopes) {
633  LVLines *Lines = ScopeInstructions.find(Scope);
634  if (Lines) {
635 
636  LLVM_DEBUG({
637  size_t Index = 0;
638  dbgs() << "\nSectionIndex: " << format_decimal(SectionIndex, 3)
639  << " Scope DIE: " << hexValue(Scope->getOffset()) << "\n"
640  << format("Instruction lines: %d\n", Lines->size());
641  for (const LVLine *Line : *Lines)
642  dbgs() << format_decimal(++Index, 5) << ": "
643  << hexValue(Line->getOffset()) << ", (" << Line->getName()
644  << ")\n";
645  });
646 
647  if (Scope->getIsArtificial()) {
648  // Add the instruction lines to their artificial scope.
649  for (LVLine *Line : *Lines)
650  Scope->addElement(Line);
651  } else {
652  DebugLines->append(*Lines);
653  }
654  Lines->clear();
655  }
656  }
657  }
658 
659  LVRange *ScopesWithRanges = getSectionRanges(SectionIndex);
660  ScopesWithRanges->startSearch();
661 
662  // Process collected lines.
663  LVScope *Scope;
664  for (LVLine *Line : *DebugLines) {
665  // Using the current line address, get its associated lexical scope and
666  // add the line information to it.
667  Scope = ScopesWithRanges->getEntry(Line->getAddress());
668  if (!Scope) {
669  // If missing scope, use the compile unit.
670  Scope = CompileUnit;
671  LLVM_DEBUG({
672  dbgs() << "Adding line to CU: " << hexValue(Line->getOffset()) << ", ("
673  << ((Line->getIsLineDebug())
674  ? Line->lineNumberAsStringStripped(/*ShowZero=*/true)
675  : Line->getName())
676  << ")\n";
677  });
678  }
679 
680  // Add line object to scope.
681  Scope->addElement(Line);
682 
683  // Report any line zero.
684  if (options().getWarningLines() && Line->getIsLineDebug() &&
685  !Line->getLineNumber())
686  CompileUnit->addLineZero(Line);
687 
688  // Some compilers generate ranges in the compile unit; other compilers
689  // only DW_AT_low_pc/DW_AT_high_pc. In order to correctly map global
690  // variables, we need to generate the map ranges for the compile unit.
691  // If we use the ranges stored at the scope level, there are cases where
692  // the address referenced by a symbol location, is not in the enclosing
693  // scope, but in an outer one. By using the ranges stored in the compile
694  // unit, we can catch all those addresses.
695  if (Line->getIsLineDebug())
696  CompileUnit->addMapping(Line, SectionIndex);
697 
698  // Resolve any given pattern.
699  patterns().resolvePatternMatch(Line);
700  }
701 
702  ScopesWithRanges->endSearch();
703 }

705 void LVBinaryReader::processLines(LVLines *DebugLines,
706  LVSectionIndex SectionIndex) {
707  assert(DebugLines && "DebugLines is null.");
708  if (DebugLines->empty() && !ScopeInstructions.findMap(SectionIndex))
709  return;
710 
711  // If the Compile Unit does not contain comdat functions, use the whole
712  // set of debug lines, as the addresses don't have conflicts.
713  if (!CompileUnit->getHasComdatScopes()) {
714  processLines(DebugLines, SectionIndex, nullptr);
715  return;
716  }
717 
718  // Find the indexes for the lines whose address is zero.
719  std::vector<size_t> AddressZero;
720  LVLines::iterator It =
721  std::find_if(std::begin(*DebugLines), std::end(*DebugLines),
722  [](LVLine *Line) { return !Line->getAddress(); });
723  while (It != std::end(*DebugLines)) {
724  AddressZero.emplace_back(std::distance(std::begin(*DebugLines), It));
725  It = std::find_if(std::next(It), std::end(*DebugLines),
726  [](LVLine *Line) { return !Line->getAddress(); });
727  }
728 
729  // If the set of debug lines does not contain any line with address zero,
730  // use the whole set. It means we are dealing with an initialization
731  // section from a fully linked binary.
732  if (AddressZero.empty()) {
733  processLines(DebugLines, SectionIndex, nullptr);
734  return;
735  }
736 
737  // The Compile unit contains comdat functions. Traverse the collected
738  // debug lines and identify logical groups based on their start and
739  // address. Each group starts with a zero address.
740  // Begin, End, Address, IsDone.
741  using LVBucket = std::tuple<size_t, size_t, LVAddress, bool>;
742  std::vector<LVBucket> Buckets;
743 
744  LVAddress Address;
745  size_t Begin = 0;
746  size_t End = 0;
747  size_t Index = 0;
748  for (Index = 0; Index < AddressZero.size() - 1; ++Index) {
749  Begin = AddressZero[Index];
750  End = AddressZero[Index + 1] - 1;
751  Address = (*DebugLines)[End]->getAddress();
752  Buckets.emplace_back(Begin, End, Address, false);
753  }
754 
755  // Add the last bucket.
756  if (Index) {
757  Begin = AddressZero[Index];
758  End = DebugLines->size() - 1;
759  Address = (*DebugLines)[End]->getAddress();
760  Buckets.emplace_back(Begin, End, Address, false);
761  }
762 
763  LLVM_DEBUG({
764  dbgs() << "\nDebug Lines buckets: " << Buckets.size() << "\n";
765  for (LVBucket &Bucket : Buckets) {
766  dbgs() << "Begin: " << format_decimal(std::get<0>(Bucket), 5) << ", "
767  << "End: " << format_decimal(std::get<1>(Bucket), 5) << ", "
768  << "Address: " << hexValue(std::get<2>(Bucket)) << "\n";
769  }
770  });
771 
772  // Traverse the sections and buckets looking for matches on the section
773  // sizes. In the unlikely event of different buckets with the same size
774  // process them in order and mark them as done.
775  LVLines Group;
776  for (LVSections::reference Entry : Sections) {
777  LVSectionIndex SectionIndex = Entry.first;
778  const object::SectionRef Section = Entry.second;
779  uint64_t Size = Section.getSize();
780  LLVM_DEBUG({
781  dbgs() << "\nSection Index: " << format_decimal(SectionIndex, 3)
782  << " , Section Size: " << hexValue(Section.getSize())
783  << " , Section Address: " << hexValue(Section.getAddress())
784  << "\n";
785  });
786 
787  for (LVBucket &Bucket : Buckets) {
788  if (std::get<3>(Bucket))
789  // Already done for previous section.
790  continue;
791  if (Size == std::get<2>(Bucket)) {
792  // We have a match on the section size.
793  Group.clear();
794  LVLines::iterator IterStart = DebugLines->begin() + std::get<0>(Bucket);
795  LVLines::iterator IterEnd =
796  DebugLines->begin() + std::get<1>(Bucket) + 1;
797  for (LVLines::iterator Iter = IterStart; Iter < IterEnd; ++Iter)
798  Group.push_back(*Iter);
799  processLines(&Group, SectionIndex, /*Function=*/nullptr);
800  std::get<3>(Bucket) = true;
801  break;
802  }
803  }
804  }
805 }
