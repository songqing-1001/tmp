38 bool MipsPreLegalizerCombinerInfo::combine(GISelChangeObserver &Observer,
39  MachineInstr &MI,
40  MachineIRBuilder &B) const {
41  CombinerHelper Helper(Observer, B, /*IsPreLegalize*/ true);
42 
43  switch (MI.getOpcode()) {
44  default:
45  return false;
46  case TargetOpcode::G_MEMCPY_INLINE:
47  return Helper.tryEmitMemcpyInline(MI);
48  case TargetOpcode::G_LOAD:
49  case TargetOpcode::G_SEXTLOAD:
50  case TargetOpcode::G_ZEXTLOAD: {
51  // Don't attempt to combine non power of 2 loads or unaligned loads when
52  // subtarget doesn't support them.
53  auto MMO = *MI.memoperands_begin();
54  const MipsSubtarget &STI = MI.getMF()->getSubtarget<MipsSubtarget>();
55  if (!isPowerOf2_64(MMO->getSize()))
56  return false;
57  bool isUnaligned = MMO->getAlign() < MMO->getSize();
58  if (!STI.systemSupportsUnalignedAccess() && isUnaligned)
59  return false;
60 
61  return Helper.tryCombineExtendingLoads(MI);
62  }
63  }
64 
65  return false;
66 }
