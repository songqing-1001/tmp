41 static bool parseInt(StringRef &input, unsigned &value) {
42  assert(value == 0);
43  if (input.empty())
44  return true;
45 
46  char next = input[0];
47  input = input.substr(1);
48  if (next < '0' || next > '9')
49  return true;
50  value = (unsigned)(next - '0');
51 
52  while (!input.empty()) {
53  next = input[0];
54  if (next < '0' || next > '9')
55  return false;
56  input = input.substr(1);
57  value = value * 10 + (unsigned)(next - '0');
58  }
59 
60  return false;
61 }
