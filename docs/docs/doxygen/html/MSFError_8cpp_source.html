20 class MSFErrorCategory : public std::error_category {
21 public:
22  const char *name() const noexcept override { return "llvm.msf"; }
23  std::string message(int Condition) const override {
24  switch (static_cast<msf_error_code>(Condition)) {
25  case msf_error_code::unspecified:
26  return "An unknown error has occurred.";
27  case msf_error_code::insufficient_buffer:
28  return "The buffer is not large enough to read the requested number of "
29  "bytes.";
30  case msf_error_code::size_overflow_4096:
31  return "Output data is larger than 4 GiB.";
32  case msf_error_code::size_overflow_8192:
33  return "Output data is larger than 8 GiB.";
34  case msf_error_code::size_overflow_16384:
35  return "Output data is larger than 16 GiB.";
36  case msf_error_code::size_overflow_32768:
37  return "Output data is larger than 32 GiB.";
38  case msf_error_code::not_writable:
39  return "The specified stream is not writable.";
40  case msf_error_code::no_stream:
41  return "The specified stream does not exist.";
42  case msf_error_code::invalid_format:
43  return "The data is in an unexpected format.";
44  case msf_error_code::block_in_use:
45  return "The block is already in use.";
46  }
47  llvm_unreachable("Unrecognized msf_error_code");
48  }
49 };
