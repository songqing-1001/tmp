45 std::string computeDataLayout(const Triple &TT, StringRef CPU,
46  const TargetOptions &Options) {
47  std::string Ret = "";
48  // M68k is Big Endian
49  Ret += "E";
50 
51  // FIXME how to wire it with the used object format?
52  Ret += "-m:e";
53 
54  // M68k pointers are always 32 bit wide even for 16-bit CPUs.
55  // The ABI only specifies 16-bit alignment.
56  // On at least the 68020+ with a 32-bit bus, there is a performance benefit
57  // to having 32-bit alignment.
58  Ret += "-p:32:16:32";
59 
60  // Bytes do not require special alignment, words are word aligned and
61  // long words are word aligned at minimum.
62  Ret += "-i8:8:8-i16:16:16-i32:16:32";
63 
64  // FIXME no floats at the moment
65 
66  // The registers can hold 8, 16, 32 bits
67  Ret += "-n8:16:32";
68 
69  Ret += "-a:0:16-S16";
70 
71  return Ret;
72 }

112 const M68kSubtarget *
113 M68kTargetMachine::getSubtargetImpl(const Function &F) const {
114  Attribute CPUAttr = F.getFnAttribute("target-cpu");
115  Attribute FSAttr = F.getFnAttribute("target-features");
116 
117  auto CPU = CPUAttr.isValid() ? CPUAttr.getValueAsString().str() : TargetCPU;
118  auto FS = FSAttr.isValid() ? FSAttr.getValueAsString().str() : TargetFS;
119 
120  auto &I = SubtargetMap[CPU + FS];
121  if (!I) {
122  // This needs to be done before we create a new subtarget since any
123  // creation will depend on the TM and the code generation flags on the
124  // function that reside in TargetOptions.
125  resetTargetOptions(F);
126  I = std::make_unique<M68kSubtarget>(TargetTriple, CPU, FS, *this);
127  }
128  return I.get();
129 }
