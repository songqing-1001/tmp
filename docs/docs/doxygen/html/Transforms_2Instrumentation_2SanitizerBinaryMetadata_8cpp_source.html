183 bool SanitizerBinaryMetadata::run() {
184  MetadataInfoSet MIS;
185 
186  for (Function &F : Mod)
187  runOn(F, MIS);
188 
189  if (MIS.empty())
190  return false;
191 
192  //
193  // Setup constructors and call all initialization functions for requested
194  // metadata features.
195  //
196 
197  auto *Int8PtrTy = IRB.getInt8PtrTy();
198  auto *Int8PtrPtrTy = PointerType::getUnqual(Int8PtrTy);
199  auto *Int32Ty = IRB.getInt32Ty();
200  const std::array<Type *, 3> InitTypes = {Int32Ty, Int8PtrPtrTy, Int8PtrPtrTy};
201  auto *Version = ConstantInt::get(Int32Ty, getVersion());
202 
203  for (const MetadataInfo *MI : MIS) {
204  const std::array<Value *, InitTypes.size()> InitArgs = {
205  Version,
206  getSectionMarker(getSectionStart(MI->SectionSuffix), Int8PtrTy),
207  getSectionMarker(getSectionEnd(MI->SectionSuffix), Int8PtrTy),
208  };
209  // We declare the _add and _del functions as weak, and only call them if
210  // there is a valid symbol linked. This allows building binaries with
211  // semantic metadata, but without having callbacks. When a tool that wants
212  // the metadata is linked which provides the callbacks, they will be called.
213  Function *Ctor =
214  createSanitizerCtorAndInitFunctions(
215  Mod, (MI->FunctionPrefix + ".module_ctor").str(),
216  (MI->FunctionPrefix + "_add").str(), InitTypes, InitArgs,
217  /*VersionCheckName=*/StringRef(), /*Weak=*/ClWeakCallbacks)
218  .first;
219  Function *Dtor =
220  createSanitizerCtorAndInitFunctions(
221  Mod, (MI->FunctionPrefix + ".module_dtor").str(),
222  (MI->FunctionPrefix + "_del").str(), InitTypes, InitArgs,
223  /*VersionCheckName=*/StringRef(), /*Weak=*/ClWeakCallbacks)
224  .first;
225  Constant *CtorComdatKey = nullptr;
226  Constant *DtorComdatKey = nullptr;
227  if (TargetTriple.supportsCOMDAT()) {
228  // Use COMDAT to deduplicate constructor/destructor function. The COMDAT
229  // key needs to be a non-local linkage.
230  Ctor->setComdat(Mod.getOrInsertComdat(Ctor->getName()));
231  Dtor->setComdat(Mod.getOrInsertComdat(Dtor->getName()));
232  Ctor->setLinkage(GlobalValue::ExternalLinkage);
233  Dtor->setLinkage(GlobalValue::ExternalLinkage);
234  // DSOs should _not_ call another constructor/destructor!
235  Ctor->setVisibility(GlobalValue::HiddenVisibility);
236  Dtor->setVisibility(GlobalValue::HiddenVisibility);
237  CtorComdatKey = Ctor;
238  DtorComdatKey = Dtor;
239  }
240  appendToGlobalCtors(Mod, Ctor, kCtorDtorPriority, CtorComdatKey);
241  appendToGlobalDtors(Mod, Dtor, kCtorDtorPriority, DtorComdatKey);
242  }
243 
244  return true;
245 }

247 void SanitizerBinaryMetadata::runOn(Function &F, MetadataInfoSet &MIS) {
248  if (F.empty())
249  return;
250  if (F.hasFnAttribute(Attribute::DisableSanitizerInstrumentation))
251  return;
252  if (Ignorelist && Ignorelist->inSection("metadata", "fun", F.getName()))
253  return;
254  // Don't touch available_externally functions, their actual body is elsewhere.
255  if (F.getLinkage() == GlobalValue::AvailableExternallyLinkage)
256  return;
257 
258  MDBuilder MDB(F.getContext());
259 
260  // The metadata features enabled for this function, stored along covered
261  // metadata (if enabled).
262  uint64_t FeatureMask = 0;
263  // Don't emit unnecessary covered metadata for all functions to save space.
264  bool RequiresCovered = false;
265 
266  if (Options.Atomics || Options.UAR) {
267  for (BasicBlock &BB : F)
268  for (Instruction &I : BB)
269  RequiresCovered |= runOn(I, MIS, MDB, FeatureMask);
270  }
271 
272  if (F.isVarArg())
273  FeatureMask &= ~kSanitizerBinaryMetadataUAR;
274  if (FeatureMask & kSanitizerBinaryMetadataUAR) {
275  RequiresCovered = true;
276  NumMetadataUAR++;
277  }
278 
279  // Covered metadata is always emitted if explicitly requested, otherwise only
280  // if some other metadata requires it to unambiguously interpret it for
281  // modules compiled with SanitizerBinaryMetadata.
282  if (Options.Covered || (FeatureMask && RequiresCovered)) {
283  NumMetadataCovered++;
284  const auto *MI = &MetadataInfo::Covered;
285  MIS.insert(MI);
286  const StringRef Section = getSectionName(MI->SectionSuffix);
287  // The feature mask will be placed after the function size.
288  Constant *CFM = IRB.getInt64(FeatureMask);
289  F.setMetadata(LLVMContext::MD_pcsections,
290  MDB.createPCSections({{Section, {CFM}}}));
291  }
292 }

294 bool isUARSafeCall(CallInst *CI) {
295  auto *F = CI->getCalledFunction();
296  // There are no intrinsic functions that leak arguments.
297  // If the called function does not return, the current function
298  // does not return as well, so no possibility of use-after-return.
299  // Sanitizer function also don't leak or don't return.
300  // It's safe to both pass pointers to local variables to them
301  // and to tail-call them.
302  return F && (F->isIntrinsic() || F->doesNotReturn() ||
303  F->getName().startswith("__asan_") ||
304  F->getName().startswith("__hwsan_") ||
305  F->getName().startswith("__ubsan_") ||
306  F->getName().startswith("__msan_") ||
307  F->getName().startswith("__tsan_"));
308 }

310 bool hasUseAfterReturnUnsafeUses(Value &V) {
311  for (User *U : V.users()) {
312  if (auto *I = dyn_cast<Instruction>(U)) {
313  if (I->isLifetimeStartOrEnd() || I->isDroppable())
314  continue;
315  if (auto *CI = dyn_cast<CallInst>(U)) {
316  if (isUARSafeCall(CI))
317  continue;
318  }
319  if (isa<LoadInst>(U))
320  continue;
321  if (auto *SI = dyn_cast<StoreInst>(U)) {
322  // If storing TO the alloca, then the address isn't taken.
323  if (SI->getOperand(1) == &V)
324  continue;
325  }
326  if (auto *GEPI = dyn_cast<GetElementPtrInst>(U)) {
327  if (!hasUseAfterReturnUnsafeUses(*GEPI))
328  continue;
329  } else if (auto *BCI = dyn_cast<BitCastInst>(U)) {
330  if (!hasUseAfterReturnUnsafeUses(*BCI))
331  continue;
332  }
333  }
334  return true;
335  }
336  return false;
337 }
338 
339 bool useAfterReturnUnsafe(Instruction &I) {
340  if (isa<AllocaInst>(I))
341  return hasUseAfterReturnUnsafeUses(I);
342  // Tail-called functions are not necessary intercepted
343  // at runtime because there is no call instruction.
344  // So conservatively mark the caller as requiring checking.
345  else if (auto *CI = dyn_cast<CallInst>(&I))
346  return CI->isTailCall() && !isUARSafeCall(CI);
347  return false;
348 }

350 bool SanitizerBinaryMetadata::pretendAtomicAccess(const Value *Addr) {
351  if (!Addr)
352  return false;
353 
354  Addr = Addr->stripInBoundsOffsets();
355  auto *GV = dyn_cast<GlobalVariable>(Addr);
356  if (!GV)
357  return false;
358 
359  // Some compiler-generated accesses are known racy, to avoid false positives
360  // in data-race analysis pretend they're atomic.
361  if (GV->hasSection()) {
362  const auto OF = Triple(Mod.getTargetTriple()).getObjectFormat();
363  const auto ProfSec =
364  getInstrProfSectionName(IPSK_cnts, OF, /*AddSegmentInfo=*/false);
365  if (GV->getSection().endswith(ProfSec))
366  return true;
367  }
368  if (GV->getName().startswith("__llvm_gcov") ||
369  GV->getName().startswith("__llvm_gcda"))
370  return true;
371 
372  return false;
373 }

375 // Returns true if the memory at `Addr` may be shared with other threads.
376 bool maybeSharedMutable(const Value *Addr) {
377  // By default assume memory may be shared.
378  if (!Addr)
379  return true;
380 
381  if (isa<AllocaInst>(getUnderlyingObject(Addr)) &&
382  !PointerMayBeCaptured(Addr, true, true))
383  return false; // Object is on stack but does not escape.
384 
385  Addr = Addr->stripInBoundsOffsets();
386  if (auto *GV = dyn_cast<GlobalVariable>(Addr)) {
387  if (GV->isConstant())
388  return false; // Shared, but not mutable.
389  }
390 
391  return true;
392 }

394 bool SanitizerBinaryMetadata::runOn(Instruction &I, MetadataInfoSet &MIS,
395  MDBuilder &MDB, uint64_t &FeatureMask) {
396  SmallVector<const MetadataInfo *, 1> InstMetadata;
397  bool RequiresCovered = false;
398 
399  // Only call if at least 1 type of metadata is requested.
400  assert(Options.UAR || Options.Atomics);
401 
402  if (Options.UAR && !(FeatureMask & kSanitizerBinaryMetadataUAR)) {
403  if (useAfterReturnUnsafe(I))
404  FeatureMask |= kSanitizerBinaryMetadataUAR;
405  }
406 
407  if (Options.Atomics) {
408  const Value *Addr = nullptr;
409  if (auto *SI = dyn_cast<StoreInst>(&I))
410  Addr = SI->getPointerOperand();
411  else if (auto *LI = dyn_cast<LoadInst>(&I))
412  Addr = LI->getPointerOperand();
413 
414  if (I.mayReadOrWriteMemory() && maybeSharedMutable(Addr)) {
415  auto SSID = getAtomicSyncScopeID(&I);
416  if ((SSID.has_value() && *SSID != SyncScope::SingleThread) ||
417  pretendAtomicAccess(Addr)) {
418  NumMetadataAtomics++;
419  InstMetadata.push_back(&MetadataInfo::Atomics);
420  }
421  FeatureMask |= kSanitizerBinaryMetadataAtomics;
422  RequiresCovered = true;
423  }
424  }
425 
426  // Attach MD_pcsections to instruction.
427  if (!InstMetadata.empty()) {
428  MIS.insert(InstMetadata.begin(), InstMetadata.end());
429  SmallVector<MDBuilder::PCSection, 1> Sections;
430  for (const auto &MI : InstMetadata)
431  Sections.push_back({getSectionName(MI->SectionSuffix), {}});
432  I.setMetadata(LLVMContext::MD_pcsections, MDB.createPCSections(Sections));
433  }
434 
435  return RequiresCovered;
436 }
