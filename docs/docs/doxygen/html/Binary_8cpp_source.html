102 Expected<OwningBinary<Binary>>
103 object::createBinary(StringRef Path, LLVMContext *Context, bool InitContent) {
104  ErrorOr<std::unique_ptr<MemoryBuffer>> FileOrErr =
105  MemoryBuffer::getFileOrSTDIN(Path, /*IsText=*/false,
106  /*RequiresNullTerminator=*/false);
107  if (std::error_code EC = FileOrErr.getError())
108  return errorCodeToError(EC);
109  std::unique_ptr<MemoryBuffer> &Buffer = FileOrErr.get();
110 
111  Expected<std::unique_ptr<Binary>> BinOrErr =
112  createBinary(Buffer->getMemBufferRef(), Context, InitContent);
113  if (!BinOrErr)
114  return BinOrErr.takeError();
115  std::unique_ptr<Binary> &Bin = BinOrErr.get();
116 
117  return OwningBinary<Binary>(std::move(Bin), std::move(Buffer));
118 }
