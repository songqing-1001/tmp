102  static void addMachOHeader(jitlink::LinkGraph &G, MachOPlatform &MOP,
103  const SymbolStringPtr &InitializerSymbol) {
104  auto &HeaderSection = G.createSection("__header", MemProt::Read);
105  auto &HeaderBlock = createHeaderBlock(G, HeaderSection);
106 
107  // Init symbol is header-start symbol.
108  G.addDefinedSymbol(HeaderBlock, 0, *InitializerSymbol,
109  HeaderBlock.getSize(), jitlink::Linkage::Strong,
110  jitlink::Scope::Default, false, true);
111  for (auto &HS : AdditionalHeaderSymbols)
112  G.addDefinedSymbol(HeaderBlock, HS.Offset, HS.Name, HeaderBlock.getSize(),
113  jitlink::Linkage::Strong, jitlink::Scope::Default,
114  false, true);
115  }

126  static jitlink::Block &createHeaderBlock(jitlink::LinkGraph &G,
127  jitlink::Section &HeaderSection) {
128  MachO::mach_header_64 Hdr;
129  Hdr.magic = MachO::MH_MAGIC_64;
130  switch (G.getTargetTriple().getArch()) {
131  case Triple::aarch64:
132  Hdr.cputype = MachO::CPU_TYPE_ARM64;
133  Hdr.cpusubtype = MachO::CPU_SUBTYPE_ARM64_ALL;
134  break;
135  case Triple::x86_64:
136  Hdr.cputype = MachO::CPU_TYPE_X86_64;
137  Hdr.cpusubtype = MachO::CPU_SUBTYPE_X86_64_ALL;
138  break;
139  default:
140  llvm_unreachable("Unrecognized architecture");
141  }
142  Hdr.filetype = MachO::MH_DYLIB; // Custom file type?
143  Hdr.ncmds = 0;
144  Hdr.sizeofcmds = 0;
145  Hdr.flags = 0;
146  Hdr.reserved = 0;
147 
148  if (G.getEndianness() != support::endian::system_endianness())
149  MachO::swapStruct(Hdr);
150 
151  auto HeaderContent = G.allocateContent(
152  ArrayRef<char>(reinterpret_cast<const char *>(&Hdr), sizeof(Hdr)));

202  void materialize(std::unique_ptr<MaterializationResponsibility> R) override {
203  using namespace jitlink;
204  auto G = createPlatformGraph(MOP, "<OrcRTCompleteBootstrap>");
205  auto &PlaceholderSection =
206  G->createSection("__orc_rt_cplt_bs", MemProt::Read);
207  auto &PlaceholderBlock =
208  G->createZeroFillBlock(PlaceholderSection, 1, ExecutorAddr(), 1, 0);
209  G->addDefinedSymbol(PlaceholderBlock, 0, *CompleteBootstrapSymbol, 1,
210  Linkage::Strong, Scope::Hidden, false, true);
211 
212  // Reserve space for the stolen actions, plus two extras.
213  G->allocActions().reserve(DeferredAAs.size() + 2);
214 
215  // 1. Bootstrap the platform support code.
216  G->allocActions().push_back(
217  {cantFail(WrapperFunctionCall::Create<SPSArgList<>>(PlatformBootstrap)),
218  cantFail(
219  WrapperFunctionCall::Create<SPSArgList<>>(PlatformShutdown))});
220 
221  // 2. Register the platform JITDylib.
222  G->allocActions().push_back(
223  {cantFail(WrapperFunctionCall::Create<
224  SPSArgList<SPSString, SPSExecutorAddr>>(
225  RegisterJITDylib, PlatformJDName, MachOHeaderAddr)),
226  cantFail(WrapperFunctionCall::Create<SPSArgList<SPSExecutorAddr>>(
227  DeregisterJITDylib, MachOHeaderAddr))});
228 
229  // 3. Add the deferred actions to the graph.
230  std::move(DeferredAAs.begin(), DeferredAAs.end(),
231  std::back_inserter(G->allocActions()));
232 
233  MOP.getObjectLinkingLayer().emit(std::move(R), std::move(G));
234  }

263  // If the target is not supported then bail out immediately.
264  if (!supportedTarget(EPC.getTargetTriple()))

266  EPC.getTargetTriple().str(),

269  // Create default aliases if the caller didn't supply any.
270  if (!RuntimeAliases)

278  if (auto Err = PlatformJD.define(absoluteSymbols(
279  {{ES.intern("___orc_rt_jit_dispatch"),

404 MachOPlatform::MachOPlatform(
405  ExecutionSession &ES, ObjectLinkingLayer &ObjLinkingLayer,
406  JITDylib &PlatformJD,
407  std::unique_ptr<DefinitionGenerator> OrcRuntimeGenerator, Error &Err)
408  : ES(ES), PlatformJD(PlatformJD), ObjLinkingLayer(ObjLinkingLayer) {
409  ErrorAsOutParameter _(&Err);
410  ObjLinkingLayer.addPlugin(std::make_unique<MachOPlatformPlugin>(*this));
411  PlatformJD.addGenerator(std::move(OrcRuntimeGenerator));
412 
413  BootstrapInfo BI;
414  Bootstrap = &BI;
415 
416  // Bootstrap process -- here be phase-ordering dragons.
417  //
418  // The MachOPlatform class uses allocation actions to register metadata
419  // sections with the ORC runtime, however the runtime contains metadata
420  // registration functions that have their own metadata that they need to
421  // register (e.g. the frame-info registration functions have frame-info).
422  // We can't use an ordinary lookup to find these registration functions
423  // because their address is needed during the link of the containing graph
424  // itself (to build the allocation actions that will call the registration
425  // functions). Further complicating the situation (a) the graph containing
426  // the registration functions is allowed to depend on other graphs (e.g. the
427  // graph containing the ORC runtime RTTI support) so we need to handle with
428  // an unknown set of dependencies during bootstrap, and (b) these graphs may
429  // be linked concurrently if the user has installed a concurrent dispatcher.
430  //
431  // We satisfy these constraint by implementing a bootstrap phase during which
432  // allocation actions generated by MachOPlatform are appended to a list of
433  // deferred allocation actions, rather than to the graphs themselves. At the
434  // end of the bootstrap process the deferred actions are attached to a final
435  // "complete-bootstrap" graph that causes them to be run.
436  //
437  // The bootstrap steps are as follows:
438  //
439  // 1. Request the graph containing the mach header. This graph is guaranteed
440  // not to have any metadata so the fact that the registration functions
441  // are not available yet is not a problem.
442  //
443  // 2. Look up the registration functions and discard the results. This will
444  // trigger linking of the graph containing these functions, and
445  // consequently any graphs that it depends on. We do not use the lookup
446  // result to find the addresses of the functions requested (as described
447  // above the lookup will return too late for that), instead we capture the
448  // addresses in a post-allocation pass injected by the platform runtime
449  // during bootstrap only.
450  //
451  // 3. During bootstrap the MachOPlatformPlugin keeps a count of the number of
452  // graphs being linked (potentially concurrently), and we block until all
453  // of these graphs have completed linking. This is to avoid a race on the
454  // deferred-actions vector: the lookup for the runtime registration
455  // functions may return while some functions (those that are being
456  // incidentally linked in, but aren't reachable via the runtime functions)
457  // are still being linked, and we need to capture any allocation actions
458  // for this incidental code before we proceed.
459  //
460  // 4. Once all active links are complete we transfer the deferred actions to
461  // a newly added CompleteBootstrap graph and then request a symbol from
462  // the CompleteBootstrap graph to trigger materialization. This will cause
463  // all deferred actions to be run, and once this lookup returns we can
464  // proceed.
465  //
466  // 5. Finally, we associate runtime support methods in MachOPlatform with
467  // the corresponding jit-dispatch tag variables in the ORC runtime to make
468  // the support methods callable. The bootstrap is now complete.
469 
470  // Step (1) Add header materialization unit and request.
471  if ((Err = PlatformJD.define(std::make_unique<MachOHeaderMaterializationUnit>(
472  *this, MachOHeaderStartSymbol))))
473  return;
474  if ((Err = ES.lookup(&PlatformJD, MachOHeaderStartSymbol).takeError()))
475  return;
476 
477  // Step (2) Request runtime registration functions to trigger
478  // materialization..
479  if ((Err = ES.lookup(makeJITDylibSearchOrder(&PlatformJD),
480  SymbolLookupSet(
481  {PlatformBootstrap.Name, PlatformShutdown.Name,
482  RegisterJITDylib.Name, DeregisterJITDylib.Name,
483  RegisterObjectPlatformSections.Name,
484  DeregisterObjectPlatformSections.Name,
485  CreatePThreadKey.Name}))
486  .takeError()))
487  return;
488 
489  // Step (3) Wait for any incidental linker work to complete.
490  {
491  std::unique_lock<std::mutex> Lock(BI.Mutex);
492  BI.CV.wait(Lock, [&]() { return BI.ActiveGraphs == 0; });
493  Bootstrap = nullptr;
494  }
495 
496  // Step (4) Add complete-bootstrap materialization unit and request.
497  auto BootstrapCompleteSymbol = ES.intern("__orc_rt_macho_complete_bootstrap");
498  if ((Err = PlatformJD.define(
499  std::make_unique<MachOPlatformCompleteBootstrapMaterializationUnit>(
500  *this, PlatformJD.getName(), BootstrapCompleteSymbol,
501  std::move(BI.DeferredAAs), PlatformBootstrap.Addr,
502  PlatformShutdown.Addr, RegisterJITDylib.Addr,
503  DeregisterJITDylib.Addr, BI.MachOHeaderAddr))))
504  return;
505  if ((Err = ES.lookup(makeJITDylibSearchOrder(
506  &PlatformJD, JITDylibLookupFlags::MatchAllSymbols),
507  std::move(BootstrapCompleteSymbol))
508  .takeError()))
509  return;
510 
511  // (5) Associate runtime support functions.
512  if ((Err = associateRuntimeSupportFunctions()))
513  return;
514 }

534 void MachOPlatform::pushInitializersLoop(
535  PushInitializersSendResultFn SendResult, JITDylibSP JD) {
536  DenseMap<JITDylib *, SymbolLookupSet> NewInitSymbols;
537  DenseMap<JITDylib *, SmallVector<JITDylib *>> JDDepMap;
538  SmallVector<JITDylib *, 16> Worklist({JD.get()});
539 
540  ES.runSessionLocked([&]() {
541  while (!Worklist.empty()) {
542  // FIXME: Check for defunct dylibs.
543 
544  auto DepJD = Worklist.back();
545  Worklist.pop_back();
546 
547  // If we've already visited this JITDylib on this iteration then continue.
548  if (JDDepMap.count(DepJD))
549  continue;
550 
551  // Add dep info.
552  auto &DM = JDDepMap[DepJD];
553  DepJD->withLinkOrderDo([&](const JITDylibSearchOrder &O) {
554  for (auto &KV : O) {
555  if (KV.first == DepJD)
556  continue;
557  DM.push_back(KV.first);
558  Worklist.push_back(KV.first);
559  }
560  });
561 
562  // Add any registered init symbols.
563  auto RISItr = RegisteredInitSymbols.find(DepJD);
564  if (RISItr != RegisteredInitSymbols.end()) {
565  NewInitSymbols[DepJD] = std::move(RISItr->second);
566  RegisteredInitSymbols.erase(RISItr);
567  }
568  }
569  });
570 
571  // If there are no further init symbols to look up then send the link order
572  // (as a list of header addresses) to the caller.
573  if (NewInitSymbols.empty()) {
574 
575  // To make the list intelligible to the runtime we need to convert all
576  // JITDylib pointers to their header addresses. Only include JITDylibs
577  // that appear in the JITDylibToHeaderAddr map (i.e. those that have been
578  // through setupJITDylib) -- bare JITDylibs aren't managed by the platform.
579  DenseMap<JITDylib *, ExecutorAddr> HeaderAddrs;
580  HeaderAddrs.reserve(JDDepMap.size());
581  {
582  std::lock_guard<std::mutex> Lock(PlatformMutex);
583  for (auto &KV : JDDepMap) {
584  auto I = JITDylibToHeaderAddr.find(KV.first);
585  if (I != JITDylibToHeaderAddr.end())
586  HeaderAddrs[KV.first] = I->second;
587  }
588  }
589 
590  // Build the dep info map to return.
591  MachOJITDylibDepInfoMap DIM;

604  }

606  }
607  SendResult(DIM);
608  return;
609  }

612  lookupInitSymbolsAsync(
613  [this, SendResult = std::move(SendResult), JD](Error Err) mutable {

615  SendResult(std::move(Err));
616  else
617  pushInitializersLoop(std::move(SendResult), JD);
618  },

718 void MachOPlatform::MachOPlatformPlugin::modifyPassConfig(
719  MaterializationResponsibility &MR, jitlink::LinkGraph &LG,
720  jitlink::PassConfiguration &Config) {
721 
722  using namespace jitlink;
723 
724  bool InBootstrapPhase =
725  &MR.getTargetJITDylib() == &MP.PlatformJD && MP.Bootstrap;
726 
727  // If we're in the bootstrap phase then increment the active graphs.
728  if (InBootstrapPhase) {
729  Config.PrePrunePasses.push_back(
730  [this](LinkGraph &G) { return bootstrapPipelineStart(G); });
731  Config.PostAllocationPasses.push_back([this](LinkGraph &G) {
732  return bootstrapPipelineRecordRuntimeFunctions(G);
733  });
734  }
735 
736  // --- Handle Initializers ---
737  if (auto InitSymbol = MR.getInitializerSymbol()) {
738 
739  // If the initializer symbol is the MachOHeader start symbol then just
740  // register it and then bail out -- the header materialization unit
741  // definitely doesn't need any other passes.
742  if (InitSymbol == MP.MachOHeaderStartSymbol && !InBootstrapPhase) {
743  Config.PostAllocationPasses.push_back([this, &MR](LinkGraph &G) {
744  return associateJITDylibHeaderSymbol(G, MR);
745  });
746  return;
747  }
748 
749  // If the object contains an init symbol other than the header start symbol
750  // then add passes to preserve, process and register the init
751  // sections/symbols.
752  Config.PrePrunePasses.push_back([this, &MR](LinkGraph &G) {
753  if (auto Err = preserveInitSections(G, MR))
754  return Err;
755  return processObjCImageInfo(G, MR);
756  });
757  }
758 
759  // Insert TLV lowering at the start of the PostPrunePasses, since we want
760  // it to run before GOT/PLT lowering.
761  Config.PostPrunePasses.insert(
762  Config.PostPrunePasses.begin(),
763  [this, &JD = MR.getTargetJITDylib()](LinkGraph &G) {
764  return fixTLVSectionsAndEdges(G, JD);
765  });
766 
767  // Add a pass to register the final addresses of any special sections in the
768  // object with the runtime.
769  Config.PostAllocationPasses.push_back(
770  [this, &JD = MR.getTargetJITDylib(), InBootstrapPhase](LinkGraph &G) {
771  return registerObjectPlatformSections(G, JD, InBootstrapPhase);
772  });
773 
774  // If we're in the bootstrap phase then steal allocation actions and then
775  // decrement the active graphs.
776  if (InBootstrapPhase)
777  Config.PostFixupPasses.push_back(
778  [this](LinkGraph &G) { return bootstrapPipelineEnd(G); });
779 }

885 Error MachOPlatform::MachOPlatformPlugin::preserveInitSections(
886  jitlink::LinkGraph &G, MaterializationResponsibility &MR) {
887 
888  JITLinkSymbolSet InitSectionSymbols;
889  for (auto &InitSectionName : MachOInitSectionNames) {
890  // Skip non-init sections.
891  auto *InitSection = G.findSectionByName(InitSectionName);
892  if (!InitSection)
893  continue;
894 
895  // Make a pass over live symbols in the section: those blocks are already
896  // preserved.
897  DenseSet<jitlink::Block *> AlreadyLiveBlocks;
898  for (auto &Sym : InitSection->symbols()) {
899  auto &B = Sym->getBlock();
900  if (Sym->isLive() && Sym->getOffset() == 0 &&
901  Sym->getSize() == B.getSize() && !AlreadyLiveBlocks.count(&B)) {
902  InitSectionSymbols.insert(Sym);
903  AlreadyLiveBlocks.insert(&B);
904  }
905  }
906 
907  // Add anonymous symbols to preserve any not-already-preserved blocks.
908  for (auto *B : InitSection->blocks())
909  if (!AlreadyLiveBlocks.count(B))
910  InitSectionSymbols.insert(
911  &G.addAnonymousSymbol(*B, 0, B->getSize(), false, true));
912  }
913 
914  if (!InitSectionSymbols.empty()) {
915  std::lock_guard<std::mutex> Lock(PluginMutex);
916  InitSymbolDeps[&MR] = std::move(InitSectionSymbols);
917  }
918 
919  return Error::success();
920 }

922 Error MachOPlatform::MachOPlatformPlugin::processObjCImageInfo(
923  jitlink::LinkGraph &G, MaterializationResponsibility &MR) {
924 
925  // If there's an ObjC imagine info then either
926  // (1) It's the first __objc_imageinfo we've seen in this JITDylib. In
927  // this case we name and record it.
928  // OR
929  // (2) We already have a recorded __objc_imageinfo for this JITDylib,
930  // in which case we just verify it.
931  auto *ObjCImageInfo = G.findSectionByName(MachOObjCImageInfoSectionName);
932  if (!ObjCImageInfo)
933  return Error::success();
934 
935  auto ObjCImageInfoBlocks = ObjCImageInfo->blocks();
936 
937  // Check that the section is not empty if present.
938  if (ObjCImageInfoBlocks.empty())
939  return make_error<StringError>("Empty " + MachOObjCImageInfoSectionName +
940  " section in " + G.getName(),
941  inconvertibleErrorCode());
942 
943  // Check that there's only one block in the section.
944  if (std::next(ObjCImageInfoBlocks.begin()) != ObjCImageInfoBlocks.end())
945  return make_error<StringError>("Multiple blocks in " +
946  MachOObjCImageInfoSectionName +
947  " section in " + G.getName(),
948  inconvertibleErrorCode());
949 
950  // Check that the __objc_imageinfo section is unreferenced.
951  // FIXME: We could optimize this check if Symbols had a ref-count.
952  for (auto &Sec : G.sections()) {
953  if (&Sec != ObjCImageInfo)
954  for (auto *B : Sec.blocks())
955  for (auto &E : B->edges())
956  if (E.getTarget().isDefined() &&
957  &E.getTarget().getBlock().getSection() == ObjCImageInfo)
958  return make_error<StringError>(MachOObjCImageInfoSectionName +
959  " is referenced within file " +
960  G.getName(),
961  inconvertibleErrorCode());
962  }
963 
964  auto &ObjCImageInfoBlock = **ObjCImageInfoBlocks.begin();
965  auto *ObjCImageInfoData = ObjCImageInfoBlock.getContent().data();
966  auto Version = support::endian::read32(ObjCImageInfoData, G.getEndianness());
967  auto Flags =
968  support::endian::read32(ObjCImageInfoData + 4, G.getEndianness());
969 
970  // Lock the mutex while we verify / update the ObjCImageInfos map.
971  std::lock_guard<std::mutex> Lock(PluginMutex);
972 
973  auto ObjCImageInfoItr = ObjCImageInfos.find(&MR.getTargetJITDylib());
974  if (ObjCImageInfoItr != ObjCImageInfos.end()) {
975  // We've already registered an __objc_imageinfo section. Verify the
976  // content of this new section matches, then delete it.
977  if (ObjCImageInfoItr->second.first != Version)
978  return make_error<StringError>(
979  "ObjC version in " + G.getName() +
980  " does not match first registered version",
981  inconvertibleErrorCode());
982  if (ObjCImageInfoItr->second.second != Flags)
983  return make_error<StringError>("ObjC flags in " + G.getName() +
984  " do not match first registered flags",
985  inconvertibleErrorCode());
986 
987  // __objc_imageinfo is valid. Delete the block.
988  for (auto *S : ObjCImageInfo->symbols())
989  G.removeDefinedSymbol(*S);
990  G.removeBlock(ObjCImageInfoBlock);
991  } else {
992  // We haven't registered an __objc_imageinfo section yet. Register and
993  // move on. The section should already be marked no-dead-strip.
994  ObjCImageInfos[&MR.getTargetJITDylib()] = std::make_pair(Version, Flags);
995  }
996 
997  return Error::success();
998 }

1000 Error MachOPlatform::MachOPlatformPlugin::fixTLVSectionsAndEdges(
1001  jitlink::LinkGraph &G, JITDylib &JD) {
1002 
1003  // Rename external references to __tlv_bootstrap to ___orc_rt_tlv_get_addr.
1004  for (auto *Sym : G.external_symbols())
1005  if (Sym->getName() == "__tlv_bootstrap") {
1006  Sym->setName("___orc_rt_macho_tlv_get_addr");
1007  break;
1008  }
1009 
1010  // Store key in __thread_vars struct fields.
1011  if (auto *ThreadDataSec = G.findSectionByName(MachOThreadVarsSectionName)) {
1012  std::optional<uint64_t> Key;
1013  {
1014  std::lock_guard<std::mutex> Lock(MP.PlatformMutex);
1015  auto I = MP.JITDylibToPThreadKey.find(&JD);
1016  if (I != MP.JITDylibToPThreadKey.end())
1017  Key = I->second;
1018  }
1019 
1020  if (!Key) {
1021  if (auto KeyOrErr = MP.createPThreadKey())
1022  Key = *KeyOrErr;
1023  else
1024  return KeyOrErr.takeError();
1025  }
1026 
1027  uint64_t PlatformKeyBits =
1028  support::endian::byte_swap(*Key, G.getEndianness());
1029 
1030  for (auto *B : ThreadDataSec->blocks()) {
1031  if (B->getSize() != 3 * G.getPointerSize())
1032  return make_error<StringError>("__thread_vars block at " +
1033  formatv("{0:x}", B->getAddress()) +
1034  " has unexpected size",
1035  inconvertibleErrorCode());
1036 
1037  auto NewBlockContent = G.allocateBuffer(B->getSize());
1038  llvm::copy(B->getContent(), NewBlockContent.data());
1039  memcpy(NewBlockContent.data() + G.getPointerSize(), &PlatformKeyBits,
1040  G.getPointerSize());
1041  B->setContent(NewBlockContent);
1042  }
1043  }
1044 
1045  // Transform any TLV edges into GOT edges.
1046  for (auto *B : G.blocks())
1047  for (auto &E : B->edges())
1048  if (E.getKind() ==
1049  jitlink::x86_64::RequestTLVPAndTransformToPCRel32TLVPLoadREXRelaxable)
1050  E.setKind(jitlink::x86_64::
1051  RequestGOTAndTransformToPCRel32GOTLoadREXRelaxable);
1052 
1053  return Error::success();
1054 }

1056 std::optional<MachOPlatform::MachOPlatformPlugin::UnwindSections>
1057 MachOPlatform::MachOPlatformPlugin::findUnwindSectionInfo(
1058  jitlink::LinkGraph &G) {
1059  using namespace jitlink;
1060 
1061  UnwindSections US;
1062 
1063  // ScanSection records a section range and adds any executable blocks that
1064  // that section points to to the CodeBlocks vector.
1065  SmallVector<Block *> CodeBlocks;
1066  auto ScanUnwindInfoSection = [&](Section &Sec, ExecutorAddrRange &SecRange) {
1067  if (Sec.blocks().empty())
1068  return;
1069  SecRange = (*Sec.blocks().begin())->getRange();
1070  for (auto *B : Sec.blocks()) {
1071  auto R = B->getRange();
1072  SecRange.Start = std::min(SecRange.Start, R.Start);
1073  SecRange.End = std::max(SecRange.End, R.End);
1074  for (auto &E : B->edges()) {
1075  if (!E.getTarget().isDefined())
1076  continue;
1077  auto &TargetBlock = E.getTarget().getBlock();
1078  auto &TargetSection = TargetBlock.getSection();
1079  if ((TargetSection.getMemProt() & MemProt::Exec) == MemProt::Exec)
1080  CodeBlocks.push_back(&TargetBlock);
1081  }
1082  }
1083  };
1084 
1085  if (Section *EHFrameSec = G.findSectionByName(MachOEHFrameSectionName))
1086  ScanUnwindInfoSection(*EHFrameSec, US.DwarfSection);
1087 
1088  if (Section *CUInfoSec =
1089  G.findSectionByName(MachOCompactUnwindInfoSectionName))
1090  ScanUnwindInfoSection(*CUInfoSec, US.CompactUnwindSection);
1091 
1092  // If we didn't find any pointed-to code-blocks then there's no need to
1093  // register any info.
1094  if (CodeBlocks.empty())
1095  return std::nullopt;
1096 
1097  // We have info to register. Sort the code blocks into address order and
1098  // build a list of contiguous address ranges covering them all.
1099  llvm::sort(CodeBlocks, [](const Block *LHS, const Block *RHS) {
1100  return LHS->getAddress() < RHS->getAddress();
1101  });
1102  for (auto *B : CodeBlocks) {
1103  if (US.CodeRanges.empty() || US.CodeRanges.back().End != B->getAddress())
1104  US.CodeRanges.push_back(B->getRange());
1105  else
1106  US.CodeRanges.back().End = B->getRange().End;
1107  }
1108 
1109  LLVM_DEBUG({
1110  dbgs() << "MachOPlatform identified unwind info in " << G.getName() << ":\n"
1111  << " DWARF: ";
1112  if (US.DwarfSection.Start)
1113  dbgs() << US.DwarfSection << "\n";
1114  else
1115  dbgs() << "none\n";
1116  dbgs() << " Compact-unwind: ";
1117  if (US.CompactUnwindSection.Start)
1118  dbgs() << US.CompactUnwindSection << "\n";
1119  else
1120  dbgs() << "none\n"
1121  << "for code ranges:\n";
1122  for (auto &CR : US.CodeRanges)
1123  dbgs() << " " << CR << "\n";
1124  if (US.CodeRanges.size() >= G.sections_size())
1125  dbgs() << "WARNING: High number of discontiguous code ranges! "
1126  "Padding may be interfering with coalescing.\n";
1127  });
1128 
1129  return US;
1130 }

1132 Error MachOPlatform::MachOPlatformPlugin::registerObjectPlatformSections(
1133  jitlink::LinkGraph &G, JITDylib &JD, bool InBootstrapPhase) {
1134 
1135  // Get a pointer to the thread data section if there is one. It will be used
1136  // below.
1137  jitlink::Section *ThreadDataSection =
1138  G.findSectionByName(MachOThreadDataSectionName);
1139 
1140  // Handle thread BSS section if there is one.
1141  if (auto *ThreadBSSSection = G.findSectionByName(MachOThreadBSSSectionName)) {
1142  // If there's already a thread data section in this graph then merge the
1143  // thread BSS section content into it, otherwise just treat the thread
1144  // BSS section as the thread data section.
1145  if (ThreadDataSection)
1146  G.mergeSections(*ThreadDataSection, *ThreadBSSSection);
1147  else
1148  ThreadDataSection = ThreadBSSSection;
1149  }
1150 
1151  SmallVector<std::pair<StringRef, ExecutorAddrRange>, 8> MachOPlatformSecs;
1152 
1153  // Collect data sections to register.
1154  StringRef DataSections[] = {MachODataDataSectionName,
1155  MachODataCommonSectionName,
1156  MachOEHFrameSectionName};
1157  for (auto &SecName : DataSections) {
1158  if (auto *Sec = G.findSectionByName(SecName)) {
1159  jitlink::SectionRange R(*Sec);
1160  if (!R.empty())
1161  MachOPlatformSecs.push_back({SecName, R.getRange()});
1162  }
1163  }
1164 
1165  // Having merged thread BSS (if present) and thread data (if present),
1166  // record the resulting section range.
1167  if (ThreadDataSection) {
1168  jitlink::SectionRange R(*ThreadDataSection);
1169  if (!R.empty())
1170  MachOPlatformSecs.push_back({MachOThreadDataSectionName, R.getRange()});
1171  }
1172
