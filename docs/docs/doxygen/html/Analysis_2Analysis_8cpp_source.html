96 LLVMBool LLVMVerifyModule(LLVMModuleRef M, LLVMVerifierFailureAction Action,
97  char **OutMessages) {
98  raw_ostream *DebugOS = Action != LLVMReturnStatusAction ? &errs() : nullptr;
99  std::string Messages;
100  raw_string_ostream MsgsOS(Messages);
101 
102  LLVMBool Result = verifyModule(*unwrap(M), OutMessages ? &MsgsOS : DebugOS);
103 
104  // Duplicate the output to stderr.
105  if (DebugOS && OutMessages)
106  *DebugOS << MsgsOS.str();
107 
108  if (Action == LLVMAbortProcessAction && Result)
109  report_fatal_error("Broken module found, compilation aborted!");
110 
111  if (OutMessages)
112  *OutMessages = strdup(MsgsOS.str().c_str());
113 
114  return Result;
115 }
