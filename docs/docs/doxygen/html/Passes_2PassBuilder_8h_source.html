312  /// Parse a textual pass pipeline description into a \c
313  /// ModulePassManager.
314  ///
315  /// The format of the textual pass pipeline description looks something like:
316  ///
317  /// module(function(instcombine,sroa),dce,cgscc(inliner,function(...)),...)
318  ///
319  /// Pass managers have ()s describing the nest structure of passes. All passes
320  /// are comma separated. As a special shortcut, if the very first pass is not
321  /// a module pass (as a module pass manager is), this will automatically form
322  /// the shortest stack of pass managers that allow inserting that first pass.
323  /// So, assuming function passes 'fpassN', CGSCC passes 'cgpassN', and loop
324  /// passes 'lpassN', all of these are valid:
325  ///
326  /// fpass1,fpass2,fpass3
327  /// cgpass1,cgpass2,cgpass3
328  /// lpass1,lpass2,lpass3
329  ///
330  /// And they are equivalent to the following (resp.):
331  ///
332  /// module(function(fpass1,fpass2,fpass3))
333  /// module(cgscc(cgpass1,cgpass2,cgpass3))
334  /// module(function(loop(lpass1,lpass2,lpass3)))
335  ///
336  /// This shortcut is especially useful for debugging and testing small pass
337  /// combinations.
338  ///
339  /// The sequence of passes aren't necessarily the exact same kind of pass.
340  /// You can mix different levels implicitly if adaptor passes are defined to
341  /// make them work. For example,
342  ///
343  /// mpass1,fpass1,fpass2,mpass2,lpass1
344  ///
345  /// This pipeline uses only one pass manager: the top-level module manager.
346  /// fpass1,fpass2 and lpass1 are added into the the top-level module manager
347  /// using only adaptor passes. No nested function/loop pass managers are
348  /// added. The purpose is to allow easy pass testing when the user
349  /// specifically want the pass to run under a adaptor directly. This is
350  /// preferred when a pipeline is largely of one type, but one or just a few
351  /// passes are of different types(See PassBuilder.cpp for examples).
352  Error parsePassPipeline(ModulePassManager &MPM, StringRef PipelineText);
353

359  ///
360  /// function(lpass)
361  Error parsePassPipeline(CGSCCPassManager &CGPM, StringRef PipelineText);

681 /// This utility template takes care of adding require<> and invalidate<>
682 /// passes for an analysis to a given \c PassManager. It is intended to be used
683 /// during parsing of a pass pipeline when parsing a single PipelineName.
684 /// When registering a new function analysis FancyAnalysis with the pass
685 /// pipeline name "fancy-analysis", a matching ParsePipelineCallback could look
686 /// like this:
687 ///
688 /// static bool parseFunctionPipeline(StringRef Name, FunctionPassManager &FPM,
689 /// ArrayRef<PipelineElement> P) {
690 /// if (parseAnalysisUtilityPasses<FancyAnalysis>("fancy-analysis", Name,
691 /// FPM))
692 /// return true;
693 /// return false;
694 /// }
695 template <typename AnalysisT, typename IRUnitT, typename AnalysisManagerT,
696  typename... ExtraArgTs>
697 bool parseAnalysisUtilityPasses(
698  StringRef AnalysisName, StringRef PipelineName,
699  PassManager<IRUnitT, AnalysisManagerT, ExtraArgTs...> &PM) {
700  if (!PipelineName.endswith(">"))
701  return false;
702  // See if this is an invalidate<> pass name
703  if (PipelineName.startswith("invalidate<")) {
704  PipelineName = PipelineName.substr(11, PipelineName.size() - 12);
705  if (PipelineName != AnalysisName)
706  return false;
707  PM.addPass(InvalidateAnalysisPass<AnalysisT>());
708  return true;
709  }
710 
711  // See if this is a require<> pass name
712  if (PipelineName.startswith("require<")) {
713  PipelineName = PipelineName.substr(8, PipelineName.size() - 9);
714  if (PipelineName != AnalysisName)
715  return false;
716  PM.addPass(RequireAnalysisPass<AnalysisT, IRUnitT, AnalysisManagerT,
717  ExtraArgTs...>());
718  return true;
719  }
720 
721  return false;
722 }
