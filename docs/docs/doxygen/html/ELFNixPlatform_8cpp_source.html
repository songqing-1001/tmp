30 class DSOHandleMaterializationUnit : public MaterializationUnit {
31 public:
32  DSOHandleMaterializationUnit(ELFNixPlatform &ENP,
33  const SymbolStringPtr &DSOHandleSymbol)
34  : MaterializationUnit(
35  createDSOHandleSectionInterface(ENP, DSOHandleSymbol)),
36  ENP(ENP) {}
37 
38  StringRef getName() const override { return "DSOHandleMU"; }
39 
40  void materialize(std::unique_ptr<MaterializationResponsibility> R) override {
41  unsigned PointerSize;
42  support::endianness Endianness;
43  jitlink::Edge::Kind EdgeKind;
44  const auto &TT =
45  ENP.getExecutionSession().getExecutorProcessControl().getTargetTriple();
46 
47  switch (TT.getArch()) {
48  case Triple::x86_64:
49  PointerSize = 8;
50  Endianness = support::endianness::little;
51  EdgeKind = jitlink::x86_64::Pointer64;
52  break;
53  case Triple::aarch64:
54  PointerSize = 8;
55  Endianness = support::endianness::little;
56  EdgeKind = jitlink::aarch64::Pointer64;
57  break;
58  default:
59  llvm_unreachable("Unrecognized architecture");
60  }
61 
62  // void *__dso_handle = &__dso_handle;
63  auto G = std::make_unique<jitlink::LinkGraph>(
64  "<DSOHandleMU>", TT, PointerSize, Endianness,
65  jitlink::getGenericEdgeKindName);
66  auto &DSOHandleSection =
67  G->createSection(".data.__dso_handle", MemProt::Read);
68  auto &DSOHandleBlock = G->createContentBlock(
69  DSOHandleSection, getDSOHandleContent(PointerSize), orc::ExecutorAddr(),
70  8, 0);
71  auto &DSOHandleSymbol = G->addDefinedSymbol(
72  DSOHandleBlock, 0, *R->getInitializerSymbol(), DSOHandleBlock.getSize(),
73  jitlink::Linkage::Strong, jitlink::Scope::Default, false, true);
74  DSOHandleBlock.addEdge(EdgeKind, 0, DSOHandleSymbol, 0);
75 
76  ENP.getObjectLinkingLayer().emit(std::move(R), std::move(G));
77  }
78 
79  void discard(const JITDylib &JD, const SymbolStringPtr &Sym) override {}
80

82  static MaterializationUnit::Interface
83  createDSOHandleSectionInterface(ELFNixPlatform &ENP,
84  const SymbolStringPtr &DSOHandleSymbol) {
85  SymbolFlagsMap SymbolFlags;
86  SymbolFlags[DSOHandleSymbol] = JITSymbolFlags::Exported;
87  return MaterializationUnit::Interface(std::move(SymbolFlags),
88  DSOHandleSymbol);
89  }

92  static const char Content[8] = {0};
93  assert(PointerSize <= sizeof Content);
94  return {Content, PointerSize};
95  }

97  ELFNixPlatform &ENP;

105 Expected<std::unique_ptr<ELFNixPlatform>> ELFNixPlatform::Create(
106  ExecutionSession &ES, ObjectLinkingLayer &ObjLinkingLayer,
107  JITDylib &PlatformJD, std::unique_ptr<DefinitionGenerator> OrcRuntime,
108  std::optional<SymbolAliasMap> RuntimeAliases) {
109 
110  auto &EPC = ES.getExecutorProcessControl();
111 
112  // If the target is not supported then bail out immediately.
113  if (!supportedTarget(EPC.getTargetTriple()))
114  return make_error<StringError>("Unsupported ELFNixPlatform triple: " +
115  EPC.getTargetTriple().str(),
116  inconvertibleErrorCode());
117 
118  // Create default aliases if the caller didn't supply any.
119  if (!RuntimeAliases) {
120  auto StandardRuntimeAliases = standardPlatformAliases(ES, PlatformJD);
121  if (!StandardRuntimeAliases)
122  return StandardRuntimeAliases.takeError();
123  RuntimeAliases = std::move(*StandardRuntimeAliases);
124  }
125 
126  // Define the aliases.
127  if (auto Err = PlatformJD.define(symbolAliases(std::move(*RuntimeAliases))))
128  return std::move(Err);
129 
130  // Add JIT-dispatch function support symbols.
131  if (auto Err = PlatformJD.define(absoluteSymbols(
132  {{ES.intern("__orc_rt_jit_dispatch"),
133  {EPC.getJITDispatchInfo().JITDispatchFunction.getValue(),
134  JITSymbolFlags::Exported}},
135  {ES.intern("__orc_rt_jit_dispatch_ctx"),
136  {EPC.getJITDispatchInfo().JITDispatchContext.getValue(),
137  JITSymbolFlags::Exported}}})))
138  return std::move(Err);
139 
140  // Create the instance.
141  Error Err = Error::success();
142  auto P = std::unique_ptr<ELFNixPlatform>(new ELFNixPlatform(
143  ES, ObjLinkingLayer, PlatformJD, std::move(OrcRuntime), Err));
144  if (Err)
145  return std::move(Err);
146  return std::move(P);
147 }

206 Expected<SymbolAliasMap>
207 ELFNixPlatform::standardPlatformAliases(ExecutionSession &ES,
208  JITDylib &PlatformJD) {
209  SymbolAliasMap Aliases;
210  addAliases(ES, Aliases, requiredCXXAliases());
211  addAliases(ES, Aliases, standardRuntimeUtilityAliases());
212 
213  // Determine whether or not the libunwind extended-API function for
214  // dynamically registering an entire .eh_frame section is available.
215  // If it is not, we assume that libgcc_s is being used, and alias to
216  // its __register_frame with the same functionality.
217  auto RTRegisterFrame = ES.intern("__orc_rt_register_eh_frame_section");
218  auto LibUnwindRegisterFrame = ES.intern("__unw_add_dynamic_eh_frame_section");
219  auto RTDeregisterFrame = ES.intern("__orc_rt_deregister_eh_frame_section");
220  auto LibUnwindDeregisterFrame =
221  ES.intern("__unw_remove_dynamic_eh_frame_section");
222  auto SM = ES.lookup(makeJITDylibSearchOrder(&PlatformJD),
223  SymbolLookupSet()
224  .add(LibUnwindRegisterFrame,
225  SymbolLookupFlags::WeaklyReferencedSymbol)
226  .add(LibUnwindDeregisterFrame,
227  SymbolLookupFlags::WeaklyReferencedSymbol));
228  if (!SM) { // Weak-ref means no "missing symbol" errors, so this must be
229  // something more serious that we should report.
230  return SM.takeError();
231  } else if (SM->size() == 2) {
232  LLVM_DEBUG({
233  dbgs() << "Using libunwind " << LibUnwindRegisterFrame
234  << " for unwind info registration\n";
235  });
236  Aliases[std::move(RTRegisterFrame)] = {LibUnwindRegisterFrame,
237  JITSymbolFlags::Exported};
238  Aliases[std::move(RTDeregisterFrame)] = {LibUnwindDeregisterFrame,
239  JITSymbolFlags::Exported};
240  } else {
241  // Since LLVM libunwind is not present, we assume that unwinding
242  // is provided by libgcc
243  LLVM_DEBUG({
244  dbgs() << "Using libgcc __register_frame"
245  << " for unwind info registration\n";
246  });
247  Aliases[std::move(RTRegisterFrame)] = {ES.intern("__register_frame"),
248  JITSymbolFlags::Exported};
249  Aliases[std::move(RTDeregisterFrame)] = {ES.intern("__deregister_frame"),
250  JITSymbolFlags::Exported};
251  }
252 
253  return Aliases;
254 }

290 ELFNixPlatform::ELFNixPlatform(
291  ExecutionSession &ES, ObjectLinkingLayer &ObjLinkingLayer,
292  JITDylib &PlatformJD,
293  std::unique_ptr<DefinitionGenerator> OrcRuntimeGenerator, Error &Err)
294  : ES(ES), ObjLinkingLayer(ObjLinkingLayer),
295  DSOHandleSymbol(ES.intern("__dso_handle")) {
296  ErrorAsOutParameter _(&Err);
297 
298  ObjLinkingLayer.addPlugin(std::make_unique<ELFNixPlatformPlugin>(*this));
299 
300  PlatformJD.addGenerator(std::move(OrcRuntimeGenerator));
301 
302  // PlatformJD hasn't been 'set-up' by the platform yet (since we're creating
303  // the platform now), so set it up.
304  if (auto E2 = setupJITDylib(PlatformJD)) {
305  Err = std::move(E2);
306  return;
307  }
308 
309  RegisteredInitSymbols[&PlatformJD].add(
310  DSOHandleSymbol, SymbolLookupFlags::WeaklyReferencedSymbol);
311 
312  // Associate wrapper function tags with JIT-side function implementations.
313  if (auto E2 = associateRuntimeSupportFunctions(PlatformJD)) {
314  Err = std::move(E2);
315  return;
316  }
317 
318  // Lookup addresses of runtime functions callable by the platform,
319  // call the platform bootstrap function to initialize the platform-state
320  // object in the executor.
321  if (auto E2 = bootstrapELFNixRuntime(PlatformJD)) {
322  Err = std::move(E2);
323  return;
324  }
325 }

327 Error ELFNixPlatform::associateRuntimeSupportFunctions(JITDylib &PlatformJD) {
328  ExecutionSession::JITDispatchHandlerAssociationMap WFs;
329 
330  using GetInitializersSPSSig =
331  SPSExpected<SPSELFNixJITDylibInitializerSequence>(SPSString);
332  WFs[ES.intern("__orc_rt_elfnix_get_initializers_tag")] =
333  ES.wrapAsyncWithSPS<GetInitializersSPSSig>(
334  this, &ELFNixPlatform::rt_getInitializers);
335 
336  using GetDeinitializersSPSSig =
337  SPSExpected<SPSELFJITDylibDeinitializerSequence>(SPSExecutorAddr);
338  WFs[ES.intern("__orc_rt_elfnix_get_deinitializers_tag")] =
339  ES.wrapAsyncWithSPS<GetDeinitializersSPSSig>(
340  this, &ELFNixPlatform::rt_getDeinitializers);
341 
342  using LookupSymbolSPSSig =
343  SPSExpected<SPSExecutorAddr>(SPSExecutorAddr, SPSString);
344  WFs[ES.intern("__orc_rt_elfnix_symbol_lookup_tag")] =
345  ES.wrapAsyncWithSPS<LookupSymbolSPSSig>(this,
346  &ELFNixPlatform::rt_lookupSymbol);
347 
348  return ES.registerJITDispatchHandlers(PlatformJD, std::move(WFs));

351 void ELFNixPlatform::getInitializersBuildSequencePhase(
352  SendInitializerSequenceFn SendResult, JITDylib &JD,
353  std::vector<JITDylibSP> DFSLinkOrder) {
354  ELFNixJITDylibInitializerSequence FullInitSeq;
355  {
356  std::lock_guard<std::mutex> Lock(PlatformMutex);
357  for (auto &InitJD : reverse(DFSLinkOrder)) {
358  LLVM_DEBUG({
359  dbgs() << "ELFNixPlatform: Appending inits for \"" << InitJD->getName()
360  << "\" to sequence\n";
361  });
362  auto ISItr = InitSeqs.find(InitJD.get());
363  if (ISItr != InitSeqs.end()) {
364  FullInitSeq.emplace_back(std::move(ISItr->second));
365  InitSeqs.erase(ISItr);
366  }
367  }
368  }
369 
370  SendResult(std::move(FullInitSeq));
371 }

373 void ELFNixPlatform::getInitializersLookupPhase(
374  SendInitializerSequenceFn SendResult, JITDylib &JD) {
375 
376  auto DFSLinkOrder = JD.getDFSLinkOrder();
377  if (!DFSLinkOrder) {
378  SendResult(DFSLinkOrder.takeError());
379  return;
380  }
381 
382  DenseMap<JITDylib *, SymbolLookupSet> NewInitSymbols;
383  ES.runSessionLocked([&]() {
384  for (auto &InitJD : *DFSLinkOrder) {
385  auto RISItr = RegisteredInitSymbols.find(InitJD.get());
386  if (RISItr != RegisteredInitSymbols.end()) {
387  NewInitSymbols[InitJD.get()] = std::move(RISItr->second);
388  RegisteredInitSymbols.erase(RISItr);
389  }
390  }
391  });
392 
393  // If there are no further init symbols to look up then move on to the next
394  // phase.
395  if (NewInitSymbols.empty()) {
396  getInitializersBuildSequencePhase(std::move(SendResult), JD,
397  std::move(*DFSLinkOrder));
398  return;
399  }
400 
401  // Otherwise issue a lookup and re-run this phase when it completes.
402  lookupInitSymbolsAsync(
403  [this, SendResult = std::move(SendResult), &JD](Error Err) mutable {
404  if (Err)
405  SendResult(std::move(Err));
406  else
407  getInitializersLookupPhase(std::move(SendResult), JD);
408  },
409  ES, std::move(NewInitSymbols));
410 }

513 Error ELFNixPlatform::bootstrapELFNixRuntime(JITDylib &PlatformJD) {
514 
515  std::pair<const char *, ExecutorAddr *> Symbols[] = {
516  {"__orc_rt_elfnix_platform_bootstrap", &orc_rt_elfnix_platform_bootstrap},
517  {"__orc_rt_elfnix_platform_shutdown", &orc_rt_elfnix_platform_shutdown},
518  {"__orc_rt_elfnix_register_object_sections",
519  &orc_rt_elfnix_register_object_sections},
520  {"__orc_rt_elfnix_create_pthread_key",
521  &orc_rt_elfnix_create_pthread_key}};
522 
523  SymbolLookupSet RuntimeSymbols;
524  std::vector<std::pair<SymbolStringPtr, ExecutorAddr *>> AddrsToRecord;
525  for (const auto &KV : Symbols) {
526  auto Name = ES.intern(KV.first);
527  RuntimeSymbols.add(Name);
528  AddrsToRecord.push_back({std::move(Name), KV.second});
529  }
530 
531  auto RuntimeSymbolAddrs = ES.lookup(
532  {{&PlatformJD, JITDylibLookupFlags::MatchAllSymbols}}, RuntimeSymbols);
533  if (!RuntimeSymbolAddrs)
534  return RuntimeSymbolAddrs.takeError();
535 
536  for (const auto &KV : AddrsToRecord) {
537  auto &Name = KV.first;
538  assert(RuntimeSymbolAddrs->count(Name) && "Missing runtime symbol?");
539  KV.second->setValue((*RuntimeSymbolAddrs)[Name].getAddress());
540  }
541 
542  auto PJDDSOHandle = ES.lookup(
543  {{&PlatformJD, JITDylibLookupFlags::MatchAllSymbols}}, DSOHandleSymbol);
544  if (!PJDDSOHandle)
545  return PJDDSOHandle.takeError();
546 
547  if (auto Err = ES.callSPSWrapper<void(uint64_t)>(
548  orc_rt_elfnix_platform_bootstrap, PJDDSOHandle->getAddress()))
549  return Err;
550 
551  // FIXME: Ordering is fuzzy here. We're probably best off saying
552  // "behavior is undefined if code that uses the runtime is added before
553  // the platform constructor returns", then move all this to the constructor.
554  RuntimeBootstrapped = true;
555  std::vector<ELFPerObjectSectionsToRegister> DeferredPOSRs;
556  {
557  std::lock_guard<std::mutex> Lock(PlatformMutex);
558  DeferredPOSRs = std::move(BootstrapPOSRs);
559  }
560 
561  for (auto &D : DeferredPOSRs)
562  if (auto Err = registerPerObjectSections(D))
563  return Err;
564 
565  return Error::success();
566 }

568 Error ELFNixPlatform::registerInitInfo(
569  JITDylib &JD, ArrayRef<jitlink::Section *> InitSections) {
570 
571  std::unique_lock<std::mutex> Lock(PlatformMutex);
572 
573  ELFNixJITDylibInitializers *InitSeq = nullptr;
574  {
575  auto I = InitSeqs.find(&JD);
576  if (I == InitSeqs.end()) {
577  // If there's no init sequence entry yet then we need to look up the
578  // header symbol to force creation of one.
579  Lock.unlock();
580 
581  auto SearchOrder =
582  JD.withLinkOrderDo([](const JITDylibSearchOrder &SO) { return SO; });
583  if (auto Err = ES.lookup(SearchOrder, DSOHandleSymbol).takeError())
584  return Err;
585 
586  Lock.lock();
587  I = InitSeqs.find(&JD);
588  assert(I != InitSeqs.end() &&
589  "Entry missing after header symbol lookup?");
590  }
591  InitSeq = &I->second;
592  }
593 
594  for (auto *Sec : InitSections) {
595  // FIXME: Avoid copy here.
596  jitlink::SectionRange R(*Sec);
597  InitSeq->InitSections[Sec->getName()].push_back(
598  {ExecutorAddr(R.getStart()), ExecutorAddr(R.getEnd())});
599  }
600 
601  return Error::success();
602 }

604 Error ELFNixPlatform::registerPerObjectSections(
605  const ELFPerObjectSectionsToRegister &POSR) {
606 
607  if (!orc_rt_elfnix_register_object_sections)

707 void ELFNixPlatform::ELFNixPlatformPlugin::addEHAndTLVSupportPasses(
708  MaterializationResponsibility &MR, jitlink::PassConfiguration &Config) {
709 
710  // Insert TLV lowering at the start of the PostPrunePasses, since we want
711  // it to run before GOT/PLT lowering.
712 
713  // TODO: Check that before the fixTLVSectionsAndEdges pass, the GOT/PLT build
714  // pass has done. Because the TLS descriptor need to be allocate in GOT.
715  Config.PostPrunePasses.push_back(
716  [this, &JD = MR.getTargetJITDylib()](jitlink::LinkGraph &G) {
717  return fixTLVSectionsAndEdges(G, JD);
718  });
719 
720  // Add a pass to register the final addresses of the eh-frame and TLV sections
721  // with the runtime.
722  Config.PostFixupPasses.push_back([this](jitlink::LinkGraph &G) -> Error {
723  ELFPerObjectSectionsToRegister POSR;
724 
725  if (auto *EHFrameSection = G.findSectionByName(ELFEHFrameSectionName)) {
726  jitlink::SectionRange R(*EHFrameSection);
727  if (!R.empty())
728  POSR.EHFrameSection = {ExecutorAddr(R.getStart()),
729  ExecutorAddr(R.getEnd())};
730  }
731 
732  // Get a pointer to the thread data section if there is one. It will be used
733  // below.
734  jitlink::Section *ThreadDataSection =
735  G.findSectionByName(ELFThreadDataSectionName);
736 
737  // Handle thread BSS section if there is one.
738  if (auto *ThreadBSSSection = G.findSectionByName(ELFThreadBSSSectionName)) {
739  // If there's already a thread data section in this graph then merge the
740  // thread BSS section content into it, otherwise just treat the thread
741  // BSS section as the thread data section.
742  if (ThreadDataSection)
743  G.mergeSections(*ThreadDataSection, *ThreadBSSSection);
744  else
745  ThreadDataSection = ThreadBSSSection;
746  }
747 
748  // Having merged thread BSS (if present) and thread data (if present),
749  // record the resulting section range.
750  if (ThreadDataSection) {
751  jitlink::SectionRange R(*ThreadDataSection);
752  if (!R.empty())
753  POSR.ThreadDataSection = {ExecutorAddr(R.getStart()),
754  ExecutorAddr(R.getEnd())};
755  }
756 
757  if (POSR.EHFrameSection.Start || POSR.ThreadDataSection.Start) {
758 
759  // If we're still bootstrapping the runtime then just record this
760  // frame for now.
761  if (!MP.RuntimeBootstrapped) {
762  std::lock_guard<std::mutex> Lock(MP.PlatformMutex);
763  MP.BootstrapPOSRs.push_back(POSR);
764  return Error::success();
765  }
766 
767  // Otherwise register it immediately.
768  if (auto Err = MP.registerPerObjectSections(POSR))
769  return Err;
770  }
771 
772  return Error::success();
773  });
774 }

776 Error ELFNixPlatform::ELFNixPlatformPlugin::preserveInitSections(
777  jitlink::LinkGraph &G, MaterializationResponsibility &MR) {
778 
779  JITLinkSymbolSet InitSectionSymbols;
780  for (auto &InitSection : G.sections()) {
781  // Skip non-init sections.
782  if (!isELFInitializerSection(InitSection.getName()))
783  continue;
784 
785  // Make a pass over live symbols in the section: those blocks are already
786  // preserved.
787  DenseSet<jitlink::Block *> AlreadyLiveBlocks;
788  for (auto &Sym : InitSection.symbols()) {
789  auto &B = Sym->getBlock();
790  if (Sym->isLive() && Sym->getOffset() == 0 &&
791  Sym->getSize() == B.getSize() && !AlreadyLiveBlocks.count(&B)) {
792  InitSectionSymbols.insert(Sym);
793  AlreadyLiveBlocks.insert(&B);
794  }
795  }
796 
797  // Add anonymous symbols to preserve any not-already-preserved blocks.
798  for (auto *B : InitSection.blocks())
799  if (!AlreadyLiveBlocks.count(B))
800  InitSectionSymbols.insert(
801  &G.addAnonymousSymbol(*B, 0, B->getSize(), false, true));
802  }
803 
804  if (!InitSectionSymbols.empty()) {
805  std::lock_guard<std::mutex> Lock(PluginMutex);
806  InitSymbolDeps[&MR] = std::move(InitSectionSymbols);
807  }
808 
809  return Error::success();
810 }

838 Error ELFNixPlatform::ELFNixPlatformPlugin::fixTLVSectionsAndEdges(
839  jitlink::LinkGraph &G, JITDylib &JD) {
840 
841  for (auto *Sym : G.external_symbols()) {
842  if (Sym->getName() == "__tls_get_addr") {
843  Sym->setName("___orc_rt_elfnix_tls_get_addr");
844  } else if (Sym->getName() == "__tlsdesc_resolver") {
845  Sym->setName("___orc_rt_elfnix_tlsdesc_resolver");
846  }
847  }
848 
849  auto *TLSInfoEntrySection = G.findSectionByName("$__TLSINFO");
850 
851  if (TLSInfoEntrySection) {
852  std::optional<uint64_t> Key;
853  {
854  std::lock_guard<std::mutex> Lock(MP.PlatformMutex);
855  auto I = MP.JITDylibToPThreadKey.find(&JD);
856  if (I != MP.JITDylibToPThreadKey.end())
857  Key = I->second;
858  }
859  if (!Key) {
860  if (auto KeyOrErr = MP.createPThreadKey())
861  Key = *KeyOrErr;
862  else
863  return KeyOrErr.takeError();
864  }
865 
866  uint64_t PlatformKeyBits =
867  support::endian::byte_swap(*Key, G.getEndianness());
868 
869  for (auto *B : TLSInfoEntrySection->blocks()) {
870  // FIXME: The TLS descriptor byte length may different with different
871  // ISA
872  assert(B->getSize() == (G.getPointerSize() * 2) &&
873  "TLS descriptor must be 2 words length");
874  auto TLSInfoEntryContent = B->getMutableContent(G);
875  memcpy(TLSInfoEntryContent.data(), &PlatformKeyBits, G.getPointerSize());
876  }
877  }
878 
879  return Error::success();
880 }
