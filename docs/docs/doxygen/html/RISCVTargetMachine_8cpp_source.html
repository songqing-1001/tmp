115 const RISCVSubtarget *
116 RISCVTargetMachine::getSubtargetImpl(const Function &F) const {
117  Attribute CPUAttr = F.getFnAttribute("target-cpu");
118  Attribute TuneAttr = F.getFnAttribute("tune-cpu");
119  Attribute FSAttr = F.getFnAttribute("target-features");
120 
121  std::string CPU =
122  CPUAttr.isValid() ? CPUAttr.getValueAsString().str() : TargetCPU;
123  std::string TuneCPU =
124  TuneAttr.isValid() ? TuneAttr.getValueAsString().str() : CPU;
125  std::string FS =
126  FSAttr.isValid() ? FSAttr.getValueAsString().str() : TargetFS;
127 
128  unsigned RVVBitsMin = RVVVectorBitsMinOpt;
129  unsigned RVVBitsMax = RVVVectorBitsMaxOpt;
130 
131  Attribute VScaleRangeAttr = F.getFnAttribute(Attribute::VScaleRange);
132  if (VScaleRangeAttr.isValid()) {
133  if (!RVVVectorBitsMinOpt.getNumOccurrences())
134  RVVBitsMin = VScaleRangeAttr.getVScaleRangeMin() * RISCV::RVVBitsPerBlock;
135  std::optional<unsigned> VScaleMax = VScaleRangeAttr.getVScaleRangeMax();
136  if (VScaleMax.has_value() && !RVVVectorBitsMaxOpt.getNumOccurrences())
137  RVVBitsMax = *VScaleMax * RISCV::RVVBitsPerBlock;
138  }
139 
140  if (RVVBitsMin != -1U) {
141  // FIXME: Change to >= 32 when VLEN = 32 is supported.
142  assert((RVVBitsMin == 0 || (RVVBitsMin >= 64 && RVVBitsMin <= 65536 &&
143  isPowerOf2_32(RVVBitsMin))) &&
144  "V or Zve* extension requires vector length to be in the range of "
145  "64 to 65536 and a power 2!");
146  assert((RVVBitsMax >= RVVBitsMin || RVVBitsMax == 0) &&
147  "Minimum V extension vector length should not be larger than its "
148  "maximum!");
149  }
150  assert((RVVBitsMax == 0 || (RVVBitsMax >= 64 && RVVBitsMax <= 65536 &&
151  isPowerOf2_32(RVVBitsMax))) &&
152  "V or Zve* extension requires vector length to be in the range of "
153  "64 to 65536 and a power 2!");
154 
155  if (RVVBitsMin != -1U) {
156  if (RVVBitsMax != 0) {
157  RVVBitsMin = std::min(RVVBitsMin, RVVBitsMax);
158  RVVBitsMax = std::max(RVVBitsMin, RVVBitsMax);
159  }
160 
161  RVVBitsMin = llvm::bit_floor(
162  (RVVBitsMin < 64 || RVVBitsMin > 65536) ? 0 : RVVBitsMin);
163  }
164  RVVBitsMax =
165  llvm::bit_floor((RVVBitsMax < 64 || RVVBitsMax > 65536) ? 0 : RVVBitsMax);
166 
167  SmallString<512> Key;
168  Key += "RVVMin";
169  Key += std::to_string(RVVBitsMin);
170  Key += "RVVMax";
171  Key += std::to_string(RVVBitsMax);
172  Key += CPU;
173  Key += TuneCPU;
174  Key += FS;
175  auto &I = SubtargetMap[Key];
176  if (!I) {
177  // This needs to be done before we create a new subtarget since any
178  // creation will depend on the TM and the code generation flags on the
179  // function that reside in TargetOptions.
180  resetTargetOptions(F);
181  auto ABIName = Options.MCOptions.getABIName();
182  if (const MDString *ModuleTargetABI = dyn_cast_or_null<MDString>(
183  F.getParent()->getModuleFlag("target-abi"))) {
184  auto TargetABI = RISCVABI::getTargetABI(ABIName);
185  if (TargetABI != RISCVABI::ABI_Unknown &&
186  ModuleTargetABI->getString() != ABIName) {
187  report_fatal_error("-target-abi option != target-abi module flag");
188  }
189  ABIName = ModuleTargetABI->getString();
190  }
191  I = std::make_unique<RISCVSubtarget>(
192  TargetTriple, CPU, TuneCPU, FS, ABIName, RVVBitsMin, RVVBitsMax, *this);
193  }
194  return I.get();
195 }

282 bool RISCVPassConfig::addPreISel() {
283  if (TM->getOptLevel() != CodeGenOpt::None) {
284  // Add a barrier before instruction selection so that we will not get
285  // deleted block address after enabling default outlining. See D99707 for
286  // more details.
287  addPass(createBarrierNoopPass());
288  }
289 
290  if (EnableGlobalMerge == cl::BOU_TRUE) {
291  addPass(createGlobalMergePass(TM, /* MaxOffset */ 2047,
292  /* OnlyOptimizeForSize */ false,
293  /* MergeExternalByDefault */ true));
294  }
295 
296  return false;
297 }
