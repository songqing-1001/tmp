77 void PassManagerBuilder::populateFunctionPassManager(
78  legacy::FunctionPassManager &FPM) {
79  // Add LibraryInfo if we have some.
80  if (LibraryInfo)
81  FPM.add(new TargetLibraryInfoWrapperPass(*LibraryInfo));
82 
83  if (OptLevel == 0) return;
84 
85  addInitialAliasAnalysisPasses(FPM);
86 
87  // Lower llvm.expect to metadata before attempting transforms.
88  // Compare/branch metadata may alter the behavior of passes like SimplifyCFG.
89  FPM.add(createLowerExpectIntrinsicPass());
90  FPM.add(createCFGSimplificationPass());
91  FPM.add(createSROAPass());
92  FPM.add(createEarlyCSEPass());
93 }

95 void PassManagerBuilder::addFunctionSimplificationPasses(
96  legacy::PassManagerBase &MPM) {
97  // Start of function pass.
98  // Break up aggregate allocas, using SSAUpdater.
99  assert(OptLevel >= 1 && "Calling function optimizer with no optimization level!");
100  MPM.add(createSROAPass());
101  MPM.add(createEarlyCSEPass(true /* Enable mem-ssa. */)); // Catch trivial redundancies
102 
103  if (OptLevel > 1) {
104  // Speculative execution if the target has divergent branches; otherwise nop.
105  MPM.add(createSpeculativeExecutionIfHasBranchDivergencePass());
106 
107  MPM.add(createJumpThreadingPass()); // Thread jumps.
108  MPM.add(createCorrelatedValuePropagationPass()); // Propagate conditionals
109  }
110  MPM.add(
111  createCFGSimplificationPass(SimplifyCFGOptions().convertSwitchRangeToICmp(
112  true))); // Merge & remove BBs
113  // Combine silly seq's
114  MPM.add(createInstructionCombiningPass());
115  if (SizeLevel == 0)
116  MPM.add(createLibCallsShrinkWrapPass());
117 
118  // TODO: Investigate the cost/benefit of tail call elimination on debugging.
119  if (OptLevel > 1)
120  MPM.add(createTailCallEliminationPass()); // Eliminate tail calls
121  MPM.add(
122  createCFGSimplificationPass(SimplifyCFGOptions().convertSwitchRangeToICmp(
123  true))); // Merge & remove BBs
124  MPM.add(createReassociatePass()); // Reassociate expressions
125 
126  // Begin the loop pass pipeline.
127 
128  // The simple loop unswitch pass relies on separate cleanup passes. Schedule
129  // them first so when we re-process a loop they run before other loop
130  // passes.
131  MPM.add(createLoopInstSimplifyPass());
132  MPM.add(createLoopSimplifyCFGPass());
133 
134  // Try to remove as much code from the loop header as possible,
135  // to reduce amount of IR that will have to be duplicated. However,
136  // do not perform speculative hoisting the first time as LICM
137  // will destroy metadata that may not need to be destroyed if run
138  // after loop rotation.
139  // TODO: Investigate promotion cap for O1.
140  MPM.add(createLICMPass(LicmMssaOptCap, LicmMssaNoAccForPromotionCap,
141  /*AllowSpeculation=*/false));
142  // Rotate Loop - disable header duplication at -Oz
143  MPM.add(createLoopRotatePass(SizeLevel == 2 ? 0 : -1, false));
144  // TODO: Investigate promotion cap for O1.
145  MPM.add(createLICMPass(LicmMssaOptCap, LicmMssaNoAccForPromotionCap,
146  /*AllowSpeculation=*/true));
147  MPM.add(createSimpleLoopUnswitchLegacyPass(OptLevel == 3));
148  // FIXME: We break the loop pass pipeline here in order to do full
149  // simplifycfg. Eventually loop-simplifycfg should be enhanced to replace the
150  // need for this.
151  MPM.add(createCFGSimplificationPass(
152  SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
153  MPM.add(createInstructionCombiningPass());
154  // We resume loop passes creating a second loop pipeline here.
155  MPM.add(createLoopIdiomPass()); // Recognize idioms like memset.
156  MPM.add(createIndVarSimplifyPass()); // Canonicalize indvars
157 
158  // Unroll small loops and perform peeling.
159  MPM.add(createSimpleLoopUnrollPass(OptLevel, DisableUnrollLoops,
160  ForgetAllSCEVInLoopUnroll));
161  // This ends the loop pass pipelines.
162 
163  // Break up allocas that may now be splittable after loop unrolling.
164  MPM.add(createSROAPass());
165 
166  if (OptLevel > 1) {
167  MPM.add(createMergedLoadStoreMotionPass()); // Merge ld/st in diamonds
168  MPM.add(createGVNPass(DisableGVNLoadPRE)); // Remove redundancies
169  }
170  MPM.add(createSCCPPass()); // Constant prop with SCCP
171 
172  // Delete dead bit computations (instcombine runs after to fold away the dead
173  // computations, and then ADCE will run later to exploit any new DCE
174  // opportunities that creates).
175  MPM.add(createBitTrackingDCEPass()); // Delete dead bit computations
176 
177  // Run instcombine after redundancy elimination to exploit opportunities
178  // opened up by them.
179  MPM.add(createInstructionCombiningPass());
180  if (OptLevel > 1) {
181  MPM.add(createJumpThreadingPass()); // Thread jumps
182  MPM.add(createCorrelatedValuePropagationPass());
183  }
184  MPM.add(createAggressiveDCEPass()); // Delete dead instructions
185 
186  MPM.add(createMemCpyOptPass()); // Remove memcpy / form memset
187  // TODO: Investigate if this is too expensive at O1.
188  if (OptLevel > 1) {
189  MPM.add(createDeadStoreEliminationPass()); // Delete dead stores
190  MPM.add(createLICMPass(LicmMssaOptCap, LicmMssaNoAccForPromotionCap,
191  /*AllowSpeculation=*/true));
192  }
193 
194  // Merge & remove BBs and sink & hoist common instructions.
195  MPM.add(createCFGSimplificationPass(
196  SimplifyCFGOptions().hoistCommonInsts(true).sinkCommonInsts(true)));
197  // Clean up after everything.
198  MPM.add(createInstructionCombiningPass());
199 }

201 /// FIXME: Should LTO cause any differences to this set of passes?
202 void PassManagerBuilder::addVectorPasses(legacy::PassManagerBase &PM,
203  bool IsFullLTO) {
204  PM.add(createLoopVectorizePass(!LoopsInterleaved, !LoopVectorize));
205 
206  if (IsFullLTO) {
207  // The vectorizer may have significantly shortened a loop body; unroll
208  // again. Unroll small loops to hide loop backedge latency and saturate any
209  // parallel execution resources of an out-of-order processor. We also then
210  // need to clean up redundancies and loop invariant code.
211  // FIXME: It would be really good to use a loop-integrated instruction
212  // combiner for cleanup here so that the unrolling and LICM can be pipelined
213  // across the loop nests.
214  PM.add(createLoopUnrollPass(OptLevel, DisableUnrollLoops,
215  ForgetAllSCEVInLoopUnroll));
216  }
217 
218  // Cleanup after the loop optimization passes.
219  PM.add(createInstructionCombiningPass());
220 
221  // Now that we've formed fast to execute loop structures, we do further
222  // optimizations. These are run afterward as they might block doing complex
223  // analyses and transforms such as what are needed for loop vectorization.
224 
225  // Cleanup after loop vectorization, etc. Simplification passes like CVP and
226  // GVN, loop transforms, and others have already run, so it's now better to
227  // convert to more optimized IR using more aggressive simplify CFG options.
228  // The extra sinking transform can create larger basic blocks, so do this
229  // before SLP vectorization.
230  PM.add(createCFGSimplificationPass(SimplifyCFGOptions()
231  .forwardSwitchCondToPhi(true)
232  .convertSwitchRangeToICmp(true)
233  .convertSwitchToLookupTable(true)
234  .needCanonicalLoops(false)
235  .hoistCommonInsts(true)
236  .sinkCommonInsts(true)));
237 
238  if (IsFullLTO) {
239  PM.add(createSCCPPass()); // Propagate exposed constants
240  PM.add(createInstructionCombiningPass()); // Clean up again
241  PM.add(createBitTrackingDCEPass());
242  }
243 
244  // Optimize parallel scalar instruction chains into SIMD instructions.
245  if (SLPVectorize) {
246  PM.add(createSLPVectorizerPass());
247  }
248 
249  if (!IsFullLTO) {
250  PM.add(createInstructionCombiningPass());
251 
252  // Unroll small loops
253  PM.add(createLoopUnrollPass(OptLevel, DisableUnrollLoops,
254  ForgetAllSCEVInLoopUnroll));
255 
256  if (!DisableUnrollLoops) {
257  // LoopUnroll may generate some redundency to cleanup.
258  PM.add(createInstructionCombiningPass());
259 
260  // Runtime unrolling will introduce runtime check in loop prologue. If the
261  // unrolled loop is a inner loop, then the prologue will be inside the
262  // outer loop. LICM pass can help to promote the runtime check out if the
263  // checked value is loop invariant.
264  PM.add(createLICMPass(LicmMssaOptCap, LicmMssaNoAccForPromotionCap,
265  /*AllowSpeculation=*/true));
266  }
267  }
268 
269  // After vectorization and unrolling, assume intrinsics may tell us more
270  // about pointer alignments.
271  PM.add(createAlignmentFromAssumptionsPass());
272 
273  if (IsFullLTO)
274  PM.add(createInstructionCombiningPass());
275 }

277 void PassManagerBuilder::populateModulePassManager(
278  legacy::PassManagerBase &MPM) {
279  MPM.add(createAnnotation2MetadataLegacyPass());
280 
281  // Allow forcing function attributes as a debugging and tuning aid.
282  MPM.add(createForceFunctionAttrsLegacyPass());
283 
284  // If all optimizations are disabled, just run the always-inline pass and,
285  // if enabled, the function merging pass.
286  if (OptLevel == 0) {
287  if (Inliner) {
288  MPM.add(Inliner);
289  Inliner = nullptr;
290  }
291 
292  return;
293  }
294 
295  // Add LibraryInfo if we have some.
296  if (LibraryInfo)
297  MPM.add(new TargetLibraryInfoWrapperPass(*LibraryInfo));
298 
299  addInitialAliasAnalysisPasses(MPM);
300 
301  // Infer attributes about declarations if possible.
302  MPM.add(createInferFunctionAttrsLegacyPass());
303 
304  if (OptLevel > 2)
305  MPM.add(createCallSiteSplittingPass());
306 
307  // Promote any localized global vars.
308  MPM.add(createPromoteMemoryToRegisterPass());
309 
310  MPM.add(createDeadArgEliminationPass()); // Dead argument elimination
311 
312  MPM.add(createInstructionCombiningPass()); // Clean up after IPCP & DAE
313  MPM.add(
314  createCFGSimplificationPass(SimplifyCFGOptions().convertSwitchRangeToICmp(
315  true))); // Clean up after IPCP & DAE
316 
317  // We add a module alias analysis pass here. In part due to bugs in the
318  // analysis infrastructure this "works" in that the analysis stays alive
319  // for the entire SCC pass run below.
320  MPM.add(createGlobalsAAWrapperPass());
321 
322  // Start of CallGraph SCC passes.
323  bool RunInliner = false;
324  if (Inliner) {
325  MPM.add(Inliner);
326  Inliner = nullptr;
327  RunInliner = true;
328  }
329 
330  MPM.add(createPostOrderFunctionAttrsLegacyPass());
331 
332  addFunctionSimplificationPasses(MPM);
333 
334  // FIXME: This is a HACK! The inliner pass above implicitly creates a CGSCC
335  // pass manager that we are specifically trying to avoid. To prevent this
336  // we must insert a no-op module pass to reset the pass manager.
337  MPM.add(createBarrierNoopPass());
338 
339  if (OptLevel > 1)
340  // Remove avail extern fns and globals definitions if we aren't
341  // compiling an object file for later LTO. For LTO we want to preserve
342  // these so they are eligible for inlining at link-time. Note if they
343  // are unreferenced they will be removed by GlobalDCE later, so
344  // this only impacts referenced available externally globals.
345  // Eventually they will be suppressed during codegen, but eliminating
346  // here enables more opportunity for GlobalDCE as it may make
347  // globals referenced by available external functions dead
348  // and saves running remaining passes on the eliminated functions.
349  MPM.add(createEliminateAvailableExternallyPass());
350 
351  // The inliner performs some kind of dead code elimination as it goes,
352  // but there are cases that are not really caught by it. We might
353  // at some point consider teaching the inliner about them, but it
354  // is OK for now to run GlobalOpt + GlobalDCE in tandem as their
355  // benefits generally outweight the cost, making the whole pipeline
356  // faster.
357  if (RunInliner) {
358  MPM.add(createGlobalDCEPass());
359  }
360 
361  // We add a fresh GlobalsModRef run at this point. This is particularly
362  // useful as the above will have inlined, DCE'ed, and function-attr
363  // propagated everything. We should at this point have a reasonably minimal
364  // and richly annotated call graph. By computing aliasing and mod/ref
365  // information for all local globals here, the late loop passes and notably
366  // the vectorizer will be able to use them to help recognize vectorizable
367  // memory operations.
368  //
369  // Note that this relies on a bug in the pass manager which preserves
370  // a module analysis into a function pass pipeline (and throughout it) so
371  // long as the first function pass doesn't invalidate the module analysis.
372  // Thus both Float2Int and LoopRotate have to preserve AliasAnalysis for
373  // this to work. Fortunately, it is trivial to preserve AliasAnalysis
374  // (doing nothing preserves it as it is required to be conservatively
375  // correct in the face of IR changes).
376  MPM.add(createGlobalsAAWrapperPass());
377 
378  MPM.add(createFloat2IntPass());
379  MPM.add(createLowerConstantIntrinsicsPass());
380 
381  // Re-rotate loops in all our loop nests. These may have fallout out of
382  // rotated form due to GVN or other transformations, and the vectorizer relies
383  // on the rotated form. Disable header duplication at -Oz.
384  MPM.add(createLoopRotatePass(SizeLevel == 2 ? 0 : -1, false));
385 
386  addVectorPasses(MPM, /* IsFullLTO */ false);
387 
388  // GlobalOpt already deletes dead functions and globals, at -O2 try a
389  // late pass of GlobalDCE. It is capable of deleting dead cycles.
390  if (OptLevel > 1) {
391  MPM.add(createGlobalDCEPass()); // Remove dead fns and globals.
392  MPM.add(createConstantMergePass()); // Merge dup global constants
393  }
394 
395  // LoopSink pass sinks instructions hoisted by LICM, which serves as a
396  // canonicalization pass that enables other optimizations. As a result,
397  // LoopSink pass needs to be a very late IR pass to avoid undoing LICM
398  // result too early.
399  MPM.add(createLoopSinkPass());
400  // Get rid of LCSSA nodes.
401  MPM.add(createInstSimplifyLegacyPass());
402 
403  // This hoists/decomposes div/rem ops. It should run after other sink/hoist
404  // passes to avoid re-sinking, but before SimplifyCFG because it can allow
405  // flattening of blocks.
406  MPM.add(createDivRemPairsPass());
407 
408  // LoopSink (and other loop passes since the last simplifyCFG) might have
409  // resulted in single-entry-single-exit or empty blocks. Clean up the CFG.
410  MPM.add(createCFGSimplificationPass(
411  SimplifyCFGOptions().convertSwitchRangeToICmp(true)));
412 }
