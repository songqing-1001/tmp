19  DWARFDie TD = getDIEForOffset(getTypeOffset() + getOffset());

21  int OffsetDumpWidth = 2 * dwarf::getDwarfOffsetByteSize(getFormat());
22 
23  if (DumpOpts.SummarizeTypes) {
24  OS << "name = '" << Name << "'"
25  << ", type_signature = " << format("0x%016" PRIx64, getTypeHash())

29  }
30

35  if (getVersion() >= 5)

37  OS << ", abbr_offset = " << format("0x%04" PRIx64, getAbbrOffset());

43  << ", type_offset = " << format("0x%04" PRIx64, getTypeOffset())
