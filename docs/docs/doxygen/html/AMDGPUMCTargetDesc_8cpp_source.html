123 class AMDGPUMCInstrAnalysis : public MCInstrAnalysis {
124 public:
125  explicit AMDGPUMCInstrAnalysis(const MCInstrInfo *Info)
126  : MCInstrAnalysis(Info) {}
127 
128  bool evaluateBranch(const MCInst &Inst, uint64_t Addr, uint64_t Size,
129  uint64_t &Target) const override {
130  if (Inst.getNumOperands() == 0 || !Inst.getOperand(0).isImm() ||
131  Info->get(Inst.getOpcode()).operands()[0].OperandType !=
132  MCOI::OPERAND_PCREL)
133  return false;
134 
135  int64_t Imm = Inst.getOperand(0).getImm();
136  // Our branches take a simm16, but we need two extra bits to account for
137  // the factor of 4.
138  APInt SignedOffset(18, Imm * 4, true);
139  Target = (SignedOffset.sext(64) + Addr + Size).getZExtValue();
140  return true;
141  }
142 };

154  for (Target *T : {&getTheAMDGPUTarget(), &getTheGCNTarget()}) {
155  RegisterMCAsmInfo<AMDGPUMCAsmInfo> X(*T);
156 
157  TargetRegistry::RegisterMCRegInfo(*T, createAMDGPUMCRegisterInfo);
158  TargetRegistry::RegisterMCSubtargetInfo(*T, createAMDGPUMCSubtargetInfo);
159  TargetRegistry::RegisterMCInstPrinter(*T, createAMDGPUMCInstPrinter);
160  TargetRegistry::RegisterMCInstrAnalysis(*T, createAMDGPUMCInstrAnalysis);
161  TargetRegistry::RegisterMCAsmBackend(*T, createAMDGPUAsmBackend);
162  TargetRegistry::RegisterELFStreamer(*T, createMCStreamer);
163  }
164 
165  // R600 specific registration
166  TargetRegistry::RegisterMCCodeEmitter(getTheAMDGPUTarget(),
167  createR600MCCodeEmitter);
168  TargetRegistry::RegisterObjectTargetStreamer(
169  getTheAMDGPUTarget(), createAMDGPUObjectTargetStreamer);

171  // GCN specific registration
172  TargetRegistry::RegisterMCCodeEmitter(getTheGCNTarget(),
173  createSIMCCodeEmitter);

179  TargetRegistry::RegisterNullTargetStreamer(getTheGCNTarget(),
180  createAMDGPUNullTargetStreamer);
181 }
