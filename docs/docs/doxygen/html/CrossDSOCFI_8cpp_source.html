48 /// Extracts a numeric type identifier from an MDNode containing type metadata.
49 ConstantInt *CrossDSOCFI::extractNumericTypeId(MDNode *MD) {
50  // This check excludes vtables for classes inside anonymous namespaces.
51  auto TM = dyn_cast<ValueAsMetadata>(MD->getOperand(1));
52  if (!TM)
53  return nullptr;
54  auto C = dyn_cast_or_null<ConstantInt>(TM->getValue());
55  if (!C) return nullptr;
56  // We are looking for i64 constants.
57  if (C->getBitWidth() != 64) return nullptr;
58 
59  return C;
60 }

62 /// buildCFICheck - emits __cfi_check for the current module.
63 void CrossDSOCFI::buildCFICheck(Module &M) {
64  // FIXME: verify that __cfi_check ends up near the end of the code section,
65  // but before the jump slots created in LowerTypeTests.
66  SetVector<uint64_t> TypeIds;
67  SmallVector<MDNode *, 2> Types;
68  for (GlobalObject &GO : M.global_objects()) {
69  Types.clear();
70  GO.getMetadata(LLVMContext::MD_type, Types);
71  for (MDNode *Type : Types)
72  if (ConstantInt *TypeId = extractNumericTypeId(Type))
73  TypeIds.insert(TypeId->getZExtValue());
74  }
75 
76  NamedMDNode *CfiFunctionsMD = M.getNamedMetadata("cfi.functions");
77  if (CfiFunctionsMD) {
78  for (auto *Func : CfiFunctionsMD->operands()) {
79  assert(Func->getNumOperands() >= 2);
80  for (unsigned I = 2; I < Func->getNumOperands(); ++I)
81  if (ConstantInt *TypeId =
82  extractNumericTypeId(cast<MDNode>(Func->getOperand(I).get())))
83  TypeIds.insert(TypeId->getZExtValue());
84  }
85  }
86 
87  LLVMContext &Ctx = M.getContext();
88  FunctionCallee C = M.getOrInsertFunction(
89  "__cfi_check", Type::getVoidTy(Ctx), Type::getInt64Ty(Ctx),
90  Type::getInt8PtrTy(Ctx), Type::getInt8PtrTy(Ctx));
91  Function *F = cast<Function>(C.getCallee());
92  // Take over the existing function. The frontend emits a weak stub so that the
93  // linker knows about the symbol; this pass replaces the function body.
94  F->deleteBody();
95  F->setAlignment(Align(4096));
96 
97  Triple T(M.getTargetTriple());
98  if (T.isARM() || T.isThumb())
99  F->addFnAttr("target-features", "+thumb-mode");
100 
101  auto args = F->arg_begin();
102  Value &CallSiteTypeId = *(args++);
103  CallSiteTypeId.setName("CallSiteTypeId");
104  Value &Addr = *(args++);
105  Addr.setName("Addr");
106  Value &CFICheckFailData = *(args++);
107  CFICheckFailData.setName("CFICheckFailData");
108  assert(args == F->arg_end());
109 
110  BasicBlock *BB = BasicBlock::Create(Ctx, "entry", F);
111  BasicBlock *ExitBB = BasicBlock::Create(Ctx, "exit", F);
112 
113  BasicBlock *TrapBB = BasicBlock::Create(Ctx, "fail", F);
114  IRBuilder<> IRBFail(TrapBB);
115  FunctionCallee CFICheckFailFn =
116  M.getOrInsertFunction("__cfi_check_fail", Type::getVoidTy(Ctx),
117  Type::getInt8PtrTy(Ctx), Type::getInt8PtrTy(Ctx));
118  IRBFail.CreateCall(CFICheckFailFn, {&CFICheckFailData, &Addr});
119  IRBFail.CreateBr(ExitBB);
120 
121  IRBuilder<> IRBExit(ExitBB);
122  IRBExit.CreateRetVoid();
123 
124  IRBuilder<> IRB(BB);
125  SwitchInst *SI = IRB.CreateSwitch(&CallSiteTypeId, TrapBB, TypeIds.size());
126  for (uint64_t TypeId : TypeIds) {
127  ConstantInt *CaseTypeId = ConstantInt::get(Type::getInt64Ty(Ctx), TypeId);
128  BasicBlock *TestBB = BasicBlock::Create(Ctx, "test", F);
129  IRBuilder<> IRBTest(TestBB);
130  Function *BitsetTestFn = Intrinsic::getDeclaration(&M, Intrinsic::type_test);
131 
132  Value *Test = IRBTest.CreateCall(
133  BitsetTestFn, {&Addr, MetadataAsValue::get(
134  Ctx, ConstantAsMetadata::get(CaseTypeId))});
135  BranchInst *BI = IRBTest.CreateCondBr(Test, ExitBB, TrapBB);
136  BI->setMetadata(LLVMContext::MD_prof, VeryLikelyWeights);
137 
138  SI->addCase(CaseTypeId, TestBB);
139  ++NumTypeIds;
140  }
141 }

143 bool CrossDSOCFI::runOnModule(Module &M) {
144  VeryLikelyWeights =
145  MDBuilder(M.getContext()).createBranchWeights((1U << 20) - 1, 1);
146  if (M.getModuleFlag("Cross-DSO CFI") == nullptr)
147  return false;
148  buildCFICheck(M);
149  return true;
150 }
