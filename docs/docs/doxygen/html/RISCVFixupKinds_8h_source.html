117  llvm_unreachable("unsupported fixup size");

122  return std::make_pair(MCFixupKind(RISCV::fixup_riscv_add_16),
123  MCFixupKind(RISCV::fixup_riscv_sub_16));
124  case 4:

126  MCFixupKind(RISCV::fixup_riscv_sub_32));
