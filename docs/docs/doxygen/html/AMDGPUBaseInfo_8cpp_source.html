39 /// \returns Bit mask for given bit \p Shift and bit \p Width.
40 unsigned getBitMask(unsigned Shift, unsigned Width) {
41  return ((1 << Width) - 1) << Shift;
42 }

44 /// Packs \p Src into \p Dst for given bit \p Shift and bit \p Width.
45 ///
46 /// \returns Packed \p Dst.
47 unsigned packBits(unsigned Src, unsigned Dst, unsigned Shift, unsigned Width) {
48  unsigned Mask = getBitMask(Shift, Width);
49  return ((Src << Shift) & Mask) | (Dst & ~Mask);
50 }

52 /// Unpacks bits from \p Src for given bit \p Shift and bit \p Width.
53 ///
54 /// \returns Unpacked bits.
55 unsigned unpackBits(unsigned Src, unsigned Shift, unsigned Width) {
56  return (Src & getBitMask(Shift, Width)) >> Shift;

245 unsigned getAddrSizeMIMGOp(const MIMGBaseOpcodeInfo *BaseOpcode,
246  const MIMGDimInfo *Dim, bool IsA16,
247  bool IsG16Supported) {
248  unsigned AddrWords = BaseOpcode->NumExtraArgs;
249  unsigned AddrComponents = (BaseOpcode->Coordinates ? Dim->NumCoords : 0) +
250  (BaseOpcode->LodOrClampOrMip ? 1 : 0);
251  if (IsA16)
252  AddrWords += divideCeil(AddrComponents, 2);
253  else
254  AddrWords += AddrComponents;
255 
256  // Note: For subtargets that support A16 but not G16, enabling A16 also
257  // enables 16 bit gradients.
258  // For subtargets that support A16 (operand) and G16 (done with a different
259  // instruction encoding), they are independent.
260 
261  if (BaseOpcode->Gradients) {
262  if ((IsA16 && !IsG16Supported) || BaseOpcode->G16)
263  // There are two gradients per coordinate, we pack them separately.
264  // For the 3d case,
265  // we get (dy/du, dx/du) (-, dz/du) (dy/dv, dx/dv) (-, dz/dv)
266  AddrWords += alignTo<2>(Dim->NumGradients / 2);
267  else
268  AddrWords += Dim->NumGradients;
269  }
270  return AddrWords;
271 }

1333 // Custom Operands.
1334 //
1335 // A table of custom operands shall describe "primary" operand names
1336 // first followed by aliases if any. It is not required but recommended
1337 // to arrange operands so that operand encoding match operand position
1338 // in the table. This will make disassembly a bit more efficient.
1339 // Unused slots in the table shall have an empty name.
1340 //
1341 //===----------------------------------------------------------------------===//

1343 template <class T>
1344 static bool isValidOpr(int Idx, const CustomOperand<T> OpInfo[], int OpInfoSize,
1345  T Context) {
1346  return 0 <= Idx && Idx < OpInfoSize && !OpInfo[Idx].Name.empty() &&
1347  (!OpInfo[Idx].Cond || OpInfo[Idx].Cond(Context));
1348 }
1349 
1350 template <class T>
1351 static int getOprIdx(std::function<bool(const CustomOperand<T> &)> Test,
1352  const CustomOperand<T> OpInfo[], int OpInfoSize,
1353  T Context) {
1354  int InvalidIdx = OPR_ID_UNKNOWN;
1355  for (int Idx = 0; Idx < OpInfoSize; ++Idx) {
1356  if (Test(OpInfo[Idx])) {
1357  if (!OpInfo[Idx].Cond || OpInfo[Idx].Cond(Context))
1358  return Idx;
1359  InvalidIdx = OPR_ID_UNSUPPORTED;
1360  }
1361  }
1362  return InvalidIdx;
1363 }

1366 static int getOprIdx(const StringRef Name, const CustomOperand<T> OpInfo[],
1367  int OpInfoSize, T Context) {

1369  return getOprIdx<T>(Test, OpInfo, OpInfoSize, Context);

1372 template <class T>
1373 static int getOprIdx(int Id, const CustomOperand<T> OpInfo[], int OpInfoSize,
1374  T Context, bool QuickCheck = true) {
1375  auto Test = [=](const CustomOperand<T> &Op) {
1376  return Op.Encoding == Id && !Op.Name.empty();
1377  };
1378  // This is an optimization that should work in most cases.
1379  // As a side effect, it may cause selection of an alias
1380  // instead of a primary operand name in case of sparse tables.
1381  if (QuickCheck && isValidOpr<T>(Id, OpInfo, OpInfoSize, Context) &&
1382  OpInfo[Id].Encoding == Id) {
1383  return Id;
1384  }
1385  return getOprIdx<T>(Test, OpInfo, OpInfoSize, Context);
1386 }

2590 // Given Imm, split it into the values to put into the SOffset and ImmOffset
2591 // fields in an MUBUF instruction. Return false if it is not possible (due to a
2592 // hardware bug needing a workaround).
2593 //
2594 // The required alignment ensures that individual address components remain
2595 // aligned if they are aligned to begin with. It also ensures that additional
2596 // offsets within the given alignment can be added to the resulting ImmOffset.
2597 bool splitMUBUFOffset(uint32_t Imm, uint32_t &SOffset, uint32_t &ImmOffset,
2598  const GCNSubtarget *Subtarget, Align Alignment) {
2599  const uint32_t MaxImm = alignDown(4095, Alignment.value());

2601 
2602  if (Imm > MaxImm) {
2603  if (Imm <= MaxImm + 64) {
2604  // Use an SOffset inline constant for 4..64
2605  Overflow = Imm - MaxImm;
2606  Imm = MaxImm;
2607  } else {
2608  // Try to keep the same value in SOffset for adjacent loads, so that
2609  // the corresponding register contents can be re-used.
2610  //
2611  // Load values with all low-bits (except for alignment bits) set into
2612  // SOffset, so that a larger range of values can be covered using
2613  // s_movk_i32.
2614  //
2615  // Atomic operations fail to work correctly when individual address
2616  // components are unaligned, even if their sum is aligned.
2617  uint32_t High = (Imm + Alignment.value()) & ~4095;
2618  uint32_t Low = (Imm + Alignment.value()) & 4095;
2619  Imm = Low;
2620  Overflow = High - Alignment.value();
2621  }
2622  }
2623 
2624  // There is a hardware bug in SI and CI which prevents address clamping in
2625  // MUBUF instructions from working correctly with SOffsets. The immediate
2626  // offset is unaffected.
2627  if (Overflow > 0 &&
2628  Subtarget->getGeneration() <= AMDGPUSubtarget::SEA_ISLANDS)
2629  return false;
2630 
2631  ImmOffset = Imm;
2632  SOffset = Overflow;
2633  return true;
