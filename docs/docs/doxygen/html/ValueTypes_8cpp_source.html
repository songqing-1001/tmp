143 TypeSize EVT::getExtendedSizeInBits() const {
144  assert(isExtended() && "Type is not extended!");
145  if (IntegerType *ITy = dyn_cast<IntegerType>(LLVMTy))
146  return TypeSize::Fixed(ITy->getBitWidth());
147  if (VectorType *VTy = dyn_cast<VectorType>(LLVMTy))
148  return VTy->getPrimitiveSizeInBits();
149  llvm_unreachable("Unrecognized extended type!");

152 /// getEVTString - This function returns value type as a string, e.g. "i32".
153 std::string EVT::getEVTString() const {
154  switch (V.SimpleTy) {
155  default:
156  if (isVector())
157  return (isScalableVector() ? "nxv" : "v") +
158  utostr(getVectorElementCount().getKnownMinValue()) +
159  getVectorElementType().getEVTString();
160  if (isInteger())
161  return "i" + utostr(getSizeInBits());
162  if (isFloatingPoint())
163  return "f" + utostr(getSizeInBits());
164  llvm_unreachable("Invalid EVT!");
165  case MVT::bf16: return "bf16";
166  case MVT::ppcf128: return "ppcf128";
167  case MVT::isVoid: return "isVoid";
168  case MVT::Other: return "ch";
169  case MVT::Glue: return "glue";
170  case MVT::x86mmx: return "x86mmx";
171  case MVT::x86amx: return "x86amx";
172  case MVT::i64x8: return "i64x8";
173  case MVT::Metadata: return "Metadata";
174  case MVT::Untyped: return "Untyped";
175  case MVT::funcref: return "funcref";
176  case MVT::externref: return "externref";
177  }
178 }

568 /// Return the value type corresponding to the specified type. This returns all
569 /// pointers as MVT::iPTR. If HandleUnknown is true, unknown types are returned
570 /// as Other, otherwise they are invalid.
571 MVT MVT::getVT(Type *Ty, bool HandleUnknown){
572  switch (Ty->getTypeID()) {
573  default:
574  if (HandleUnknown) return MVT(MVT::Other);
575  llvm_unreachable("Unknown type!");
576  case Type::VoidTyID:
577  return MVT::isVoid;
578  case Type::IntegerTyID:
579  return getIntegerVT(cast<IntegerType>(Ty)->getBitWidth());
580  case Type::HalfTyID: return MVT(MVT::f16);
581  case Type::BFloatTyID: return MVT(MVT::bf16);
582  case Type::FloatTyID: return MVT(MVT::f32);
583  case Type::DoubleTyID: return MVT(MVT::f64);
584  case Type::X86_FP80TyID: return MVT(MVT::f80);
585  case Type::X86_MMXTyID: return MVT(MVT::x86mmx);
586  case Type::X86_AMXTyID: return MVT(MVT::x86amx);
587  case Type::FP128TyID: return MVT(MVT::f128);
588  case Type::PPC_FP128TyID: return MVT(MVT::ppcf128);
589  case Type::PointerTyID: return MVT(MVT::iPTR);
590  case Type::FixedVectorTyID:
591  case Type::ScalableVectorTyID: {
592  VectorType *VTy = cast<VectorType>(Ty);
593  return getVectorVT(
594  getVT(VTy->getElementType(), /*HandleUnknown=*/ false),
595  VTy->getElementCount());
596  }
597  }
598 }

602 /// are returned as Other, otherwise they are invalid.
603 EVT EVT::getEVT(Type *Ty, bool HandleUnknown){
604  switch (Ty->getTypeID()) {

615  }
