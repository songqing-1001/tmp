435  /// Size examples:
436  ///
437  /// Type SizeInBits StoreSizeInBits AllocSizeInBits[*]
438  /// ---- ---------- --------------- ---------------
439  /// i1 1 8 8

441  /// i19 19 24 32
442  /// i32 32 32 32
443  /// i100 100 104 128
444  /// i128 128 128 128

446  /// Double 64 64 64

449  /// [*] The alloc size depends on the alignment, and thus on the target.

451 
452  /// Returns the number of bits necessary to hold the specified type.
453  ///
454  /// If Ty is a scalable vector type, the scalable property will be set and
455  /// the runtime size will be a positive integer multiple of the base size.
456  ///
457  /// For example, returns 36 for i36 and 80 for x86_fp80. The type passed must
458  /// have a size (Type::isSized() must return true).
459  TypeSize getTypeSizeInBits(Type *Ty) const;
460 
461  /// Returns the maximum number of bytes that may be overwritten by
462  /// storing the specified type.
463  ///
464  /// If Ty is a scalable vector type, the scalable property will be set and
465  /// the runtime size will be a positive integer multiple of the base size.
466  ///
467  /// For example, returns 5 for i36 and 10 for x86_fp80.
468  TypeSize getTypeStoreSize(Type *Ty) const {
469  TypeSize BaseSize = getTypeSizeInBits(Ty);
470  return {divideCeil(BaseSize.getKnownMinValue(), 8), BaseSize.isScalable()};
471  }
472 
473  /// Returns the maximum number of bits that may be overwritten by
474  /// storing the specified type; always a multiple of 8.
475  ///
476  /// If Ty is a scalable vector type, the scalable property will be set and
477  /// the runtime size will be a positive integer multiple of the base size.
478  ///
479  /// For example, returns 40 for i36 and 80 for x86_fp80.
480  TypeSize getTypeStoreSizeInBits(Type *Ty) const {
481  return 8 * getTypeStoreSize(Ty);

669 inline TypeSize DataLayout::getTypeSizeInBits(Type *Ty) const {
670  assert(Ty->isSized() && "Cannot getTypeInfo() on a type that is unsized!");
671  switch (Ty->getTypeID()) {
672  case Type::LabelTyID:
673  return TypeSize::Fixed(getPointerSizeInBits(0));
674  case Type::PointerTyID:
675  return TypeSize::Fixed(getPointerSizeInBits(Ty->getPointerAddressSpace()));
676  case Type::ArrayTyID: {

679  getTypeAllocSizeInBits(ATy->getElementType());
680  }
681  case Type::StructTyID:

685  case Type::IntegerTyID:

687  case Type::HalfTyID:
688  case Type::BFloatTyID:

690  case Type::FloatTyID:
691  return TypeSize::Fixed(32);
692  case Type::DoubleTyID:
693  case Type::X86_MMXTyID:
694  return TypeSize::Fixed(64);
695  case Type::PPC_FP128TyID:
696  case Type::FP128TyID:
697  return TypeSize::Fixed(128);
698  case Type::X86_AMXTyID:
699  return TypeSize::Fixed(8192);
700  // In memory objects this is always aligned to a higher boundary, but
701  // only 80 bits contain information.
702  case Type::X86_FP80TyID:
703  return TypeSize::Fixed(80);
704  case Type::FixedVectorTyID:
705  case Type::ScalableVectorTyID: {
706  VectorType *VTy = cast<VectorType>(Ty);
707  auto EltCnt = VTy->getElementCount();
708  uint64_t MinBits = EltCnt.getKnownMinValue() *
709  getTypeSizeInBits(VTy->getElementType()).getFixedValue();
710  return TypeSize(MinBits, EltCnt.isScalable());
711  }
712  case Type::TargetExtTyID: {
713  Type *LayoutTy = cast<TargetExtType>(Ty)->getLayoutType();
714  return getTypeSizeInBits(LayoutTy);
715  }
716  default:
717  llvm_unreachable("DataLayout::getTypeSizeInBits(): Unsupported type");
718  }
