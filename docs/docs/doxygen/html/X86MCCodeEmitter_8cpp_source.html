44 class X86OpcodePrefixHelper {
45  // REX (1 byte)
46  // +-----+ +------+
47  // | 40H | | WRXB |
48  // +-----+ +------+
49 
50  // XOP (3-byte)
51  // +-----+ +--------------+ +-------------------+
52  // | 8Fh | | RXB | m-mmmm | | W | vvvv | L | pp |
53  // +-----+ +--------------+ +-------------------+
54 
55  // VEX2 (2 bytes)
56  // +-----+ +-------------------+
57  // | C5h | | R | vvvv | L | pp |
58  // +-----+ +-------------------+
59 
60  // VEX3 (3 bytes)
61  // +-----+ +--------------+ +-------------------+
62  // | C4h | | RXB | m-mmmm | | W | vvvv | L | pp |
63  // +-----+ +--------------+ +-------------------+
64 
65  // VEX_R: opcode externsion equivalent to REX.R in
66  // 1's complement (inverted) form
67  //
68  // 1: Same as REX_R=0 (must be 1 in 32-bit mode)
69  // 0: Same as REX_R=1 (64 bit mode only)
70 
71  // VEX_X: equivalent to REX.X, only used when a
72  // register is used for index in SIB Byte.
73  //
74  // 1: Same as REX.X=0 (must be 1 in 32-bit mode)
75  // 0: Same as REX.X=1 (64-bit mode only)
76 
77  // VEX_B:
78  // 1: Same as REX_B=0 (ignored in 32-bit mode)
79  // 0: Same as REX_B=1 (64 bit mode only)
80 
81  // VEX_W: opcode specific (use like REX.W, or used for
82  // opcode extension, or ignored, depending on the opcode byte)
83 
84  // VEX_5M (VEX m-mmmmm field):
85  //
86  // 0b00000: Reserved for future use
87  // 0b00001: implied 0F leading opcode
88  // 0b00010: implied 0F 38 leading opcode bytes
89  // 0b00011: implied 0F 3A leading opcode bytes
90  // 0b00100: Reserved for future use
91  // 0b00101: VEX MAP5
92  // 0b00110: VEX MAP6
93  // 0b00111-0b11111: Reserved for future use
94  // 0b01000: XOP map select - 08h instructions with imm byte
95  // 0b01001: XOP map select - 09h instructions with no imm byte
96  // 0b01010: XOP map select - 0Ah instructions with imm dword
97 
98  // VEX_4V (VEX vvvv field): a register specifier
99  // (in 1's complement form) or 1111 if unused.
100 
101  // VEX_PP: opcode extension providing equivalent
102  // functionality of a SIMD prefix
103  // 0b00: None
104  // 0b01: 66
105  // 0b10: F3
106  // 0b11: F2
107 
108  // EVEX (4 bytes)
109  // +-----+ +--------------+ +-------------------+ +------------------------+
110  // | 62h | | RXBR' | 00mm | | W | vvvv | 1 | pp | | z | L'L | b | v' | aaa |
111  // +-----+ +--------------+ +-------------------+ +------------------------+
112 
113  // EVEX_L2/VEX_L (Vector Length):
114  // L2 L
115  // 0 0: scalar or 128-bit vector
116  // 0 1: 256-bit vector
117  // 1 0: 512-bit vector
118

194  switch (Kind) {
195  case None:
196  Kind = (W | R | X | B) ? REX : None;
197  break;
198  case REX:
199  case XOP:
200  case VEX3:
201  case EVEX:
202  break;

210  void emit(raw_ostream &OS) const {
211  uint8_t FirstPayload =
212  ((~R) & 0x1) << 7 | ((~X) & 0x1) << 6 | ((~B) & 0x1) << 5;
213  uint8_t LastPayload = ((~VEX_4V) & 0xf) << 3 | VEX_L << 2 | VEX_PP;
214  switch (Kind) {
215  case None:
216  return;
217  case REX:
218  emitByte(0x40 | W << 3 | R << 2 | X << 1 | B, OS);
219  return;
220  case VEX2:
221  emitByte(0xC5, OS);
222  emitByte(((~R) & 1) << 7 | LastPayload, OS);
223  return;
224  case VEX3:
225  case XOP:
226  emitByte(Kind == VEX3 ? 0xC4 : 0x8F, OS);
227  emitByte(FirstPayload | VEX_5M, OS);
228  emitByte(W << 7 | LastPayload, OS);
229  return;
230  case EVEX:
231  assert(VEX_5M & 0x7 &&
232  "More than 3 significant bits in VEX.m-mmmm fields for EVEX!");
233  emitByte(0x62, OS);
234  emitByte(FirstPayload | ((~EVEX_R2) & 0x1) << 4 | VEX_5M, OS);
235  emitByte(W << 7 | ((~VEX_4V) & 0xf) << 3 | 1 << 2 | VEX_PP, OS);
236  emitByte(EVEX_z << 7 | EVEX_L2 << 6 | VEX_L << 5 | EVEX_b << 4 |
237  ((~EVEX_V2) & 0x1) << 3 | EVEX_aaa,
238  OS);
239  return;
240  }

315 /// Determine if this immediate can fit in a disp8 or a compressed disp8 for
316 /// EVEX instructions. \p will be set to the value to pass to the ImmOffset
317 /// parameter of emitImmediate.
318 static bool isDispOrCDisp8(uint64_t TSFlags, int Value, int &ImmOffset) {
319  bool HasEVEX = (TSFlags & X86II::EncodingMask) == X86II::EVEX;
320 
321  int CD8_Scale =
322  (TSFlags & X86II::CD8_Scale_Mask) >> X86II::CD8_Scale_Shift;
323  if (!HasEVEX || CD8_Scale == 0)
324  return isInt<8>(Value);
325 
326  assert(isPowerOf2_32(CD8_Scale) && "Unexpected CD8 scale!");
327  if (Value & (CD8_Scale - 1)) // Unaligned offset
328  return false;
329 
330  int CDisp8 = Value / CD8_Scale;
331  if (!isInt<8>(CDisp8))
332  return false;
333 
334  // ImmOffset will be added to Value in emitImmediate leaving just CDisp8.
335  ImmOffset = CDisp8 - Value;
336  return true;
337 }

340 /// instruction with the specified TSFlags.
341 static MCFixupKind getImmFixupKind(uint64_t TSFlags) {
342  unsigned Size = X86II::getSizeOfImm(TSFlags);
343  bool isPCRel = X86II::isImmPCRel(TSFlags);

356 enum GlobalOffsetTableExprKind { GOT_None, GOT_Normal, GOT_SymDiff };
357 
358 /// Check if this expression starts with _GLOBAL_OFFSET_TABLE_ and if it is
359 /// of the form _GLOBAL_OFFSET_TABLE_-symbol. This is needed to support PIC on
360 /// ELF i386 as _GLOBAL_OFFSET_TABLE_ is magical. We check only simple case that
361 /// are know to be used: _GLOBAL_OFFSET_TABLE_ by itself or at the start of a
362 /// binary expression.
363 static GlobalOffsetTableExprKind
364 startsWithGlobalOffsetTable(const MCExpr *Expr) {
365  const MCExpr *RHS = nullptr;
366  if (Expr->getKind() == MCExpr::Binary) {
367  const MCBinaryExpr *BE = static_cast<const MCBinaryExpr *>(Expr);
368  Expr = BE->getLHS();
369  RHS = BE->getRHS();
370  }
371 
372  if (Expr->getKind() != MCExpr::SymbolRef)
373  return GOT_None;
374 
375  const MCSymbolRefExpr *Ref = static_cast<const MCSymbolRefExpr *>(Expr);
376  const MCSymbol &S = Ref->getSymbol();
377  if (S.getName() != "_GLOBAL_OFFSET_TABLE_")
378  return GOT_None;
379  if (RHS && RHS->getKind() == MCExpr::SymbolRef)
380  return GOT_SymDiff;
381  return GOT_Normal;
382 }

418 void X86MCCodeEmitter::emitImmediate(const MCOperand &DispOp, SMLoc Loc,
419  unsigned Size, MCFixupKind FixupKind,
420  uint64_t StartByte, raw_ostream &OS,
421  SmallVectorImpl<MCFixup> &Fixups,
422  int ImmOffset) const {
423  const MCExpr *Expr = nullptr;
424  if (DispOp.isImm()) {
425  // If this is a simple integer displacement that doesn't require a
426  // relocation, emit it now.
427  if (FixupKind != FK_PCRel_1 && FixupKind != FK_PCRel_2 &&
428  FixupKind != FK_PCRel_4) {
429  emitConstant(DispOp.getImm() + ImmOffset, Size, OS);
430  return;
431  }
432  Expr = MCConstantExpr::create(DispOp.getImm(), Ctx);
433  } else {
434  Expr = DispOp.getExpr();
435  }
436 
437  // If we have an immoffset, add it to the expression.
438  if ((FixupKind == FK_Data_4 || FixupKind == FK_Data_8 ||
439  FixupKind == MCFixupKind(X86::reloc_signed_4byte))) {
440  GlobalOffsetTableExprKind Kind = startsWithGlobalOffsetTable(Expr);
441  if (Kind != GOT_None) {
442  assert(ImmOffset == 0);
443 
444  if (Size == 8) {
445  FixupKind = MCFixupKind(X86::reloc_global_offset_table8);
446  } else {
447  assert(Size == 4);
448  FixupKind = MCFixupKind(X86::reloc_global_offset_table);
449  }
450 
451  if (Kind == GOT_Normal)
452  ImmOffset = static_cast<int>(OS.tell() - StartByte);
453  } else if (Expr->getKind() == MCExpr::SymbolRef) {
454  if (hasSecRelSymbolRef(Expr)) {
455  FixupKind = MCFixupKind(FK_SecRel_4);
456  }
457  } else if (Expr->getKind() == MCExpr::Binary) {
458  const MCBinaryExpr *Bin = static_cast<const MCBinaryExpr *>(Expr);
459  if (hasSecRelSymbolRef(Bin->getLHS()) ||
460  hasSecRelSymbolRef(Bin->getRHS())) {
461  FixupKind = MCFixupKind(FK_SecRel_4);
462  }
463  }
464  }
465 
466  // If the fixup is pc-relative, we need to bias the value to be relative to
467  // the start of the field, not the end of the field.
468  if (FixupKind == FK_PCRel_4 ||
469  FixupKind == MCFixupKind(X86::reloc_riprel_4byte) ||
470  FixupKind == MCFixupKind(X86::reloc_riprel_4byte_movq_load) ||
471  FixupKind == MCFixupKind(X86::reloc_riprel_4byte_relax) ||
472  FixupKind == MCFixupKind(X86::reloc_riprel_4byte_relax_rex) ||
473  FixupKind == MCFixupKind(X86::reloc_branch_4byte_pcrel)) {
474  ImmOffset -= 4;
475  // If this is a pc-relative load off _GLOBAL_OFFSET_TABLE_:
476  // leaq _GLOBAL_OFFSET_TABLE_(%rip), %r15
477  // this needs to be a GOTPC32 relocation.
478  if (startsWithGlobalOffsetTable(Expr) != GOT_None)
479  FixupKind = MCFixupKind(X86::reloc_global_offset_table);
480  }
481  if (FixupKind == FK_PCRel_2)
482  ImmOffset -= 2;
483  if (FixupKind == FK_PCRel_1)
484  ImmOffset -= 1;
485 
486  if (ImmOffset)
487  Expr = MCBinaryExpr::createAdd(Expr, MCConstantExpr::create(ImmOffset, Ctx),
488  Ctx);
489 
490  // Emit a symbolic constant as a fixup and 4 zeros.
491  Fixups.push_back(MCFixup::create(static_cast<uint32_t>(OS.tell() - StartByte),
492  Expr, FixupKind, Loc));
493  emitConstant(0, Size, OS);
494 }

508 void X86MCCodeEmitter::emitMemModRMByte(const MCInst &MI, unsigned Op,
509  unsigned RegOpcodeField,
510  uint64_t TSFlags, PrefixKind Kind,
511  uint64_t StartByte, raw_ostream &OS,
512  SmallVectorImpl<MCFixup> &Fixups,
513  const MCSubtargetInfo &STI,
514  bool ForceSIB) const {
515  const MCOperand &Disp = MI.getOperand(Op + X86::AddrDisp);
516  const MCOperand &Base = MI.getOperand(Op + X86::AddrBaseReg);
517  const MCOperand &Scale = MI.getOperand(Op + X86::AddrScaleAmt);
518  const MCOperand &IndexReg = MI.getOperand(Op + X86::AddrIndexReg);
519  unsigned BaseReg = Base.getReg();
520 
521  // Handle %rip relative addressing.
522  if (BaseReg == X86::RIP ||
523  BaseReg == X86::EIP) { // [disp32+rIP] in X86-64 mode
524  assert(STI.hasFeature(X86::Is64Bit) &&
525  "Rip-relative addressing requires 64-bit mode");
526  assert(IndexReg.getReg() == 0 && !ForceSIB &&
527  "Invalid rip-relative address");
528  emitByte(modRMByte(0, RegOpcodeField, 5), OS);
529 
530  unsigned Opcode = MI.getOpcode();
531  unsigned FixupKind = [&]() {
532  // Enable relaxed relocation only for a MCSymbolRefExpr. We cannot use a
533  // relaxed relocation if an offset is present (e.g. x@GOTPCREL+4).
534  if (!(Disp.isExpr() && isa<MCSymbolRefExpr>(Disp.getExpr())))
535  return X86::reloc_riprel_4byte;
536 
537  // Certain loads for GOT references can be relocated against the symbol
538  // directly if the symbol ends up in the same linkage unit.
539  switch (Opcode) {
540  default:
541  return X86::reloc_riprel_4byte;
542  case X86::MOV64rm:
543  // movq loads is a subset of reloc_riprel_4byte_relax_rex. It is a
544  // special case because COFF and Mach-O don't support ELF's more
545  // flexible R_X86_64_REX_GOTPCRELX relaxation.
546  assert(Kind == REX);
547  return X86::reloc_riprel_4byte_movq_load;
548  case X86::ADC32rm:
549  case X86::ADD32rm:
550  case X86::AND32rm:
551  case X86::CMP32rm:
552  case X86::MOV32rm:
553  case X86::OR32rm:
554  case X86::SBB32rm:
555  case X86::SUB32rm:
556  case X86::TEST32mr:
557  case X86::XOR32rm:
558  case X86::CALL64m:
559  case X86::JMP64m:
560  case X86::TAILJMPm64:
561  case X86::TEST64mr:
562  case X86::ADC64rm:
563  case X86::ADD64rm:
564  case X86::AND64rm:
565  case X86::CMP64rm:
566  case X86::OR64rm:
567  case X86::SBB64rm:
568  case X86::SUB64rm:
569  case X86::XOR64rm:
570  return Kind == REX ? X86::reloc_riprel_4byte_relax_rex
571  : X86::reloc_riprel_4byte_relax;
572  }
573  }();
574 
575  // rip-relative addressing is actually relative to the *next* instruction.
576  // Since an immediate can follow the mod/rm byte for an instruction, this
577  // means that we need to bias the displacement field of the instruction with
578  // the size of the immediate field. If we have this case, add it into the
579  // expression to emit.
580  // Note: rip-relative addressing using immediate displacement values should
581  // not be adjusted, assuming it was the user's intent.
582  int ImmSize = !Disp.isImm() && X86II::hasImm(TSFlags)
583  ? X86II::getSizeOfImm(TSFlags)
584  : 0;
585 
586  emitImmediate(Disp, MI.getLoc(), 4, MCFixupKind(FixupKind), StartByte, OS,
587  Fixups, -ImmSize);
588  return;
589  }
590 
591  unsigned BaseRegNo = BaseReg ? getX86RegNum(Base) : -1U;
592 
593  // 16-bit addressing forms of the ModR/M byte have a different encoding for
594  // the R/M field and are far more limited in which registers can be used.
595  if (X86_MC::is16BitMemOperand(MI, Op, STI)) {
596  if (BaseReg) {
597  // For 32-bit addressing, the row and column values in Table 2-2 are
598  // basically the same. It's AX/CX/DX/BX/SP/BP/SI/DI in that order, with
599  // some special cases. And getX86RegNum reflects that numbering.
600  // For 16-bit addressing it's more fun, as shown in the SDM Vol 2A,
601  // Table 2-1 "16-Bit Addressing Forms with the ModR/M byte". We can only
602  // use SI/DI/BP/BX, which have "row" values 4-7 in no particular order,
603  // while values 0-3 indicate the allowed combinations (base+index) of
604  // those: 0 for BX+SI, 1 for BX+DI, 2 for BP+SI, 3 for BP+DI.
605  //
606  // R16Table[] is a lookup from the normal RegNo, to the row values from
607  // Table 2-1 for 16-bit addressing modes. Where zero means disallowed.
608  static const unsigned R16Table[] = {0, 0, 0, 7, 0, 6, 4, 5};
609  unsigned RMfield = R16Table[BaseRegNo];
610 
611  assert(RMfield && "invalid 16-bit base register");
612 
613  if (IndexReg.getReg()) {
614  unsigned IndexReg16 = R16Table[getX86RegNum(IndexReg)];
615 
616  assert(IndexReg16 && "invalid 16-bit index register");
617  // We must have one of SI/DI (4,5), and one of BP/BX (6,7).
618  assert(((IndexReg16 ^ RMfield) & 2) &&
619  "invalid 16-bit base/index register combination");
620  assert(Scale.getImm() == 1 &&
621  "invalid scale for 16-bit memory reference");
622 
623  // Allow base/index to appear in either order (although GAS doesn't).
624  if (IndexReg16 & 2)
625  RMfield = (RMfield & 1) | ((7 - IndexReg16) << 1);
626  else
627  RMfield = (IndexReg16 & 1) | ((7 - RMfield) << 1);
628  }
629 
630  if (Disp.isImm() && isInt<8>(Disp.getImm())) {
631  if (Disp.getImm() == 0 && RMfield != 6) {
632  // There is no displacement; just the register.
633  emitByte(modRMByte(0, RegOpcodeField, RMfield), OS);
634  return;
635  }
636  // Use the [REG]+disp8 form, including for [BP] which cannot be encoded.
637  emitByte(modRMByte(1, RegOpcodeField, RMfield), OS);
638  emitImmediate(Disp, MI.getLoc(), 1, FK_Data_1, StartByte, OS, Fixups);
639  return;
640  }
641  // This is the [REG]+disp16 case.
642  emitByte(modRMByte(2, RegOpcodeField, RMfield), OS);
643  } else {
644  assert(IndexReg.getReg() == 0 && "Unexpected index register!");
645  // There is no BaseReg; this is the plain [disp16] case.
646  emitByte(modRMByte(0, RegOpcodeField, 6), OS);
647  }
648 
649  // Emit 16-bit displacement for plain disp16 or [REG]+disp16 cases.
650  emitImmediate(Disp, MI.getLoc(), 2, FK_Data_2, StartByte, OS, Fixups);
651  return;
652  }
653 
654  // Check for presence of {disp8} or {disp32} pseudo prefixes.
655  bool UseDisp8 = MI.getFlags() & X86::IP_USE_DISP8;
656  bool UseDisp32 = MI.getFlags() & X86::IP_USE_DISP32;
657 
658  // We only allow no displacement if no pseudo prefix is present.
659  bool AllowNoDisp = !UseDisp8 && !UseDisp32;
660  // Disp8 is allowed unless the {disp32} prefix is present.
661  bool AllowDisp8 = !UseDisp32;
662 
663  // Determine whether a SIB byte is needed.
664  if (// The SIB byte must be used if there is an index register or the
665  // encoding requires a SIB byte.
666  !ForceSIB && IndexReg.getReg() == 0 &&
667  // The SIB byte must be used if the base is ESP/RSP/R12, all of which
668  // encode to an R/M value of 4, which indicates that a SIB byte is
669  // present.
670  BaseRegNo != N86::ESP &&
671  // If there is no base register and we're in 64-bit mode, we need a SIB
672  // byte to emit an addr that is just 'disp32' (the non-RIP relative form).
673  (!STI.hasFeature(X86::Is64Bit) || BaseReg != 0)) {
674 
675  if (BaseReg == 0) { // [disp32] in X86-32 mode
676  emitByte(modRMByte(0, RegOpcodeField, 5), OS);
677  emitImmediate(Disp, MI.getLoc(), 4, FK_Data_4, StartByte, OS, Fixups);
678  return;
679  }
680 
681  // If the base is not EBP/ESP/R12/R13 and there is no displacement, use
682  // simple indirect register encoding, this handles addresses like [EAX].
683  // The encoding for [EBP] or[R13] with no displacement means [disp32] so we
684  // handle it by emitting a displacement of 0 later.
685  if (BaseRegNo != N86::EBP) {
686  if (Disp.isImm() && Disp.getImm() == 0 && AllowNoDisp) {
687  emitByte(modRMByte(0, RegOpcodeField, BaseRegNo), OS);
688  return;
689  }
690 
691  // If the displacement is @tlscall, treat it as a zero.
692  if (Disp.isExpr()) {
693  auto *Sym = dyn_cast<MCSymbolRefExpr>(Disp.getExpr());
694  if (Sym && Sym->getKind() == MCSymbolRefExpr::VK_TLSCALL) {
695  // This is exclusively used by call *a@tlscall(base). The relocation
696  // (R_386_TLSCALL or R_X86_64_TLSCALL) applies to the beginning.
697  Fixups.push_back(MCFixup::create(0, Sym, FK_NONE, MI.getLoc()));
698  emitByte(modRMByte(0, RegOpcodeField, BaseRegNo), OS);
699  return;
700  }
701  }
702  }
703 
704  // Otherwise, if the displacement fits in a byte, encode as [REG+disp8].
705  // Including a compressed disp8 for EVEX instructions that support it.
706  // This also handles the 0 displacement for [EBP] or [R13]. We can't use
707  // disp8 if the {disp32} pseudo prefix is present.
708  if (Disp.isImm() && AllowDisp8) {
709  int ImmOffset = 0;
710  if (isDispOrCDisp8(TSFlags, Disp.getImm(), ImmOffset)) {
711  emitByte(modRMByte(1, RegOpcodeField, BaseRegNo), OS);
712  emitImmediate(Disp, MI.getLoc(), 1, FK_Data_1, StartByte, OS, Fixups,
713  ImmOffset);
714  return;
715  }
716  }
717 
718  // Otherwise, emit the most general non-SIB encoding: [REG+disp32].
719  // Displacement may be 0 for [EBP] or [R13] case if {disp32} pseudo prefix
720  // prevented using disp8 above.
721  emitByte(modRMByte(2, RegOpcodeField, BaseRegNo), OS);
722  unsigned Opcode = MI.getOpcode();
723  unsigned FixupKind = Opcode == X86::MOV32rm ? X86::reloc_signed_4byte_relax
724  : X86::reloc_signed_4byte;
725  emitImmediate(Disp, MI.getLoc(), 4, MCFixupKind(FixupKind), StartByte, OS,
726  Fixups);
727  return;
728  }
729 
730  // We need a SIB byte, so start by outputting the ModR/M byte first
731  assert(IndexReg.getReg() != X86::ESP && IndexReg.getReg() != X86::RSP &&
732  "Cannot use ESP as index reg!");
733 
734  bool ForceDisp32 = false;
735  bool ForceDisp8 = false;
736  int ImmOffset = 0;
737  if (BaseReg == 0) {
738  // If there is no base register, we emit the special case SIB byte with
739  // MOD=0, BASE=5, to JUST get the index, scale, and displacement.
740  BaseRegNo = 5;
741  emitByte(modRMByte(0, RegOpcodeField, 4), OS);
742  ForceDisp32 = true;
743  } else if (Disp.isImm() && Disp.getImm() == 0 && AllowNoDisp &&
744  // Base reg can't be EBP/RBP/R13 as that would end up with '5' as
745  // the base field, but that is the magic [*] nomenclature that
746  // indicates no base when mod=0. For these cases we'll emit a 0
747  // displacement instead.
748  BaseRegNo != N86::EBP) {
749  // Emit no displacement ModR/M byte
750  emitByte(modRMByte(0, RegOpcodeField, 4), OS);
751  } else if (Disp.isImm() && AllowDisp8 &&
752  isDispOrCDisp8(TSFlags, Disp.getImm(), ImmOffset)) {
753  // Displacement fits in a byte or matches an EVEX compressed disp8, use
754  // disp8 encoding. This also handles EBP/R13 base with 0 displacement unless
755  // {disp32} pseudo prefix was used.
756  emitByte(modRMByte(1, RegOpcodeField, 4), OS);
757  ForceDisp8 = true;
758  } else {
759  // Otherwise, emit the normal disp32 encoding.
760  emitByte(modRMByte(2, RegOpcodeField, 4), OS);
761  ForceDisp32 = true;
762  }
763 
764  // Calculate what the SS field value should be...
765  static const unsigned SSTable[] = {~0U, 0, 1, ~0U, 2, ~0U, ~0U, ~0U, 3};
766  unsigned SS = SSTable[Scale.getImm()];
767 
768  unsigned IndexRegNo = IndexReg.getReg() ? getX86RegNum(IndexReg) : 4;
769 
770  emitSIBByte(SS, IndexRegNo, BaseRegNo, OS);
771 
772  // Do we need to output a displacement?
773  if (ForceDisp8)
774  emitImmediate(Disp, MI.getLoc(), 1, FK_Data_1, StartByte, OS, Fixups,
775  ImmOffset);
776  else if (ForceDisp32)
777  emitImmediate(Disp, MI.getLoc(), 4, MCFixupKind(X86::reloc_signed_4byte),
778  StartByte, OS, Fixups);
779 }

781 /// Emit all instruction prefixes.
782 ///
783 /// \returns one of the REX, XOP, VEX2, VEX3, EVEX if any of them is used,
784 /// otherwise returns None.
785 PrefixKind X86MCCodeEmitter::emitPrefixImpl(unsigned &CurOp, const MCInst &MI,
786  const MCSubtargetInfo &STI,
787  raw_ostream &OS) const {
788  uint64_t TSFlags = MCII.get(MI.getOpcode()).TSFlags;
789  // Determine where the memory operand starts, if present.
790  int MemoryOperand = X86II::getMemoryOperandNo(TSFlags);
791  // Emit segment override opcode prefix as needed.
792  if (MemoryOperand != -1) {
793  MemoryOperand += CurOp;
794  emitSegmentOverridePrefix(MemoryOperand + X86::AddrSegmentReg, MI, OS);
795  }
796 
797  // Emit the repeat opcode prefix as needed.
798  unsigned Flags = MI.getFlags();
799  if (TSFlags & X86II::REP || Flags & X86::IP_HAS_REPEAT)
800  emitByte(0xF3, OS);
801  if (Flags & X86::IP_HAS_REPEAT_NE)
802  emitByte(0xF2, OS);
803 
804  // Emit the address size opcode prefix as needed.
805  if (X86_MC::needsAddressSizeOverride(MI, STI, MemoryOperand, TSFlags) ||
806  Flags & X86::IP_HAS_AD_SIZE)
807  emitByte(0x67, OS);
808 
809  uint64_t Form = TSFlags & X86II::FormMask;
810  switch (Form) {
811  default:
812  break;
813  case X86II::RawFrmDstSrc: {
814  // Emit segment override opcode prefix as needed (not for %ds).
815  if (MI.getOperand(2).getReg() != X86::DS)
816  emitSegmentOverridePrefix(2, MI, OS);
817  CurOp += 3; // Consume operands.
818  break;
819  }
820  case X86II::RawFrmSrc: {
821  // Emit segment override opcode prefix as needed (not for %ds).
822  if (MI.getOperand(1).getReg() != X86::DS)
823  emitSegmentOverridePrefix(1, MI, OS);
824  CurOp += 2; // Consume operands.
825  break;
826  }
827  case X86II::RawFrmDst: {
828  ++CurOp; // Consume operand.
829  break;
830  }
831  case X86II::RawFrmMemOffs: {
832  // Emit segment override opcode prefix as needed.
833  emitSegmentOverridePrefix(1, MI, OS);
834  break;
835  }
836  }
837 
838  // REX prefix is optional, but if used must be immediately before the opcode
839  // Encoding type for this instruction.
840  return (TSFlags & X86II::EncodingMask)
841  ? emitVEXOpcodePrefix(MemoryOperand, MI, OS)
842  : emitOpcodePrefix(MemoryOperand, MI, STI, OS);
843 }

845 // AVX instructions are encoded using an encoding scheme that combines
846 // prefix bytes, opcode extension field, operand encoding fields, and vector
847 // length encoding capability into a new prefix, referred to as VEX.
848 
849 // The majority of the AVX-512 family of instructions (operating on
850 // 512/256/128-bit vector register operands) are encoded using a new prefix
851 // (called EVEX).
852 
853 // XOP is a revised subset of what was originally intended as SSE5. It was
854 // changed to be similar but not overlapping with AVX.
855 
856 /// Emit XOP, VEX2, VEX3 or EVEX prefix.
857 /// \returns the used prefix.
858 PrefixKind X86MCCodeEmitter::emitVEXOpcodePrefix(int MemOperand,
859  const MCInst &MI,
860  raw_ostream &OS) const {
861  const MCInstrDesc &Desc = MCII.get(MI.getOpcode());
862  uint64_t TSFlags = Desc.TSFlags;
863 
864  assert(!(TSFlags & X86II::LOCK) && "Can't have LOCK VEX.");
865 
866  X86OpcodePrefixHelper Prefix(*Ctx.getRegisterInfo());
867  switch (TSFlags & X86II::EncodingMask) {
868  default:
869  break;
870  case X86II::XOP:
871  Prefix.setLowerBound(XOP);
872  break;
873  case X86II::VEX:
874  // VEX can be 2 byte or 3 byte, not determined yet if not explicit
875  Prefix.setLowerBound(MI.getFlags() & X86::IP_USE_VEX3 ? VEX3 : VEX2);
876  break;
877  case X86II::EVEX:
878  Prefix.setLowerBound(EVEX);
879  break;
880  }
881 
882  Prefix.setW(TSFlags & X86II::VEX_W);
883 
884  bool HasEVEX_K = TSFlags & X86II::EVEX_K;
885  bool HasVEX_4V = TSFlags & X86II::VEX_4V;
886  bool HasEVEX_RC = TSFlags & X86II::EVEX_RC;
887 
888  switch (TSFlags & X86II::OpMapMask) {
889  default:
890  llvm_unreachable("Invalid prefix!");
891  case X86II::TB:
892  Prefix.set5M(0x1); // 0F
893  break;
894  case X86II::T8:
895  Prefix.set5M(0x2); // 0F 38
896  break;
897  case X86II::TA:
898  Prefix.set5M(0x3); // 0F 3A
899  break;
900  case X86II::XOP8:
901  Prefix.set5M(0x8);
902  break;
903  case X86II::XOP9:
904  Prefix.set5M(0x9);
905  break;
906  case X86II::XOPA:
907  Prefix.set5M(0xA);
908  break;
909  case X86II::T_MAP5:
910  Prefix.set5M(0x5);
911  break;
912  case X86II::T_MAP6:
913  Prefix.set5M(0x6);
914  break;
915  }
916 
917  Prefix.setL(TSFlags & X86II::VEX_L);
918  Prefix.setL2(TSFlags & X86II::EVEX_L2);
919  switch (TSFlags & X86II::OpPrefixMask) {
920  case X86II::PD:
921  Prefix.setPP(0x1); // 66
922  break;
923  case X86II::XS:
924  Prefix.setPP(0x2); // F3
925  break;
926  case X86II::XD:
927  Prefix.setPP(0x3); // F2
928  break;
929  }
930 
931  Prefix.setZ(HasEVEX_K && (TSFlags & X86II::EVEX_Z));
932  Prefix.setEVEX_b(TSFlags & X86II::EVEX_B);
933 
934  bool EncodeRC = false;
935  uint8_t EVEX_rc = 0;
936  unsigned CurOp = X86II::getOperandBias(Desc);
937 
938  switch (TSFlags & X86II::FormMask) {
939  default:
940  llvm_unreachable("Unexpected form in emitVEXOpcodePrefix!");
941  case X86II::MRMDestMem4VOp3CC: {
942  // MemAddr, src1(ModR/M), src2(VEX_4V)
943  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
944  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
945  CurOp += X86::AddrNumOperands;
946  Prefix.setR(MI, ++CurOp);
947  Prefix.set4V(MI, CurOp++);
948  break;
949  }
950  case X86II::MRM_C0:
951  case X86II::RawFrm:
952  case X86II::PrefixByte:
953  break;
954  case X86II::MRMDestMemFSIB:
955  case X86II::MRMDestMem: {
956  // MRMDestMem instructions forms:
957  // MemAddr, src1(ModR/M)
958  // MemAddr, src1(VEX_4V), src2(ModR/M)
959  // MemAddr, src1(ModR/M), imm8
960  //
961  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
962  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
963  if (!HasVEX_4V) // Only needed with VSIB which don't use VVVV.
964  Prefix.setV2(MI, MemOperand + X86::AddrIndexReg);
965 
966  CurOp += X86::AddrNumOperands;
967 
968  if (HasEVEX_K)
969  Prefix.setAAA(MI, CurOp++);
970 
971  if (HasVEX_4V)
972  Prefix.set4VV2(MI, CurOp++);
973 
974  Prefix.setRR2(MI, CurOp++);
975  break;
976  }
977  case X86II::MRMSrcMemFSIB:
978  case X86II::MRMSrcMem: {
979  // MRMSrcMem instructions forms:
980  // src1(ModR/M), MemAddr
981  // src1(ModR/M), src2(VEX_4V), MemAddr
982  // src1(ModR/M), MemAddr, imm8
983  // src1(ModR/M), MemAddr, src2(Imm[7:4])
984  //
985  // FMA4:
986  // dst(ModR/M.reg), src1(VEX_4V), src2(ModR/M), src3(Imm[7:4])
987  Prefix.setRR2(MI, CurOp++);
988 
989  if (HasEVEX_K)
990  Prefix.setAAA(MI, CurOp++);
991 
992  if (HasVEX_4V)
993  Prefix.set4VV2(MI, CurOp++);
994 
995  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
996  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
997  if (!HasVEX_4V) // Only needed with VSIB which don't use VVVV.
998  Prefix.setV2(MI, MemOperand + X86::AddrIndexReg);
999 
1000  break;
1001  }
1002  case X86II::MRMSrcMem4VOp3: {
1003  // Instruction format for 4VOp3:
1004  // src1(ModR/M), MemAddr, src3(VEX_4V)
1005  Prefix.setR(MI, CurOp++);
1006  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
1007  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
1008  Prefix.set4V(MI, CurOp + X86::AddrNumOperands);
1009  break;
1010  }
1011  case X86II::MRMSrcMemOp4: {
1012  // dst(ModR/M.reg), src1(VEX_4V), src2(Imm[7:4]), src3(ModR/M),
1013  Prefix.setR(MI, CurOp++);
1014  Prefix.set4V(MI, CurOp++);
1015  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
1016  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
1017  break;
1018  }
1019  case X86II::MRM0m:
1020  case X86II::MRM1m:
1021  case X86II::MRM2m:
1022  case X86II::MRM3m:
1023  case X86II::MRM4m:
1024  case X86II::MRM5m:
1025  case X86II::MRM6m:
1026  case X86II::MRM7m: {
1027  // MRM[0-9]m instructions forms:
1028  // MemAddr
1029  // src1(VEX_4V), MemAddr
1030  if (HasVEX_4V)
1031  Prefix.set4VV2(MI, CurOp++);
1032 
1033  if (HasEVEX_K)
1034  Prefix.setAAA(MI, CurOp++);
1035 
1036  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
1037  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
1038  if (!HasVEX_4V) // Only needed with VSIB which don't use VVVV.
1039  Prefix.setV2(MI, MemOperand + X86::AddrIndexReg);
1040 
1041  break;
1042  }
1043  case X86II::MRMSrcReg: {
1044  // MRMSrcReg instructions forms:
1045  // dst(ModR/M), src1(VEX_4V), src2(ModR/M), src3(Imm[7:4])
1046  // dst(ModR/M), src1(ModR/M)
1047  // dst(ModR/M), src1(ModR/M), imm8
1048  //
1049  // FMA4:
1050  // dst(ModR/M.reg), src1(VEX_4V), src2(Imm[7:4]), src3(ModR/M),
1051  Prefix.setRR2(MI, CurOp++);
1052 
1053  if (HasEVEX_K)
1054  Prefix.setAAA(MI, CurOp++);
1055 
1056  if (HasVEX_4V)
1057  Prefix.set4VV2(MI, CurOp++);
1058 
1059  Prefix.setB(MI, CurOp);
1060  Prefix.setX(MI, CurOp, 4);
1061  ++CurOp;
1062 
1063  if (TSFlags & X86II::EVEX_B) {
1064  if (HasEVEX_RC) {
1065  unsigned NumOps = Desc.getNumOperands();
1066  unsigned RcOperand = NumOps - 1;
1067  assert(RcOperand >= CurOp);
1068  EVEX_rc = MI.getOperand(RcOperand).getImm();
1069  assert(EVEX_rc <= 3 && "Invalid rounding control!");
1070  }
1071  EncodeRC = true;
1072  }
1073  break;
1074  }
1075  case X86II::MRMSrcReg4VOp3: {
1076  // Instruction format for 4VOp3:
1077  // src1(ModR/M), src2(ModR/M), src3(VEX_4V)
1078  Prefix.setR(MI, CurOp++);
1079  Prefix.setB(MI, CurOp++);
1080  Prefix.set4V(MI, CurOp++);
1081  break;
1082  }
1083  case X86II::MRMSrcRegOp4: {
1084  // dst(ModR/M.reg), src1(VEX_4V), src2(Imm[7:4]), src3(ModR/M),
1085  Prefix.setR(MI, CurOp++);
1086  Prefix.set4V(MI, CurOp++);
1087  // Skip second register source (encoded in Imm[7:4])
1088  ++CurOp;
1089 
1090  Prefix.setB(MI, CurOp);
1091  Prefix.setX(MI, CurOp, 4);
1092  ++CurOp;
1093  break;
1094  }
1095  case X86II::MRMDestReg: {
1096  // MRMDestReg instructions forms:
1097  // dst(ModR/M), src(ModR/M)
1098  // dst(ModR/M), src(ModR/M), imm8
1099  // dst(ModR/M), src1(VEX_4V), src2(ModR/M)
1100  Prefix.setB(MI, CurOp);
1101  Prefix.setX(MI, CurOp, 4);
1102  ++CurOp;
1103 
1104  if (HasEVEX_K)
1105  Prefix.setAAA(MI, CurOp++);
1106 
1107  if (HasVEX_4V)
1108  Prefix.set4VV2(MI, CurOp++);
1109 
1110  Prefix.setRR2(MI, CurOp++);
1111  if (TSFlags & X86II::EVEX_B)
1112  EncodeRC = true;
1113  break;
1114  }
1115  case X86II::MRMr0: {
1116  // MRMr0 instructions forms:
1117  // 11:rrr:000
1118  // dst(ModR/M)
1119  Prefix.setRR2(MI, CurOp++);
1120  break;
1121  }
1122  case X86II::MRM0r:
1123  case X86II::MRM1r:
1124  case X86II::MRM2r:
1125  case X86II::MRM3r:
1126  case X86II::MRM4r:
1127  case X86II::MRM5r:
1128  case X86II::MRM6r:
1129  case X86II::MRM7r: {
1130  // MRM0r-MRM7r instructions forms:
1131  // dst(VEX_4V), src(ModR/M), imm8
1132  if (HasVEX_4V)
1133  Prefix.set4VV2(MI, CurOp++);
1134 
1135  if (HasEVEX_K)
1136  Prefix.setAAA(MI, CurOp++);
1137 
1138  Prefix.setB(MI, CurOp);
1139  Prefix.setX(MI, CurOp, 4);
1140  ++CurOp;
1141  break;
1142  }
1143  }
1144  if (EncodeRC) {
1145  Prefix.setL(EVEX_rc & 0x1);
1146  Prefix.setL2(EVEX_rc & 0x2);
1147  }
1148  PrefixKind Kind = Prefix.determineOptimalKind();
1149  Prefix.emit(OS);
1150  return Kind;

1153 /// Emit REX prefix which specifies
1154 /// 1) 64-bit instructions,
1155 /// 2) non-default operand size, and
1156 /// 3) use of X86-64 extended registers.
1157 ///
1158 /// \returns the used prefix (REX or None).
1159 PrefixKind X86MCCodeEmitter::emitREXPrefix(int MemOperand, const MCInst &MI,
1160  const MCSubtargetInfo &STI,
1161  raw_ostream &OS) const {
1162  if (!STI.hasFeature(X86::Is64Bit))
1163  return None;
1164  X86OpcodePrefixHelper Prefix(*Ctx.getRegisterInfo());
1165  bool UsesHighByteReg = false;
1166  const MCInstrDesc &Desc = MCII.get(MI.getOpcode());
1167  uint64_t TSFlags = Desc.TSFlags;
1168  Prefix.setW(TSFlags & X86II::REX_W);
1169  unsigned NumOps = MI.getNumOperands();
1170  if (!NumOps) {
1171  PrefixKind Kind = Prefix.determineOptimalKind();
1172  Prefix.emit(OS);
1173  return Kind;
1174  }
1175  unsigned CurOp = X86II::getOperandBias(Desc);
1176  for (unsigned i = CurOp; i != NumOps; ++i) {
1177  const MCOperand &MO = MI.getOperand(i);
1178  if (MO.isReg()) {
1179  unsigned Reg = MO.getReg();
1180  if (Reg == X86::AH || Reg == X86::BH || Reg == X86::CH || Reg == X86::DH)
1181  UsesHighByteReg = true;
1182  // If it accesses SPL, BPL, SIL, or DIL, then it requires a REX prefix.
1183  if (X86II::isX86_64NonExtLowByteReg(Reg))
1184  Prefix.setLowerBound(REX);
1185  } else if (MO.isExpr() && STI.getTargetTriple().isX32()) {
1186  // GOTTPOFF and TLSDESC relocations require a REX prefix to allow
1187  // linker optimizations: even if the instructions we see may not require
1188  // any prefix, they may be replaced by instructions that do. This is
1189  // handled as a special case here so that it also works for hand-written
1190  // assembly without the user needing to write REX, as with GNU as.
1191  const auto *Ref = dyn_cast<MCSymbolRefExpr>(MO.getExpr());
1192  if (Ref && (Ref->getKind() == MCSymbolRefExpr::VK_GOTTPOFF ||
1193  Ref->getKind() == MCSymbolRefExpr::VK_TLSDESC)) {
1194  Prefix.setLowerBound(REX);
1195  }
1196  }
1197  }
1198  switch (TSFlags & X86II::FormMask) {
1199  case X86II::AddRegFrm:
1200  Prefix.setB(MI, CurOp++);
1201  break;
1202  case X86II::MRMSrcReg:
1203  case X86II::MRMSrcRegCC:
1204  Prefix.setR(MI, CurOp++);
1205  Prefix.setB(MI, CurOp++);
1206  break;
1207  case X86II::MRMSrcMem:
1208  case X86II::MRMSrcMemCC:
1209  Prefix.setR(MI, CurOp++);
1210  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
1211  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
1212  CurOp += X86::AddrNumOperands;
1213  break;
1214  case X86II::MRMDestReg:
1215  Prefix.setB(MI, CurOp++);
1216  Prefix.setR(MI, CurOp++);
1217  break;
1218  case X86II::MRMDestMem:
1219  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
1220  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
1221  CurOp += X86::AddrNumOperands;
1222  Prefix.setR(MI, CurOp++);
1223  break;
1224  case X86II::MRMXmCC:
1225  case X86II::MRMXm:
1226  case X86II::MRM0m:
1227  case X86II::MRM1m:
1228  case X86II::MRM2m:
1229  case X86II::MRM3m:
1230  case X86II::MRM4m:
1231  case X86II::MRM5m:
1232  case X86II::MRM6m:
1233  case X86II::MRM7m:
1234  Prefix.setB(MI, MemOperand + X86::AddrBaseReg);
1235  Prefix.setX(MI, MemOperand + X86::AddrIndexReg);
1236  break;
1237  case X86II::MRMXrCC:
1238  case X86II::MRMXr:
1239  case X86II::MRM0r:
1240  case X86II::MRM1r:
1241  case X86II::MRM2r:
1242  case X86II::MRM3r:
1243  case X86II::MRM4r:
1244  case X86II::MRM5r:
1245  case X86II::MRM6r:
1246  case X86II::MRM7r:
1247  Prefix.setB(MI, CurOp++);
1248  break;
1249  case X86II::MRMr0:
1250  Prefix.setR(MI, CurOp++);
1251  break;
1252  case X86II::MRMDestMemFSIB:
1253  llvm_unreachable("FSIB format never need REX prefix!");
1254  }
1255  PrefixKind Kind = Prefix.determineOptimalKind();
1256  if (Kind && UsesHighByteReg)
1257  report_fatal_error(
1258  "Cannot encode high byte register in REX-prefixed instruction");
1259  Prefix.emit(OS);
1260  return Kind;
1261 }

1272 /// Emit all instruction prefixes prior to the opcode.
1273 ///
1274 /// \param MemOperand the operand # of the start of a memory operand if present.
1275 /// If not present, it is -1.
1276 ///
1277 /// \returns the used prefix (REX or None).
1278 PrefixKind X86MCCodeEmitter::emitOpcodePrefix(int MemOperand, const MCInst &MI,
1279  const MCSubtargetInfo &STI,
1280  raw_ostream &OS) const {
1281  const MCInstrDesc &Desc = MCII.get(MI.getOpcode());
1282  uint64_t TSFlags = Desc.TSFlags;
1283 
1284  // Emit the operand size opcode prefix as needed.
1285  if ((TSFlags & X86II::OpSizeMask) ==
1286  (STI.hasFeature(X86::Is16Bit) ? X86II::OpSize32 : X86II::OpSize16))
1287  emitByte(0x66, OS);
1288 
1289  // Emit the LOCK opcode prefix.
1290  if (TSFlags & X86II::LOCK || MI.getFlags() & X86::IP_HAS_LOCK)
1291  emitByte(0xF0, OS);
1292 
1293  // Emit the NOTRACK opcode prefix.
1294  if (TSFlags & X86II::NOTRACK || MI.getFlags() & X86::IP_HAS_NOTRACK)
1295  emitByte(0x3E, OS);
1296 
1297  switch (TSFlags & X86II::OpPrefixMask) {
1298  case X86II::PD: // 66
1299  emitByte(0x66, OS);
1300  break;
1301  case X86II::XS: // F3
1302  emitByte(0xF3, OS);
1303  break;
1304  case X86II::XD: // F2
1305  emitByte(0xF2, OS);
1306  break;
1307  }
1308 
1309  // Handle REX prefix.
1310  assert((STI.hasFeature(X86::Is64Bit) || !(TSFlags & X86II::REX_W)) &&
1311  "REX.W requires 64bit mode.");
1312  PrefixKind Kind = emitREXPrefix(MemOperand, MI, STI, OS);
1313 
1314  // 0x0F escape code must be emitted just before the opcode.
1315  switch (TSFlags & X86II::OpMapMask) {
1316  case X86II::TB: // Two-byte opcode map
1317  case X86II::T8: // 0F 38
1318  case X86II::TA: // 0F 3A
1319  case X86II::ThreeDNow: // 0F 0F, second 0F emitted by caller.
1320  emitByte(0x0F, OS);
1321  break;
1322  }
1323 
1324  switch (TSFlags & X86II::OpMapMask) {
1325  case X86II::T8: // 0F 38
1326  emitByte(0x38, OS);
1327  break;
1328  case X86II::TA: // 0F 3A
1329  emitByte(0x3A, OS);
1330  break;
1331  }
1332 
1333  return Kind;
1334 }

1351 void X86MCCodeEmitter::encodeInstruction(const MCInst &MI, raw_ostream &OS,
1352  SmallVectorImpl<MCFixup> &Fixups,
1353  const MCSubtargetInfo &STI) const {
1354  unsigned Opcode = MI.getOpcode();
1355  const MCInstrDesc &Desc = MCII.get(Opcode);
1356  uint64_t TSFlags = Desc.TSFlags;
1357 
1358  // Pseudo instructions don't get encoded.
1359  if (X86II::isPseudo(TSFlags))
1360  return;
1361 
1362  unsigned NumOps = Desc.getNumOperands();
1363  unsigned CurOp = X86II::getOperandBias(Desc);
1364 
1365  uint64_t StartByte = OS.tell();
1366 
1367  PrefixKind Kind = emitPrefixImpl(CurOp, MI, STI, OS);
1368 
1369  // It uses the VEX.VVVV field?
1370  bool HasVEX_4V = TSFlags & X86II::VEX_4V;
1371  bool HasVEX_I8Reg = (TSFlags & X86II::ImmMask) == X86II::Imm8Reg;
1372 
1373  // It uses the EVEX.aaa field?
1374  bool HasEVEX_K = TSFlags & X86II::EVEX_K;
1375  bool HasEVEX_RC = TSFlags & X86II::EVEX_RC;
1376 
1377  // Used if a register is encoded in 7:4 of immediate.
1378  unsigned I8RegNum = 0;
1379 
1380  uint8_t BaseOpcode = X86II::getBaseOpcodeFor(TSFlags);
1381 
1382  if ((TSFlags & X86II::OpMapMask) == X86II::ThreeDNow)
1383  BaseOpcode = 0x0F; // Weird 3DNow! encoding.
1384 
1385  unsigned OpcodeOffset = 0;
1386 
1387  uint64_t Form = TSFlags & X86II::FormMask;
1388  switch (Form) {
1389  default:
1390  errs() << "FORM: " << Form << "\n";
1391  llvm_unreachable("Unknown FormMask value in X86MCCodeEmitter!");
1392  case X86II::Pseudo:
1393  llvm_unreachable("Pseudo instruction shouldn't be emitted");
1394  case X86II::RawFrmDstSrc:
1395  case X86II::RawFrmSrc:
1396  case X86II::RawFrmDst:
1397  case X86II::PrefixByte:
1398  emitByte(BaseOpcode, OS);
1399  break;
1400  case X86II::AddCCFrm: {
1401  // This will be added to the opcode in the fallthrough.
1402  OpcodeOffset = MI.getOperand(NumOps - 1).getImm();
1403  assert(OpcodeOffset < 16 && "Unexpected opcode offset!");
1404  --NumOps; // Drop the operand from the end.
1405  [[fallthrough]];
1406  case X86II::RawFrm:
1407  emitByte(BaseOpcode + OpcodeOffset, OS);
1408 
1409  if (!STI.hasFeature(X86::Is64Bit) || !isPCRel32Branch(MI, MCII))
1410  break;
1411 
1412  const MCOperand &Op = MI.getOperand(CurOp++);
1413  emitImmediate(Op, MI.getLoc(), X86II::getSizeOfImm(TSFlags),
1414  MCFixupKind(X86::reloc_branch_4byte_pcrel), StartByte, OS,
1415  Fixups);
1416  break;
1417  }
1418  case X86II::RawFrmMemOffs:
1419  emitByte(BaseOpcode, OS);
1420  emitImmediate(MI.getOperand(CurOp++), MI.getLoc(),
1421  X86II::getSizeOfImm(TSFlags), getImmFixupKind(TSFlags),
1422  StartByte, OS, Fixups);
1423  ++CurOp; // skip segment operand
1424  break;
1425  case X86II::RawFrmImm8:
1426  emitByte(BaseOpcode, OS);
1427  emitImmediate(MI.getOperand(CurOp++), MI.getLoc(),
1428  X86II::getSizeOfImm(TSFlags), getImmFixupKind(TSFlags),
1429  StartByte, OS, Fixups);
1430  emitImmediate(MI.getOperand(CurOp++), MI.getLoc(), 1, FK_Data_1, StartByte,
1431  OS, Fixups);
1432  break;
1433  case X86II::RawFrmImm16:
1434  emitByte(BaseOpcode, OS);
1435  emitImmediate(MI.getOperand(CurOp++), MI.getLoc(),
1436  X86II::getSizeOfImm(TSFlags), getImmFixupKind(TSFlags),
1437  StartByte, OS, Fixups);
1438  emitImmediate(MI.getOperand(CurOp++), MI.getLoc(), 2, FK_Data_2, StartByte,
1439  OS, Fixups);
1440  break;
1441 
1442  case X86II::AddRegFrm:
1443  emitByte(BaseOpcode + getX86RegNum(MI.getOperand(CurOp++)), OS);
1444  break;
1445 
1446  case X86II::MRMDestReg: {
1447  emitByte(BaseOpcode, OS);
1448  unsigned SrcRegNum = CurOp + 1;
1449 
1450  if (HasEVEX_K) // Skip writemask
1451  ++SrcRegNum;
1452 
1453  if (HasVEX_4V) // Skip 1st src (which is encoded in VEX_VVVV)
1454  ++SrcRegNum;
1455 
1456  emitRegModRMByte(MI.getOperand(CurOp),
1457  getX86RegNum(MI.getOperand(SrcRegNum)), OS);
1458  CurOp = SrcRegNum + 1;
1459  break;
1460  }
1461  case X86II::MRMDestMem4VOp3CC: {
1462  unsigned CC = MI.getOperand(8).getImm();
1463  emitByte(BaseOpcode + CC, OS);
1464  unsigned SrcRegNum = CurOp + X86::AddrNumOperands;
1465  emitMemModRMByte(MI, CurOp + 1, getX86RegNum(MI.getOperand(0)), TSFlags,
1466  Kind, StartByte, OS, Fixups, STI, false);
1467  CurOp = SrcRegNum + 3; // skip reg, VEX_V4 and CC
1468  break;
1469  }
1470  case X86II::MRMDestMemFSIB:
1471  case X86II::MRMDestMem: {
1472  emitByte(BaseOpcode, OS);
1473  unsigned SrcRegNum = CurOp + X86::AddrNumOperands;
1474 
1475  if (HasEVEX_K) // Skip writemask
1476  ++SrcRegNum;
1477 
1478  if (HasVEX_4V) // Skip 1st src (which is encoded in VEX_VVVV)
1479  ++SrcRegNum;
1480 
1481  bool ForceSIB = (Form == X86II::MRMDestMemFSIB);
1482  emitMemModRMByte(MI, CurOp, getX86RegNum(MI.getOperand(SrcRegNum)), TSFlags,
1483  Kind, StartByte, OS, Fixups, STI, ForceSIB);
1484  CurOp = SrcRegNum + 1;
1485  break;
1486  }
1487  case X86II::MRMSrcReg: {
1488  emitByte(BaseOpcode, OS);
1489  unsigned SrcRegNum = CurOp + 1;
1490 
1491  if (HasEVEX_K) // Skip writemask
1492  ++SrcRegNum;
1493 
1494  if (HasVEX_4V) // Skip 1st src (which is encoded in VEX_VVVV)
1495  ++SrcRegNum;
1496 
1497  emitRegModRMByte(MI.getOperand(SrcRegNum),
1498  getX86RegNum(MI.getOperand(CurOp)), OS);
1499  CurOp = SrcRegNum + 1;
1500  if (HasVEX_I8Reg)
1501  I8RegNum = getX86RegEncoding(MI, CurOp++);
1502  // do not count the rounding control operand
1503  if (HasEVEX_RC)
1504  --NumOps;
1505  break;
1506  }
1507  case X86II::MRMSrcReg4VOp3: {
1508  emitByte(BaseOpcode, OS);
1509  unsigned SrcRegNum = CurOp + 1;
1510 
1511  emitRegModRMByte(MI.getOperand(SrcRegNum),
1512  getX86RegNum(MI.getOperand(CurOp)), OS);
1513  CurOp = SrcRegNum + 1;
1514  ++CurOp; // Encoded in VEX.VVVV
1515  break;
1516  }

1518  emitByte(BaseOpcode, OS);
1519  unsigned SrcRegNum = CurOp + 1;
1520

1522  ++SrcRegNum;

1532  }

1558  TSFlags, Kind, StartByte, OS, Fixups, STI, ForceSIB);
1559  CurOp = FirstMemOp + X86::AddrNumOperands;
1560  if (HasVEX_I8Reg)
1561  I8RegNum = getX86RegEncoding(MI, CurOp++);
1562  break;
1563  }
1564  case X86II::MRMSrcMem4VOp3: {
1565  unsigned FirstMemOp = CurOp + 1;
1566 
1567  emitByte(BaseOpcode, OS);
1568 
1569  emitMemModRMByte(MI, FirstMemOp, getX86RegNum(MI.getOperand(CurOp)),
1570  TSFlags, Kind, StartByte, OS, Fixups, STI);
1571  CurOp = FirstMemOp + X86::AddrNumOperands;

1573  break;
1574  }
1575  case X86II::MRMSrcMemOp4: {

1577

1581  assert(HasVEX_I8Reg && "MRMSrcRegOp4 should imply VEX_I8Reg");

1584  emitByte(BaseOpcode, OS);
1585 
1586  emitMemModRMByte(MI, FirstMemOp, getX86RegNum(MI.getOperand(CurOp)),
1587  TSFlags, Kind, StartByte, OS, Fixups, STI);
1588  CurOp = FirstMemOp + X86::AddrNumOperands;
1589  break;
1590  }
1591  case X86II::MRMSrcMemCC: {
1592  unsigned RegOp = CurOp++;
1593  unsigned FirstMemOp = CurOp;
1594  CurOp = FirstMemOp + X86::AddrNumOperands;
1595 
1596  unsigned CC = MI.getOperand(CurOp++).getImm();
1597  emitByte(BaseOpcode + CC, OS);
1598 
1599  emitMemModRMByte(MI, FirstMemOp, getX86RegNum(MI.getOperand(RegOp)),

1602  }
1603

1606 
1607  unsigned CC = MI.getOperand(CurOp++).getImm();

1612 
1613  case X86II::MRMXr:
1614  case X86II::MRM0r:
1615  case X86II::MRM1r:
1616  case X86II::MRM2r:
1617  case X86II::MRM3r:
