31 const RegisterBank &
32 PPCRegisterBankInfo::getRegBankFromRegClass(const TargetRegisterClass &RC,
33  LLT Ty) const {
34  switch (RC.getID()) {
35  case PPC::G8RCRegClassID:
36  case PPC::G8RC_NOX0RegClassID:
37  case PPC::G8RC_and_G8RC_NOX0RegClassID:
38  case PPC::GPRCRegClassID:
39  case PPC::GPRC_NOR0RegClassID:
40  case PPC::GPRC_and_GPRC_NOR0RegClassID:
41  return getRegBank(PPC::GPRRegBankID);
42  case PPC::VSFRCRegClassID:
43  case PPC::SPILLTOVSRRC_and_VSFRCRegClassID:
44  case PPC::SPILLTOVSRRC_and_VFRCRegClassID:
45  case PPC::SPILLTOVSRRC_and_F4RCRegClassID:
46  case PPC::F8RCRegClassID:
47  case PPC::VFRCRegClassID:
48  case PPC::VSSRCRegClassID:
49  case PPC::F4RCRegClassID:
50  return getRegBank(PPC::FPRRegBankID);
51  case PPC::CRRCRegClassID:
52  case PPC::CRBITRCRegClassID:
53  return getRegBank(PPC::CRRegBankID);
54  default:
55  llvm_unreachable("Unexpected register class");
56  }

59 const RegisterBankInfo::InstructionMapping &
60 PPCRegisterBankInfo::getInstrMapping(const MachineInstr &MI) const {
61  const unsigned Opc = MI.getOpcode();
62 
63  // Try the default logic for non-generic instructions that are either copies
64  // or already have some operands assigned to banks.
65  if (!isPreISelGenericOpcode(Opc) || Opc == TargetOpcode::G_PHI) {
66  const RegisterBankInfo::InstructionMapping &Mapping =
67  getInstrMappingImpl(MI);
68  if (Mapping.isValid())
69  return Mapping;
70  }
71 
72  const MachineFunction &MF = *MI.getParent()->getParent();
73  const MachineRegisterInfo &MRI = MF.getRegInfo();
74  const TargetSubtargetInfo &STI = MF.getSubtarget();
75  const TargetRegisterInfo &TRI = *STI.getRegisterInfo();
76 
77  unsigned NumOperands = MI.getNumOperands();
78  const ValueMapping *OperandsMapping = nullptr;
79  unsigned Cost = 1;
80  unsigned MappingID = DefaultMappingID;
81 
82  switch (Opc) {
83  // Arithmetic ops.
84  case TargetOpcode::G_ADD:
85  case TargetOpcode::G_SUB:
86  // Bitwise ops.
87  case TargetOpcode::G_AND:
88  case TargetOpcode::G_OR:
89  case TargetOpcode::G_XOR:
90  // Extension ops.
91  case TargetOpcode::G_SEXT:
92  case TargetOpcode::G_ZEXT:
93  case TargetOpcode::G_ANYEXT:
94  assert(NumOperands <= 3 &&
95  "This code is for instructions with 3 or less operands");
96  OperandsMapping = getValueMapping(PMI_GPR64);
97  break;
98  case TargetOpcode::G_FADD:
99  case TargetOpcode::G_FSUB:
100  case TargetOpcode::G_FMUL:
101  case TargetOpcode::G_FDIV: {
102  Register SrcReg = MI.getOperand(1).getReg();
103  unsigned Size = getSizeInBits(SrcReg, MRI, TRI);
104 
105  assert((Size == 32 || Size == 64) && "Unsupported floating point types!\n");
106  OperandsMapping = getValueMapping(Size == 32 ? PMI_FPR32 : PMI_FPR64);
107  break;
108  }
109  case TargetOpcode::G_FCMP: {
110  unsigned CmpSize = MRI.getType(MI.getOperand(2).getReg()).getSizeInBits();
111 
112  OperandsMapping = getOperandsMapping(
113  {getValueMapping(PMI_CR), nullptr,
114  getValueMapping(CmpSize == 32 ? PMI_FPR32 : PMI_FPR64),
115  getValueMapping(CmpSize == 32 ? PMI_FPR32 : PMI_FPR64)});
116  break;
117  }
118  case TargetOpcode::G_CONSTANT:
119  OperandsMapping = getOperandsMapping({getValueMapping(PMI_GPR64), nullptr});
120  break;
121  case TargetOpcode::G_CONSTANT_POOL:
122  OperandsMapping = getOperandsMapping({getValueMapping(PMI_GPR64), nullptr});
123  break;
124  case TargetOpcode::G_FPTOUI:
125  case TargetOpcode::G_FPTOSI: {
126  Register SrcReg = MI.getOperand(1).getReg();
127  unsigned Size = getSizeInBits(SrcReg, MRI, TRI);
128 
129  OperandsMapping = getOperandsMapping(
130  {getValueMapping(PMI_GPR64),
131  getValueMapping(Size == 32 ? PMI_FPR32 : PMI_FPR64)});
132  break;
133  }
134  case TargetOpcode::G_UITOFP:
135  case TargetOpcode::G_SITOFP: {
136  Register SrcReg = MI.getOperand(0).getReg();
137  unsigned Size = getSizeInBits(SrcReg, MRI, TRI);
138 
139  OperandsMapping =
140  getOperandsMapping({getValueMapping(Size == 32 ? PMI_FPR32 : PMI_FPR64),
141  getValueMapping(PMI_GPR64)});
142  break;
143  }
144  case TargetOpcode::G_LOAD: {
145  unsigned Size = MRI.getType(MI.getOperand(0).getReg()).getSizeInBits();
146  // Check if that load feeds fp instructions.
147  if (any_of(MRI.use_nodbg_instructions(MI.getOperand(0).getReg()),
148  [&](const MachineInstr &UseMI) {
149  // If we have at least one direct use in a FP instruction,
150  // assume this was a floating point load in the IR. If it was
151  // not, we would have had a bitcast before reaching that
152  // instruction.
153  //
154  // Int->FP conversion operations are also captured in
155  // onlyDefinesFP().
156  return onlyUsesFP(UseMI, MRI, TRI);
157  }))
158  OperandsMapping = getOperandsMapping(
159  {getValueMapping(Size == 64 ? PMI_FPR64 : PMI_FPR32),
160  getValueMapping(PMI_GPR64)});
161  else
162  OperandsMapping = getOperandsMapping(
163  {getValueMapping(Size == 64 ? PMI_GPR64 : PMI_GPR32),
164  getValueMapping(PMI_GPR64)});
165  break;
166  }
167  case TargetOpcode::G_STORE: {
168  // Check if the store is fed by fp instructions.
169  MachineInstr *DefMI = MRI.getVRegDef(MI.getOperand(0).getReg());
170  unsigned Size = MRI.getType(MI.getOperand(0).getReg()).getSizeInBits();
171  if (onlyDefinesFP(*DefMI, MRI, TRI))
172  OperandsMapping = getOperandsMapping(
173  {getValueMapping(Size == 64 ? PMI_FPR64 : PMI_FPR32),
174  getValueMapping(PMI_GPR64)});
175  else
176  OperandsMapping = getOperandsMapping(
177  {getValueMapping(Size == 64 ? PMI_GPR64 : PMI_GPR32),
178  getValueMapping(PMI_GPR64)});
179  break;
180  }
181  case TargetOpcode::G_INTRINSIC_W_SIDE_EFFECTS: {
182  // FIXME: We have to check every operand in this MI and compute value
183  // mapping accordingly.
184  SmallVector<const ValueMapping *, 8> OpdsMapping(NumOperands);
185  OperandsMapping = getOperandsMapping(OpdsMapping);
186  break;
187  }
188  default:
189  return getInvalidInstructionMapping();
190  }
191 
192  return getInstructionMapping(MappingID, Cost, OperandsMapping, NumOperands);
193 }

239 /// FIXME: this is copied from target AArch64. Needs some code refactor here to
240 /// put this function in class RegisterBankInfo.
241 bool PPCRegisterBankInfo::hasFPConstraints(const MachineInstr &MI,
242  const MachineRegisterInfo &MRI,
243  const TargetRegisterInfo &TRI,
244  unsigned Depth) const {
245  unsigned Op = MI.getOpcode();
246  if (Op == TargetOpcode::G_INTRINSIC && isFPIntrinsic(MI.getIntrinsicID()))
247  return true;
248 
249  // Do we have an explicit floating point instruction?
250  if (isPreISelGenericFloatingPointOpcode(Op))
251  return true;
252 
253  // No. Check if we have a copy-like instruction. If we do, then we could
254  // still be fed by floating point instructions.
255  if (Op != TargetOpcode::COPY && !MI.isPHI() &&
256  !isPreISelGenericOptimizationHint(Op))
257  return false;
258 
259  // Check if we already know the register bank.
260  auto *RB = getRegBank(MI.getOperand(0).getReg(), MRI, TRI);
261  if (RB == &PPC::FPRRegBank)
262  return true;
263  if (RB == &PPC::GPRRegBank)
264  return false;
265 
266  // We don't know anything.
267  //
268  // If we have a phi, we may be able to infer that it will be assigned a FPR
269  // based off of its inputs.
270  if (!MI.isPHI() || Depth > MaxFPRSearchDepth)
271  return false;
272 
273  return any_of(MI.explicit_uses(), [&](const MachineOperand &Op) {
274  return Op.isReg() &&
275  onlyDefinesFP(*MRI.getVRegDef(Op.getReg()), MRI, TRI, Depth + 1);
276  });
277 }
