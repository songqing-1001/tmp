412  bool isMultiStoresCheaperThanBitsMerge(EVT LTy, EVT HTy) const override {
413  // If the pair to store is a mixture of float and int values, we will
414  // save two bitwise instructions and one float-to-int instruction and
415  // increase one store instruction. There is potentially a more
416  // significant benefit because it avoids the float->int domain switch
417  // for input value. So It is more likely a win.
418  if ((LTy.isFloatingPoint() && HTy.isInteger()) ||
419  (LTy.isInteger() && HTy.isFloatingPoint()))
420  return true;
421  // If the pair only contains int values, we will save two bitwise
422  // instructions and increase one store instruction (costing one more
423  // store buffer). Since the benefit is more blurred we leave such a pair
424  // out until we get testcase to prove it is a win.
425  return false;
426  }
