61 static DecodeStatus DecodeGPRRegisterClass(MCInst &Inst, uint32_t RegNo,
62  uint64_t Address,
63  const MCDisassembler *Decoder) {
64  const FeatureBitset &FeatureBits =
65  Decoder->getSubtargetInfo().getFeatureBits();

67 
68  if (RegNo >= 32 || (IsRV32E && RegNo >= 16))
69  return MCDisassembler::Fail;
70 
71  MCRegister Reg = RISCV::X0 + RegNo;

261 // Add implied SP operand for instructions *SP compressed instructions. The SP
262 // operand isn't explicitly encoded in the instruction.
263 static void addImplySP(MCInst &Inst, int64_t Address,
264  const MCDisassembler *Decoder) {
265  if (Inst.getOpcode() == RISCV::C_LWSP || Inst.getOpcode() == RISCV::C_SWSP ||
266  Inst.getOpcode() == RISCV::C_LDSP || Inst.getOpcode() == RISCV::C_SDSP ||
267  Inst.getOpcode() == RISCV::C_FLWSP ||
268  Inst.getOpcode() == RISCV::C_FSWSP ||
269  Inst.getOpcode() == RISCV::C_FLDSP ||
270  Inst.getOpcode() == RISCV::C_FSDSP ||
271  Inst.getOpcode() == RISCV::C_ADDI4SPN) {
272  DecodeGPRRegisterClass(Inst, 2, Address, Decoder);
273  }
274  if (Inst.getOpcode() == RISCV::C_ADDI16SP) {
275  DecodeGPRRegisterClass(Inst, 2, Address, Decoder);
276  DecodeGPRRegisterClass(Inst, 2, Address, Decoder);
277  }
278 }

432 DecodeStatus RISCVDisassembler::getInstruction(MCInst &MI, uint64_t &Size,
433  ArrayRef<uint8_t> Bytes,
434  uint64_t Address,
435  raw_ostream &CS) const {
436  // TODO: This will need modification when supporting instruction set
437  // extensions with instructions > 32-bits (up to 176 bits wide).
438  uint32_t Insn;
439  DecodeStatus Result;
440 
441  // It's a 32 bit instruction if bit 0 and 1 are 1.
442  if ((Bytes[0] & 0x3) == 0x3) {
443  if (Bytes.size() < 4) {
444  Size = 0;
445  return MCDisassembler::Fail;
446  }
447  Size = 4;
448 
449  Insn = support::endian::read32le(Bytes.data());
450 
451  if (STI.getFeatureBits()[RISCV::FeatureStdExtZdinx] &&
452  !STI.getFeatureBits()[RISCV::Feature64Bit]) {
453  LLVM_DEBUG(dbgs() << "Trying RV32Zdinx table (Double in Integer and"
454  "rv32)\n");
455  Result = decodeInstruction(DecoderTableRV32Zdinx32, MI, Insn, Address,
456  this, STI);
457  if (Result != MCDisassembler::Fail)
458  return Result;
459  }
460  if (STI.getFeatureBits()[RISCV::FeatureStdExtZfinx]) {
461  LLVM_DEBUG(dbgs() << "Trying RVZfinx table (Float in Integer):\n");
462  Result = decodeInstruction(DecoderTableRVZfinx32, MI, Insn, Address, this,
463  STI);
464  if (Result != MCDisassembler::Fail)
465  return Result;
466  }
467  if (STI.getFeatureBits()[RISCV::FeatureVendorXVentanaCondOps]) {
468  LLVM_DEBUG(dbgs() << "Trying Ventana custom opcode table:\n");
469  Result = decodeInstruction(DecoderTableVentana32, MI, Insn, Address, this,
470  STI);
471  if (Result != MCDisassembler::Fail)
472  return Result;
473  }
474  if (STI.getFeatureBits()[RISCV::FeatureVendorXTHeadBa]) {
475  LLVM_DEBUG(dbgs() << "Trying XTHeadBa custom opcode table:\n");
476  Result = decodeInstruction(DecoderTableTHeadBa32, MI, Insn, Address, this,
477  STI);
478  if (Result != MCDisassembler::Fail)
479  return Result;
480  }
481  if (STI.getFeatureBits()[RISCV::FeatureVendorXTHeadBb]) {
482  LLVM_DEBUG(dbgs() << "Trying XTHeadBb custom opcode table:\n");
483  Result = decodeInstruction(DecoderTableTHeadBb32, MI, Insn, Address, this,
484  STI);
485  if (Result != MCDisassembler::Fail)
486  return Result;
487  }
488  if (STI.getFeatureBits()[RISCV::FeatureVendorXTHeadBs]) {
489  LLVM_DEBUG(dbgs() << "Trying XTHeadBs custom opcode table:\n");
490  Result = decodeInstruction(DecoderTableTHeadBs32, MI, Insn, Address, this,
491  STI);
492  if (Result != MCDisassembler::Fail)
493  return Result;
494  }
495  if (STI.getFeatureBits()[RISCV::FeatureVendorXTHeadMac]) {
496  LLVM_DEBUG(dbgs() << "Trying XTHeadMac custom opcode table:\n");
497  Result = decodeInstruction(DecoderTableTHeadMac32, MI, Insn, Address,
498  this, STI);
499  if (Result != MCDisassembler::Fail)
500  return Result;
501  }
502  if (STI.getFeatureBits()[RISCV::FeatureVendorXTHeadVdot]) {
503  LLVM_DEBUG(dbgs() << "Trying XTHeadVdot custom opcode table:\n");
504  Result =
505  decodeInstruction(DecoderTableTHeadV32, MI, Insn, Address, this, STI);
506  if (Result != MCDisassembler::Fail)
507  return Result;
508  }
509 
510  LLVM_DEBUG(dbgs() << "Trying RISCV32 table :\n");
511  return decodeInstruction(DecoderTable32, MI, Insn, Address, this, STI);
512  }
513 
514  if (Bytes.size() < 2) {
515  Size = 0;
516  return MCDisassembler::Fail;
517  }
518  Size = 2;
519 
520  Insn = support::endian::read16le(Bytes.data());
521 
522  if (!STI.getFeatureBits()[RISCV::Feature64Bit]) {
523  LLVM_DEBUG(
524  dbgs() << "Trying RISCV32Only_16 table (16-bit Instruction):\n");
525  // Calling the auto-generated decoder function.
526  Result = decodeInstruction(DecoderTableRISCV32Only_16, MI, Insn, Address,
527  this, STI);
528  if (Result != MCDisassembler::Fail)
529  return Result;
530  }
531 
532  LLVM_DEBUG(dbgs() << "Trying RISCV_C table (16-bit Instruction):\n");
533  // Calling the auto-generated decoder function.
534  return decodeInstruction(DecoderTable16, MI, Insn, Address, this, STI);
