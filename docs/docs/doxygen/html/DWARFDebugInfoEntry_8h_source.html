37  /// Extracts a debug info entry, which is a child of a given unit,
38  /// starting at a given offset. If DIE can't be extracted, returns false and
39  /// doesn't change OffsetPtr.

47  /// Returns index of the parent die.

51

53  }
54

68  }
69
