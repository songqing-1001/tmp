97 uint64_t MSP430AsmBackend::adjustFixupValue(const MCFixup &Fixup,
98  uint64_t Value,
99  MCContext &Ctx) const {
100  unsigned Kind = Fixup.getKind();
101  switch (Kind) {
102  case MSP430::fixup_10_pcrel: {
103  if (Value & 0x1)
104  Ctx.reportError(Fixup.getLoc(), "fixup value must be 2-byte aligned");
105 
106  // Offset is signed
107  int16_t Offset = Value;
108  // Jumps are in words
109  Offset >>= 1;
110  // PC points to the next instruction so decrement by one
111  --Offset;
112 
113  if (Offset < -512 || Offset > 511)
114  Ctx.reportError(Fixup.getLoc(), "fixup value out of range");
115 
116  // Mask 10 bits
117  Offset &= 0x3ff;
118 
119  return Offset;
120  }
121  default:
122  return Value;
123  }
124 }

126 void MSP430AsmBackend::applyFixup(const MCAssembler &Asm, const MCFixup &Fixup,
127  const MCValue &Target,
128  MutableArrayRef<char> Data,
129  uint64_t Value, bool IsResolved,
130  const MCSubtargetInfo *STI) const {
131  Value = adjustFixupValue(Fixup, Value, Asm.getContext());
132  MCFixupKindInfo Info = getFixupKindInfo(Fixup.getKind());
133  if (!Value)
134  return; // Doesn't change encoding.
135 
136  // Shift the value into position.
137  Value <<= Info.TargetOffset;
138 
139  unsigned Offset = Fixup.getOffset();
140  unsigned NumBytes = alignTo(Info.TargetSize + Info.TargetOffset, 8) / 8;
141 
142  assert(Offset + NumBytes <= Data.size() && "Invalid fixup offset!");
143 
144  // For each byte of the fragment that the fixup touches, mask in the
145  // bits from the fixup value.
146  for (unsigned i = 0; i != NumBytes; ++i) {
147  Data[Offset + i] |= uint8_t((Value >> (i * 8)) & 0xff);
148  }
149 }

151 bool MSP430AsmBackend::writeNopData(raw_ostream &OS, uint64_t Count,
152  const MCSubtargetInfo *STI) const {
153  if ((Count % 2) != 0)
154  return false;
155 
156  // The canonical nop on MSP430 is mov #0, r3
157  uint64_t NopCount = Count / 2;
158  while (NopCount--)
159  OS.write("\x03\x43", 2);
160 
161  return true;
162 }
