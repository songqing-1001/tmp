111 bool HardwareLoopInfo::isHardwareLoopCandidate(ScalarEvolution &SE,
112  LoopInfo &LI, DominatorTree &DT,
113  bool ForceNestedLoop,
114  bool ForceHardwareLoopPHI) {
115  SmallVector<BasicBlock *, 4> ExitingBlocks;
116  L->getExitingBlocks(ExitingBlocks);
117 
118  for (BasicBlock *BB : ExitingBlocks) {
119  // If we pass the updated counter back through a phi, we need to know
120  // which latch the updated value will be coming from.
121  if (!L->isLoopLatch(BB)) {
122  if (ForceHardwareLoopPHI || CounterInReg)
123  continue;
124  }
125 
126  const SCEV *EC = SE.getExitCount(L, BB);
127  if (isa<SCEVCouldNotCompute>(EC))
128  continue;
129  if (const SCEVConstant *ConstEC = dyn_cast<SCEVConstant>(EC)) {
130  if (ConstEC->getValue()->isZero())
131  continue;
132  } else if (!SE.isLoopInvariant(EC, L))
133  continue;
134 
135  if (SE.getTypeSizeInBits(EC->getType()) > CountType->getBitWidth())
136  continue;
137 
138  // If this exiting block is contained in a nested loop, it is not eligible
139  // for insertion of the branch-and-decrement since the inner loop would
140  // end up messing up the value in the CTR.
141  if (!IsNestingLegal && LI.getLoopFor(BB) != L && !ForceNestedLoop)
142  continue;
143 
144  // We now have a loop-invariant count of loop iterations (which is not the
145  // constant zero) for which we know that this loop will not exit via this
146  // existing block.
147 
148  // We need to make sure that this block will run on every loop iteration.
149  // For this to be true, we must dominate all blocks with backedges. Such
150  // blocks are in-loop predecessors to the header block.
151  bool NotAlways = false;
152  for (BasicBlock *Pred : predecessors(L->getHeader())) {
153  if (!L->contains(Pred))
154  continue;
155 
156  if (!DT.dominates(BB, Pred)) {
157  NotAlways = true;
158  break;
159  }
160  }
161 
162  if (NotAlways)
163  continue;
164 
165  // Make sure this blocks ends with a conditional branch.
166  Instruction *TI = BB->getTerminator();
167  if (!TI)
168  continue;
169 
170  if (BranchInst *BI = dyn_cast<BranchInst>(TI)) {
171  if (!BI->isConditional())
172  continue;
173 
174  ExitBranch = BI;
175  } else
176  continue;
177 
178  // Note that this block may not be the loop latch block, even if the loop
179  // has a latch block.
180  ExitBlock = BB;
181  ExitCount = EC;
182  break;
183  }
184 
185  if (!ExitBlock)
186  return false;
187  return true;
188 }

742 TargetTransformInfo::OperandValueInfo
743 TargetTransformInfo::getOperandInfo(const Value *V) {
744  OperandValueKind OpInfo = OK_AnyValue;
745  OperandValueProperties OpProps = OP_None;
746 
747  if (isa<ConstantInt>(V) || isa<ConstantFP>(V)) {
748  if (const auto *CI = dyn_cast<ConstantInt>(V)) {
749  if (CI->getValue().isPowerOf2())
750  OpProps = OP_PowerOf2;
751  else if (CI->getValue().isNegatedPowerOf2())
752  OpProps = OP_NegatedPowerOf2;
753  }
754  return {OK_UniformConstantValue, OpProps};
755  }
756 
757  // A broadcast shuffle creates a uniform value.
758  // TODO: Add support for non-zero index broadcasts.
759  // TODO: Add support for different source vector width.
760  if (const auto *ShuffleInst = dyn_cast<ShuffleVectorInst>(V))
761  if (ShuffleInst->isZeroEltSplat())
762  OpInfo = OK_UniformValue;
763 
764  const Value *Splat = getSplatValue(V);
765 
766  // Check for a splat of a constant or for a non uniform vector of constants
767  // and check if the constant(s) are all powers of two.
768  if (isa<ConstantVector>(V) || isa<ConstantDataVector>(V)) {
769  OpInfo = OK_NonUniformConstantValue;
770  if (Splat) {
771  OpInfo = OK_UniformConstantValue;
772  if (auto *CI = dyn_cast<ConstantInt>(Splat)) {
773  if (CI->getValue().isPowerOf2())
774  OpProps = OP_PowerOf2;
775  else if (CI->getValue().isNegatedPowerOf2())
776  OpProps = OP_NegatedPowerOf2;
777  }
778  } else if (const auto *CDS = dyn_cast<ConstantDataSequential>(V)) {
779  bool AllPow2 = true, AllNegPow2 = true;
780  for (unsigned I = 0, E = CDS->getNumElements(); I != E; ++I) {
781  if (auto *CI = dyn_cast<ConstantInt>(CDS->getElementAsConstant(I))) {
782  AllPow2 &= CI->getValue().isPowerOf2();
783  AllNegPow2 &= CI->getValue().isNegatedPowerOf2();
784  if (AllPow2 || AllNegPow2)
785  continue;
786  }
787  AllPow2 = AllNegPow2 = false;
788  break;
789  }
790  OpProps = AllPow2 ? OP_PowerOf2 : OpProps;
791  OpProps = AllNegPow2 ? OP_NegatedPowerOf2 : OpProps;
792  }
793  }
794 
795  // Check for a splat of a uniform value. This is not loop aware, so return
796  // true only for the obviously uniform cases (argument, globalvalue)
797  if (Splat && (isa<Argument>(Splat) || isa<GlobalValue>(Splat)))
798  OpInfo = OK_UniformValue;
799 
800  return {OpInfo, OpProps};
801 }

825 TTI::CastContextHint
826 TargetTransformInfo::getCastContextHint(const Instruction *I) {
827  if (!I)
828  return CastContextHint::None;
829 
830  auto getLoadStoreKind = [](const Value *V, unsigned LdStOp, unsigned MaskedOp,
831  unsigned GatScatOp) {
832  const Instruction *I = dyn_cast<Instruction>(V);
833  if (!I)
834  return CastContextHint::None;
835 
836  if (I->getOpcode() == LdStOp)
837  return CastContextHint::Normal;
838 
839  if (const IntrinsicInst *II = dyn_cast<IntrinsicInst>(I)) {
840  if (II->getIntrinsicID() == MaskedOp)
841  return TTI::CastContextHint::Masked;
842  if (II->getIntrinsicID() == GatScatOp)
843  return TTI::CastContextHint::GatherScatter;
844  }
845 
846  return TTI::CastContextHint::None;
847  };
848 
849  switch (I->getOpcode()) {
850  case Instruction::ZExt:
851  case Instruction::SExt:
852  case Instruction::FPExt:
853  return getLoadStoreKind(I->getOperand(0), Instruction::Load,
854  Intrinsic::masked_load, Intrinsic::masked_gather);
855  case Instruction::Trunc:
856  case Instruction::FPTrunc:
857  if (I->hasOneUse())
858  return getLoadStoreKind(*I->user_begin(), Instruction::Store,
859  Intrinsic::masked_store,
860  Intrinsic::masked_scatter);
861  break;
862  default:
863  return CastContextHint::None;
864  }
865 
866  return TTI::CastContextHint::None;
867 }
