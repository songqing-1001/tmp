163 void DebugInfoFinder::processModule(const Module &M) {
164  for (auto *CU : M.debug_compile_units())
165  processCompileUnit(CU);
166  for (auto &F : M.functions()) {
167  if (auto *SP = cast_or_null<DISubprogram>(F.getSubprogram()))
168  processSubprogram(SP);
169  // There could be subprograms from inlined functions referenced from
170  // instructions only. Walk the function to find them.
171  for (const BasicBlock &BB : F)
172  for (const Instruction &I : BB)
173  processInstruction(M, I);
174  }

177 void DebugInfoFinder::processCompileUnit(DICompileUnit *CU) {
178  if (!addCompileUnit(CU))
179  return;
180  for (auto *DIG : CU->getGlobalVariables()) {
181  if (!addGlobalVariable(DIG))
182  continue;
183  auto *GV = DIG->getVariable();
184  processScope(GV->getScope());
185  processType(GV->getType());
186  }
187  for (auto *ET : CU->getEnumTypes())
188  processType(ET);
189  for (auto *RT : CU->getRetainedTypes())
190  if (auto *T = dyn_cast<DIType>(RT))
191  processType(T);
192  else
193  processSubprogram(cast<DISubprogram>(RT));
194  for (auto *Import : CU->getImportedEntities()) {
195  auto *Entity = Import->getEntity();
196  if (auto *T = dyn_cast<DIType>(Entity))
197  processType(T);
198  else if (auto *SP = dyn_cast<DISubprogram>(Entity))
199  processSubprogram(SP);
200  else if (auto *NS = dyn_cast<DINamespace>(Entity))
201  processScope(NS->getScope());
202  else if (auto *M = dyn_cast<DIModule>(Entity))
203  processScope(M->getScope());
204  }

273 void DebugInfoFinder::processSubprogram(DISubprogram *SP) {
274  if (!addSubprogram(SP))
275  return;
276  processScope(SP->getScope());
277  // Some of the users, e.g. CloneFunctionInto / CloneModule, need to set up a
278  // ValueMap containing identity mappings for all of the DICompileUnit's, not
279  // just DISubprogram's, referenced from anywhere within the Function being
280  // cloned prior to calling MapMetadata / RemapInstruction to avoid their
281  // duplication later as DICompileUnit's are also directly referenced by
282  // llvm.dbg.cu list. Thefore we need to collect DICompileUnit's here as well.
283  // Also, DICompileUnit's may reference DISubprogram's too and therefore need
284  // to be at least looked through.
285  processCompileUnit(SP->getUnit());
286  processType(SP->getType());
287  for (auto *Element : SP->getTemplateParams()) {
288  if (auto *TType = dyn_cast<DITemplateTypeParameter>(Element)) {
289  processType(TType->getType());
290  } else if (auto *TVal = dyn_cast<DITemplateValueParameter>(Element)) {
291  processType(TVal->getType());
292  }
293  }

365 static MDNode *updateLoopMetadataDebugLocationsImpl(
366  MDNode *OrigLoopID, function_ref<Metadata *(Metadata *)> Updater) {
367  assert(OrigLoopID && OrigLoopID->getNumOperands() > 0 &&
368  "Loop ID needs at least one operand");
369  assert(OrigLoopID && OrigLoopID->getOperand(0).get() == OrigLoopID &&
370  "Loop ID should refer to itself");
371 
372  // Save space for the self-referential LoopID.
373  SmallVector<Metadata *, 4> MDs = {nullptr};
374 
375  for (unsigned i = 1; i < OrigLoopID->getNumOperands(); ++i) {
376  Metadata *MD = OrigLoopID->getOperand(i);
377  if (!MD)
378  MDs.push_back(nullptr);
379  else if (Metadata *NewMD = Updater(MD))
380  MDs.push_back(NewMD);
381  }
382 
383  MDNode *NewLoopID = MDNode::getDistinct(OrigLoopID->getContext(), MDs);
384  // Insert the self-referential LoopID.
385  NewLoopID->replaceOperandWith(0, NewLoopID);
386  return NewLoopID;
387 }

398 /// Return true if a node is a DILocation or if a DILocation is
399 /// indirectly referenced by one of the node's children.
400 static bool isDILocationReachable(SmallPtrSetImpl<Metadata *> &Visited,
401  SmallPtrSetImpl<Metadata *> &Reachable,
402  Metadata *MD) {
403  MDNode *N = dyn_cast_or_null<MDNode>(MD);
404  if (!N)
405  return false;
406  if (isa<DILocation>(N) || Reachable.count(N))
407  return true;
408  if (!Visited.insert(N).second)
409  return false;
410  for (auto &OpIt : N->operands()) {
411  Metadata *Op = OpIt.get();
412  if (isDILocationReachable(Visited, Reachable, Op)) {
413  // Don't return just yet as we want to visit all MD's children to
414  // initialize DILocationReachable in stripDebugLocFromLoopID
415  Reachable.insert(N);
416  }
417  }
418  return Reachable.count(N);
419 }

421 static bool isAllDILocation(SmallPtrSetImpl<Metadata *> &Visited,
422  SmallPtrSetImpl<Metadata *> &AllDILocation,
423  const SmallPtrSetImpl<Metadata *> &DIReachable,
424  Metadata *MD) {
425  MDNode *N = dyn_cast_or_null<MDNode>(MD);
426  if (!N)
427  return false;
428  if (isa<DILocation>(N) || AllDILocation.count(N))
429  return true;
430  if (!DIReachable.count(N))
431  return false;
432  if (!Visited.insert(N).second)
433  return false;
434  for (auto &OpIt : N->operands()) {
435  Metadata *Op = OpIt.get();
436  if (Op == MD)
437  continue;
438  if (!isAllDILocation(Visited, AllDILocation, DIReachable, Op)) {
439  return false;
440  }
441  }
442  AllDILocation.insert(N);
443  return true;

446 static Metadata *
447 stripLoopMDLoc(const SmallPtrSetImpl<Metadata *> &AllDILocation,
448  const SmallPtrSetImpl<Metadata *> &DIReachable, Metadata *MD) {
449  if (isa<DILocation>(MD) || AllDILocation.count(MD))
450  return nullptr;
451 
452  if (!DIReachable.count(MD))
453  return MD;
454 
455  MDNode *N = dyn_cast_or_null<MDNode>(MD);
456  if (!N)
457  return MD;
458 
459  SmallVector<Metadata *, 4> Args;
460  bool HasSelfRef = false;
461  for (unsigned i = 0; i < N->getNumOperands(); ++i) {
462  Metadata *A = N->getOperand(i);
463  if (!A) {
464  Args.push_back(nullptr);
465  } else if (A == MD) {
466  assert(i == 0 && "expected i==0 for self-reference");
467  HasSelfRef = true;
468  Args.push_back(nullptr);
469  } else if (Metadata *NewArg =
470  stripLoopMDLoc(AllDILocation, DIReachable, A)) {
471  Args.push_back(NewArg);
472  }
473  }
474  if (Args.empty() || (HasSelfRef && Args.size() == 1))
475  return nullptr;
476 
477  MDNode *NewMD = N->isDistinct() ? MDNode::getDistinct(N->getContext(), Args)
478  : MDNode::get(N->getContext(), Args);
479  if (HasSelfRef)
480  NewMD->replaceOperandWith(0, NewMD);
481  return NewMD;
482 }

484 static MDNode *stripDebugLocFromLoopID(MDNode *N) {
485  assert(!N->operands().empty() && "Missing self reference?");
486  SmallPtrSet<Metadata *, 8> Visited, DILocationReachable, AllDILocation;
487  // If we already visited N, there is nothing to do.
488  if (!Visited.insert(N).second)
489  return N;
490 
491  // If there is no debug location, we do not have to rewrite this
492  // MDNode. This loop also initializes DILocationReachable, later
493  // needed by updateLoopMetadataDebugLocationsImpl; the use of
494  // count_if avoids an early exit.
495  if (!llvm::count_if(llvm::drop_begin(N->operands()),
496  [&Visited, &DILocationReachable](const MDOperand &Op) {
497  return isDILocationReachable(
498  Visited, DILocationReachable, Op.get());
499  }))
500  return N;
501 
502  Visited.clear();
503  // If there is only the debug location without any actual loop metadata, we
504  // can remove the metadata.
505  if (llvm::all_of(llvm::drop_begin(N->operands()),
506  [&Visited, &AllDILocation,
507  &DILocationReachable](const MDOperand &Op) {
508  return isAllDILocation(Visited, AllDILocation,
509  DILocationReachable, Op.get());
510  }))
511  return nullptr;
512 
513  return updateLoopMetadataDebugLocationsImpl(
514  N, [&AllDILocation, &DILocationReachable](Metadata *MD) -> Metadata * {
515  return stripLoopMDLoc(AllDILocation, DILocationReachable, MD);
516  });
517 }

519 bool llvm::stripDebugInfo(Function &F) {
520  bool Changed = false;
521  if (F.hasMetadata(LLVMContext::MD_dbg)) {
522  Changed = true;
523  F.setSubprogram(nullptr);
524  }
525 
526  DenseMap<MDNode *, MDNode *> LoopIDsMap;
527  for (BasicBlock &BB : F) {
528  for (Instruction &I : llvm::make_early_inc_range(BB)) {
529  if (isa<DbgInfoIntrinsic>(&I)) {
530  I.eraseFromParent();
531  Changed = true;
532  continue;
533  }
534  if (I.getDebugLoc()) {
535  Changed = true;
536  I.setDebugLoc(DebugLoc());
537  }
538  if (auto *LoopID = I.getMetadata(LLVMContext::MD_loop)) {
539  auto *NewLoopID = LoopIDsMap.lookup(LoopID);
540  if (!NewLoopID)
541  NewLoopID = LoopIDsMap[LoopID] = stripDebugLocFromLoopID(LoopID);
542  if (NewLoopID != LoopID)
543  I.setMetadata(LLVMContext::MD_loop, NewLoopID);
544  }
545  // Strip other attachments that are or use debug info.
546  if (I.hasMetadataOtherThanDebugLoc()) {
547  // Heapallocsites point into the DIType system.
548  I.setMetadata("heapallocsite", nullptr);
549  // DIAssignID are debug info metadata primitives.
550  I.setMetadata(LLVMContext::MD_DIAssignID, nullptr);
551  }
552  }
553  }
554  return Changed;
555 }

558  bool Changed = false;
559

629  DISubprogram *getReplacementSubprogram(DISubprogram *MDS) {
630  auto *FileAndScope = cast_or_null<DIFile>(map(MDS->getFile()));
631  StringRef LinkageName = MDS->getName().empty() ? MDS->getLinkageName() : "";
632  DISubprogram *Declaration = nullptr;
633  auto *Type = cast_or_null<DISubroutineType>(map(MDS->getType()));
634  DIType *ContainingType =
635  cast_or_null<DIType>(map(MDS->getContainingType()));
636  auto *Unit = cast_or_null<DICompileUnit>(map(MDS->getUnit()));
637  auto Variables = nullptr;
638  auto TemplateParams = nullptr;
639 
640  // Make a distinct DISubprogram, for situations that warrent it.
641  auto distinctMDSubprogram = [&]() {
642  return DISubprogram::getDistinct(
643  MDS->getContext(), FileAndScope, MDS->getName(), LinkageName,
644  FileAndScope, MDS->getLine(), Type, MDS->getScopeLine(),
645  ContainingType, MDS->getVirtualIndex(), MDS->getThisAdjustment(),
646  MDS->getFlags(), MDS->getSPFlags(), Unit, TemplateParams, Declaration,
647  Variables);
648  };
649 
650  if (MDS->isDistinct())
651  return distinctMDSubprogram();
652 
653  auto *NewMDS = DISubprogram::get(
654  MDS->getContext(), FileAndScope, MDS->getName(), LinkageName,
655  FileAndScope, MDS->getLine(), Type, MDS->getScopeLine(), ContainingType,
656  MDS->getVirtualIndex(), MDS->getThisAdjustment(), MDS->getFlags(),
657  MDS->getSPFlags(), Unit, TemplateParams, Declaration, Variables);
658 
659  StringRef OldLinkageName = MDS->getLinkageName();
660 
661  // See if we need to make a distinct one.
662  auto OrigLinkage = NewToLinkageName.find(NewMDS);
663  if (OrigLinkage != NewToLinkageName.end()) {
664  if (OrigLinkage->second == OldLinkageName)
665  // We're good.
666  return NewMDS;
667 
668  // Otherwise, need to make a distinct one.
669  // TODO: Query the map to see if we already have one.
670  return distinctMDSubprogram();
671  }
672 
673  NewToLinkageName.insert({NewMDS, MDS->getLinkageName()});
674  return NewMDS;
675  }

719  /// Attempt to re-map N to a newly created node.
720  void remap(MDNode *N) {
721  if (Replacements.count(N))
722  return;
723 
724  auto doRemap = [&](MDNode *N) -> MDNode * {
725  if (!N)
726  return nullptr;
727  if (auto *MDSub = dyn_cast<DISubprogram>(N)) {
728  remap(MDSub->getUnit());
729  return getReplacementSubprogram(MDSub);
730  }
731  if (isa<DISubroutineType>(N))
732  return EmptySubroutineType;
733  if (auto *CU = dyn_cast<DICompileUnit>(N))
734  return getReplacementCU(CU);
735  if (isa<DIFile>(N))
736  return N;
737  if (auto *MDLB = dyn_cast<DILexicalBlockBase>(N))
738  // Remap to our referenced scope (recursively).
739  return mapNode(MDLB->getScope());
740  if (auto *MLD = dyn_cast<DILocation>(N))
741  return getReplacementMDLocation(MLD);
742 
743  // Otherwise, if we see these, just drop them now. Not strictly necessary,
744  // but this speeds things up a little.
745  if (isa<DINode>(N))
746  return nullptr;
747 
748  return getReplacementMDNode(N);
749  };
750  Replacements[N] = doRemap(N);
751  }

759 void DebugTypeInfoRemoval::traverse(MDNode *N) {
760  if (!N || Replacements.count(N))
761  return;
762 
763  // To avoid cycles, as well as for efficiency sake, we will sometimes prune
764  // parts of the graph.
765  auto prune = [](MDNode *Parent, MDNode *Child) {
766  if (auto *MDS = dyn_cast<DISubprogram>(Parent))
767  return Child == MDS->getRetainedNodes().get();
768  return false;
769  };
770 
771  SmallVector<MDNode *, 16> ToVisit;
772  DenseSet<MDNode *> Opened;
773 
774  // Visit each node starting at N in post order, and map them.
775  ToVisit.push_back(N);
776  while (!ToVisit.empty()) {
777  auto *N = ToVisit.back();
778  if (!Opened.insert(N).second) {
779  // Close it.
780  remap(N);
781  ToVisit.pop_back();
782  continue;
783  }
784  for (auto &I : N->operands())
785  if (auto *MDN = dyn_cast_or_null<MDNode>(I))
786  if (!Opened.count(MDN) && !Replacements.count(MDN) && !prune(N, MDN) &&
787  !isa<DICompileUnit>(MDN))
788  ToVisit.push_back(MDN);
789  }
790 }

792 bool llvm::stripNonLineTableDebugInfo(Module &M) {
793  bool Changed = false;
794 
795  // First off, delete the debug intrinsics.
796  auto RemoveUses = [&](StringRef Name) {
797  if (auto *DbgVal = M.getFunction(Name)) {
798  while (!DbgVal->use_empty())
799  cast<Instruction>(DbgVal->user_back())->eraseFromParent();
800  DbgVal->eraseFromParent();
801  Changed = true;
802  }
803  };
804  RemoveUses("llvm.dbg.addr");
805  RemoveUses("llvm.dbg.declare");
806  RemoveUses("llvm.dbg.label");
807  RemoveUses("llvm.dbg.value");
808 
809  // Delete non-CU debug info named metadata nodes.
810  for (auto NMI = M.named_metadata_begin(), NME = M.named_metadata_end();
811  NMI != NME;) {
812  NamedMDNode *NMD = &*NMI;
813  ++NMI;
814  // Specifically keep dbg.cu around.
815  if (NMD->getName() == "llvm.dbg.cu")
816  continue;
817  }
818 
819  // Drop all dbg attachments from global variables.
820  for (auto &GV : M.globals())
821  GV.eraseMetadata(LLVMContext::MD_dbg);
822 
823  DebugTypeInfoRemoval Mapper(M.getContext());
824  auto remap = [&](MDNode *Node) -> MDNode * {
825  if (!Node)
826  return nullptr;
827  Mapper.traverseAndRemap(Node);
828  auto *NewNode = Mapper.mapNode(Node);
829  Changed |= Node != NewNode;
830  Node = NewNode;
831  return NewNode;
832  };
833 
834  // Rewrite the DebugLocs to be equivalent to what
835  // -gline-tables-only would have created.
836  for (auto &F : M) {
837  if (auto *SP = F.getSubprogram()) {
838  Mapper.traverseAndRemap(SP);
839  auto *NewSP = cast<DISubprogram>(Mapper.mapNode(SP));
840  Changed |= SP != NewSP;
841  F.setSubprogram(NewSP);
842  }
843  for (auto &BB : F) {
844  for (auto &I : BB) {
845  auto remapDebugLoc = [&](const DebugLoc &DL) -> DebugLoc {
846  auto *Scope = DL.getScope();
847  MDNode *InlinedAt = DL.getInlinedAt();
848  Scope = remap(Scope);
849  InlinedAt = remap(InlinedAt);
850  return DILocation::get(M.getContext(), DL.getLine(), DL.getCol(),
851  Scope, InlinedAt);
852  };
853 
854  if (I.getDebugLoc() != DebugLoc())
855  I.setDebugLoc(remapDebugLoc(I.getDebugLoc()));
856 
857  // Remap DILocations in llvm.loop attachments.
858  updateLoopMetadataDebugLocations(I, [&](Metadata *MD) -> Metadata * {
859  if (auto *Loc = dyn_cast_or_null<DILocation>(MD))
860  return remapDebugLoc(Loc).get();
861  return MD;
862  });
863 
864  // Strip heapallocsite attachments, they point into the DIType system.
865  if (I.hasMetadataOtherThanDebugLoc())
866  I.setMetadata("heapallocsite", nullptr);
867  }
868  }
869  }
870 
871  // Create a new llvm.dbg.cu, which is equivalent to the one
872  // -gline-tables-only would have created.
873  for (auto &NMD : M.named_metadata()) {
874  SmallVector<MDNode *, 8> Ops;
875  for (MDNode *Op : NMD.operands())
876  Ops.push_back(remap(Op));
877 
878  if (!Changed)
879  continue;
880 
881  NMD.clearOperands();
882  for (auto *Op : Ops)
883  if (Op)
884  NMD.addOperand(Op);
885  }
886  return Changed;
887 }

901 void Instruction::mergeDIAssignID(
902  ArrayRef<const Instruction *> SourceInstructions) {
903  // Replace all uses (and attachments) of all the DIAssignIDs
904  // on SourceInstructions with a single merged value.
905  assert(getFunction() && "Uninserted instruction merged");
906  // Collect up the DIAssignID tags.
907  SmallVector<DIAssignID *, 4> IDs;
908  for (const Instruction *I : SourceInstructions) {
909  if (auto *MD = I->getMetadata(LLVMContext::MD_DIAssignID))
910  IDs.push_back(cast<DIAssignID>(MD));
911  assert(getFunction() == I->getFunction() &&
912  "Merging with instruction from another function not allowed");
913  }
914 
915  // Add this instruction's DIAssignID too, if it has one.
916  if (auto *MD = getMetadata(LLVMContext::MD_DIAssignID))
917  IDs.push_back(cast<DIAssignID>(MD));
918 
919  if (IDs.empty())
920  return; // No DIAssignID tags to process.
921 
922  DIAssignID *MergeID = IDs[0];
923  for (auto It = std::next(IDs.begin()), End = IDs.end(); It != End; ++It) {
924  if (*It != MergeID)
925  at::RAUW(*It, MergeID);
926  }
927  setMetadata(LLVMContext::MD_DIAssignID, MergeID);
928 }

932 void Instruction::dropLocation() {
933  const DebugLoc &DL = getDebugLoc();
934  if (!DL)
935  return;
936 
937  // If this isn't a call, drop the location to allow a location from a
938  // preceding instruction to propagate.
939  bool MayLowerToCall = false;
940  if (isa<CallBase>(this)) {
941  auto *II = dyn_cast<IntrinsicInst>(this);
942  MayLowerToCall =
943  !II || IntrinsicInst::mayLowerToFunctionCall(II->getIntrinsicID());
944  }
945 
946  if (!MayLowerToCall) {
947  setDebugLoc(DebugLoc());
948  return;
949  }
950 
951  // Set a line 0 location for calls to preserve scope information in case
952  // inlining occurs.
953  DISubprogram *SP = getFunction()->getSubprogram();
954  if (SP)
955  // If a function scope is available, set it on the line 0 location. When
956  // hoisting a call to a predecessor block, using the function scope avoids
957  // making it look like the callee was reached earlier than it should be.
958  setDebugLoc(DILocation::get(getContext(), 0, 0, SP));
959  else
960  // The parent function has no scope. Go ahead and drop the location. If
961  // the parent function is inlined, and the callee has a subprogram, the
962  // inliner will attach a location to the call.
963  //
964  // One alternative is to set a line 0 location with the existing scope and
965  // inlinedAt info. The location might be sensitive to when inlining occurs.
966  setDebugLoc(DebugLoc());
967 }

1875 void at::trackAssignments(Function::iterator Start, Function::iterator End,
1876  const StorageToVarsMap &Vars, const DataLayout &DL,
1877  bool DebugPrints) {
1878  // Early-exit if there are no interesting variables.
1879  if (Vars.empty())
1880  return;
1881 
1882  auto &Ctx = Start->getContext();
1883  auto &Module = *Start->getModule();
1884 
1885  // Undef type doesn't matter, so long as it isn't void. Let's just use i1.
1886  auto *Undef = UndefValue::get(Type::getInt1Ty(Ctx));
1887  DIBuilder DIB(Module, /*AllowUnresolved*/ false);
1888 
1889  // Scan the instructions looking for stores to local variables' storage.
1890  LLVM_DEBUG(errs() << "# Scanning instructions\n");
1891  for (auto BBI = Start; BBI != End; ++BBI) {
1892  for (Instruction &I : *BBI) {
1893 
1894  std::optional<AssignmentInfo> Info;
1895  Value *ValueComponent = nullptr;
1896  Value *DestComponent = nullptr;
1897  if (auto *AI = dyn_cast<AllocaInst>(&I)) {
1898  // We want to track the variable's stack home from its alloca's
1899  // position onwards so we treat it as an assignment (where the stored
1900  // value is Undef).
1901  Info = getAssignmentInfo(DL, AI);
1902  ValueComponent = Undef;
1903  DestComponent = AI;
1904  } else if (auto *SI = dyn_cast<StoreInst>(&I)) {
1905  Info = getAssignmentInfo(DL, SI);
1906  ValueComponent = SI->getValueOperand();
1907  DestComponent = SI->getPointerOperand();
1908  } else if (auto *MI = dyn_cast<MemTransferInst>(&I)) {
1909  Info = getAssignmentInfo(DL, MI);
1910  // May not be able to represent this value easily.
1911  ValueComponent = Undef;
1912  DestComponent = MI->getOperand(0);
1913  } else if (auto *MI = dyn_cast<MemSetInst>(&I)) {
1914  Info = getAssignmentInfo(DL, MI);
1915  // If we're zero-initing we can state the assigned value is zero,
1916  // otherwise use undef.
1917  auto *ConstValue = dyn_cast<ConstantInt>(MI->getOperand(1));
1918  if (ConstValue && ConstValue->isZero())
1919  ValueComponent = ConstValue;
1920  else
1921  ValueComponent = Undef;
1922  DestComponent = MI->getOperand(0);
1923  } else {
1924  // Not a store-like instruction.
1925  continue;
1926  }
1927 
1928  assert(ValueComponent && DestComponent);
1929  LLVM_DEBUG(errs() << "SCAN: Found store-like: " << I << "\n");
1930 
1931  // Check if getAssignmentInfo failed to understand this store.
1932  if (!Info.has_value()) {
1933  LLVM_DEBUG(
1934  errs()
1935  << " | SKIP: Untrackable store (e.g. through non-const gep)\n");
1936  continue;
1937  }
1938  LLVM_DEBUG(errs() << " | BASE: " << *Info->Base << "\n");
1939 
1940  // Check if the store destination is a local variable with debug info.
1941  auto LocalIt = Vars.find(Info->Base);
1942  if (LocalIt == Vars.end()) {
1943  LLVM_DEBUG(
1944  errs()
1945  << " | SKIP: Base address not associated with local variable\n");
1946  continue;
1947  }
1948 
1949  DIAssignID *ID =
1950  cast_or_null<DIAssignID>(I.getMetadata(LLVMContext::MD_DIAssignID));
1951  if (!ID) {
1952  ID = DIAssignID::getDistinct(Ctx);
1953  I.setMetadata(LLVMContext::MD_DIAssignID, ID);
1954  }
1955 
1956  for (const VarRecord &R : LocalIt->second) {
1957  auto *Assign =
1958  emitDbgAssign(*Info, ValueComponent, DestComponent, I, R, DIB);
1959  (void)Assign;
1960  LLVM_DEBUG(errs() << " > INSERT: " << *Assign << "\n");
1961  }
1962  }
1963  }
1964 }

1966 void AssignmentTrackingPass::runOnFunction(Function &F) {
1967  // Collect a map of {backing storage : dbg.declares} (currently "backing
1968  // storage" is limited to Allocas). We'll use this to find dbg.declares to
1969  // delete after running `trackAssignments`.
1970  DenseMap<const AllocaInst *, SmallPtrSet<DbgDeclareInst *, 2>> DbgDeclares;
1971  // Create another similar map of {storage : variables} that we'll pass to
1972  // trackAssignments.
1973  StorageToVarsMap Vars;
1974  for (auto &BB : F) {
1975  for (auto &I : BB) {
1976  DbgDeclareInst *DDI = dyn_cast<DbgDeclareInst>(&I);
1977  if (!DDI)
1978  continue;
1979  // FIXME: trackAssignments doesn't let you specify any modifiers to the
1980  // variable (e.g. fragment) or location (e.g. offset), so we have to
1981  // leave dbg.declares with non-empty expressions in place.
1982  if (DDI->getExpression()->getNumElements() != 0)
1983  continue;
1984  if (AllocaInst *Alloca =
1985  dyn_cast<AllocaInst>(DDI->getAddress()->stripPointerCasts())) {
1986  DbgDeclares[Alloca].insert(DDI);
1987  Vars[Alloca].insert(VarRecord(DDI));
1988  }
1989  }
1990  }
1991 
1992  auto DL = std::make_unique<DataLayout>(F.getParent());
1993  // FIXME: Locals can be backed by caller allocas (sret, byval).
1994  // Note: trackAssignments doesn't respect dbg.declare's IR positions (as it
1995  // doesn't "understand" dbg.declares). However, this doesn't appear to break
1996  // any rules given this description of dbg.declare from
1997  // llvm/docs/SourceLevelDebugging.rst:
1998  //
1999  // It is not control-dependent, meaning that if a call to llvm.dbg.declare
2000  // exists and has a valid location argument, that address is considered to
2001  // be the true home of the variable across its entire lifetime.
2002  trackAssignments(F.begin(), F.end(), Vars, *DL);
2003 
2004  // Delete dbg.declares for variables now tracked with assignment tracking.
2005  for (auto &P : DbgDeclares) {
2006  const AllocaInst *Alloca = P.first;
2007  auto Markers = at::getAssignmentMarkers(Alloca);
2008  (void)Markers;
2009  for (DbgDeclareInst *DDI : P.second) {
2010  // Assert that the alloca that DDI uses is now linked to a dbg.assign
2011  // describing the same variable (i.e. check that this dbg.declare
2012  // has been replaced by a dbg.assign).
2013  assert(llvm::any_of(Markers, [DDI](DbgAssignIntrinsic *DAI) {
2014  return DebugVariable(DAI) == DebugVariable(DDI);
2015  }));
2016  // Delete DDI because the variable location is now tracked using
2017  // assignment tracking.
2018  DDI->eraseFromParent();
2019  }
2020  }
2021 }
