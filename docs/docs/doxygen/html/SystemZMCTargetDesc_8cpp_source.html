131 unsigned SystemZMC::getFirstReg(unsigned Reg) {
132  static unsigned Map[SystemZ::NUM_TARGET_REGS];
133  static bool Initialized = false;
134  if (!Initialized) {
135  for (unsigned I = 0; I < 16; ++I) {
136  Map[GR32Regs[I]] = I;
137  Map[GRH32Regs[I]] = I;
138  Map[GR64Regs[I]] = I;
139  Map[GR128Regs[I]] = I;
140  Map[FP128Regs[I]] = I;
141  Map[AR32Regs[I]] = I;
142  }
143  for (unsigned I = 0; I < 32; ++I) {
144  Map[VR32Regs[I]] = I;
145  Map[VR64Regs[I]] = I;
146  Map[VR128Regs[I]] = I;
147  }
148  }
149  assert(Reg < SystemZ::NUM_TARGET_REGS);
150  return Map[Reg];
151 }

153 static MCAsmInfo *createSystemZMCAsmInfo(const MCRegisterInfo &MRI,
154  const Triple &TT,
155  const MCTargetOptions &Options) {
156  if (TT.isOSzOS())
157  return new SystemZMCAsmInfoGOFF(TT);
158 
159  MCAsmInfo *MAI = new SystemZMCAsmInfoELF(TT);
160  MCCFIInstruction Inst = MCCFIInstruction::cfiDefCfa(
161  nullptr, MRI.getDwarfRegNum(SystemZ::R15D, true),
162  SystemZMC::ELFCFAOffsetFromInitialSP);
163  MAI->addInitialFrameState(Inst);
164  return MAI;
165 }

192 void SystemZTargetStreamer::emitConstantPools() {
193  // Emit EXRL target instructions.
194  if (EXRLTargets2Sym.empty())
195  return;
196  // Switch to the .text section.
197  const MCObjectFileInfo &OFI = *Streamer.getContext().getObjectFileInfo();
198  Streamer.switchSection(OFI.getTextSection());
199  for (auto &I : EXRLTargets2Sym) {
200  Streamer.emitLabel(I.second);
201  const MCInstSTIPair &MCI_STI = I.first;
202  Streamer.emitInstruction(MCI_STI.first, *MCI_STI.second);
203  }
204  EXRLTargets2Sym.clear();
205 }
