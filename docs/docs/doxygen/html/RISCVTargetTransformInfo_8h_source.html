231  TargetTransformInfo::VPLegalization
232  getVPLegalizationStrategy(const VPIntrinsic &PI) const {
233  using VPLegalization = TargetTransformInfo::VPLegalization;
234  if (!ST->hasVInstructions() ||
235  (PI.getIntrinsicID() == Intrinsic::vp_reduce_mul &&
236  cast<VectorType>(PI.getArgOperand(1)->getType())
237  ->getElementType()
238  ->getIntegerBitWidth() != 1))
239  return VPLegalization(VPLegalization::Discard, VPLegalization::Convert);
240  return VPLegalization(VPLegalization::Legal, VPLegalization::Legal);

243  bool isLegalToVectorizeReduction(const RecurrenceDescriptor &RdxDesc,
244  ElementCount VF) const {
245  if (!VF.isScalable())
246  return true;
247 
248  Type *Ty = RdxDesc.getRecurrenceType();
249  if (!TLI->isLegalElementTypeForRVV(Ty))
250  return false;
251 
252  switch (RdxDesc.getRecurrenceKind()) {
253  case RecurKind::Add:
254  case RecurKind::FAdd:
255  case RecurKind::And:
256  case RecurKind::Or:
257  case RecurKind::Xor:
258  case RecurKind::SMin:
259  case RecurKind::SMax:
260  case RecurKind::UMin:
261  case RecurKind::UMax:
262  case RecurKind::FMin:
263  case RecurKind::FMax:
264  case RecurKind::SelectICmp:
265  case RecurKind::SelectFCmp:
266  case RecurKind::FMulAdd:
267  return true;
268  default:
269  return false;
270  }

279  enum RISCVRegisterClass { GPRRC, FPRRC, VRRC };
280  unsigned getNumberOfRegisters(unsigned ClassID) const {
281  switch (ClassID) {
282  case RISCVRegisterClass::GPRRC:
283  // 31 = 32 GPR - x0 (zero register)
284  // FIXME: Should we exclude fixed registers like SP, TP or GP?
285  return 31;
286  case RISCVRegisterClass::FPRRC:
287  if (ST->hasStdExtF())
288  return 32;
289  return 0;
290  case RISCVRegisterClass::VRRC:
291  // Although there are 32 vector registers, v0 is special in that it is the
292  // only register that can be used to hold a mask.
293  // FIXME: Should we conservatively return 31 as the number of usable
294  // vector registers?
295  return ST->hasVInstructions() ? 32 : 0;
296  }
297  llvm_unreachable("unknown register class");
298  }
299 
300  unsigned getRegisterClassForType(bool Vector, Type *Ty = nullptr) const {
301  if (Vector)
302  return RISCVRegisterClass::VRRC;
303  if (!Ty)
304  return RISCVRegisterClass::GPRRC;
305 
306  Type *ScalarTy = Ty->getScalarType();
307  if ((ScalarTy->isHalfTy() && ST->hasStdExtZfhOrZfhmin()) ||
308  (ScalarTy->isFloatTy() && ST->hasStdExtF()) ||
309  (ScalarTy->isDoubleTy() && ST->hasStdExtD())) {
310  return RISCVRegisterClass::FPRRC;
311  }
312 
313  return RISCVRegisterClass::GPRRC;
314  }
