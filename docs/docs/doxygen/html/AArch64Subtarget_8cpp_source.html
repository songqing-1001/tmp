308  InlineAsmLoweringInfo.reset(new InlineAsmLowering(getTargetLowering()));
309  Legalizer.reset(new AArch64LegalizerInfo(*this));
310

312 
313  // FIXME: At this point, we can't rely on Subtarget having RBI.
314  // It's awkward to mix passing RBI and the Subtarget; should we pass
315  // TII/TRI as well?
316  InstSelector.reset(createAArch64InstructionSelector(
317  *static_cast<const AArch64TargetMachine *>(&TM), *this, *RBI));
318 
319  RegBankInfo.reset(RBI);
320 
321  auto TRI = getRegisterInfo();
322  StringSet<> ReservedRegNames;
323  ReservedRegNames.insert(ReservedRegsForRA.begin(), ReservedRegsForRA.end());
324  for (unsigned i = 0; i < 29; ++i) {
325  if (ReservedRegNames.count(TRI->getName(AArch64::X0 + i)))
326  ReserveXRegisterForRA.set(i);
327  }
328  // X30 is named LR, so we can't use TRI->getName to check X30.
329  if (ReservedRegNames.count("X30") || ReservedRegNames.count("LR"))
330  ReserveXRegisterForRA.set(30);
331  // X29 is named FP, so we can't use TRI->getName to check X29.
332  if (ReservedRegNames.count("X29") || ReservedRegNames.count("FP"))
333  ReserveXRegisterForRA.set(29);
334 }

356 /// Find the target operand flags that describe how a global value should be
357 /// referenced for the current subtarget.
358 unsigned
359 AArch64Subtarget::ClassifyGlobalReference(const GlobalValue *GV,
360  const TargetMachine &TM) const {
361  // MachO large model always goes via a GOT, simply to get a single 8-byte
362  // absolute relocation on all global addresses.
363  if (TM.getCodeModel() == CodeModel::Large && isTargetMachO())
364  return AArch64II::MO_GOT;
365 
366  // All globals dynamically protected by MTE must have their address tags
367  // synthesized. This is done by having the loader stash the tag in the GOT
368  // entry. Force all tagged globals (even ones with internal linkage) through
369  // the GOT.
370  if (GV->isTagged())
371  return AArch64II::MO_GOT;
372 
373  if (!TM.shouldAssumeDSOLocal(*GV->getParent(), GV)) {
374  if (GV->hasDLLImportStorageClass()) {
375  if (isWindowsArm64EC() && GV->getValueType()->isFunctionTy())
376  return AArch64II::MO_GOT | AArch64II::MO_DLLIMPORTAUX;
377  return AArch64II::MO_GOT | AArch64II::MO_DLLIMPORT;
378  }
379  if (getTargetTriple().isOSWindows())
380  return AArch64II::MO_GOT | AArch64II::MO_COFFSTUB;
381  return AArch64II::MO_GOT;
382  }
383 
384  // The small code model's direct accesses use ADRP, which cannot
385  // necessarily produce the value 0 (if the code is above 4GB).
386  // Same for the tiny code model, where we have a pc relative LDR.
387  if ((useSmallAddressing() || TM.getCodeModel() == CodeModel::Tiny) &&
388  GV->hasExternalWeakLinkage())
389  return AArch64II::MO_GOT;
390 
391  // References to tagged globals are marked with MO_NC | MO_TAGGED to indicate
392  // that their nominal addresses are tagged and outside of the code model. In
393  // AArch64ExpandPseudo::expandMI we emit an additional instruction to set the
394  // tag if necessary based on MO_TAGGED.
395  if (AllowTaggedGlobals && !isa<FunctionType>(GV->getValueType()))
396  return AArch64II::MO_NC | AArch64II::MO_TAGGED;
397 
398  return AArch64II::MO_NO_FLAG;
399 }

401 unsigned AArch64Subtarget::classifyGlobalFunctionReference(
402  const GlobalValue *GV, const TargetMachine &TM) const {
403  // MachO large model always goes via a GOT, because we don't have the
404  // relocations available to do anything else..
405  if (TM.getCodeModel() == CodeModel::Large && isTargetMachO() &&
406  !GV->hasInternalLinkage())
407  return AArch64II::MO_GOT;
408 
409  // NonLazyBind goes via GOT unless we know it's available locally.
410  auto *F = dyn_cast<Function>(GV);
411  if (UseNonLazyBind && F && F->hasFnAttribute(Attribute::NonLazyBind) &&
412  !TM.shouldAssumeDSOLocal(*GV->getParent(), GV))
413  return AArch64II::MO_GOT;
414 
415  if (getTargetTriple().isOSWindows()) {
416  if (isWindowsArm64EC() && GV->getValueType()->isFunctionTy() &&
417  GV->hasDLLImportStorageClass()) {
418  // On Arm64EC, if we're calling a function directly, use MO_DLLIMPORT,
419  // not MO_DLLIMPORTAUX.
420  return AArch64II::MO_GOT | AArch64II::MO_DLLIMPORT;
421  }
422 
423  // Use ClassifyGlobalReference for setting MO_DLLIMPORT/MO_COFFSTUB.
424  return ClassifyGlobalReference(GV, TM);
425  }
426 
427  return AArch64II::MO_NO_FLAG;
428 }
