41  M.getModuleFlag("sign-return-address"))) {

43  if (const auto *All = mdconst::extract_or_null<ConstantInt>(
44  M.getModuleFlag("sign-return-address-all")))
45  return {true, All->getZExtValue()};
46  return {true, false};
47  }
