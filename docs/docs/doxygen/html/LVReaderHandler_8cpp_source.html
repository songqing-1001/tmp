35 Error LVReaderHandler::createReader(StringRef Filename, LVReaders &Readers,
36  PdbOrObj &Input, StringRef FileFormatName,
37  StringRef ExePath) {
38  auto CreateOneReader = [&]() -> std::unique_ptr<LVReader> {
39  if (Input.is<ObjectFile *>()) {
40  ObjectFile &Obj = *Input.get<ObjectFile *>();
41  if (Obj.isELF() || Obj.isMachO())
42  return std::make_unique<LVELFReader>(Filename, FileFormatName, Obj, W);
43  }
44  return nullptr;
45  };
46 
47  std::unique_ptr<LVReader> ReaderObj = CreateOneReader();
48  if (!ReaderObj)
49  return createStringError(errc::invalid_argument,
50  "unable to create reader for: '%s'",
51  Filename.str().c_str());
52 
53  LVReader *Reader = ReaderObj.get();
54  Readers.emplace_back(std::move(ReaderObj));
55  return Reader->doLoad();
56 }

58 Error LVReaderHandler::handleArchive(LVReaders &Readers, StringRef Filename,
59  Archive &Arch) {
60  Error Err = Error::success();
61  for (const Archive::Child &Child : Arch.children(Err)) {
62  Expected<MemoryBufferRef> BuffOrErr = Child.getMemoryBufferRef();
63  if (Error Err = BuffOrErr.takeError())
64  return createStringError(errorToErrorCode(std::move(Err)), "%s",
65  Filename.str().c_str());
66  Expected<StringRef> NameOrErr = Child.getName();
67  if (Error Err = NameOrErr.takeError())
68  return createStringError(errorToErrorCode(std::move(Err)), "%s",
69  Filename.str().c_str());
70  std::string Name = (Filename + "(" + NameOrErr.get() + ")").str();
71  if (Error Err = handleBuffer(Readers, Name, BuffOrErr.get()))
72  return createStringError(errorToErrorCode(std::move(Err)), "%s",
73  Filename.str().c_str());
74  }
75 
76  return Error::success();
77 }

90 Error LVReaderHandler::handleFile(LVReaders &Readers, StringRef Filename,
91  StringRef ExePath) {
92  // Convert any Windows backslashes into forward slashes to get the path.
93  std::string ConvertedPath =
94  sys::path::convert_to_slash(Filename, sys::path::Style::windows);
95  ErrorOr<std::unique_ptr<MemoryBuffer>> BuffOrErr =
96  MemoryBuffer::getFileOrSTDIN(ConvertedPath);
97  if (BuffOrErr.getError()) {
98  return createStringError(errc::bad_file_descriptor,
99  "File '%s' does not exist.",
100  ConvertedPath.c_str());
101  }
102  std::unique_ptr<MemoryBuffer> Buffer = std::move(BuffOrErr.get());
103  return handleBuffer(Readers, ConvertedPath, *Buffer, ExePath);

106 Error LVReaderHandler::handleMach(LVReaders &Readers, StringRef Filename,
107  MachOUniversalBinary &Mach) {
108  for (const MachOUniversalBinary::ObjectForArch &ObjForArch : Mach.objects()) {
109  std::string ObjName = (Twine(Filename) + Twine("(") +
110  Twine(ObjForArch.getArchFlagName()) + Twine(")"))
111  .str();
112  if (Expected<std::unique_ptr<MachOObjectFile>> MachOOrErr =
113  ObjForArch.getAsObjectFile()) {
114  MachOObjectFile &Obj = **MachOOrErr;
115  PdbOrObj Input = &Obj;
116  if (Error Err =
117  createReader(Filename, Readers, Input, Obj.getFileFormatName()))
118  return Err;
119  continue;
120  } else
121  consumeError(MachOOrErr.takeError());
122  if (Expected<std::unique_ptr<Archive>> ArchiveOrErr =
123  ObjForArch.getAsArchive()) {
124  if (Error Err = handleArchive(Readers, ObjName, *ArchiveOrErr.get()))
125  return Err;
126  continue;
127  } else
128  consumeError(ArchiveOrErr.takeError());
129  }
130  return Error::success();
131 }

133 Error LVReaderHandler::handleObject(LVReaders &Readers, StringRef Filename,
134  Binary &Binary) {
135  if (PdbOrObj Input = dyn_cast<ObjectFile>(&Binary))
136  return createReader(Filename, Readers, Input,
137  Input.get<ObjectFile *>()->getFileFormatName());
138 
139  if (MachOUniversalBinary *Fat = dyn_cast<MachOUniversalBinary>(&Binary))
140  return handleMach(Readers, Filename, *Fat);
141 
142  if (Archive *Arch = dyn_cast<Archive>(&Binary))
143  return handleArchive(Readers, Filename, *Arch);
144 
145  return createStringError(errc::not_supported,
146  "Binary object format in '%s' is not supported.",
147  Filename.str().c_str());

174 Error LVReaderHandler::compareReaders() {
175  LLVM_DEBUG(dbgs() << "compareReaders\n");
176  size_t ReadersCount = TheReaders.size();
177  if (options().getCompareExecute() && ReadersCount >= 2) {
178  // If we have more than 2 readers, compare them by pairs.
179  size_t ViewPairs = ReadersCount / 2;
180  LVCompare Compare(OS);
181  for (size_t Pair = 0, Index = 0; Pair < ViewPairs; ++Pair) {
182  if (Error Err = Compare.execute(TheReaders[Index].get(),
183  TheReaders[Index + 1].get()))
184  return Err;
185  Index += 2;
186  }
187  }
188 
189  return Error::success();
190 }
