310 /// Scope for local variables.
311 ///
312 /// When the scope is destroyed, instructions are emitted to tear down
313 /// all variables declared in this scope.
314 template <class Emitter> class LocalScope : public VariableScope<Emitter> {
315 public:
316  LocalScope(ByteCodeExprGen<Emitter> *Ctx) : VariableScope<Emitter>(Ctx) {}
317 
318  ~LocalScope() override { this->emitDestruction(); }
319 
320  void addLocal(const Scope::Local &Local) override {
321  if (!Idx) {
322  Idx = this->Ctx->Descriptors.size();
323  this->Ctx->Descriptors.emplace_back();
324  }
325 
326  this->Ctx->Descriptors[*Idx].emplace_back(Local);
327  }
328 
329  void emitDestruction() override {
330  if (!Idx)
331  return;
332  this->Ctx->emitDestroy(*Idx, SourceInfo{});
333  }
334 
335 protected:
336  /// Index of the scope in the chain.
337  std::optional<unsigned> Idx;
338 };
