69 void Flang::addPicOptions(const ArgList &Args, ArgStringList &CmdArgs) const {
70  // ParsePICArgs parses -fPIC/-fPIE and their variants and returns a tuple of
71  // (RelocationModel, PICLevel, IsPIE).
72  llvm::Reloc::Model RelocationModel;
73  unsigned PICLevel;
74  bool IsPIE;
75  std::tie(RelocationModel, PICLevel, IsPIE) =
76  ParsePICArgs(getToolChain(), Args);
77 
78  if (auto *RMName = RelocationModelName(RelocationModel)) {
79  CmdArgs.push_back("-mrelocation-model");
80  CmdArgs.push_back(RMName);
81  }
82  if (PICLevel > 0) {
83  CmdArgs.push_back("-pic-level");
84  CmdArgs.push_back(PICLevel == 1 ? "1" : "2");
85  if (IsPIE)
86  CmdArgs.push_back("-pic-is-pie");
87  }
88 }

116 static void addFloatingPointOptions(const Driver &D, const ArgList &Args,
117  ArgStringList &CmdArgs) {
118  StringRef FPContract;
119  bool HonorINFs = true;
120  bool HonorNaNs = true;
121  bool ApproxFunc = false;
122  bool SignedZeros = true;
123  bool AssociativeMath = false;
124  bool ReciprocalMath = false;
125 
126  if (const Arg *A = Args.getLastArg(options::OPT_ffp_contract)) {
127  const StringRef Val = A->getValue();
128  if (Val == "fast" || Val == "off") {
129  FPContract = Val;
130  } else if (Val == "on") {
131  // Warn instead of error because users might have makefiles written for
132  // gfortran (which accepts -ffp-contract=on)
133  D.Diag(diag::warn_drv_unsupported_option_for_flang)
134  << Val << A->getOption().getName() << "off";
135  FPContract = "off";
136  } else
137  // Clang's "fast-honor-pragmas" option is not supported because it is
138  // non-standard
139  D.Diag(diag::err_drv_unsupported_option_argument)
140  << A->getSpelling() << Val;
141  }
142 
143  for (const Arg *A : Args) {
144  auto optId = A->getOption().getID();
145  switch (optId) {
146  // if this isn't an FP option, skip the claim below
147  default:
148  continue;
149 
150  case options::OPT_fhonor_infinities:
151  HonorINFs = true;
152  break;
153  case options::OPT_fno_honor_infinities:
154  HonorINFs = false;
155  break;
156  case options::OPT_fhonor_nans:
157  HonorNaNs = true;
158  break;
159  case options::OPT_fno_honor_nans:
160  HonorNaNs = false;
161  break;
162  case options::OPT_fapprox_func:
163  ApproxFunc = true;
164  break;
165  case options::OPT_fno_approx_func:
166  ApproxFunc = false;
167  break;
168  case options::OPT_fsigned_zeros:
169  SignedZeros = true;
170  break;
171  case options::OPT_fno_signed_zeros:
172  SignedZeros = false;
173  break;
174  case options::OPT_fassociative_math:
175  AssociativeMath = true;
176  break;
177  case options::OPT_fno_associative_math:
178  AssociativeMath = false;
179  break;
180  case options::OPT_freciprocal_math:
181  ReciprocalMath = true;
182  break;
183  case options::OPT_fno_reciprocal_math:
184  ReciprocalMath = false;
185  break;
186  case options::OPT_Ofast:
187  [[fallthrough]];
188  case options::OPT_ffast_math:
189  HonorINFs = false;
190  HonorNaNs = false;
191  AssociativeMath = true;
192  ReciprocalMath = true;
193  ApproxFunc = true;
194  SignedZeros = false;
195  FPContract = "fast";
196  break;
197  case options::OPT_fno_fast_math:
198  HonorINFs = true;
199  HonorNaNs = true;
200  AssociativeMath = false;
201  ReciprocalMath = false;
202  ApproxFunc = false;
203  SignedZeros = true;
204  // -fno-fast-math should undo -ffast-math so I return FPContract to the
205  // default. It is important to check it is "fast" (the default) so that
206  // --ffp-contract=off -fno-fast-math --> -ffp-contract=off
207  if (FPContract == "fast")
208  FPContract = "";
209  break;
210  }
211 
212  // If we handled this option claim it
213  A->claim();
214  }
215 
216  if (!HonorINFs && !HonorNaNs && AssociativeMath && ReciprocalMath &&
217  ApproxFunc && !SignedZeros &&
218  (FPContract == "fast" || FPContract == "")) {
219  CmdArgs.push_back("-ffast-math");
220  return;
221  }
222 
223  if (!FPContract.empty())
224  CmdArgs.push_back(Args.MakeArgString("-ffp-contract=" + FPContract));
225 
226  if (!HonorINFs)
227  CmdArgs.push_back("-menable-no-infs");
228 
229  if (!HonorNaNs)
230  CmdArgs.push_back("-menable-no-nans");
231 
232  if (ApproxFunc)
233  CmdArgs.push_back("-fapprox-func");
234 
235  if (!SignedZeros)
236  CmdArgs.push_back("-fno-signed-zeros");
237 
238  if (AssociativeMath && !SignedZeros)
239  CmdArgs.push_back("-mreassociate");
240 
241  if (ReciprocalMath)
242  CmdArgs.push_back("-freciprocal-math");
243 }

245 void Flang::ConstructJob(Compilation &C, const JobAction &JA,
246  const InputInfo &Output, const InputInfoList &Inputs,
247  const ArgList &Args, const char *LinkingOutput) const {
248  const auto &TC = getToolChain();
249  const llvm::Triple &Triple = TC.getEffectiveTriple();
250  const std::string &TripleStr = Triple.getTriple();
251 
252  const Driver &D = TC.getDriver();
253  ArgStringList CmdArgs;
254 
255  // Invoke ourselves in -fc1 mode.
256  CmdArgs.push_back("-fc1");
257 
258  // Add the "effective" target triple.
259  CmdArgs.push_back("-triple");
260  CmdArgs.push_back(Args.MakeArgString(TripleStr));
261 
262  if (isa<PreprocessJobAction>(JA)) {
263  CmdArgs.push_back("-E");
264  } else if (isa<CompileJobAction>(JA) || isa<BackendJobAction>(JA)) {
265  if (JA.getType() == types::TY_Nothing) {
266  CmdArgs.push_back("-fsyntax-only");
267  } else if (JA.getType() == types::TY_AST) {
268  CmdArgs.push_back("-emit-ast");
269  } else if (JA.getType() == types::TY_LLVM_IR ||
270  JA.getType() == types::TY_LTO_IR) {
271  CmdArgs.push_back("-emit-llvm");
272  } else if (JA.getType() == types::TY_LLVM_BC ||
273  JA.getType() == types::TY_LTO_BC) {
274  CmdArgs.push_back("-emit-llvm-bc");
275  } else if (JA.getType() == types::TY_PP_Asm) {
276  CmdArgs.push_back("-S");
277  } else {
278  assert(false && "Unexpected output type!");
279  }
280  } else if (isa<AssembleJobAction>(JA)) {
281  CmdArgs.push_back("-emit-obj");
282  } else {
283  assert(false && "Unexpected action class for Flang tool.");
284  }
285 
286  const InputInfo &Input = Inputs[0];
287  types::ID InputType = Input.getType();
288 
289  // Add preprocessing options like -I, -D, etc. if we are using the
290  // preprocessor (i.e. skip when dealing with e.g. binary files).
291  if (types::getPreprocessedType(InputType) != types::TY_INVALID)
292  addPreprocessingOptions(Args, CmdArgs);
293 
294  addFortranDialectOptions(Args, CmdArgs);
295 
296  // Color diagnostics are parsed by the driver directly from argv and later
297  // re-parsed to construct this job; claim any possible color diagnostic here
298  // to avoid warn_drv_unused_argument.
299  Args.getLastArg(options::OPT_fcolor_diagnostics,
300  options::OPT_fno_color_diagnostics);
301  if (D.getDiags().getDiagnosticOptions().ShowColors)
302  CmdArgs.push_back("-fcolor-diagnostics");
303 
304  // -fPIC and related options.
305  addPicOptions(Args, CmdArgs);
306 
307  // Floating point related options
308  addFloatingPointOptions(D, Args, CmdArgs);
309 
310  // Add target args, features, etc.
311  addTargetOptions(Args, CmdArgs);
312 
313  // Add other compile options
314  addOtherOptions(Args, CmdArgs);
315 
316  // Forward -Xflang arguments to -fc1
317  Args.AddAllArgValues(CmdArgs, options::OPT_Xflang);
318 
319  // Forward -mllvm options to the LLVM option parser. In practice, this means
320  // forwarding to `-fc1` as that's where the LLVM parser is run.
321  for (const Arg *A : Args.filtered(options::OPT_mllvm)) {
322  A->claim();
323  A->render(Args, CmdArgs);
324  }
325 
326  for (const Arg *A : Args.filtered(options::OPT_mmlir)) {
327  A->claim();
328  A->render(Args, CmdArgs);
329  }
330 
331  // Optimization level for CodeGen.
332  if (const Arg *A = Args.getLastArg(options::OPT_O_Group)) {
333  if (A->getOption().matches(options::OPT_O4)) {
334  CmdArgs.push_back("-O3");
335  D.Diag(diag::warn_O4_is_O3);
336  } else if (A->getOption().matches(options::OPT_Ofast)) {
337  CmdArgs.push_back("-O3");
338  } else {
339  A->render(Args, CmdArgs);
340  }
341  }
342 
343  if (Output.isFilename()) {
344  CmdArgs.push_back("-o");
345  CmdArgs.push_back(Output.getFilename());
346  } else {
347  assert(Output.isNothing() && "Invalid output.");
348  }
349 
350  assert(Input.isFilename() && "Invalid input.");
351 
352  addDashXForInput(Args, Input, CmdArgs);
353 
354  CmdArgs.push_back(Input.getFilename());
355 
356  // TODO: Replace flang-new with flang once the new driver replaces the
357  // throwaway driver
358  const char *Exec = Args.MakeArgString(D.GetProgramPath("flang-new", TC));
359  C.addCommand(std::make_unique<Command>(JA, *this,
360  ResponseFileSupport::AtFileUTF8(),
361  Exec, CmdArgs, Inputs, Output));
362 }
