67 static bool CheckActive(InterpState &S, CodePtr OpPC, const Pointer &Ptr,
68  AccessKinds AK) {
69  if (Ptr.isActive())
70  return true;
71 
72  // Get the inactive field descriptor.
73  const FieldDecl *InactiveField = Ptr.getField();
74 
75  // Walk up the pointer chain to find the union which is not active.
76  Pointer U = Ptr.getBase();
77  while (!U.isActive()) {
78  U = U.getBase();
79  }
80 
81  // Find the active field of the union.
82  Record *R = U.getRecord();
83  assert(R && R->isUnion() && "Not a union");
84  const FieldDecl *ActiveField = nullptr;
85  for (unsigned I = 0, N = R->getNumFields(); I < N; ++I) {
86  const Pointer &Field = U.atField(R->getField(I)->Offset);
87  if (Field.isActive()) {
88  ActiveField = Field.getField();
89  break;
90  }
91  }
92 
93  const SourceInfo &Loc = S.Current->getSource(OpPC);
94  S.FFDiag(Loc, diag::note_constexpr_access_inactive_union_member)
95  << AK << InactiveField << !ActiveField << ActiveField;
96  return false;
97 }

100  AccessKinds AK) {

158  AccessKinds AK) {
159  if (Ptr.isZero()) {
160  const auto &Src = S.Current->getSource(OpPC);
161

163  S.FFDiag(Src, diag::note_constexpr_null_subobject) << CSK_Field;
164  else
165  S.FFDiag(Src, diag::note_constexpr_access_null) << AK;

173 
174  S.FFDiag(Src, diag::note_constexpr_lifetime_ended, 1) << AK << !IsTemp;
175

177  S.Note(Ptr.getDeclLoc(), diag::note_constexpr_temporary_here);
178  else
179  S.Note(Ptr.getDeclLoc(), diag::note_declared_at);

181  return false;
182  }

289 bool CheckCallable(InterpState &S, CodePtr OpPC, const Function *F) {
290 
291  if (F->isVirtual()) {
292  if (!S.getLangOpts().CPlusPlus20) {
293  const SourceLocation &Loc = S.Current->getLocation(OpPC);
294  S.CCEDiag(Loc, diag::note_constexpr_virtual_call);
295  return false;
296  }
297  }
298 
299  if (!F->isConstexpr()) {
300  // Don't emit anything if we're checking for a potential constant
301  // expression. That will happen later when actually executing.
302  if (S.checkingPotentialConstantExpression())
303  return false;
304 
305  const SourceLocation &Loc = S.Current->getLocation(OpPC);
306  if (S.getLangOpts().CPlusPlus11) {
307  const FunctionDecl *DiagDecl = F->getDecl();
308 
309  // If this function is not constexpr because it is an inherited
310  // non-constexpr constructor, diagnose that directly.
311  auto *CD = dyn_cast<CXXConstructorDecl>(DiagDecl);
312  if (CD && CD->isInheritingConstructor()) {
313  auto *Inherited = CD->getInheritedConstructor().getConstructor();
314  if (!Inherited->isConstexpr())
315  DiagDecl = CD = Inherited;
316  }
317 
318  // FIXME: If DiagDecl is an implicitly-declared special member function
319  // or an inheriting constructor, we should be much more explicit about why
320  // it's not constexpr.
321  if (CD && CD->isInheritingConstructor())
322  S.FFDiag(Loc, diag::note_constexpr_invalid_inhctor, 1)
323  << CD->getInheritedConstructor().getConstructor()->getParent();
324  else
325  S.FFDiag(Loc, diag::note_constexpr_invalid_function, 1)
326  << DiagDecl->isConstexpr() << (bool)CD << DiagDecl;
327  S.Note(DiagDecl->getLocation(), diag::note_declared_at);
328  } else {
329  S.FFDiag(Loc, diag::note_invalid_subexpr_in_const_expr);
330  }
331  return false;
332  }
333 
334  return true;
335 }

375 static bool CheckArrayInitialized(InterpState &S, CodePtr OpPC,
376  const Pointer &BasePtr,
377  const ConstantArrayType *CAT) {
378  bool Result = true;
379  size_t NumElems = CAT->getSize().getZExtValue();
380  QualType ElemType = CAT->getElementType();
381 
382  if (isa<RecordType>(ElemType.getTypePtr())) {
383  const Record *R = BasePtr.getElemRecord();
384  for (size_t I = 0; I != NumElems; ++I) {
385  Pointer ElemPtr = BasePtr.atIndex(I).narrow();
386  Result &= CheckFieldsInitialized(S, OpPC, ElemPtr, R);
387  }
388  } else if (auto *ElemCAT = dyn_cast<ConstantArrayType>(ElemType)) {
389  for (size_t I = 0; I != NumElems; ++I) {
390  Pointer ElemPtr = BasePtr.atIndex(I).narrow();
391  Result &= CheckArrayInitialized(S, OpPC, ElemPtr, ElemCAT);
392  }
393  } else {
394  for (size_t I = 0; I != NumElems; ++I) {
395  if (!BasePtr.atIndex(I).isInitialized()) {
396  DiagnoseUninitializedSubobject(S, S.Current->getSource(OpPC), ElemType,
397  BasePtr.getFieldDesc()->getLocation());
398  Result = false;
399  }
400  }
401  }
402 
403  return Result;
404 }

406 static bool CheckFieldsInitialized(InterpState &S, CodePtr OpPC,
407  const Pointer &BasePtr, const Record *R) {
408  assert(R);
409  bool Result = true;
410  // Check all fields of this record are initialized.
411  for (const Record::Field &F : R->fields()) {
412  Pointer FieldPtr = BasePtr.atField(F.Offset);
413  QualType FieldType = F.Decl->getType();
414 
415  if (FieldType->isRecordType()) {
416  Result &= CheckFieldsInitialized(S, OpPC, FieldPtr, FieldPtr.getRecord());
417  } else if (FieldType->isArrayType()) {
418  const auto *CAT =
419  cast<ConstantArrayType>(FieldType->getAsArrayTypeUnsafe());
420  Result &= CheckArrayInitialized(S, OpPC, FieldPtr, CAT);
421  } else if (!FieldPtr.isInitialized()) {
422  DiagnoseUninitializedSubobject(S, S.Current->getSource(OpPC),
423  F.Decl->getType(), F.Decl->getLocation());
424  Result = false;
425  }
426  }
427  return Result;
428 }

436 bool CheckFloatResult(InterpState &S, CodePtr OpPC, APFloat::opStatus Status) {
437  // In a constant context, assume that any dynamic rounding mode or FP
438  // exception state matches the default floating-point environment.
439  if (S.inConstantContext())
440  return true;
441 
442  const SourceInfo &E = S.Current->getSource(OpPC);
443  FPOptions FPO = E.asExpr()->getFPFeaturesInEffect(S.Ctx.getLangOpts());
444 
445  if ((Status & APFloat::opInexact) &&
446  FPO.getRoundingMode() == llvm::RoundingMode::Dynamic) {
447  // Inexact result means that it depends on rounding mode. If the requested
448  // mode is dynamic, the evaluation cannot be made in compile time.
449  S.FFDiag(E, diag::note_constexpr_dynamic_rounding);
450  return false;
451  }
452 
453  if ((Status != APFloat::opOK) &&
454  (FPO.getRoundingMode() == llvm::RoundingMode::Dynamic ||
455  FPO.getExceptionMode() != LangOptions::FPE_Ignore ||
456  FPO.getAllowFEnvAccess())) {
457  S.FFDiag(E, diag::note_constexpr_float_arithmetic_strict);
458  return false;
459  }
460 
461  if ((Status & APFloat::opStatus::opInvalidOp) &&
462  FPO.getExceptionMode() != LangOptions::FPE_Ignore) {
463  // There is no usefully definable result.
464  S.FFDiag(E);
465  return false;
466  }
467 
468  return true;
469 }
