63 template <class Emitter>
64 bool ByteCodeExprGen<Emitter>::VisitCastExpr(const CastExpr *CE) {
65  auto *SubExpr = CE->getSubExpr();
66  switch (CE->getCastKind()) {
67 
68  case CK_LValueToRValue: {
69  return dereference(
70  CE->getSubExpr(), DerefKind::Read,
71  [](PrimType) {
72  // Value loaded - nothing to do here.
73  return true;
74  },
75  [this, CE](PrimType T) {
76  // Pointer on stack - dereference it.
77  if (!this->emitLoadPop(T, CE))
78  return false;
79  return DiscardResult ? this->emitPop(T, CE) : true;
80  });
81  }
82 
83  case CK_UncheckedDerivedToBase:
84  case CK_DerivedToBase: {
85  if (!this->visit(SubExpr))
86  return false;
87 
88  const CXXRecordDecl *FromDecl = getRecordDecl(SubExpr);
89  assert(FromDecl);
90  const CXXRecordDecl *ToDecl = getRecordDecl(CE);
91  assert(ToDecl);
92  const Record *R = getRecord(FromDecl);
93  const Record::Base *ToBase = R->getBase(ToDecl);
94  assert(ToBase);
95 
96  return this->emitGetPtrBasePop(ToBase->Offset, CE);
97  }
98 
99  case CK_FloatingCast: {
100  if (!this->visit(SubExpr))
101  return false;
102  const auto *TargetSemantics =
103  &Ctx.getASTContext().getFloatTypeSemantics(CE->getType());
104  return this->emitCastFP(TargetSemantics, getRoundingMode(CE), CE);
105  }
106 
107  case CK_IntegralToFloating: {
108  std::optional<PrimType> FromT = classify(SubExpr->getType());
109  if (!FromT)
110  return false;
111 
112  if (!this->visit(SubExpr))
113  return false;
114 
115  const auto *TargetSemantics =
116  &Ctx.getASTContext().getFloatTypeSemantics(CE->getType());
117  llvm::RoundingMode RM = getRoundingMode(CE);
118  return this->emitCastIntegralFloating(*FromT, TargetSemantics, RM, CE);
119  }
120 
121  case CK_FloatingToBoolean:
122  case CK_FloatingToIntegral: {
123  std::optional<PrimType> ToT = classify(CE->getType());
124 
125  if (!ToT)
126  return false;
127 
128  if (!this->visit(SubExpr))
129  return false;
130 
131  return this->emitCastFloatingIntegral(*ToT, CE);
132  }
133 
134  case CK_ArrayToPointerDecay:
135  case CK_AtomicToNonAtomic:
136  case CK_ConstructorConversion:
137  case CK_FunctionToPointerDecay:
138  case CK_NonAtomicToAtomic:
139  case CK_NoOp:
140  case CK_UserDefinedConversion:
141  case CK_NullToPointer:
142  return this->visit(SubExpr);
143 
144  case CK_IntegralToBoolean:
145  case CK_IntegralCast: {
146  std::optional<PrimType> FromT = classify(SubExpr->getType());
147  std::optional<PrimType> ToT = classify(CE->getType());
148  if (!FromT || !ToT)
149  return false;
150 
151  if (!this->visit(SubExpr))
152  return false;
153 
154  // TODO: Emit this only if FromT != ToT.
155  return this->emitCast(*FromT, *ToT, CE);
156  }
157 
158  case CK_ToVoid:
159  return discard(SubExpr);
160 
161  default:
162  assert(false && "Cast not implemented");
163  }
164  llvm_unreachable("Unhandled clang::CastKind enum");
165 }

188 template <class Emitter>
189 bool ByteCodeExprGen<Emitter>::VisitBinaryOperator(const BinaryOperator *BO) {
190  // Need short-circuiting for these.
191  if (BO->isLogicalOp())
192  return this->VisitLogicalBinOp(BO);
193 
194  const Expr *LHS = BO->getLHS();
195  const Expr *RHS = BO->getRHS();
196 
197  // Deal with operations which have composite or void types.
198  switch (BO->getOpcode()) {
199  case BO_Comma:
200  if (!discard(LHS))
201  return false;
202  if (!this->visit(RHS))
203  return false;
204  return true;
205  default:
206  break;
207  }
208 
209  // Typecheck the args.
210  std::optional<PrimType> LT = classify(LHS->getType());
211  std::optional<PrimType> RT = classify(RHS->getType());
212  std::optional<PrimType> T = classify(BO->getType());
213  if (!LT || !RT || !T) {
214  return this->bail(BO);
215  }
216 
217  auto Discard = [this, T, BO](bool Result) {
218  if (!Result)
219  return false;
220  return DiscardResult ? this->emitPop(*T, BO) : true;
221  };
222 
223  // Pointer arithmetic special case.
224  if (BO->getOpcode() == BO_Add || BO->getOpcode() == BO_Sub) {
225  if (*T == PT_Ptr || (*LT == PT_Ptr && *RT == PT_Ptr))
226  return this->VisitPointerArithBinOp(BO);
227  }
228 
229  if (!visit(LHS) || !visit(RHS))
230  return false;
231 
232  switch (BO->getOpcode()) {
233  case BO_EQ:
234  return Discard(this->emitEQ(*LT, BO));
235  case BO_NE:
236  return Discard(this->emitNE(*LT, BO));
237  case BO_LT:
238  return Discard(this->emitLT(*LT, BO));
239  case BO_LE:
240  return Discard(this->emitLE(*LT, BO));
241  case BO_GT:
242  return Discard(this->emitGT(*LT, BO));
243  case BO_GE:
244  return Discard(this->emitGE(*LT, BO));
245  case BO_Sub:
246  if (BO->getType()->isFloatingType())
247  return Discard(this->emitSubf(getRoundingMode(BO), BO));
248  return Discard(this->emitSub(*T, BO));
249  case BO_Add:
250  if (BO->getType()->isFloatingType())
251  return Discard(this->emitAddf(getRoundingMode(BO), BO));
252  return Discard(this->emitAdd(*T, BO));
253  case BO_Mul:
254  if (BO->getType()->isFloatingType())
255  return Discard(this->emitMulf(getRoundingMode(BO), BO));
256  return Discard(this->emitMul(*T, BO));
257  case BO_Rem:
258  return Discard(this->emitRem(*T, BO));
259  case BO_Div:
260  if (BO->getType()->isFloatingType())
261  return Discard(this->emitDivf(getRoundingMode(BO), BO));
262  return Discard(this->emitDiv(*T, BO));
263  case BO_Assign:
264  if (DiscardResult)
265  return this->emitStorePop(*T, BO);
266  return this->emitStore(*T, BO);
267  case BO_And:
268  return Discard(this->emitBitAnd(*T, BO));
269  case BO_Or:
270  return Discard(this->emitBitOr(*T, BO));
271  case BO_Shl:
272  return Discard(this->emitShl(*LT, *RT, BO));
273  case BO_Shr:
274  return Discard(this->emitShr(*LT, *RT, BO));
275  case BO_Xor:
276  return Discard(this->emitBitXor(*T, BO));
277  case BO_LOr:
278  case BO_LAnd:
279  llvm_unreachable("Already handled earlier");
280  default:
281  return this->bail(BO);
282  }
283 
284  llvm_unreachable("Unhandled binary op");
285 }

287 /// Perform addition/subtraction of a pointer and an integer or
288 /// subtraction of two pointers.
289 template <class Emitter>
290 bool ByteCodeExprGen<Emitter>::VisitPointerArithBinOp(const BinaryOperator *E) {
291  BinaryOperatorKind Op = E->getOpcode();
292  const Expr *LHS = E->getLHS();
293  const Expr *RHS = E->getRHS();
294 
295  if ((Op != BO_Add && Op != BO_Sub) ||
296  (!LHS->getType()->isPointerType() && !RHS->getType()->isPointerType()))
297  return false;
298 
299  std::optional<PrimType> LT = classify(LHS);
300  std::optional<PrimType> RT = classify(RHS);
301 
302  if (!LT || !RT)
303  return false;
304 
305  if (LHS->getType()->isPointerType() && RHS->getType()->isPointerType()) {
306  if (Op != BO_Sub)
307  return false;
308 
309  assert(E->getType()->isIntegerType());
310  if (!visit(RHS) || !visit(LHS))
311  return false;
312 
313  return this->emitSubPtr(classifyPrim(E->getType()), E);
314  }
315 
316  PrimType OffsetType;
317  if (LHS->getType()->isIntegerType()) {
318  if (!visit(RHS) || !visit(LHS))
319  return false;
320  OffsetType = *LT;
321  } else if (RHS->getType()->isIntegerType()) {
322  if (!visit(LHS) || !visit(RHS))
323  return false;
324  OffsetType = *RT;
325  } else {
326  return false;
327  }
328 
329  if (Op == BO_Add)
330  return this->emitAddOffset(OffsetType, E);
331  else if (Op == BO_Sub)
332  return this->emitSubOffset(OffsetType, E);
333 
334  return this->bail(E);
335 }

337 template <class Emitter>
338 bool ByteCodeExprGen<Emitter>::VisitLogicalBinOp(const BinaryOperator *E) {
339  assert(E->isLogicalOp());
340  BinaryOperatorKind Op = E->getOpcode();
341  const Expr *LHS = E->getLHS();
342  const Expr *RHS = E->getRHS();
343 
344  if (Op == BO_LOr) {
345  // Logical OR. Visit LHS and only evaluate RHS if LHS was FALSE.
346  LabelTy LabelTrue = this->getLabel();
347  LabelTy LabelEnd = this->getLabel();
348 
349  if (!this->visit(LHS))
350  return false;
351  if (!this->jumpTrue(LabelTrue))
352  return false;
353 
354  if (!this->visit(RHS))
355  return false;
356  if (!this->jump(LabelEnd))
357  return false;
358 
359  this->emitLabel(LabelTrue);
360  this->emitConstBool(true, E);
361  this->fallthrough(LabelEnd);
362  this->emitLabel(LabelEnd);
363 
364  if (DiscardResult)
365  return this->emitPopBool(E);
366 
367  return true;
368  }
369 
370  // Logical AND.
371  // Visit LHS. Only visit RHS if LHS was TRUE.
372  LabelTy LabelFalse = this->getLabel();
373  LabelTy LabelEnd = this->getLabel();
374 
375  if (!this->visit(LHS))
376  return false;
377  if (!this->jumpFalse(LabelFalse))
378  return false;
379 
380  if (!this->visit(RHS))
381  return false;
382  if (!this->jump(LabelEnd))
383  return false;
384 
385  this->emitLabel(LabelFalse);
386  this->emitConstBool(false, E);
387  this->fallthrough(LabelEnd);
388  this->emitLabel(LabelEnd);
389 
390  if (DiscardResult)
391  return this->emitPopBool(E);
392 
393  return true;
394 }

458 static CharUnits AlignOfType(QualType T, const ASTContext &ASTCtx,
459  UnaryExprOrTypeTrait Kind) {
460  bool AlignOfReturnsPreferred =
461  ASTCtx.getLangOpts().getClangABICompat() <= LangOptions::ClangABI::Ver7;
462 
463  // C++ [expr.alignof]p3:
464  // When alignof is applied to a reference type, the result is the
465  // alignment of the referenced type.
466  if (const auto *Ref = T->getAs<ReferenceType>())
467  T = Ref->getPointeeType();
468 
469  // __alignof is defined to return the preferred alignment.
470  // Before 8, clang returned the preferred alignment for alignof and
471  // _Alignof as well.
472  if (Kind == UETT_PreferredAlignOf || AlignOfReturnsPreferred)
473  return ASTCtx.toCharUnitsFromBits(ASTCtx.getPreferredTypeAlign(T));
474 
475  return ASTCtx.getTypeAlignInChars(T);
476 }

478 template <class Emitter>
479 bool ByteCodeExprGen<Emitter>::VisitUnaryExprOrTypeTraitExpr(
480  const UnaryExprOrTypeTraitExpr *E) {
481  UnaryExprOrTypeTrait Kind = E->getKind();
482  ASTContext &ASTCtx = Ctx.getASTContext();
483 
484  if (Kind == UETT_SizeOf) {
485  QualType ArgType = E->getTypeOfArgument();
486  CharUnits Size;
487  if (ArgType->isVoidType() || ArgType->isFunctionType())
488  Size = CharUnits::One();
489  else {
490  if (ArgType->isDependentType() || !ArgType->isConstantSizeType())
491  return false;
492 
493  Size = ASTCtx.getTypeSizeInChars(ArgType);
494  }
495 
496  return this->emitConst(Size.getQuantity(), E);
497  }
498 
499  if (Kind == UETT_AlignOf || Kind == UETT_PreferredAlignOf) {
500  CharUnits Size;
501 
502  if (E->isArgumentType()) {
503  QualType ArgType = E->getTypeOfArgument();
504 
505  Size = AlignOfType(ArgType, ASTCtx, Kind);
506  } else {
507  // Argument is an expression, not a type.
508  const Expr *Arg = E->getArgumentExpr()->IgnoreParens();
509 
510  // The kinds of expressions that we have special-case logic here for
511  // should be kept up to date with the special checks for those
512  // expressions in Sema.
513 
514  // alignof decl is always accepted, even if it doesn't make sense: we
515  // default to 1 in those cases.
516  if (const auto *DRE = dyn_cast<DeclRefExpr>(Arg))
517  Size = ASTCtx.getDeclAlign(DRE->getDecl(),
518  /*RefAsPointee*/ true);
519  else if (const auto *ME = dyn_cast<MemberExpr>(Arg))
520  Size = ASTCtx.getDeclAlign(ME->getMemberDecl(),
521  /*RefAsPointee*/ true);
522  else
523  Size = AlignOfType(Arg->getType(), ASTCtx, Kind);
524  }
525 
526  return this->emitConst(Size.getQuantity(), E);
527  }
528 
529  return false;
530 }

617 template <class Emitter>
618 bool ByteCodeExprGen<Emitter>::VisitFloatCompoundAssignOperator(
619  const CompoundAssignOperator *E) {
620  assert(E->getType()->isFloatingType());
621 
622  const Expr *LHS = E->getLHS();
623  const Expr *RHS = E->getRHS();
624  llvm::RoundingMode RM = getRoundingMode(E);
625  QualType LHSComputationType = E->getComputationLHSType();
626  QualType ResultType = E->getComputationResultType();
627  std::optional<PrimType> LT = classify(LHSComputationType);
628  std::optional<PrimType> RT = classify(ResultType);
629 
630  if (!LT || !RT)
631  return false;
632 
633  // First, visit LHS.
634  if (!visit(LHS))
635  return false;
636 
637  if (!this->emitLoad(*LT, E))
638  return false;
639 
640  // If necessary, convert LHS to its computation type.
641  if (LHS->getType() != LHSComputationType) {
642  const auto *TargetSemantics =
643  &Ctx.getASTContext().getFloatTypeSemantics(LHSComputationType);
644 
645  if (!this->emitCastFP(TargetSemantics, RM, E))
646  return false;
647  }
648 
649  // Now load RHS.
650  if (!visit(RHS))
651  return false;
652 
653  switch (E->getOpcode()) {
654  case BO_AddAssign:
655  if (!this->emitAddf(RM, E))
656  return false;
657  break;
658  case BO_SubAssign:
659  if (!this->emitSubf(RM, E))
660  return false;
661  break;
662  case BO_MulAssign:
663  if (!this->emitMulf(RM, E))
664  return false;
665  break;
666  case BO_DivAssign:
667  if (!this->emitDivf(RM, E))
668  return false;
669  break;
670  default:
671  return false;
672  }
673 
674  // If necessary, convert result to LHS's type.
675  if (LHS->getType() != ResultType) {
676  const auto *TargetSemantics =
677  &Ctx.getASTContext().getFloatTypeSemantics(LHS->getType());
678 
679  if (!this->emitCastFP(TargetSemantics, RM, E))
680  return false;
681  }
682 
683  if (DiscardResult)
684  return this->emitStorePop(*LT, E);
685  return this->emitStore(*LT, E);
686 }

723 template <class Emitter>
724 bool ByteCodeExprGen<Emitter>::VisitCompoundAssignOperator(
725  const CompoundAssignOperator *E) {
726 
727  // Handle floating point operations separately here, since they
728  // require special care.
729  if (E->getType()->isFloatingType())
730  return VisitFloatCompoundAssignOperator(E);
731 
732  if (E->getType()->isPointerType())
733  return VisitPointerCompoundAssignOperator(E);
734 
735  const Expr *LHS = E->getLHS();
736  const Expr *RHS = E->getRHS();
737  std::optional<PrimType> LHSComputationT =
738  classify(E->getComputationLHSType());
739  std::optional<PrimType> LT = classify(LHS->getType());
740  std::optional<PrimType> RT = classify(E->getComputationResultType());
741  std::optional<PrimType> ResultT = classify(E->getType());
742 
743  if (!LT || !RT || !ResultT || !LHSComputationT)
744  return false;
745 
746  assert(!E->getType()->isPointerType() && "Handled above");
747  assert(!E->getType()->isFloatingType() && "Handled above");
748 
749  // Get LHS pointer, load its value and get RHS value.
750  if (!visit(LHS))
751  return false;
752  if (!this->emitLoad(*LT, E))
753  return false;
754  // If necessary, cast LHS to its computation type.
755  if (*LT != *LHSComputationT) {
756  if (!this->emitCast(*LT, *LHSComputationT, E))
757  return false;
758  }
759 
760  if (!visit(RHS))
761  return false;
762 
763  // Perform operation.
764  switch (E->getOpcode()) {
765  case BO_AddAssign:
766  if (!this->emitAdd(*LHSComputationT, E))
767  return false;
768  break;
769  case BO_SubAssign:
770  if (!this->emitSub(*LHSComputationT, E))
771  return false;
772  break;
773  case BO_MulAssign:
774  if (!this->emitMul(*LHSComputationT, E))
775  return false;
776  break;
777  case BO_DivAssign:
778  if (!this->emitDiv(*LHSComputationT, E))
779  return false;
780  break;
781  case BO_RemAssign:
782  if (!this->emitRem(*LHSComputationT, E))
783  return false;
784  break;
785  case BO_ShlAssign:
786  if (!this->emitShl(*LHSComputationT, *RT, E))
787  return false;
788  break;
789  case BO_ShrAssign:
790  if (!this->emitShr(*LHSComputationT, *RT, E))
791  return false;
792  break;
793  case BO_AndAssign:
794  if (!this->emitBitAnd(*LHSComputationT, E))
795  return false;
796  break;
797  case BO_XorAssign:
798  if (!this->emitBitXor(*LHSComputationT, E))
799  return false;
800  break;
801  case BO_OrAssign:
802  if (!this->emitBitOr(*LHSComputationT, E))
803  return false;
804  break;
805  default:
806  llvm_unreachable("Unimplemented compound assign operator");
807  }
808 
809  // And now cast from LHSComputationT to ResultT.
810  if (*ResultT != *LHSComputationT) {
811  if (!this->emitCast(*LHSComputationT, *ResultT, E))
812  return false;
813  }
814 
815  // And store the result in LHS.
816  if (DiscardResult)
817  return this->emitStorePop(*ResultT, E);
818  return this->emitStore(*ResultT, E);
819 }

1232  if (!this->emitConstUint64(I, SubExpr))

1685 bool ByteCodeExprGen<Emitter>::VisitUnaryOperator(const UnaryOperator *E) {

1687  std::optional<PrimType> T = classify(SubExpr->getType());
1688

1690  case UO_PostInc: { // x++
1691  if (!this->visit(SubExpr))
1692  return false;

1695  if (!this->emitIncPtr(E))
1696  return false;
1697 
1698  return DiscardResult ? this->emitPopPtr(E) : true;
1699  }
1700 
1701  return DiscardResult ? this->emitIncPop(*T, E) : this->emitInc(*T, E);
1702  }
1703  case UO_PostDec: { // x--
1704  if (!this->visit(SubExpr))
1705  return false;
1706 
1707  if (T == PT_Ptr) {
1708  if (!this->emitDecPtr(E))
1709  return false;
1710

1712  }
1713 
1714  return DiscardResult ? this->emitDecPop(*T, E) : this->emitDec(*T, E);
1715  }
1716  case UO_PreInc: { // ++x
1717  if (!this->visit(SubExpr))
1718  return false;
1719 
1720  if (T == PT_Ptr) {
1721  this->emitLoadPtr(E);
1722  this->emitConstUint8(1, E);
1723  this->emitAddOffsetUint8(E);
1724  return DiscardResult ? this->emitStorePopPtr(E) : this->emitStorePtr(E);
1725  }
1726 
1727  // Post-inc and pre-inc are the same if the value is to be discarded.
1728  if (DiscardResult)
1729  return this->emitIncPop(*T, E);
1730 
1731  this->emitLoad(*T, E);
1732  this->emitConst(1, E);
1733  this->emitAdd(*T, E);
1734  return this->emitStore(*T, E);
1735  }
1736  case UO_PreDec: { // --x
1737  if (!this->visit(SubExpr))
1738  return false;
1739 
1740  if (T == PT_Ptr) {
1741  this->emitLoadPtr(E);
1742  this->emitConstUint8(1, E);
1743  this->emitSubOffsetUint8(E);
1744  return DiscardResult ? this->emitStorePopPtr(E) : this->emitStorePtr(E);
1745  }
1746 
1747  // Post-dec and pre-dec are the same if the value is to be discarded.
1748  if (DiscardResult)
1749  return this->emitDecPop(*T, E);
1750 
1751  this->emitLoad(*T, E);
1752  this->emitConst(1, E);
1753  this->emitSub(*T, E);
1754  return this->emitStore(*T, E);
1755  }
1756  case UO_LNot: // !x
1757  if (!this->visit(SubExpr))
1758  return false;
1759  // The Inv doesn't change anything, so skip it if we don't need the result.
1760  return DiscardResult ? this->emitPop(*T, E) : this->emitInvBool(E);
1761  case UO_Minus: // -x
1762  if (!this->visit(SubExpr))
1763  return false;
1764  return DiscardResult ? this->emitPop(*T, E) : this->emitNeg(*T, E);
1765  case UO_Plus: // +x
1766  if (!this->visit(SubExpr)) // noop
1767  return false;
1768  return DiscardResult ? this->emitPop(*T, E) : true;
1769  case UO_AddrOf: // &x
1770  // We should already have a pointer when we get here.
1771  if (!this->visit(SubExpr))
1772  return false;
1773  return DiscardResult ? this->emitPop(*T, E) : true;
1774  case UO_Deref: // *x
1775  return dereference(
1776  SubExpr, DerefKind::Read,
1777  [](PrimType) {
1778  llvm_unreachable("Dereferencing requires a pointer");
1779  return false;
1780  },
1781  [this, E](PrimType T) {
1782  return DiscardResult ? this->emitPop(T, E) : true;
1783  });
1784  case UO_Not: // ~x
1785  if (!this->visit(SubExpr))
1786  return false;

1788  case UO_Real: // __real x
1789  case UO_Imag: // __imag x
1790  case UO_Extension:
1791  case UO_Coawait:
1792  assert(false && "Unhandled opcode");
1793  }
1794

1796 }
