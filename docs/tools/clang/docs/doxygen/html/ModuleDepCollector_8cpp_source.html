23 static void optimizeHeaderSearchOpts(HeaderSearchOptions &Opts,
24  ASTReader &Reader,
25  const serialization::ModuleFile &MF) {
26  // Only preserve search paths that were used during the dependency scan.
27  std::vector<HeaderSearchOptions::Entry> Entries = Opts.UserEntries;
28  Opts.UserEntries.clear();
29 
30  llvm::BitVector SearchPathUsage(Entries.size());
31  llvm::DenseSet<const serialization::ModuleFile *> Visited;
32  std::function<void(const serialization::ModuleFile *)> VisitMF =
33  [&](const serialization::ModuleFile *MF) {
34  SearchPathUsage |= MF->SearchPathUsage;
35  Visited.insert(MF);
36  for (const serialization::ModuleFile *Import : MF->Imports)
37  if (!Visited.contains(Import))
38  VisitMF(Import);
39  };
40  VisitMF(&MF);
41 
42  for (auto Idx : SearchPathUsage.set_bits())
43  Opts.UserEntries.push_back(Entries[Idx]);
44 }

81 CompilerInvocation
82 ModuleDepCollector::makeInvocationForModuleBuildWithoutOutputs(
83  const ModuleDeps &Deps,
84  llvm::function_ref<void(CompilerInvocation &)> Optimize) const {
85  // Make a deep copy of the original Clang invocation.
86  CompilerInvocation CI(OriginalInvocation);
87 
88  CI.resetNonModularOptions();
89  CI.clearImplicitModuleBuildOptions();
90 
91  // Remove options incompatible with explicit module build or are likely to
92  // differ between identical modules discovered from different translation
93  // units.
94  CI.getFrontendOpts().Inputs.clear();
95  CI.getFrontendOpts().OutputFile.clear();
96 
97  // TODO: Figure out better way to set options to their default value.
98  CI.getCodeGenOpts().MainFileName.clear();
99  CI.getCodeGenOpts().DwarfDebugFlags.clear();
100  if (!CI.getLangOpts()->ModulesCodegen) {
101  CI.getCodeGenOpts().DebugCompilationDir.clear();
102  CI.getCodeGenOpts().CoverageCompilationDir.clear();
103  }
104 
105  // Map output paths that affect behaviour to "-" so their existence is in the
106  // context hash. The final path will be computed in addOutputPaths.
107  if (!CI.getDiagnosticOpts().DiagnosticSerializationFile.empty())
108  CI.getDiagnosticOpts().DiagnosticSerializationFile = "-";
109  if (!CI.getDependencyOutputOpts().OutputFile.empty())
110  CI.getDependencyOutputOpts().OutputFile = "-";
111  CI.getDependencyOutputOpts().Targets.clear();
112 
113  CI.getFrontendOpts().ProgramAction = frontend::GenerateModule;
114  CI.getFrontendOpts().ARCMTAction = FrontendOptions::ARCMT_None;
115  CI.getFrontendOpts().ObjCMTAction = FrontendOptions::ObjCMT_None;
116  CI.getFrontendOpts().MTMigrateDir.clear();
117  CI.getLangOpts()->ModuleName = Deps.ID.ModuleName;
118  CI.getFrontendOpts().IsSystemModule = Deps.IsSystem;
119 
120  // Inputs
121  InputKind ModuleMapInputKind(CI.getFrontendOpts().DashX.getLanguage(),
122  InputKind::Format::ModuleMap);
123  CI.getFrontendOpts().Inputs.emplace_back(Deps.ClangModuleMapFile,
124  ModuleMapInputKind);
125 
126  auto CurrentModuleMapEntry =
127  ScanInstance.getFileManager().getFile(Deps.ClangModuleMapFile);
128  assert(CurrentModuleMapEntry && "module map file entry not found");
129 
130  auto DepModuleMapFiles = collectModuleMapFiles(Deps.ClangModuleDeps);
131  for (StringRef ModuleMapFile : Deps.ModuleMapFileDeps) {
132  // TODO: Track these as `FileEntryRef` to simplify the equality check below.
133  auto ModuleMapEntry = ScanInstance.getFileManager().getFile(ModuleMapFile);
134  assert(ModuleMapEntry && "module map file entry not found");
135 
136  // Don't report module maps describing eagerly-loaded dependency. This
137  // information will be deserialized from the PCM.
138  // TODO: Verify this works fine when modulemap for module A is eagerly
139  // loaded from A.pcm, and module map passed on the command line contains
140  // definition of a submodule: "explicit module A.Private { ... }".
141  if (EagerLoadModules && DepModuleMapFiles.contains(*ModuleMapEntry))
142  continue;
143 
144  // Don't report module map file of the current module unless it also
145  // describes a dependency (for symmetry).
146  if (*ModuleMapEntry == *CurrentModuleMapEntry &&
147  !DepModuleMapFiles.contains(*ModuleMapEntry))
148  continue;
149 
150  CI.getFrontendOpts().ModuleMapFiles.emplace_back(ModuleMapFile);
151  }
152 
153  // Report the prebuilt modules this module uses.
154  for (const auto &PrebuiltModule : Deps.PrebuiltModuleDeps)
155  CI.getFrontendOpts().ModuleFiles.push_back(PrebuiltModule.PCMFile);
156 
157  // Add module file inputs from dependencies.
158  addModuleFiles(CI, Deps.ClangModuleDeps);
159 
160  // Remove any macro definitions that are explicitly ignored.
161  if (!CI.getHeaderSearchOpts().ModulesIgnoreMacros.empty()) {
162  llvm::erase_if(
163  CI.getPreprocessorOpts().Macros,
164  [&CI](const std::pair<std::string, bool> &Def) {
165  StringRef MacroDef = Def.first;
166  return CI.getHeaderSearchOpts().ModulesIgnoreMacros.contains(
167  llvm::CachedHashString(MacroDef.split('=').first));
168  });
169  // Remove the now unused option.
170  CI.getHeaderSearchOpts().ModulesIgnoreMacros.clear();
171  }
172 
173  Optimize(CI);
174 
175  return CI;
176 }

228 void ModuleDepCollector::applyDiscoveredDependencies(CompilerInvocation &CI) {
229  CI.clearImplicitModuleBuildOptions();
230 
231  if (llvm::any_of(CI.getFrontendOpts().Inputs, needsModules)) {
232  Preprocessor &PP = ScanInstance.getPreprocessor();
233  if (Module *CurrentModule = PP.getCurrentModuleImplementation())
234  if (OptionalFileEntryRef CurrentModuleMap =
235  PP.getHeaderSearchInfo()
236  .getModuleMap()
237  .getModuleMapFileForUniquing(CurrentModule))
238  CI.getFrontendOpts().ModuleMapFiles.emplace_back(
239  CurrentModuleMap->getNameAsRequested());
240 
241  SmallVector<ModuleID> DirectDeps;
242  for (const auto &KV : ModularDeps)
243  if (KV.second->ImportedByMainFile)
244  DirectDeps.push_back(KV.second->ID);
245 
246  // TODO: Report module maps the same way it's done for modular dependencies.
247  addModuleMapFiles(CI, DirectDeps);
248 
249  addModuleFiles(CI, DirectDeps);
250 
251  for (const auto &KV : DirectPrebuiltModularDeps)
252  CI.getFrontendOpts().ModuleFiles.push_back(KV.second.PCMFile);
253  }

256 static std::string getModuleContextHash(const ModuleDeps &MD,
257  const CompilerInvocation &CI,
258  bool EagerLoadModules) {
259  llvm::HashBuilder<llvm::TruncatedBLAKE3<16>,
260  llvm::support::endianness::native>
261  HashBuilder;
262  SmallString<32> Scratch;
263 
264  // Hash the compiler version and serialization version to ensure the module
265  // will be readable.
266  HashBuilder.add(getClangFullRepositoryVersion());
267  HashBuilder.add(serialization::VERSION_MAJOR, serialization::VERSION_MINOR);
268 
269  // Hash the BuildInvocation without any input files.
270  SmallVector<const char *, 32> Args;
271  llvm::BumpPtrAllocator Alloc;
272  llvm::StringSaver Saver(Alloc);
273  CI.generateCC1CommandLine(
274  Args, [&](const Twine &Arg) { return Saver.save(Arg).data(); });
275  HashBuilder.addRange(Args);
276 
277  // Hash the module dependencies. These paths may differ even if the invocation
278  // is identical if they depend on the contents of the files in the TU -- for
279  // example, case-insensitive paths to modulemap files. Usually such a case
280  // would indicate a missed optimization to canonicalize, but it may be
281  // difficult to canonicalize all cases when there is a VFS.
282  for (const auto &ID : MD.ClangModuleDeps) {
283  HashBuilder.add(ID.ModuleName);
284  HashBuilder.add(ID.ContextHash);
285  }
286 
287  HashBuilder.add(EagerLoadModules);
288 
289  llvm::BLAKE3Result<16> Hash = HashBuilder.final();
290  std::array<uint64_t, 2> Words;
291  static_assert(sizeof(Hash) == sizeof(Words), "Hash must match Words");
292  std::memcpy(Words.data(), Hash.data(), sizeof(Hash));
293  return toString(llvm::APInt(sizeof(Words) * 8, Words), 36, /*Signed=*/false);
294 }

368 void ModuleDepCollectorPP::EndOfMainFile() {
369  FileID MainFileID = MDC.ScanInstance.getSourceManager().getMainFileID();
370  MDC.MainFile = std::string(MDC.ScanInstance.getSourceManager()
371  .getFileEntryForID(MainFileID)
372  ->getName());
373 
374  auto &PP = MDC.ScanInstance.getPreprocessor();
375  if (PP.isInNamedModule()) {
376  P1689ModuleInfo ProvidedModule;
377  ProvidedModule.ModuleName = PP.getNamedModuleName();
378  ProvidedModule.Type = P1689ModuleInfo::ModuleType::NamedCXXModule;
379  ProvidedModule.IsStdCXXModuleInterface = PP.isInNamedInterfaceUnit();
380  // Don't put implementation (non partition) unit as Provide.
381  // Put the module as required instead. Since the implementation
382  // unit will import the primary module implicitly.
383  if (PP.isInImplementationUnit())
384  MDC.RequiredStdCXXModules.push_back(ProvidedModule);
385  else
386  MDC.ProvidedStdCXXModule = ProvidedModule;
387  }
388 
389  if (!MDC.ScanInstance.getPreprocessorOpts().ImplicitPCHInclude.empty())
390  MDC.addFileDep(MDC.ScanInstance.getPreprocessorOpts().ImplicitPCHInclude);
391 
392  for (const Module *M :
393  MDC.ScanInstance.getPreprocessor().getAffectingClangModules())
394  if (!MDC.isPrebuiltModule(M))
395  DirectModularDeps.insert(M);
396 
397  for (const Module *M : DirectModularDeps)
398  handleTopLevelModule(M);
399 
400  MDC.Consumer.handleDependencyOutputOpts(*MDC.Opts);
401 
402  if (MDC.IsStdModuleP1689Format)
403  MDC.Consumer.handleProvidedAndRequiredStdCXXModules(
404  MDC.ProvidedStdCXXModule, MDC.RequiredStdCXXModules);
405 
406  for (auto &&I : MDC.ModularDeps)
407  MDC.Consumer.handleModuleDependency(*I.second);
408 
409  for (auto &&I : MDC.FileDeps)
410  MDC.Consumer.handleFileDependency(I);
411 
412  for (auto &&I : MDC.DirectPrebuiltModularDeps)
413  MDC.Consumer.handlePrebuiltModuleDependency(I.second);
414 }

416 std::optional<ModuleID>
417 ModuleDepCollectorPP::handleTopLevelModule(const Module *M) {
418  assert(M == M->getTopLevelModule() && "Expected top level module!");
419 
420  // A top-level module might not be actually imported as a module when
421  // -fmodule-name is used to compile a translation unit that imports this
422  // module. In that case it can be skipped. The appropriate header
423  // dependencies will still be reported as expected.
424  if (!M->getASTFile())
425  return {};
426 
427  // If this module has been handled already, just return its ID.
428  auto ModI = MDC.ModularDeps.insert({M, nullptr});
429  if (!ModI.second)
430  return ModI.first->second->ID;
431 
432  ModI.first->second = std::make_unique<ModuleDeps>();
433  ModuleDeps &MD = *ModI.first->second;
434 
435  MD.ID.ModuleName = M->getFullModuleName();
436  MD.ImportedByMainFile = DirectModularDeps.contains(M);
437  MD.IsSystem = M->IsSystem;
438 
439  ModuleMap &ModMapInfo =
440  MDC.ScanInstance.getPreprocessor().getHeaderSearchInfo().getModuleMap();
441 
442  OptionalFileEntryRef ModuleMap = ModMapInfo.getModuleMapFileForUniquing(M);
443 
444  if (ModuleMap) {
445  SmallString<128> Path = ModuleMap->getNameAsRequested();
446  ModMapInfo.canonicalizeModuleMapPath(Path);
447  MD.ClangModuleMapFile = std::string(Path);
448  }
449 
450  serialization::ModuleFile *MF =
451  MDC.ScanInstance.getASTReader()->getModuleManager().lookup(
452  M->getASTFile());
453  MDC.ScanInstance.getASTReader()->visitInputFiles(
454  *MF, true, true, [&](const serialization::InputFile &IF, bool isSystem) {
455  // __inferred_module.map is the result of the way in which an implicit
456  // module build handles inferred modules. It adds an overlay VFS with
457  // this file in the proper directory and relies on the rest of Clang to
458  // handle it like normal. With explicitly built modules we don't need
459  // to play VFS tricks, so replace it with the correct module map.
460  if (IF.getFile()->getName().endswith("__inferred_module.map")) {
461  MDC.addFileDep(MD, ModuleMap->getName());
462  return;
463  }
464  MDC.addFileDep(MD, IF.getFile()->getName());
465  });
466 
467  llvm::DenseSet<const Module *> SeenDeps;
468  addAllSubmodulePrebuiltDeps(M, MD, SeenDeps);
469  addAllSubmoduleDeps(M, MD, SeenDeps);
470  addAllAffectingClangModules(M, MD, SeenDeps);
471 
472  MDC.ScanInstance.getASTReader()->visitTopLevelModuleMaps(
473  *MF, [&](FileEntryRef FE) {
474  if (FE.getNameAsRequested().endswith("__inferred_module.map"))
475  return;
476  MD.ModuleMapFileDeps.emplace_back(FE.getNameAsRequested());
477  });
478 
479  CompilerInvocation CI = MDC.makeInvocationForModuleBuildWithoutOutputs(
480  MD, [&](CompilerInvocation &BuildInvocation) {
481  if (MDC.OptimizeArgs)
482  optimizeHeaderSearchOpts(BuildInvocation.getHeaderSearchOpts(),
483  *MDC.ScanInstance.getASTReader(), *MF);
484  });
485 
486  MDC.associateWithContextHash(CI, MD);
487 
488  // Finish the compiler invocation. Requires dependencies and the context hash.
489  MDC.addOutputPaths(CI, MD);
490 
491  MD.BuildArguments = CI.getCC1CommandLine();
492 
493  return MD.ID;
494 }

496 static void forEachSubmoduleSorted(const Module *M,
497  llvm::function_ref<void(const Module *)> F) {
498  // Submodule order depends on order of header includes for inferred submodules
499  // we don't care about the exact order, so sort so that it's consistent across
500  // TUs to improve sharing.
501  SmallVector<const Module *> Submodules(M->submodule_begin(),
502  M->submodule_end());
503  llvm::stable_sort(Submodules, [](const Module *A, const Module *B) {
504  return A->Name < B->Name;
505  });
506  for (const Module *SubM : Submodules)
507  F(SubM);
508 }

562 void ModuleDepCollectorPP::addAffectingClangModule(
563  const Module *M, ModuleDeps &MD,
564  llvm::DenseSet<const Module *> &AddedModules) {
565  for (const Module *Affecting : M->AffectingClangModules) {
